  endforeach()

  endforeach()
endif()

if(BUILD_DOTNET_SAMPLES)
  file(GLOB DOTNET_SRCS "*.cs")
  foreach(SAMPLE IN LISTS DOTNET_SRCS)
    add_dotnet_sample(${SAMPLE})
  endforeach()
endif()
