14 // Minimal example to call the GLOP solver.
15 // [START program]
16 using System;
17 using Google.OrTools.LinearSolver;
18 
19 public class SimpleLpProgram

26  // [END solver]
27 
28  // [START variables]
29  // Create the variables x and y.

32

34  // [END variables]
35

41

44
