128 public long Value(LinearExpr e)
129 {
130 List<LinearExpr> exprs = new List<LinearExpr>();
131 List<long> coeffs = new List<long>();
132 exprs.Add(e);
133 coeffs.Add(1L);
134 long constant = 0;
135
136 while (exprs.Count > 0)
137 {
138 LinearExpr expr = exprs[0];
139 exprs.RemoveAt(0);
140 long coeff = coeffs[0];
141 coeffs.RemoveAt(0);
142 if (coeff == 0)
143 continue;
144
145 if (expr is ProductCst)
146 {
147 ProductCst p = (ProductCst)expr;
148 if (p.Coeff != 0)
149 {
150 exprs.Add(p.Expr);
151 coeffs.Add(p.Coeff * coeff);
152 }
153 }
154 else if (expr is SumArray)
155 {
156 SumArray a = (SumArray)expr;
157 constant += coeff * a.Offset;
158 foreach (LinearExpr sub in a.Expressions)
159 {
160 exprs.Add(sub);
161 coeffs.Add(coeff);
162 }
163 }
164 else if (expr is IntVar)
165 {
166 int index = expr.Index;
167 long value = index >= 0 ? response_.Solution[index] : -response_.Solution[-index - 1];
168 constant += coeff * value;
169 }
170 else if (expr is NotBooleanVariable)
171 {
172 throw new ArgumentException("Cannot evaluate a literal in an integer expression.");
173 }
174 else
175 {
176 throw new ArgumentException("Cannot evaluate '" + expr.ToString() + "' in an integer expression");
177 }
178 }
179 return constant;
180 }

182 public Boolean BooleanValue(ILiteral literal)
183 {

185 {
186 int index = literal.GetIndex();
187 if (index >= 0)

190 }
191 else
192 {
193 return response_.Solution[-index - 1] == 0;

195 }
