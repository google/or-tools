89class BasePathFilter : public IntVarLocalSearchFilter {
90 public:
91 BasePathFilter(const std::vector<IntVar*>& nexts, int next_domain_size);
92 ~BasePathFilter() override {}
93 bool Accept(const Assignment* delta, const Assignment* deltadelta,
94 int64_t objective_min, int64_t objective_max) override;
95 void OnSynchronize(const Assignment* delta) override;
96
97 protected:
98 static const int64_t kUnassigned;
99
100 int64_t GetNext(int64_t node) const {
101 return (new_nexts_[node] == kUnassigned)
102 ? (IsVarSynced(node) ? Value(node) : kUnassigned)
103 : new_nexts_[node];
104 }
105 int NumPaths() const { return starts_.size(); }
106 int64_t Start(int i) const { return starts_[i]; }
107 int GetPath(int64_t node) const { return paths_[node]; }
108 int Rank(int64_t node) const { return ranks_[node]; }
109 bool IsDisabled() const { return status_ == DISABLED; }
110 const std::vector<int64_t>& GetTouchedPathStarts() const {
111 return touched_paths_.PositionsSetAtLeastOnce();
112 }
113 const std::vector<int64_t>& GetNewSynchronizedUnperformedNodes() const {
114 return new_synchronized_unperformed_nodes_.PositionsSetAtLeastOnce();
115 }
116
117 private:
118 enum Status { UNKNOWN, ENABLED, DISABLED };
119
120 virtual bool DisableFiltering() const { return false; }
121 virtual void OnBeforeSynchronizePaths() {}
122 virtual void OnAfterSynchronizePaths() {}
123 virtual void OnSynchronizePathFromStart(int64_t start) {}
124 virtual void InitializeAcceptPath() {}
125 virtual bool AcceptPath(int64_t path_start, int64_t chain_start,
126 int64_t chain_end) = 0;
127 virtual bool FinalizeAcceptPath(const Assignment* delta,
128 int64_t objective_min,
129 int64_t objective_max) {
130 return true;
131 }
133 void ComputePathStarts(std::vector<int64_t>* path_starts,
134 std::vector<int>* index_to_path);
135 bool HavePathsChanged();
136 void SynchronizeFullAssignment();
137 void UpdateAllRanks();
138 void UpdatePathRanksFromStart(int start);
139
140 std::vector<int64_t> node_path_starts_;
141 std::vector<int64_t> starts_;
142 std::vector<int> paths_;
143 SparseBitset<int64_t> new_synchronized_unperformed_nodes_;
144 std::vector<int64_t> new_nexts_;
145 std::vector<int> delta_touched_;
146 SparseBitset<> touched_paths_;
147 // clang-format off
148 std::vector<std::pair<int64_t, int64_t> > touched_path_chain_start_ends_;
149 // clang-format on
150 std::vector<int> ranks_;
151
152 Status status_;
153};
