72 // For easing reading the code, we declare these types:

80 static constexpr double kInf = std::numeric_limits<double>::infinity();
81 // Data associated with each constraint. With it we know if the underlying
82 // representation is either:
83 // linear_terms <= upper_bound (if lower bound <= -GRB_INFINITY)
84 // linear_terms >= lower_bound (if upper bound >= GRB_INFINTY)
85 // linear_terms == xxxxx_bound (if upper_bound == lower_bound)
86 // linear_term - slack == 0 (with slack bounds equal to xxxxx_bound)
87 struct ConstraintData {
88 GurobiLinearConstraintIndex constraint_index = kUnspecifiedConstraint;
89 // only valid for true ranged constraints.
90 GurobiVariableIndex slack_index = kUnspecifiedIndex;
91 double lower_bound = -kInf;
92 double upper_bound = kInf;
93 };

175 // Note: Gurobi environments CAN be shared across several models, however
176 // there are some caveats:
177 // - Environments are not thread-safe.
178 // - Once a gurobi_model_ is created, it makes an internal copy of the
179 // "master" environment, so, later changes to that environment will not
180 // be reflected in the gurobi_model_, for that reason we also keep
181 // around a pointer to the gurobi_model_ environment in the
182 // `active_env_` (which should not be freed).
183 // - Every "master" environment counts as a "use" of a Gurobi License.
184 // This means that if you have a limited usage count of licenses, this
185 // implementation will be consuming more licenses. On the other hand, if
186 // you have a machine license, a site license, or an academic license,
187 // this disadvantage goes away.
188 //
189 // TODO(user) implement a sharing master Gurobi environment mode.
190 // This would be akin to the `default environment` of Gurobi in python.
191 GRBenv* master_env_ = nullptr;
192 GRBenv* active_env_ = nullptr;
193 GRBmodel* gurobi_model_ = nullptr;
194 // Note that we use linked_hash_map because the index of the gurobi_model_
195 // variables/constraints is exactly the order in which they are added to the
196 // model.
197 // Internal correspondence from variable proto IDs to Gurobi-numbered
198 // variables.
199 gtl::linked_hash_map<VariableId, GurobiVariableIndex> variables_map_;
200 // Internal correspondence from linear constraint proto IDs to
201 // Gurobi-numbered linear constraint and extra information.
202 gtl::linked_hash_map<LinearConstraintId, ConstraintData>
203 linear_constraints_map_;
204 // For those constraints that need a slack in the gurobi_model_
205 // representation (i.e. those with
206 // ConstraintData.slack_index != kUnspecifiedIndex), we keep a hash_map of
207 // those LinearConstraintId's to a reference of the ConstraintData stored in
208 // the linear_constraints_map_. This means that we only have one location
209 // were we store information about constraints, but two places from where we
210 // can access it.
211 // Furthermore, the internal slack_index associated with each
212 // actual slack variable is increasing in the order of them in this
213 // slack_map_, and is the main reason why we choose to use
214 // gtl::linked_hash_map, as the order of insertion (of the remaining
215 // elements) is preserved after removals.
216 // Note that after deletions, the renumbering of variables must be done
217 // simultaneously for variables and slacks following the merged-order of the
218 // variable_index and slack_map_.slack_index.
219 gtl::linked_hash_map<LinearConstraintId, ConstraintData&> slack_map_;
220 // Number of Gurobi variables, this internal quantity is updated when we
221 // actually add or remove variables or constraints in the underlying Gurobi
222 // model. Furthermore, since 'ModelUpdate' can trigger both creation and
223 // deletion of variables and constraints, we batch those changes in two
224 // steps: first add all new variables and constraints, and then delete all
225 // variables and constraints that need deletion. Finally flush changes at
226 // the gurobi model level (if any deletion was performed).
227 int num_gurobi_variables_ = 0;
228
229 static constexpr int kGrbBasicConstraint = 0;
230 static constexpr int kGrbNonBasicConstraint = -1;
231};
