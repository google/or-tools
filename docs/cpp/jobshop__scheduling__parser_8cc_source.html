54 bool JsspParser::ParseFile(const std::string& filename) {
55  problem_.Clear();
56  // Try to detect the type of the data file.
57  // - fjs suffix -> Flexible Jobshop
58  // - txt suffix -> Taillard or time dependent scheduling.
59  if (absl::EndsWith(filename, "fjs")) {
60  problem_type_ = FLEXIBLE;
61  } else if (absl::EndsWith(filename, ".txt")) {
62  problem_type_ = TAILLARD;
63  } else {
64  problem_type_ = JSSP;
65  }
66  for (const std::string& line : FileLines(filename)) {
67  if (line.empty()) {
68  continue;
69  }
70  switch (problem_type_) {
71  case JSSP: {
72  ProcessJsspLine(line);
73  break;
74  }
75  case TAILLARD: {
76  ProcessTaillardLine(line);
77  break;
78  }
79  case FLEXIBLE: {
80  ProcessFlexibleLine(line);
81  break;
82  }
83  case SDST: {
84  ProcessSdstLine(line);
85  break;
86  }
87  case TARDINESS: {
88  ProcessTardinessLine(line);
89  break;
90  }
91  case PSS: {
92  ProcessPssLine(line);
93  break;
94  }
95  case EARLY_TARDY: {
96  ProcessEarlyTardyLine(line);
97  break;
98  }
99  default: {
100  LOG(FATAL) << "Should not be here.";
101  break;
102  }
103  }
104  }
105  return parser_state_ != PARSING_ERROR;
106 }

108 void JsspParser::ProcessJsspLine(const std::string& line) {
109  const std::vector<std::string> words =
110  absl::StrSplit(line, ' ', absl::SkipEmpty());
111  switch (parser_state_) {
112  case START: {
113  if (words.size() == 2 && words[0] == "instance") {
114  problem_.set_name(words[1]);
115  parser_state_ = NAME_READ;
116  current_job_index_ = 0;
117  } else if (words.size() == 1 && words[0] == "1") {
118  problem_type_ = PSS;
119  } else if (words.size() == 2) {
120  SetJobs(strtoint32(words[0]));
121  SetMachines(strtoint32(words[1]));
122  problem_type_ = EARLY_TARDY;
123  parser_state_ = JOB_COUNT_READ;
124  }
125  break;
126  }
127  case NAME_READ: {
128  if (words.size() == 2) {
129  SetJobs(strtoint32(words[0]));
130  SetMachines(strtoint32(words[1]));
131  problem_.set_makespan_cost_per_time_unit(1L);
132  parser_state_ = JOB_COUNT_READ;
133  }
134  break;
135  }
136  case JOB_COUNT_READ: {
137  CHECK_GE(words.size(), declared_machine_count_ * 2);
138  Job* const job = problem_.mutable_jobs(current_job_index_);
139  for (int i = 0; i < declared_machine_count_; ++i) {
140  const int machine_id = strtoint32(words[2 * i]);
141  const int64_t duration = strtoint64(words[2 * i + 1]);
142  Task* const task = job->add_tasks();
143  task->add_machine(machine_id);
144  task->add_duration(duration);
145  }
146  if (words.size() == declared_machine_count_ * 2 + 3) {
147  // Early Tardy problem in JET format.
148  const int due_date = strtoint32(words[declared_machine_count_ * 2]);
149  const int early_cost =
150  strtoint32(words[declared_machine_count_ * 2 + 1]);
151  const int late_cost =
152  strtoint32(words[declared_machine_count_ * 2 + 2]);
153  job->set_early_due_date(due_date);
154  job->set_late_due_date(due_date);
155  job->set_earliness_cost_per_time_unit(early_cost);
156  job->set_lateness_cost_per_time_unit(late_cost);
157  }
158  current_job_index_++;
159  if (current_job_index_ == declared_job_count_) {
160  parser_state_ = DONE;
161  }
162  break;
163  }
164  default: {
165  LOG(FATAL) << "Should not be here with state " << parser_state_;
166  }
167  }

170 void JsspParser::ProcessTaillardLine(const std::string& line) {
171  const std::vector<std::string> words =
172  absl::StrSplit(line, ' ', absl::SkipEmpty());
173 
174  switch (parser_state_) {
175  case START: {
176  if (words.size() == 2) { // Switch to SDST parser.
177  problem_type_ = SDST;
178  ProcessSdstLine(line);
179  return;
180  } else if (words.size() == 3) { // Switch to TARDINESS parser.
181  problem_type_ = TARDINESS;
182  ProcessTardinessLine(line);
183  return;
184  }
185  if (words.size() == 1 && strtoint32(words[0]) > 0) {
186  parser_state_ = JOB_COUNT_READ;
187  SetJobs(strtoint32(words[0]));
188  }
189  break;
190  }
191  case JOB_COUNT_READ: {
192  CHECK_EQ(1, words.size());
193  SetMachines(strtoint32(words[0]));
194  problem_.set_makespan_cost_per_time_unit(1L);
195  parser_state_ = MACHINE_COUNT_READ;
196  break;
197  }
198  case MACHINE_COUNT_READ: {
199  CHECK_EQ(1, words.size());
200  const int seed = strtoint32(words[0]);
201  problem_.set_seed(seed);
202  parser_state_ = SEED_READ;
203  break;
204  }
205  case SEED_READ:
206  ABSL_FALLTHROUGH_INTENDED;
207  case JOB_READ: {
208  CHECK_EQ(1, words.size());
209  current_job_index_ = strtoint32(words[0]);
210  parser_state_ = JOB_ID_READ;
211  break;
212  }
213  case JOB_ID_READ: {
214  CHECK_EQ(1, words.size());
215  parser_state_ = JOB_LENGTH_READ;
216  break;
217  }
218  case JOB_LENGTH_READ: {
219  CHECK_EQ(declared_machine_count_, words.size());
220  Job* const job = problem_.mutable_jobs(current_job_index_);
221  for (int i = 0; i < declared_machine_count_; ++i) {
222  const int64_t duration = strtoint64(words[i]);
223  Task* const task = job->add_tasks();
224  task->add_machine(i);
225  task->add_duration(duration);
226  }
227  parser_state_ =
228  current_job_index_ == declared_job_count_ - 1 ? DONE : JOB_READ;
229  break;
230  }
231  default: {
232  LOG(FATAL) << "Should not be here with state " << parser_state_;
233  }
234  }

236 void JsspParser::ProcessFlexibleLine(const std::string& line) {
237  const std::vector<std::string> words =
238  absl::StrSplit(line, ' ', absl::SkipEmpty());
239  switch (parser_state_) {
240  case START: {
241  CHECK_GE(words.size(), 2);
242  SetJobs(strtoint32(words[0]));
243  SetMachines(strtoint32(words[1]));
244  problem_.set_makespan_cost_per_time_unit(1L);
245  parser_state_ = JOB_COUNT_READ;
246  break;
247  }
248  case JOB_COUNT_READ: {
249  const int operations_count = strtoint32(words[0]);
250  int index = 1;
251  Job* const job = problem_.mutable_jobs(current_job_index_);
252  for (int operation = 0; operation < operations_count; ++operation) {
253  const int alternatives_count = strtoint32(words[index++]);
254  Task* const task = job->add_tasks();
255  for (int alt = 0; alt < alternatives_count; alt++) {
256  // Machine id are 1 based.
257  const int machine_id = strtoint32(words[index++]) - 1;
258  const int64_t duration = strtoint64(words[index++]);
259  task->add_machine(machine_id);
260  task->add_duration(duration);
261  }
262  }
263  CHECK_LE(index, words.size()); // Ignore CR at the end of the line.
264  current_job_index_++;
265  if (current_job_index_ == declared_job_count_) {
266  parser_state_ = DONE;
267  }
268  break;
269  }
270  default: {
271  LOG(FATAL) << "Should not be here with state " << parser_state_;
272  }
273  }

275 void JsspParser::ProcessSdstLine(const std::string& line) {
276  const std::vector<std::string> words =
277  absl::StrSplit(line, ' ', absl::SkipEmpty());
278  switch (parser_state_) {
279  case START: {
280  if (words.size() == 2) {
281  SetJobs(strtoint32(words[0]));
282  SetMachines(strtoint32(words[1]));
283  problem_.set_makespan_cost_per_time_unit(1L);
284  parser_state_ = JOB_COUNT_READ;
285  current_machine_index_ = 0;
286  }
287  break;
288  }
289  case JOB_COUNT_READ: {
290  CHECK_EQ(words.size(), declared_machine_count_ * 2);
291  Job* const job = problem_.mutable_jobs(current_job_index_);
292  for (int i = 0; i < declared_machine_count_; ++i) {
293  const int machine_id = strtoint32(words[2 * i]);
294  const int64_t duration = strtoint64(words[2 * i + 1]);
295  Task* const task = job->add_tasks();
296  task->add_machine(machine_id);
297  task->add_duration(duration);
298  }
299  current_job_index_++;
300  if (current_job_index_ == declared_job_count_) {
301  parser_state_ = JOBS_READ;
302  }
303  break;
304  }
305  case JOBS_READ: {
306  CHECK_EQ(1, words.size());
307  CHECK_EQ("SSD", words[0]);
308  parser_state_ = SSD_READ;
309  break;
310  }
311  case SSD_READ: {
312  CHECK_EQ(1, words.size());
313  CHECK_EQ(words[0], absl::StrCat("M", current_machine_index_)) << line;
314  current_job_index_ = 0;
315  parser_state_ = MACHINE_READ;
316  break;
317  }
318  case MACHINE_READ: {
319  CHECK_EQ(declared_job_count_, words.size());
320  Machine* const machine =
321  problem_.mutable_machines(current_machine_index_);
322  for (const std::string& w : words) {
323  const int64_t t = strtoint64(w);
324  machine->mutable_transition_time_matrix()->add_transition_time(t);
325  }
326  if (++current_job_index_ == declared_job_count_) {
327  parser_state_ = ++current_machine_index_ == declared_machine_count_
328  ? DONE
329  : SSD_READ;
330  }
331  break;
332  }
333  default: {
334  LOG(FATAL) << "Should not be here with state " << parser_state_
335  << "with line " << line;
336  }
337  }

340 void JsspParser::ProcessTardinessLine(const std::string& line) {
341  const std::vector<std::string> words =
342  absl::StrSplit(line, ' ', absl::SkipEmpty());
343  switch (parser_state_) {
344  case START: {
345  CHECK_EQ(3, words.size());
346  SetJobs(strtoint32(words[0]));
347  SetMachines(strtoint32(words[1]));
348  parser_state_ = JOB_COUNT_READ;
349  current_job_index_ = 0;
350  break;
351  }
352  case JOB_COUNT_READ: {
353  CHECK_GE(words.size(), 6);
354  Job* const job = problem_.mutable_jobs(current_job_index_);
355  const int64_t est = strtoint64(words[0]);
356  if (est != 0L) {
357  job->mutable_earliest_start()->set_value(est);
358  }
359  job->set_late_due_date(strtoint64(words[1]));
360  const double weight = std::stod(words[2]);
361  const int64_t tardiness = static_cast<int64_t>(
362  round(weight * absl::GetFlag(FLAGS_jssp_scaling_up_factor)));
363  job->set_lateness_cost_per_time_unit(tardiness);
364  const int num_operations = strtoint32(words[3]);
365  for (int i = 0; i < num_operations; ++i) {
366  const int machine_id = strtoint32(words[4 + 2 * i]) - 1; // 1 based.
367  const int64_t duration = strtoint64(words[5 + 2 * i]);
368  Task* const task = job->add_tasks();
369  task->add_machine(machine_id);
370  task->add_duration(duration);
371  }
372  current_job_index_++;
373  if (current_job_index_ == declared_job_count_) {
374  // Fix tardiness weights if all integer from start.
375  bool all_integral = true;
376  for (const Job& job : problem_.jobs()) {
377  if (job.lateness_cost_per_time_unit() %
378  absl::GetFlag(FLAGS_jssp_scaling_up_factor) !=
379  0) {
380  all_integral = false;
381  break;
382  }
383  }
384  if (all_integral) {
385  for (Job& job : *problem_.mutable_jobs()) {
386  job.set_lateness_cost_per_time_unit(
387  job.lateness_cost_per_time_unit() /
388  absl::GetFlag(FLAGS_jssp_scaling_up_factor));
389  }
390  } else {
391  problem_.mutable_scaling_factor()->set_value(
392  1.0L / absl::GetFlag(FLAGS_jssp_scaling_up_factor));
393  }
394  parser_state_ = DONE;
395  }
396  break;
397  }
398  default: {
399  LOG(FATAL) << "Should not be here with state " << parser_state_
400  << "with line " << line;
401  }
402  }

405 void JsspParser::ProcessPssLine(const std::string& line) {
406  const std::vector<std::string> words =
407  absl::StrSplit(line, ' ', absl::SkipEmpty());
408  switch (parser_state_) {
409  case START: {
410  problem_.set_makespan_cost_per_time_unit(1L);
411  CHECK_EQ(1, words.size());
412  SetJobs(strtoint32(words[0]));
413  parser_state_ = JOB_COUNT_READ;
414  break;
415  }
416  case JOB_COUNT_READ: {
417  CHECK_EQ(1, words.size());
418  SetMachines(strtoint32(words[0]));
419  parser_state_ = MACHINE_COUNT_READ;
420  current_job_index_ = 0;
421  break;
422  }
423  case MACHINE_COUNT_READ: {
424  CHECK_EQ(1, words.size());
425  CHECK_EQ(declared_machine_count_, strtoint32(words[0]));
426  if (++current_job_index_ == declared_job_count_) {
427  parser_state_ = JOB_LENGTH_READ;
428  current_job_index_ = 0;
429  current_machine_index_ = 0;
430  }
431  break;
432  }
433  case JOB_LENGTH_READ: {
434  CHECK_EQ(4, words.size());
435  CHECK_EQ(0, strtoint32(words[2]));
436  CHECK_EQ(0, strtoint32(words[3]));
437  const int machine_id = strtoint32(words[0]) - 1;
438  const int duration = strtoint32(words[1]);
439  Job* const job = problem_.mutable_jobs(current_job_index_);
440  Task* const task = job->add_tasks();
441  task->add_machine(machine_id);
442  task->add_duration(duration);
443  if (++current_machine_index_ == declared_machine_count_) {
444  current_machine_index_ = 0;
445  if (++current_job_index_ == declared_job_count_) {
446  current_job_index_ = -1;
447  current_machine_index_ = 0;
448  parser_state_ = JOBS_READ;
449  transition_index_ = 0;
450  for (int m = 0; m < declared_machine_count_; ++m) {
451  Machine* const machine = problem_.mutable_machines(m);
452  for (int i = 0; i < declared_job_count_ * declared_job_count_;
453  ++i) {
454  machine->mutable_transition_time_matrix()->add_transition_time(0);
455  }
456  }
457  }
458  }
459  break;
460  }
461  case JOBS_READ: {
462  CHECK_EQ(1, words.size());
463  const int index = transition_index_++;
464  const int size = declared_job_count_ * declared_machine_count_ + 1;
465  const int t1 = index / size;
466  const int t2 = index % size;
467  if (t1 == 0 || t2 == 0) { // Dummy task.
468  break;
469  }
470  const int item1 = t1 - 1;
471  const int item2 = t2 - 1;
472  const int job1 = item1 / declared_machine_count_;
473  const int task1 = item1 % declared_machine_count_;
474  const int m1 = problem_.jobs(job1).tasks(task1).machine(0);
475  const int job2 = item2 / declared_machine_count_;
476  const int task2 = item2 % declared_machine_count_;
477  const int m2 = problem_.jobs(job2).tasks(task2).machine(0);
478  if (m1 != m2) { // We are only interested in same machine transitions.
479  break;
480  }
481  const int transition = strtoint32(words[0]);
482  Machine* const machine = problem_.mutable_machines(m1);
483  machine->mutable_transition_time_matrix()->set_transition_time(
484  job1 * declared_job_count_ + job2, transition);
485  if (transition_index_ == size * size) {
486  parser_state_ = DONE;
487  }
488  break;
489  }
490  default: {
491  LOG(FATAL) << "Should not be here with state " << parser_state_
492  << "with line " << line;
493  }
494  }
