200 bool SearchForGurobiDynamicLibrary() {
201  const char *gurobi_home_from_env = getenv("GUROBI_HOME");
202 #if defined(_MSC_VER) // Windows
203  if (!gurobi_library_path.empty() &&
204  LoadSpecificGurobiLibrary(gurobi_library_path)) {
205  return true;
206  }
207  if (gurobi_home_from_env != nullptr &&
208  LoadSpecificGurobiLibrary(
209  absl::StrCat(gurobi_home_from_env, "\\bin\\gurobi90.dll"))) {
210  return true;
211  }
212  if (LoadSpecificGurobiLibrary(
213  "C:\\Program Files\\gurobi902\\win64\\bin\\gurobi90.dll")) {
214  return true;
215  }
216 #elif defined(__APPLE__) // OS X
217  if (!gurobi_library_path.empty() &&
218  LoadSpecificGurobiLibrary(gurobi_library_path)) {
219  return true;
220  }
221  if (gurobi_home_from_env != nullptr &&
222  LoadSpecificGurobiLibrary(
223  absl::StrCat(gurobi_home_from_env, "/lib/libgurobi90.dylib"))) {
224  return true;
225  }
226  if (LoadSpecificGurobiLibrary(
227  "/Library/gurobi902/mac64/lib/libgurobi90.dylib")) {
228  return true;
229  }
230 #elif defined(__GNUC__) // Linux
231  if (gurobi_home_from_env != nullptr &&
232  LoadSpecificGurobiLibrary(
233  absl::StrCat(gurobi_home_from_env, "/lib/libgurobi90.so"))) {
234  return true;
235  }
236  if (!gurobi_library_path.empty() &&
237  LoadSpecificGurobiLibrary(gurobi_library_path)) {
238  return true;
239  }
240 #endif
241 
242  return false;
243 }

245 bool MPSolver::LoadGurobiSharedLibrary() {
246  if (gurobi_dynamic_library.get() != nullptr) {
247  return gurobi_dynamic_library->LibraryIsLoaded();
248  }
249 
250  gurobi_dynamic_library.reset(new DynamicLibrary());
251 
252  if (SearchForGurobiDynamicLibrary()) {
253  LoadGurobiFunctions();
254  return true;
255  }
256 
257  return false;
258 }
