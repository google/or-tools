14// IntType is a simple template class mechanism for defining "logical"
15// integer-like class types that support many of the same functionalities
16// as native integer types, but which prevent assignment, construction, and
17// other operations from other similar integer-like types. Essentially, the
18// template class IntType<IntTypeName, ValueType> (where ValueType assumes
19// valid scalar types such as int, uint, int32_t, etc) has the additional
20// property that it cannot be assigned to or constructed from other IntTypes
21// or native integer types of equal or implicitly convertible type.
22//
23// The class is useful for preventing mingling of integer variables with
24// different logical roles or units. Unfortunately, C++ provides relatively
25// good type-safety for user-defined classes but not for integer types. It is
26// essentially up to the user to use nice variable names and comments to prevent
27// accidental mismatches, such as confusing a user-index with a group-index or a
28// time-in-milliseconds with a time-in-seconds. The use of typedefs are limited
29// in that regard as they do not enforce type-safety.
30//
31// USAGE -----------------------------------------------------------------------
32//
33// DEFINE_INT_TYPE(IntTypeName, ValueType);
34//
35// where:
36// IntTypeName: is the desired (unique) name for the "logical" integer type.
37// ValueType: is one of the integral types as defined by std::is_integral
38// (see <type_traits>).
39//
40// DISALLOWED OPERATIONS / TYPE-SAFETY ENFORCEMENT -----------------------------
41//
42// Consider these definitions and variable declarations:
43// DEFINE_INT_TYPE(GlobalDocID, int64_t);
44// DEFINE_INT_TYPE(LocalDocID, int64_t);
45// GlobalDocID global;
46// LocalDocID local;
47//
48// The class IntType prevents:
49//
50// 1) Assignments of other IntTypes with different IntTypeNames.
51//
52// global = local; <-- Fails to compile!
53// local = global; <-- Fails to compile!
54//
55// 2) Explicit/implicit conversion from an IntType to another IntType.
56//
57// LocalDocID l(global); <-- Fails to compile!
58// LocalDocID l = global; <-- Fails to compile!
59//
60// void GetGlobalDoc(GlobalDocID global) { }
61// GetGlobalDoc(global); <-- Compiles fine, types match!
62// GetGlobalDoc(local); <-- Fails to compile!
63//
64// 3) Implicit conversion from an IntType to a native integer type.
65//
66// void GetGlobalDoc(int64_t global) { ...
67// GetGlobalDoc(global); <-- Fails to compile!
68// GetGlobalDoc(local); <-- Fails to compile!
69//
70// void GetLocalDoc(int32_t local) { ...
71// GetLocalDoc(global); <-- Fails to compile!
72// GetLocalDoc(local); <-- Fails to compile!
73//
74//
75// SUPPORTED OPERATIONS --------------------------------------------------------
76//
77// The following operators are supported: unary: ++ (both prefix and postfix),
78// +, -, ! (logical not), ~ (one's complement); comparison: ==, !=, <, <=, >,
79// >=; numerical: +, -, *, /; assignment: =, +=, -=, /=, *=; stream: <<. Each
80// operator allows the same IntTypeName and the ValueType to be used on
81// both left- and right-hand sides.
82//
83// It also supports an accessor value() returning the stored value as ValueType,
84// and a templatized accessor value<T>() method that serves as syntactic sugar
85// for static_cast<T>(var.value()). These accessors are useful when assigning
86// the stored value into protocol buffer fields and using it as printf args.
87//
88// The class also defines a hash functor that allows the IntType to be used
89// as key to hashable containers such as hash_map and hash_set.
90//
91// FixedArray and STL vector (see int-type-indexed-container.h) if an IntType is
92// intended to be used as an index into these containers. These wrappers are
93// indexed in a type-safe manner using IntTypes to ensure type-safety.
94//
95// NB: this implementation does not attempt to abide by or enforce dimensional
96// analysis on these scalar types.
97//
98// EXAMPLES --------------------------------------------------------------------
99//
100// DEFINE_INT_TYPE(GlobalDocID, int64_t);
101// GlobalDocID global = 3;
102// std::cout << global; <-- Prints 3 to stdout.
103//
104// for (GlobalDocID i(0); i < global; ++i) {
105// std::cout << i;
106// } <-- Print(ln)s 0 1 2 to stdout
107//
108// DEFINE_INT_TYPE(LocalDocID, int64_t);
109// LocalDocID local;
110// std::cout << local; <-- Prints 0 to stdout it
111// default
112// initializes the value to 0.
113//
114// local = 5;
115// local *= 2;
116// LocalDocID l(local);
117// std::cout << l + local; <-- Prints 20 to stdout.
118//
119// GenericSearchRequest request;
120// request.set_doc_id(global.value()); <-- Uses value() to extract the value
121// from the IntType class.
122//
123// REMARKS ---------------------------------------------------------------------
124//
125// The following bad usage is permissible although discouraged. Essentially, it
126// involves using the value*() accessors to extract the native integer type out
127// of the IntType class. Keep in mind that the primary reason for the IntType
128// class is to prevent *accidental* mingling of similar logical integer types --
129// and not type casting from one type to another.
130//
131// DEFINE_INT_TYPE(GlobalDocID, int64_t);
132// DEFINE_INT_TYPE(LocalDocID, int64_t);
133// GlobalDocID global;
134// LocalDocID local;
135//
136// global = local.value(); <-- Compiles fine.
137//
138// void GetGlobalDoc(GlobalDocID global) { ...
139// GetGlobalDoc(local.value()); <-- Compiles fine.
140//
141// void GetGlobalDoc(int64_t global) { ...
142// GetGlobalDoc(local.value()); <-- Compiles fine.
143

162// Defines the IntType using value_type and typedefs it to int_type_name.
163// The struct int_type_name ## _tag_ trickery is needed to ensure that a new
164// type is created per int_type_name.
165#define DEFINE_INT_TYPE(int_type_name, value_type) \
166 struct int_type_name##_tag_ { \
167 static constexpr absl::string_view TypeName() { return #int_type_name; } \
168 }; \
169 typedef ::gtl::IntType<int_type_name##_tag_, value_type> int_type_name;

171// Holds a integral value (of type ValueType) and behaves as a
172// ValueType by exposing assignment, unary, comparison, and arithmetic
173// operators.
174//
175// The template parameter IntTypeName defines the name for the int type and must
176// be unique within a binary (the convenient DEFINE_INT_TYPE macro at the end of
177// the file generates a unique IntTypeName). The parameter ValueType defines
178// the integer type value (see supported list above).
179//
180// This class is NOT thread-safe.
181template <typename IntTypeName, typename _ValueType>
182class IntType {
183 public:
184 typedef _ValueType ValueType; // for non-member operators
185 typedef IntType<IntTypeName, ValueType> ThisType; // Syntactic sugar.
186
187 static constexpr absl::string_view TypeName() {
188 return IntTypeName::TypeName();
189 }
190
191 // Note that this may change from time to time without notice.
192 struct Hasher {
193 size_t operator()(const IntType& arg) const {
194 return static_cast<size_t>(arg.value());
195 }
196 };
197
198 public:
199 // Default c'tor initializing value_ to 0.
200 constexpr IntType() : value_(0) {}
201 // C'tor explicitly initializing from a ValueType.
202 constexpr explicit IntType(ValueType value) : value_(value) {}
203
204 // IntType uses the default copy constructor, destructor and assign operator.
205 // The defaults are sufficient and omitting them allows the compiler to add
206 // the move constructor/assignment.
207
208 // -- ACCESSORS --------------------------------------------------------------
209 // The class provides a value() accessor returning the stored ValueType value_
210 // as well as a templatized accessor that is just a syntactic sugar for
211 // static_cast<T>(var.value());
212 constexpr ValueType value() const { return value_; }
213
214 template <typename ValType>
215 constexpr ValType value() const {
216 return static_cast<ValType>(value_);
217 }
218
219 // -- UNARY OPERATORS --------------------------------------------------------
220 ThisType& operator++() { // prefix ++
221 ++value_;
222 return *this;
223 }
224 const ThisType operator++(int v) { // postfix ++
225 ThisType temp(*this);
226 ++value_;
227 return temp;
228 }
229 ThisType& operator--() { // prefix --
230 --value_;
231 return *this;
232 }
233 const ThisType operator--(int v) { // postfix --
234 ThisType temp(*this);
235 --value_;
236 return temp;
237 }
238
239 constexpr bool operator!() const { return value_ == 0; }
240 constexpr const ThisType operator+() const { return ThisType(value_); }
241 constexpr const ThisType operator-() const { return ThisType(-value_); }
242 constexpr const ThisType operator~() const { return ThisType(~value_); }
243
244 // -- ASSIGNMENT OPERATORS ---------------------------------------------------
245 // We support the following assignment operators: =, +=, -=, *=, /=, <<=, >>=
246 // and %= for both ThisType and ValueType.
247#define INT_TYPE_ASSIGNMENT_OP(op) \
248 ThisType& operator op(const ThisType& arg_value) { \
249 value_ op arg_value.value(); \
250 return *this; \
251 } \
252 ThisType& operator op(ValueType arg_value) { \
253 value_ op arg_value; \
254 return *this; \
255 }
256 INT_TYPE_ASSIGNMENT_OP(+=);
257 INT_TYPE_ASSIGNMENT_OP(-=);
258 INT_TYPE_ASSIGNMENT_OP(*=);
259 INT_TYPE_ASSIGNMENT_OP(/=);
260 INT_TYPE_ASSIGNMENT_OP(<<=); // NOLINT
261 INT_TYPE_ASSIGNMENT_OP(>>=); // NOLINT
262 INT_TYPE_ASSIGNMENT_OP(%=);
263#undef INT_TYPE_ASSIGNMENT_OP
264
265 ThisType& operator=(ValueType arg_value) {
266 value_ = arg_value;
267 return *this;
268 }
269
270 private:
271 // The integer value of type ValueType.
272 ValueType value_;
273
274 COMPILE_ASSERT(std::is_integral<ValueType>::value,
275 invalid_integer_type_for_id_type_);
276} ABSL_ATTRIBUTE_PACKED;

287// -- NON-MEMBER ARITHMETIC OPERATORS ------------------------------------------
288// We support only the +, -, *, and / operators with the same IntType and
289// ValueType types. The reason is to allow simple manipulation on these IDs
290// when used as indices in vectors and arrays.
291//
292// NB: Although it is possible to do IntType * IntType and IntType / IntType,
293// it is probably non-sensical from a dimensionality analysis perspective.
294#define INT_TYPE_ARITHMETIC_OP(op) \
295 template <typename IntTypeName, typename ValueType> \
296 constexpr IntType<IntTypeName, ValueType> operator op( \
297 IntType<IntTypeName, ValueType> id_1, \
298 IntType<IntTypeName, ValueType> id_2) { \
299 return IntType<IntTypeName, ValueType>(id_1.value() op id_2.value()); \
300 } \
301 template <typename IntTypeName, typename ValueType> \
302 constexpr IntType<IntTypeName, ValueType> operator op( \
303 IntType<IntTypeName, ValueType> id, \
304 typename IntType<IntTypeName, ValueType>::ValueType arg_val) { \
305 return IntType<IntTypeName, ValueType>(id.value() op arg_val); \
306 } \
307 template <typename IntTypeName, typename ValueType> \
308 constexpr IntType<IntTypeName, ValueType> operator op( \
309 typename IntType<IntTypeName, ValueType>::ValueType arg_val, \
310 IntType<IntTypeName, ValueType> id) { \
311 return IntType<IntTypeName, ValueType>(arg_val op id.value()); \
312 }
313INT_TYPE_ARITHMETIC_OP(+);
314INT_TYPE_ARITHMETIC_OP(-);
315INT_TYPE_ARITHMETIC_OP(*);
316INT_TYPE_ARITHMETIC_OP(/);
317INT_TYPE_ARITHMETIC_OP(<<); // NOLINT
318INT_TYPE_ARITHMETIC_OP(>>); // NOLINT

322// -- NON-MEMBER COMPARISON OPERATORS ------------------------------------------
323// Static inline comparison operators. We allow all comparison operators among
324// the following types (OP \in [==, !=, <, <=, >, >=]:
325// IntType<IntTypeName, ValueType> OP IntType<IntTypeName, ValueType>
326// IntType<IntTypeName, ValueType> OP ValueType
327// ValueType OP IntType<IntTypeName, ValueType>
328#define INT_TYPE_COMPARISON_OP(op) \
329 template <typename IntTypeName, typename ValueType> \
330 static inline constexpr bool operator op( \
331 IntType<IntTypeName, ValueType> id_1, \
332 IntType<IntTypeName, ValueType> id_2) { \
333 return id_1.value() op id_2.value(); \
334 } \
335 template <typename IntTypeName, typename ValueType> \
336 static inline constexpr bool operator op( \
337 IntType<IntTypeName, ValueType> id, \
338 typename IntType<IntTypeName, ValueType>::ValueType val) { \
339 return id.value() op val; \
340 } \
341 template <typename IntTypeName, typename ValueType> \
342 static inline constexpr bool operator op( \
343 typename IntType<IntTypeName, ValueType>::ValueType val, \
344 IntType<IntTypeName, ValueType> id) { \
345 return val op id.value(); \
346 }
