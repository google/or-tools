52std::string BoolVar::DebugString() const {
53 if (index_ < 0) {
54 return absl::StrFormat("Not(%s)", Not().DebugString());
55 } else {
56 std::string output;
57 const IntegerVariableProto& var_proto = builder_->Proto().variables(index_);
58 // Special case for constant variables without names.
59 if (var_proto.name().empty() && var_proto.domain_size() == 2 &&
60 var_proto.domain(0) == var_proto.domain(1)) {
61 output.append(var_proto.domain(0) == 0 ? "false" : "true");
62 } else {
63 if (var_proto.name().empty()) {
64 absl::StrAppendFormat(&output, "BoolVar%i(", index_);
65 } else {
66 absl::StrAppendFormat(&output, "%s(", var_proto.name());
67 }
68 if (var_proto.domain(0) == var_proto.domain(1)) {
69 output.append(var_proto.domain(0) == 0 ? "false)" : "true)");
70 } else {
71 absl::StrAppend(&output, var_proto.domain(0), ", ", var_proto.domain(1),
72 ")");
73 }
74 }
75 return output;
76 }
77}

316std::string LinearExpr::DebugString(const CpModelProto* proto) const {
317 std::string result;
318 for (int i = 0; i < variables_.size(); ++i) {
319 const int64_t coeff = coefficients_[i];
320 const std::string var_string = proto == nullptr
321 ? absl::StrCat("V", variables_[i])
322 : VarDebugString(*proto, variables_[i]);
323 if (i == 0) {
324 if (coeff == 1) {
325 absl::StrAppend(&result, var_string);
326 } else if (coeff == -1) {
327 absl::StrAppend(&result, "-", var_string);
328 } else if (coeff != 0) {
329 absl::StrAppend(&result, coeff, " * ", var_string);
330 }
331 } else if (coeff == 1) {
332 absl::StrAppend(&result, " + ", var_string);
333 } else if (coeff > 0) {
334 absl::StrAppend(&result, " + ", coeff, " * ", var_string);
335 } else if (coeff == -1) {
336 absl::StrAppend(&result, " - ", var_string);
337 } else if (coeff < 0) {
338 absl::StrAppend(&result, " - ", -coeff, " * ", var_string);
339 }
340 }
341
342 if (constant_ != 0) {
343 if (variables_.empty()) {
344 return absl::StrCat(constant_);
345 } else if (constant_ > 0) {
346 absl::StrAppend(&result, " + ", constant_);
347 } else {
348 absl::StrAppend(&result, " - ", -constant_);
349 }
350 }
351 return result;
352}

695 IntegerVariableProto* const var_proto = cp_model_.add_variables();

703int CpModelBuilder::GetOrCreateIntegerIndex(int index) {
704 if (index >= 0) {
705 return index;
706 }
707 if (!bool_to_integer_index_map_.contains(index)) {
708 const int var = PositiveRef(index);
709 const IntegerVariableProto& old_var = cp_model_.variables(var);
710 const int new_index = cp_model_.variables_size();
711 IntegerVariableProto* const new_var = cp_model_.add_variables();
712 new_var->add_domain(0);
713 new_var->add_domain(1);
714 if (!old_var.name().empty()) {
715 new_var->set_name(absl::StrCat("Not(", old_var.name(), ")"));
716 }
717 AddEquality(IntVar(new_index, this), BoolVar(index, this));
718 bool_to_integer_index_map_[index] = new_index;
719 return new_index;
720 }
721 return bool_to_integer_index_map_[index];
722}

724IntVar CpModelBuilder::NewIntVar(const Domain& domain) {
725 const int index = cp_model_.variables_size();
726 IntegerVariableProto* const var_proto = cp_model_.add_variables();
