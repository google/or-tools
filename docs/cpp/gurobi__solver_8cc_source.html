63inline BasisStatus ConvertVariableStatus(const int status) {
64 switch (status) {
65 case GRB_BASIC:
66 return BasisStatus::BASIC;
67 case GRB_NONBASIC_LOWER:
68 return BasisStatus::AT_LOWER_BOUND;
69 case GRB_NONBASIC_UPPER:
70 return BasisStatus::AT_UPPER_BOUND;
71 case GRB_SUPERBASIC:
72 return BasisStatus::FREE;
73 default:
74 return BasisStatus::INVALID;
75 }
76}
77
78inline int GrbVariableStatus(const BasisStatus status) {
79 switch (status) {
80 case BasisStatus::BASIC:
81 return GRB_BASIC;
82 case BasisStatus::AT_LOWER_BOUND:
83 case BasisStatus::FIXED_VALUE:
84 return GRB_NONBASIC_LOWER;
85 case BasisStatus::AT_UPPER_BOUND:
86 return GRB_NONBASIC_UPPER;
87 case BasisStatus::FREE:
88 return GRB_SUPERBASIC;
89 case BasisStatus::INVALID:
90 default:
91 LOG(FATAL) << "Unexpected invalid initial_basis.";
92 return 0;
93 }
94}

98GurobiParametersProto MergeParameters(
99 const CommonSolveParametersProto& common_parameters,
100 const GurobiParametersProto& gurobi_parameters) {
101 GurobiParametersProto merged_parameters;
102 if (common_parameters.has_enable_output()) {
103 // TODO(user): Gurobi allows for custom display callbacks. This
104 // allows redirecting output to a file, or parsing it and retrieve key
105 // information, or redirect it to IDE outputs (such as in Jupiter
106 // Notebooks). We should install such a callback to manage output if this is
107 // enabled.
108 const int enable_output = common_parameters.enable_output() ? 1 : 0;
109 GurobiParametersProto::Parameter* const parameter =
110 merged_parameters.add_parameters();
111 parameter->set_name(GRB_INT_PAR_LOGTOCONSOLE);
112 parameter->set_value(absl::StrCat(enable_output));
113 }
114 if (common_parameters.has_time_limit()) {
115 const double time_limit = absl::ToDoubleSeconds(
116 util_time::DecodeGoogleApiProto(common_parameters.time_limit())
117 .value());
118 GurobiParametersProto::Parameter* const parameter =
119 merged_parameters.add_parameters();
120 parameter->set_name(GRB_DBL_PAR_TIMELIMIT);
121 parameter->set_value(absl::StrCat(time_limit));
122 }
123 if (common_parameters.has_threads()) {
124 const int threads = common_parameters.threads();
125 GurobiParametersProto::Parameter* const parameter =
126 merged_parameters.add_parameters();
127 parameter->set_name(GRB_INT_PAR_THREADS);
128 parameter->set_value(absl::StrCat(threads));
129 }
130 if (common_parameters.has_random_seed()) {
131 const int random_seed =
132 std::min(GRB_MAXINT, std::max(common_parameters.random_seed(), 0));
133 GurobiParametersProto::Parameter* const parameter =
134 merged_parameters.add_parameters();
135 parameter->set_name(GRB_INT_PAR_SEED);
136 parameter->set_value(absl::StrCat(random_seed));
137 }
138 if (common_parameters.lp_algorithm() != LP_ALGORITHM_UNSPECIFIED) {
139 GurobiParametersProto::Parameter* const parameter =
140 merged_parameters.add_parameters();
141 parameter->set_name(GRB_INT_PAR_METHOD);
142 switch (common_parameters.lp_algorithm()) {
143 case LP_ALGORITHM_PRIMAL_SIMPLEX:
144 parameter->set_value(absl::StrCat(GRB_METHOD_PRIMAL));
145 break;
146 case LP_ALGORITHM_DUAL_SIMPLEX:
147 parameter->set_value(absl::StrCat(GRB_METHOD_DUAL));
148 break;
149 case LP_ALGORITHM_BARRIER:
150 parameter->set_value(absl::StrCat(GRB_METHOD_BARRIER));
151 break;
152 default:
153 LOG(FATAL) << "LPAlgorithm: "
154 << ProtoEnumToString(common_parameters.lp_algorithm())
155 << " unknown, error setting Gurobi parameters";
156 }
157 }
158 if (common_parameters.scaling() != EMPHASIS_UNSPECIFIED) {
159 GurobiParametersProto::Parameter* const parameter =
160 merged_parameters.add_parameters();
161 parameter->set_name(GRB_INT_PAR_SCALEFLAG);
162 switch (common_parameters.scaling()) {
163 case EMPHASIS_OFF:
164 parameter->set_value(absl::StrCat(0));
165 break;
166 case EMPHASIS_LOW:
167 case EMPHASIS_MEDIUM:
168 parameter->set_value(absl::StrCat(1));
169 break;
170 case EMPHASIS_HIGH:
171 parameter->set_value(absl::StrCat(2));
172 break;
173 case EMPHASIS_VERY_HIGH:
174 parameter->set_value(absl::StrCat(3));
175 break;
176 default:
177 LOG(FATAL) << "Scaling emphasis: "
178 << ProtoEnumToString(common_parameters.scaling())
179 << " unknown, error setting Gurobi parameters";
180 }
181 }
182 if (common_parameters.cuts() != EMPHASIS_UNSPECIFIED) {
183 GurobiParametersProto::Parameter* const parameter =
184 merged_parameters.add_parameters();
185 parameter->set_name(GRB_INT_PAR_CUTS);
186 switch (common_parameters.cuts()) {
187 case EMPHASIS_OFF:
188 parameter->set_value(absl::StrCat(0));
189 break;
190 case EMPHASIS_LOW:
191 case EMPHASIS_MEDIUM:
192 parameter->set_value(absl::StrCat(1));
193 break;
194 case EMPHASIS_HIGH:
195 parameter->set_value(absl::StrCat(2));
196 break;
197 case EMPHASIS_VERY_HIGH:
198 parameter->set_value(absl::StrCat(3));
199 break;
200 default:
201 LOG(FATAL) << "Cuts emphasis: "
202 << ProtoEnumToString(common_parameters.cuts())
203 << " unknown, error setting Gurobi parameters";
204 }
205 }
206 if (common_parameters.heuristics() != EMPHASIS_UNSPECIFIED) {
207 GurobiParametersProto::Parameter* const parameter =
208 merged_parameters.add_parameters();
209 parameter->set_name(GRB_DBL_PAR_HEURISTICS);
210 switch (common_parameters.heuristics()) {
211 case EMPHASIS_OFF:
212 parameter->set_value(absl::StrCat(0.0));
213 break;
214 case EMPHASIS_LOW:
215 parameter->set_value(absl::StrCat(0.025));
216 break;
217 case EMPHASIS_MEDIUM:
218 // As of Gurobi 9.1 this is the default value.
219 // https://www.gurobi.com/documentation/9.1/refman/heuristics.html
220 parameter->set_value(absl::StrCat(0.05));
221 break;
222 case EMPHASIS_HIGH:
223 parameter->set_value(absl::StrCat(0.1));
224 break;
225 case EMPHASIS_VERY_HIGH:
226 parameter->set_value(absl::StrCat(0.2));
227 break;
228 default:
229 LOG(FATAL) << "Heuristics emphasis: "
230 << ProtoEnumToString(common_parameters.heuristics())
231 << " unknown, error setting Gurobi parameters";
232 }
233 }
234
235 if (common_parameters.presolve() != EMPHASIS_UNSPECIFIED) {
236 GurobiParametersProto::Parameter* const parameter =
237 merged_parameters.add_parameters();
238 parameter->set_name(GRB_INT_PAR_PRESOLVE);
239 switch (common_parameters.presolve()) {
240 case EMPHASIS_OFF:
241 parameter->set_value(absl::StrCat(0));
242 break;
243 case EMPHASIS_LOW:
244 case EMPHASIS_MEDIUM:
245 parameter->set_value(absl::StrCat(1));
246 break;
247 case EMPHASIS_HIGH:
248 case EMPHASIS_VERY_HIGH:
249 parameter->set_value(absl::StrCat(2));
250 break;
251 default:
252 LOG(FATAL) << "Presolve emphasis: "
253 << ProtoEnumToString(common_parameters.presolve())
254 << " unknown, error setting Gurobi parameters";
255 }
256 }
257
258 for (const GurobiParametersProto_Parameter& parameter :
259 gurobi_parameters.parameters()) {
260 *merged_parameters.add_parameters() = parameter;
261 }
262 return merged_parameters;
263}

285std::string GurobiSolver::GurobiErrorMessage(int error_code) const {
286 if (error_code == kGrbOk) {
287 return {};
288 }
289 if (active_env_ == nullptr) {
290 if (error_code == GRB_ERROR_NO_LICENSE) {
291 return absl::StrCat("Gurobi error code:", error_code,
292 " (Failed to obtain a valid license)");
293 } else {
294 return absl::StrCat("Gurboi error code:", error_code,
295 " (No environment is available)");
296 }
297 }
298 return absl::StrCat("Gurobi error code: ", error_code, ", ",
299 GRBgeterrormsg(active_env_));
300}

302std::string GurobiSolver::LogGurobiCode(
303 int error_code, const char* source_file, int source_line,
304 const char* statement, absl::string_view extra_message = "") const {
305 if (error_code == kGrbOk) {
306 return "";
307 }
308 const std::string error_message =
309 absl::StrFormat("%s:%d, on '%s' : %s%s", source_file, source_line,
310 statement, GurobiErrorMessage(error_code), extra_message);
311 VLOG(1) << error_message;
312 return error_message;
313}

355 }
356 return absl::OkStatus();

389absl::StatusOr<std::pair<SolveResultProto::TerminationReason, std::string>>
390GurobiSolver::ConvertTerminationReason(const int gurobi_status,
391 const bool has_feasible_solution) {
392 switch (gurobi_status) {
393 case GRB_OPTIMAL:
394 return std::make_pair(SolveResultProto::OPTIMAL, "");
395 case GRB_INFEASIBLE:
396 return std::make_pair(SolveResultProto::INFEASIBLE, "");
397 case GRB_UNBOUNDED: {
398 if (has_feasible_solution) {
399 return std::make_pair(SolveResultProto::UNBOUNDED, "");
400 } else {
401 return std::make_pair(SolveResultProto::DUAL_INFEASIBLE,
402 "Gurobi status GRB_UNBOUNDED, but no feasible "
403 "point found, only primal ray.");
404 }
405 }
406 case GRB_INF_OR_UNBD:
407 return std::make_pair(SolveResultProto::DUAL_INFEASIBLE,
408 "Gurobi status GRB_INF_OR_UNBD.");
409 case GRB_CUTOFF:
410 return std::make_pair(SolveResultProto::OBJECTIVE_LIMIT,
411 "Gurobi status GRB_CUTOFF.");
412 case GRB_ITERATION_LIMIT:
413 return std::make_pair(SolveResultProto::ITERATION_LIMIT, "");
414 case GRB_NODE_LIMIT:
415 return std::make_pair(SolveResultProto::NODE_LIMIT, "");
416 case GRB_TIME_LIMIT:
417 return std::make_pair(SolveResultProto::TIME_LIMIT, "");
418 case GRB_SOLUTION_LIMIT:
419 return std::make_pair(SolveResultProto::SOLUTION_LIMIT, "");
420 case GRB_INTERRUPTED:
421 return std::make_pair(SolveResultProto::INTERRUPTED, "");
422 case GRB_NUMERIC:
423 return std::make_pair(SolveResultProto::NUMERICAL_ERROR, "");
424 case GRB_SUBOPTIMAL:
425 return std::make_pair(SolveResultProto::IMPRECISE, "");
426 case GRB_USER_OBJ_LIMIT:
427 return std::make_pair(SolveResultProto::OBJECTIVE_LIMIT,
428 "Gurobi status GRB_USR_OBJ_LIMIT.");
429 case GRB_LOADED:
430 return absl::InternalError(
431 "Error creating termination reason, unexpected gurobi status code "
432 "GRB_LOADED.");
433 case GRB_INPROGRESS:
434 return absl::InternalError(
435 "Error creating termination reason, unexpected gurobi status code "
436 "GRB_INPROGRESS.");
437 default:
438 return absl::InternalError(absl::StrCat(
439 "Missing Gurobi optimization status code case: ", gurobi_status));
440 }

472absl::Status GurobiSolver::SetGurobiBasis(const BasisProto& basis) {
473 std::vector<int> gurobi_variable_basis_status(num_gurobi_variables_);
474 for (const auto [id, value] : MakeView(basis.variable_status())) {
475 gurobi_variable_basis_status[variables_map_.at(id)] =
476 GrbVariableStatus(static_cast<BasisStatus>(value));
477 }
478
479 std::vector<int> gurobi_constraint_basis_status;
480 gurobi_constraint_basis_status.reserve(num_gurobi_constraints());
481 for (const auto [id, value] : MakeView(basis.constraint_status())) {
482 const ConstraintData& constraint_data = linear_constraints_map_.at(id);
483 // Non-ranged constraints
484 if (constraint_data.slack_index == kUnspecifiedIndex) {
485 if (value == BasisStatus::BASIC) {
486 gurobi_constraint_basis_status.push_back(kGrbBasicConstraint);
487 } else {
488 gurobi_constraint_basis_status.push_back(kGrbNonBasicConstraint);
489 }
490 // Ranged constraints
491 } else if (value == BasisStatus::BASIC) {
492 // Either constraint or MathOpt slack is basic, but not both (because
493 // columns for MathOpt slack and internal Gurobi slack are linearly
494 // dependent). We choose the MathOpt slack to be basic.
495 gurobi_variable_basis_status[constraint_data.slack_index] = GRB_BASIC;
496 gurobi_constraint_basis_status.push_back(kGrbNonBasicConstraint);
497 } else {
498 gurobi_variable_basis_status[constraint_data.slack_index] =
499 GrbVariableStatus(static_cast<BasisStatus>(value));
500 gurobi_constraint_basis_status.push_back(kGrbNonBasicConstraint);
501 }
502 }
503 RETURN_IF_GUROBI_ERROR(GRBsetintattrarray(
504 /*model=*/gurobi_model_, /*attrname=*/GRB_INT_ATTR_VBASIS,
505 /*first=*/0, /*len=*/gurobi_variable_basis_status.size(),
506 /*newvalues=*/gurobi_variable_basis_status.data()));
507 RETURN_IF_GUROBI_ERROR(GRBsetintattrarray(
508 /*model=*/gurobi_model_, /*attrname=*/GRB_INT_ATTR_CBASIS,
509 /*first=*/0, /*len=*/gurobi_constraint_basis_status.size(),
510 /*newvalues=*/gurobi_constraint_basis_status.data()));
511
512 return absl::OkStatus();
513}

515absl::StatusOr<BasisProto> GurobiSolver::GetGurobiBasis() {
516 BasisProto basis;
517
518 std::vector<int> gurobi_variable_basis_status(num_gurobi_variables_);
519 RETURN_IF_ERROR(GetIntAttrArray(
520 GRB_INT_ATTR_VBASIS, absl::MakeSpan(gurobi_variable_basis_status)));
521
522 for (auto [variable_id, gurobi_variable_index] : variables_map_) {
523 basis.mutable_variable_status()->add_ids(variable_id);
524 const BasisStatus variable_status = ConvertVariableStatus(
525 gurobi_variable_basis_status[gurobi_variable_index]);
526 if (variable_status == BasisStatus::INVALID) {
527 return absl::InternalError(
528 absl::StrCat("Invalid Gurobi variable basis status: ",
529 gurobi_variable_basis_status[gurobi_variable_index]));
530 }
531 basis.mutable_variable_status()->add_values(variable_status);
532 }
533
534 std::vector<int> gurobi_constraint_basis_status(num_gurobi_constraints());
535 RETURN_IF_ERROR(GetIntAttrArray(
536 GRB_INT_ATTR_CBASIS, absl::MakeSpan(gurobi_constraint_basis_status)));
537 for (auto [constraint_id, gurobi_data] : linear_constraints_map_) {
538 basis.mutable_constraint_status()->add_ids(constraint_id);
539 const int gurobi_constraint_status =
540 gurobi_constraint_basis_status[gurobi_data.constraint_index];
541 if ((gurobi_constraint_status != kGrbBasicConstraint) &&
542 (gurobi_constraint_status != kGrbNonBasicConstraint)) {
543 return absl::InternalError(
544 absl::StrCat("Invalid Gurobi constraint basis status: ",
545 gurobi_constraint_status));
546 }
547 // linear_terms <= upper_bound
548 if (gurobi_data.lower_bound <= -GRB_INFINITY &&
549 gurobi_data.upper_bound < GRB_INFINITY) {
550 if (gurobi_constraint_status == kGrbBasicConstraint) {
551 basis.mutable_constraint_status()->add_values(BasisStatus::BASIC);
552 } else {
553 basis.mutable_constraint_status()->add_values(
554 BasisStatus::AT_UPPER_BOUND);
555 }
556 // linear_terms >= lower_bound
557 } else if (gurobi_data.lower_bound > -GRB_INFINITY &&
558 gurobi_data.upper_bound >= GRB_INFINITY) {
559 if (gurobi_constraint_status == kGrbBasicConstraint) {
560 basis.mutable_constraint_status()->add_values(BasisStatus::BASIC);
561 } else {
562 basis.mutable_constraint_status()->add_values(
563 BasisStatus::AT_LOWER_BOUND);
564 }
565 // linear_terms == xxxxx_bound
566 } else if (gurobi_data.lower_bound == gurobi_data.upper_bound) {
567 if (gurobi_constraint_status == kGrbBasicConstraint) {
568 basis.mutable_constraint_status()->add_values(BasisStatus::BASIC);
569 } else {
570 // TODO(user): consider refining this to
571 // AT_LOWER_BOUND/AT_UPPER_BOUND using the sign of the dual variable.
572 basis.mutable_constraint_status()->add_values(BasisStatus::FIXED_VALUE);
573 }
574 // linear_term - slack == 0 (ranged constraint)
575 } else {
576 const BasisStatus slack_status = ConvertVariableStatus(
577 gurobi_variable_basis_status[gurobi_data.slack_index]);
578 if (slack_status == BasisStatus::INVALID) {
579 return absl::InternalError(absl::StrCat(
580 "Invalid Gurobi slack variable basis status: ", slack_status));
581 }
582 if ((gurobi_constraint_status == kGrbBasicConstraint) ||
583 (slack_status == BasisStatus::BASIC)) {
584 basis.mutable_constraint_status()->add_values(BasisStatus::BASIC);
585 } else {
586 basis.mutable_constraint_status()->add_values(slack_status);
587 }
588 }
589 }
590 return basis;
591}

593// See go/mathopt-dev-transformations#gurobi-inf for details of this
594// transformation, comments inside the function refer to the notation there.
595absl::StatusOr<DualRayProto> GurobiSolver::GetGurobiDualRay(
596 const SparseVectorFilterProto& linear_constraints_filter,
597 const SparseVectorFilterProto& variables_filter, const bool is_maximize) {
598 // farkas_dual = lambda
599 std::vector<double> farkas_dual(num_gurobi_constraints());
600 RETURN_IF_ERROR(
601 GetDoubleAttrArray(GRB_DBL_ATTR_FARKASDUAL, absl::MakeSpan(farkas_dual)));
602 DualRayProto dual_ray;
603
604 // Compute y = -lambda
605 {
606 SparseVectorFilterPredicate predicate(linear_constraints_filter);
607 for (auto [constraint_id, gurobi_data] : linear_constraints_map_) {
608 // constraint_dual_value = y[constraint_id]
609 const double value = -farkas_dual[gurobi_data.constraint_index];
610 if (predicate.AcceptsAndUpdate(constraint_id, value)) {
611 dual_ray.mutable_dual_values()->add_ids(constraint_id);
612 if (is_maximize) {
613 dual_ray.mutable_dual_values()->add_values(-value);
614 } else {
615 dual_ray.mutable_dual_values()->add_values(value);
616 }
617 }
618 }
619 }
620
621 // Compute r = \bar{a} = A^T lambda
622 std::vector<int> constraint_indices(num_gurobi_constraints());
623 std::vector<double> coefficients(num_gurobi_constraints());
624 {
625 SparseVectorFilterPredicate predicate(variables_filter);
626 for (auto [var_id, gurobi_variable_index] : variables_map_) {
627 // reduced_cost_value = r[gurobi_variable_index]
628 // = \bar{a}[gurobi_variable_index]
629 double reduced_cost_value = 0.0;
630 int result_size;
631 int vbeg;
632 // coefficients = column gurobi_variable_index of A
633 RETURN_IF_GUROBI_ERROR(GRBgetvars(
634 gurobi_model_, &result_size, &vbeg, constraint_indices.data(),
635 coefficients.data(), gurobi_variable_index, 1));
636 for (int i = 0; i < result_size; ++i) {
637 reduced_cost_value +=
638 farkas_dual[constraint_indices[i]] * coefficients[i];
639 }
640 if (predicate.AcceptsAndUpdate(var_id, reduced_cost_value)) {
641 dual_ray.mutable_reduced_costs()->add_ids(var_id);
642 if (is_maximize) {
643 dual_ray.mutable_reduced_costs()->add_values(-reduced_cost_value);
644 } else {
645 dual_ray.mutable_reduced_costs()->add_values(reduced_cost_value);
646 }
647 }
648 }
649 }
650 return dual_ray;
651}

655 const ModelSolveParametersProto& model_parameters) {

657 if (GRBisattravailable(gurobi_model_, GRB_INT_ATTR_SOLCOUNT)) {

659 }

661 ASSIGN_OR_RETURN(const int grb_termination,
662 GetIntAttr(GRB_INT_ATTR_STATUS));
663

665
666 ASSIGN_OR_RETURN(
667 const auto reason_and_detail,
668 ConvertTerminationReason(grb_termination, has_feasible_solution));
669 result.set_termination_reason(reason_and_detail.first);
670 result.set_termination_detail(reason_and_detail.second);
671 }

673 ASSIGN_OR_RETURN(const double obj_val, GetDoubleAttr(GRB_DBL_ATTR_OBJVAL));
674 result.mutable_solve_stats()->set_best_primal_bound(obj_val);
675 } else {
676 result.mutable_solve_stats()->set_best_primal_bound(is_maximize ? -kInf
677 : kInf);
678 }
679 if (GRBisattravailable(gurobi_model_, GRB_DBL_ATTR_OBJBOUND)) {
680 ASSIGN_OR_RETURN(const double obj_bound,
681 GetDoubleAttr(GRB_DBL_ATTR_OBJBOUND));
682 result.mutable_solve_stats()->set_best_dual_bound(obj_bound);

686 }

693 }
694 if (GRBisattravailable(gurobi_model_, GRB_INT_ATTR_BARITERCOUNT)) {
695 ASSIGN_OR_RETURN(const int barrier_iters,
696 GetIntAttr(GRB_INT_ATTR_BARITERCOUNT));
697 result.mutable_solve_stats()->set_barrier_iterations(barrier_iters);
698 }
699 if (GRBisattravailable(gurobi_model_, GRB_DBL_ATTR_NODECOUNT)) {
700 ASSIGN_OR_RETURN(const double nodes_double,
701 GetDoubleAttr(GRB_DBL_ATTR_NODECOUNT));
702 ASSIGN_OR_RETURN(const int64_t nodes, SafeInt64FromDouble(nodes_double));
703 result.mutable_solve_stats()->set_node_count(nodes);
704 }
705
706 for (int i = 0; i < num_solutions; ++i) {
707 PrimalSolutionProto* const primal_solution = result.add_primal_solutions();
708 double sol_val;
709 std::vector<double> grb_var_values(num_gurobi_variables_);
710 // TODO(user): there seems to be some kind of issue with Gurobi where
711 // GRB_DBL_ATTR_POOLOBJVAL is not always filled in when when there is
712 // one, solution, probably a bug.
713 if (i == 0) {
714 ASSIGN_OR_RETURN(sol_val, GetDoubleAttr(GRB_DBL_ATTR_OBJVAL));
715 RETURN_IF_ERROR(
716 GetDoubleAttrArray(GRB_DBL_ATTR_X, absl::MakeSpan(grb_var_values)));
717 } else {
718 RETURN_IF_GUROBI_ERROR(
719 GRBsetintparam(active_env_, GRB_INT_PAR_SOLUTIONNUMBER, i))

722 RETURN_IF_ERROR(
723 GetDoubleAttrArray(GRB_DBL_ATTR_XN, absl::MakeSpan(grb_var_values)));
724 }
725 primal_solution->set_objective_value(sol_val);

727 grb_var_values, variables_map_,
728 *primal_solution->mutable_variable_values(),
729 model_parameters.primal_variables_filter());
730 }
731 // TODO(user): support getting infeasibility proofs, and basis.
732 // If model is LP, get dual solutions
733 // Note that we can ignore the reduced costs of the slack variables for ranged
734 // constraints because of
735 // go/mathopt-dev-transformations#slack-var-range-constraint

738 GRBisattravailable(gurobi_model_, GRB_DBL_ATTR_RC)) {
739 std::vector<double> grb_constraint_duals(num_gurobi_constraints());
740 RETURN_IF_ERROR(GetDoubleAttrArray(GRB_DBL_ATTR_PI,

742 DualSolutionProto* const dual_solution = result.add_dual_solutions();
743 ASSIGN_OR_RETURN(const double obj_val, GetDoubleAttr(GRB_DBL_ATTR_OBJVAL));
744 dual_solution->set_objective_value(obj_val);
745 GurobiVectorToSparseDoubleVector(
746 grb_constraint_duals, linear_constraints_map_,
747 *dual_solution->mutable_dual_values(),
748 model_parameters.dual_linear_constraints_filter());
749
750 std::vector<double> grb_reduced_cost_values(num_gurobi_variables_);
751 RETURN_IF_ERROR(GetDoubleAttrArray(
752 GRB_DBL_ATTR_RC, absl::MakeSpan(grb_reduced_cost_values)));
753 GurobiVectorToSparseDoubleVector(grb_reduced_cost_values, variables_map_,
754 *dual_solution->mutable_reduced_costs(),
755 model_parameters.dual_variables_filter());
756 }
757 if (is_lp && GRBisattravailable(gurobi_model_, GRB_DBL_ATTR_UNBDRAY)) {
758 std::vector<double> grb_ray_var_values(num_gurobi_variables_);
759 RETURN_IF_ERROR(GetDoubleAttrArray(GRB_DBL_ATTR_UNBDRAY,
760 absl::MakeSpan(grb_ray_var_values)));
761 PrimalRayProto* const primal_ray = result.add_primal_rays();
762 GurobiVectorToSparseDoubleVector(
763 grb_ray_var_values, variables_map_,
764 *primal_ray->mutable_variable_values(),
765 model_parameters.primal_variables_filter());
766 }
767 if (is_lp && GRBisattravailable(gurobi_model_, GRB_DBL_ATTR_FARKASDUAL)) {
768 ASSIGN_OR_RETURN(
769 DualRayProto dual_ray,
770 GetGurobiDualRay(model_parameters.dual_linear_constraints_filter(),

772 is_maximize));

776 GRBisattravailable(gurobi_model_, GRB_INT_ATTR_CBASIS)) {
777 ASSIGN_OR_RETURN(BasisProto basis, GetGurobiBasis());
778 result.mutable_basis()->Add(std::move(basis));
779 }
780 return absl::OkStatus();
781}

849// Given a vector of pairs<LinearConstraintId,ConstraintData&> add a slack
850// variable for each of the constraints in the underlying `gurobi_model_`
851// using the referenced bounds.
852absl::Status GurobiSolver::AddNewSlacks(
853 const std::vector<SlackInfo>& new_slacks) {
854 // Note that we are really adding the sub-matrix
855 // D * slack
856 // to the set of linear constraints, and the D matrix is stored in compressed
857 // sparse column (CSC) format. In our particular case, D is a diagonal matrix
858 // with -1.0 coefficients for each new slack in the row indicated in the
859 // row_indices vector.
860 const int num_slacks = new_slacks.size();
861 if (num_slacks == 0) {
862 return absl::OkStatus();
863 }
864 // Build the D matrix in CSC format.
865 std::vector<double> column_non_zeros(num_slacks, -1.0);
866 std::vector<double> lower_bounds;
867 std::vector<double> upper_bounds;
868 std::vector<GurobiLinearConstraintIndex> row_indices;
869 std::vector<int> column_non_zero_begin;
870 column_non_zero_begin.reserve(num_slacks);
871 row_indices.reserve(num_slacks);
872 lower_bounds.reserve(num_slacks);
873 upper_bounds.reserve(num_slacks);
874 for (int k = 0; k < num_slacks; ++k) {
875 auto& [id, constraint_data] = new_slacks[k];
876 gtl::InsertOrDie(&slack_map_, id, constraint_data);
877 row_indices.emplace_back(constraint_data.constraint_index);
878 lower_bounds.emplace_back(constraint_data.lower_bound);
879 upper_bounds.emplace_back(constraint_data.upper_bound);
880 column_non_zero_begin.emplace_back(k);
881 }
882 // Add variables to the underlying model.
883 RETURN_IF_GUROBI_ERROR(GRBaddvars(
884 /*model=*/gurobi_model_, /*numvars=*/num_slacks, /*numnz=*/num_slacks,
885 /*vbeg=*/column_non_zero_begin.data(), /*vind=*/row_indices.data(),
886 /*vval=*/column_non_zeros.data(), /*obj=*/nullptr,
887 /*lb=*/lower_bounds.data(), /*ub=*/upper_bounds.data(),
888 /*vtype=*/nullptr, /*varnames=*/nullptr));
889 num_gurobi_variables_ += num_slacks;
890 return absl::OkStatus();
891}
892
893absl::Status GurobiSolver::AddNewConstraints(
894 const LinearConstraintsProto& constraints) {
895 CHECK(gurobi_model_ != nullptr);
896 const int num_model_constraints = num_gurobi_constraints();
897 const int num_new_constraints = constraints.lower_bounds().size();
898
899 const bool has_constraint_names = !constraints.names().empty();
900 std::vector<char*> constraint_names;
901 if (has_constraint_names) {
902 constraint_names.reserve(num_new_constraints);
903 }
904 // Constraints are translated into:
905 // 1. ax <= upper_bound (if lower bound <= -GRB_INFINITY, and upper_bound
906 // is finite and less than GRB_INFINITY)
907 // 2. ax >= lower_bound (if upper bound >= GRB_INFINITY, and lower_bound is
908 // finite and greater than -GRB_INFINITY)
909 // 3. ax == xxxxx_bound (if both bounds are finite, equal, and their
910 // absolute values less than GRB_INFINITY)
911 // 4. ax - slack = 0.0 (otherwise,
912 // slack bounds == [lower_bound, upper_bound])
913 std::vector<double> constraint_rhs;
914 std::vector<char> constraint_sense;
915 std::vector<SlackInfo> new_slacks;
916 constraint_rhs.reserve(num_new_constraints);
917 constraint_sense.reserve(num_new_constraints);
918 new_slacks.reserve(num_new_constraints);
919 for (int i = 0; i < num_new_constraints; ++i) {
920 const int64_t id = constraints.ids(i);
921 ConstraintData& constraint_data =
922 gtl::InsertKeyOrDie(&linear_constraints_map_, id);
923 constraint_data.lower_bound = constraints.lower_bounds(i);
924 constraint_data.upper_bound = constraints.upper_bounds(i);
925 constraint_data.constraint_index = i + num_model_constraints;
926 char sense = GRB_EQUAL;
927 double rhs = 0.0;
928 // Detect the type of constraint to add and store RHS and bounds.
929 if (constraint_data.lower_bound <= -GRB_INFINITY &&
930 constraint_data.upper_bound < GRB_INFINITY) {
931 rhs = constraint_data.upper_bound;
932 sense = GRB_LESS_EQUAL;
933 } else if (constraint_data.lower_bound > -GRB_INFINITY &&
934 constraint_data.upper_bound >= GRB_INFINITY) {
935 rhs = constraint_data.lower_bound;
936 sense = GRB_GREATER_EQUAL;
937 } else if (constraint_data.lower_bound == constraint_data.upper_bound) {
938 rhs = constraint_data.lower_bound;
939 sense = GRB_EQUAL;
940 } else {
941 // Note that constraints where the lower bound and the upper bound are
942 // -+infinity translate into a range constraint with an unbounded slack.
943 constraint_data.slack_index = new_slacks.size() + num_gurobi_variables_;
944 new_slacks.emplace_back(id, constraint_data);
945 }
946 constraint_rhs.emplace_back(rhs);
947 constraint_sense.emplace_back(sense);
948 if (has_constraint_names) {
949 constraint_names.emplace_back(
950 const_cast<char*>(constraints.names(i).c_str()));
951 }
952 }
953 // Add all constraints in one call.
954 RETURN_IF_GUROBI_ERROR(GRBaddconstrs(
955 gurobi_model_, /*numconstrs=*/num_new_constraints,
956 /*numnz=*/0, /*cbeg=*/nullptr, /*cind=*/nullptr, /*cval=*/nullptr,
957 /*sense=*/constraint_sense.data(), /*rhs=*/constraint_rhs.data(),
958 /*constrnames=*/
959 has_constraint_names ? constraint_names.data() : nullptr));
960
961 // Add slacks for true ranged constraints (if needed)
962 if (!new_slacks.empty()) {
963 RETURN_IF_ERROR(AddNewSlacks(new_slacks));
964 }
965 return absl::OkStatus();
966}

1048// Bound changes in constraints can induce new variables, and also remove
1049// some slacks. We first add all new variables, and queue all deletions to be
1050// dealt with later on.
1051absl::Status GurobiSolver::UpdateLinearConstraints(
1052 const LinearConstraintUpdatesProto& constraints_update,
1053 std::vector<GurobiVariableIndex>& deleted_variables_index) {
1054 const SparseDoubleVectorProto& constraint_lower_bounds =
1055 constraints_update.lower_bounds();
1056 const SparseDoubleVectorProto& constraint_upper_bounds =
1057 constraints_update.upper_bounds();
1058
1059 // If no update, just return.
1060 if (constraint_lower_bounds.ids().empty() &&
1061 constraint_upper_bounds.ids().empty()) {
1062 return absl::OkStatus();
1063 }
1064
1065 // We want to avoid changing the right-hand-side, sense, or slacks of each
1066 // constraint more than once. Since we can refer to the same constraint ID
1067 // both in the `constraint_upper_bounds` and `constraint_lower_bounds` sparse
1068 // vectors, we collect all changes into a single structure:
1069 struct UpdateConstraintData {
1070 LinearConstraintId constraint_id;
1071 ConstraintData& source;
1072 double new_lower_bound;
1073 double new_upper_bound;
1074 UpdateConstraintData(const LinearConstraintId id, ConstraintData& reference)
1075 : constraint_id(id),
1076 source(reference),
1077 new_lower_bound(reference.lower_bound),
1078 new_upper_bound(reference.upper_bound) {}
1079 };
1080 const int upper_bounds_size = constraint_upper_bounds.ids().size();
1081 const int lower_bounds_size = constraint_lower_bounds.ids().size();
1082 std::vector<UpdateConstraintData> update_vector;
1083 update_vector.reserve(upper_bounds_size + lower_bounds_size);
1084 // We exploit the fact that IDs are sorted in increasing order to merge
1085 // changes into a vector of aggregated changes.
1086 for (int lower_index = 0, upper_index = 0;
1087 lower_index < lower_bounds_size || upper_index < upper_bounds_size;) {
1088 VariableId lower_id = std::numeric_limits<int64_t>::max();
1089 if (lower_index < lower_bounds_size) {
1090 lower_id = constraint_lower_bounds.ids(lower_index);
1091 }
1092 VariableId upper_id = std::numeric_limits<int64_t>::max();
1093 if (upper_index < upper_bounds_size) {
1094 upper_id = constraint_upper_bounds.ids(upper_index);
1095 }
1096 const VariableId id = std::min(lower_id, upper_id);
1097 DCHECK(id < std::numeric_limits<int64_t>::max());
1098 UpdateConstraintData update(id, linear_constraints_map_.at(id));
1099 if (lower_id == upper_id) {
1100 update.new_lower_bound = constraint_lower_bounds.values(lower_index++);
1101 update.new_upper_bound = constraint_upper_bounds.values(upper_index++);
1102 } else if (lower_id < upper_id) {
1103 update.new_lower_bound = constraint_lower_bounds.values(lower_index++);
1104 } else { /* upper_id < lower_id */
1105 update.new_upper_bound = constraint_upper_bounds.values(upper_index++);
1106 }
1107 update_vector.emplace_back(update);
1108 }
1109
1110 // We have grouped all changes in update_vector, now generate changes in
1111 // slack bounds, rhs, senses, new slacks, and deleted_slacks (to be dealt
1112 // with later, outside this function).
1113 // These three vectors keep changes to right-hand-side and senses.
1114 std::vector<char> sense_data;
1115 std::vector<double> rhs_data;
1116 std::vector<GurobiLinearConstraintIndex> rhs_index;
1117 // These three vectors keep changes to bounds on existing slack.
1118 std::vector<double> lower_bound_data;
1119 std::vector<double> upper_bound_data;
1120 std::vector<GurobiVariableIndex> bound_index;
1121 // This vector keep newly introduced slacks.
1122 std::vector<SlackInfo> new_slacks;
1123 // Iterate on the changes, and populate the three possible changes.
1124 for (UpdateConstraintData& update_data : update_vector) {
1125 const bool same_lower_bound =
1126 (update_data.source.lower_bound == update_data.new_lower_bound) ||
1127 ((update_data.source.lower_bound <= -GRB_INFINITY) &&
1128 (update_data.new_lower_bound <= -GRB_INFINITY));
1129 const bool same_upper_bound =
1130 (update_data.source.upper_bound == update_data.new_upper_bound) ||
1131 ((update_data.source.upper_bound >= GRB_INFINITY) &&
1132 (update_data.new_upper_bound >= GRB_INFINITY));
1133 if (same_upper_bound && same_lower_bound) continue;
1134 // Save into linear_constraints_map_[id] the new bounds for the linear
1135 // constraint.
1136 update_data.source.lower_bound = update_data.new_lower_bound;
1137 update_data.source.upper_bound = update_data.new_upper_bound;
1138 bool delete_slack = false;
1139 // Detect the type of constraint to add and store RHS and bounds.
1140 if (update_data.new_lower_bound <= -GRB_INFINITY &&
1141 update_data.new_upper_bound < GRB_INFINITY) {
1142 delete_slack = true;
1143 rhs_index.emplace_back(update_data.source.constraint_index);
1144 rhs_data.emplace_back(update_data.new_upper_bound);
1145 sense_data.emplace_back(GRB_LESS_EQUAL);
1146 } else if (update_data.new_lower_bound > -GRB_INFINITY &&
1147 update_data.new_upper_bound >= GRB_INFINITY) {
1148 delete_slack = true;
1149 rhs_index.emplace_back(update_data.source.constraint_index);
1150 rhs_data.emplace_back(update_data.new_lower_bound);
1151 sense_data.emplace_back(GRB_GREATER_EQUAL);
1152 } else if (update_data.new_lower_bound == update_data.new_upper_bound) {
1153 delete_slack = true;
1154 rhs_index.emplace_back(update_data.source.constraint_index);
1155 rhs_data.emplace_back(update_data.new_lower_bound);
1156 sense_data.emplace_back(GRB_EQUAL);
1157 } else {
1158 // Note that constraints where the lower bound and the upper bound are
1159 // -+infinity translated into a range constraint with an unbounded
1160 // slack.
1161 if (update_data.source.slack_index != kUnspecifiedIndex) {
1162 bound_index.emplace_back(update_data.source.slack_index);
1163 lower_bound_data.emplace_back(update_data.new_lower_bound);
1164 upper_bound_data.emplace_back(update_data.new_upper_bound);
1165 } else {
1166 // Note that if we add a new slack, we must both reset the sense and
1167 // right hand side for the inequality.
1168 rhs_index.emplace_back(update_data.source.constraint_index);
1169 rhs_data.emplace_back(0.0);
1170 sense_data.emplace_back(GRB_EQUAL);
1171 // Update the slack_index in the linear_constraints_map_[id]
1172 update_data.source.slack_index =
1173 new_slacks.size() + num_gurobi_variables_;
1174 // Save the data needed to add the new slack.
1175 new_slacks.emplace_back(update_data.constraint_id, update_data.source);
1176 }
1177 }
1178 // If the constraint had a slack, and now is marked for deletion, we reset
1179 // the stored slack_index in linear_constraints_map_[id] and save the index
1180 // in the list of variables to be deleted later on.
1181 if (delete_slack && update_data.source.slack_index != kUnspecifiedIndex) {
1182 deleted_variables_index.emplace_back(update_data.source.slack_index);
1183 update_data.source.slack_index = kUnspecifiedIndex;
1184 }
1185 }
1186
1187 // Pass down changes to Gurobi.
1188 if (!rhs_index.empty()) {
1189 RETURN_IF_GUROBI_ERROR(GRBsetdblattrlist(
1190 /*model=*/gurobi_model_, /*attrname=*/GRB_DBL_ATTR_RHS,
1191 /*len=*/rhs_index.size(), /*ind=*/rhs_index.data(),
1192 /*newvalues=*/rhs_data.data()));
1193 RETURN_IF_GUROBI_ERROR(GRBsetcharattrlist(
1194 /*model=*/gurobi_model_, /*attrname=*/GRB_CHAR_ATTR_SENSE,
1195 /*len=*/rhs_index.size(), /*ind=*/rhs_index.data(),
1196 /*newvalues=*/sense_data.data()));
1197 } // rhs changes
1198 if (!bound_index.empty()) {
1199 RETURN_IF_GUROBI_ERROR(GRBsetdblattrlist(
1200 /*model=*/gurobi_model_, /*attrname=*/GRB_DBL_ATTR_LB,
1201 /*len=*/bound_index.size(), /*ind=*/bound_index.data(),
1202 /*newvalues=*/lower_bound_data.data()));
1203 RETURN_IF_GUROBI_ERROR(GRBsetdblattrlist(
1204 /*model=*/gurobi_model_, /*attrname=*/GRB_DBL_ATTR_UB,
1205 /*len=*/bound_index.size(), /*ind=*/bound_index.data(),
1206 /*newvalues=*/upper_bound_data.data()));
1207 } // Slack bound changes.
1208
1209 if (!new_slacks.empty()) {
1210 RETURN_IF_ERROR(AddNewSlacks(new_slacks));
1211 }
1212 return absl::OkStatus();
1213}
1214
1215// This function re-assign indices for variables and constraints after
1216// deletion. The updated indices are computed from the previous indices, sorted
1217// in incremental form, but re-assigned so that all indices are contiguous
1218// between [0, num_variables-1] and [0, num_linear_constraints-1].
1219// This implementation exploit the fact that gtl::linked_hash_map preserves the
1220// insertion order of whatever elements remain in the hash tables.
1221absl::Status GurobiSolver::UpdateGurobiIndices() {
1222 { // Recover index of variables.
1223 GurobiVariableIndex next_index = 0;
1224 GurobiVariableIndex prev_index = kUnspecifiedIndex;
1225 auto variable_it = variables_map_.begin();
1226 auto slack_it = slack_map_.begin();
1227 while (variable_it != variables_map_.end() ||
1228 slack_it != slack_map_.end()) {
1229 GurobiVariableIndex variable_index = std::numeric_limits<int32_t>::max();
1230 if (variable_it != variables_map_.end()) {
1231 variable_index = variable_it->second;
1232 }
1233 GurobiVariableIndex slack_index = std::numeric_limits<int32_t>::max();
1234 if (slack_it != slack_map_.end()) {
1235 slack_index = slack_it->second.slack_index;
1236 }
1237 DCHECK_LT(prev_index, variable_index);
1238 DCHECK_LT(prev_index, slack_index);
1239 DCHECK_NE(variable_index, slack_index);
1240 if (slack_index < variable_index) {
1241 prev_index = slack_index;
1242 slack_it->second.slack_index = next_index++;
1243 ++slack_it;
1244 } else {
1245 prev_index = variable_index;
1246 variable_it->second = next_index++;
1247 ++variable_it;
1248 }
1249 }
1250 DCHECK_EQ(next_index, num_gurobi_variables_);
1251 }
1252 { // Recover index of constraints.
1253 GurobiLinearConstraintIndex next_constraint = 0;
1254 GurobiLinearConstraintIndex prev_constraint = kUnspecifiedConstraint;
1255 for (auto& constraint_iterator : linear_constraints_map_) {
1256 DCHECK_LT(prev_constraint, constraint_iterator.second.constraint_index);
1257 prev_constraint = constraint_iterator.second.constraint_index;
1258 constraint_iterator.second.constraint_index = next_constraint++;
1259 }
1260 DCHECK_EQ(next_constraint, num_gurobi_constraints());
1261 }
1262 return absl::OkStatus();
1263}

1265absl::Status GurobiSolver::Update(const ModelUpdateProto& model_update) {
1266 CHECK(gurobi_model_ != nullptr);
1267
1268 RETURN_IF_ERROR(AddNewVariables(model_update.new_variables()));
1269
1270 RETURN_IF_ERROR(AddNewConstraints(model_update.new_linear_constraints()));
1271
1272 RETURN_IF_ERROR(
1273 ChangeCoefficients(model_update.linear_constraint_matrix_updates()));
1274
1275 if (model_update.objective_updates().has_direction_update()) {
1276 const int model_sense = model_update.objective_updates().direction_update()
1277 ? GRB_MAXIMIZE
1278 : GRB_MINIMIZE;
1279 RETURN_IF_GUROBI_ERROR(
1280 GRBsetintattr(gurobi_model_, GRB_INT_ATTR_MODELSENSE, model_sense));
1281 }
1282
1283 if (model_update.objective_updates().has_offset_update()) {
1284 RETURN_IF_GUROBI_ERROR(
1285 GRBsetdblattr(gurobi_model_, GRB_DBL_ATTR_OBJCON,
1286 model_update.objective_updates().offset_update()));
1287 }
1288
1289 RETURN_IF_ERROR(UpdateDoubleListAttribute(
1290 model_update.objective_updates().linear_coefficients(), GRB_DBL_ATTR_OBJ,
1291 variables_map_));
1292
1293 RETURN_IF_ERROR(
1294 UpdateDoubleListAttribute(model_update.variable_updates().lower_bounds(),
1295 GRB_DBL_ATTR_LB, variables_map_));
1296
1297 RETURN_IF_ERROR(
1298 UpdateDoubleListAttribute(model_update.variable_updates().upper_bounds(),
1299 GRB_DBL_ATTR_UB, variables_map_));
1300
1301 if (model_update.variable_updates().has_integers()) {
1302 const SparseBoolVectorProto& update =
1303 model_update.variable_updates().integers();
1304 const int sparse_length = update.ids().size();
1305 std::vector<GurobiVariableIndex> index(sparse_length);
1306 std::vector<char> value(sparse_length);
1307 for (int k = 0; k < sparse_length; ++k) {
1308 index[k] = variables_map_.at(update.ids(k));
1309 value[k] = update.values(k) ? GRB_INTEGER : GRB_CONTINUOUS;
1310 }
1311 RETURN_IF_GUROBI_ERROR(GRBsetcharattrlist(
1312 gurobi_model_, /*attrname=*/GRB_CHAR_ATTR_VTYPE, /*len=*/sparse_length,
1313 /*ind=*/index.data(), /*newvalues=*/value.data()));
1314 }
1315
1316 // We cache all Gurobi variables and constraint indices that must be deleted,
1317 // and perform deletions at the end of the update call.
1318 std::vector<GurobiVariableIndex> deleted_variables_index;
1319 std::vector<GurobiLinearConstraintIndex> deleted_constraints_index;
1320
1321 RETURN_IF_ERROR(UpdateLinearConstraints(
1322 model_update.linear_constraint_updates(), deleted_variables_index));
1323
1324 for (const VariableId id : model_update.deleted_variable_ids()) {
1325 deleted_variables_index.emplace_back(variables_map_.at(id));
1326 variables_map_.erase(id);
1327 }
1328
1329 for (const LinearConstraintId id :
1330 model_update.deleted_linear_constraint_ids()) {
1331 ConstraintData& constraint_data = linear_constraints_map_.at(id);
1332 deleted_constraints_index.emplace_back(constraint_data.constraint_index);
1333 if (constraint_data.slack_index != kUnspecifiedIndex) {
1334 deleted_variables_index.emplace_back(constraint_data.slack_index);
1335 constraint_data.slack_index = kUnspecifiedIndex;
1336 slack_map_.erase(id);
1337 }
1338 linear_constraints_map_.erase(id);
1339 }
1340
1341 // If no cached deletions, we are done.
1342 if (deleted_variables_index.empty() && deleted_constraints_index.empty()) {
1343 return absl::OkStatus();
1344 }
1345 // If we are removing variables or constraints we remove them after adding
1346 // any variable or constraint. This is to avoid problems with
1347 // the numbering of possibly new variables and constraints.
1348 // After that we must update the model so that sequence of updates don't
1349 // interfere with one-another.
1350 if (!deleted_constraints_index.empty()) {
1351 RETURN_IF_GUROBI_ERROR(
1352 GRBdelconstrs(/*model=*/gurobi_model_,
1353 /*len=*/deleted_constraints_index.size(),
1354 /*ind=*/deleted_constraints_index.data()));
1355 }
1356
1357 if (!deleted_variables_index.empty()) {
1358 RETURN_IF_GUROBI_ERROR(GRBdelvars(/*model=*/gurobi_model_,
1359 /*len=*/deleted_variables_index.size(),
1360 /*ind=*/deleted_variables_index.data()));
1361 num_gurobi_variables_ -= deleted_variables_index.size();
1362 }
1363
1364 // If we removed variables or constraints we must flush all pending changes
1365 // to synchronize the number of variables and constraints with the Gurobi
1366 // model.
1367 RETURN_IF_GUROBI_ERROR(GRBupdatemodel(gurobi_model_));
1368 // Regenerate indices.
1369 RETURN_IF_ERROR(UpdateGurobiIndices());
1370
1371 return absl::OkStatus();
1372}

1435absl::StatusOr<SolveResultProto> GurobiSolver::Solve(
1436 const SolveParametersProto& parameters,
1437 const ModelSolveParametersProto& model_parameters,
1438 const CallbackRegistrationProto& callback_registration, const Callback cb) {
1439 const absl::Time start = absl::Now();
1440 SolveResultProto solve_result;
1441 // We must set the parameters before calling RegisterCallback since it changes
1442 // some parameters depending on the callback registration.
1443 const std::vector<absl::Status> param_errors = SetParameters(parameters);
1444 if (!param_errors.empty()) {
1445 if (parameters.common_parameters().strictness().bad_parameter()) {
1446 return absl::InvalidArgumentError(JoinErrors(param_errors));
1447 } else {
1448 for (const absl::Status& param_error : param_errors) {
1449 solve_result.add_warnings(std::string(param_error.message()));
1450 }

1452 }
1453
1454 // TODO(user): any warnings above will be lost if we get an error Status
1455 // below, reconsider the Solve API.
1456 std::unique_ptr<GurobiCallbackData> gurobi_cb_data;
1457 if (cb != nullptr) {
1458 ASSIGN_OR_RETURN(gurobi_cb_data,
1459 RegisterCallback(callback_registration, cb, start));
1460 RETURN_IF_GUROBI_ERROR(GRBsetcallbackfunc(gurobi_model_, GurobiCallback,
1461 gurobi_cb_data.get()));
1462 }
1463 auto callback_cleanup = absl::MakeCleanup([&]() {
1464 if (cb != nullptr) {
1465 GRBsetcallbackfunc(gurobi_model_, nullptr, nullptr);
1466 }
1467 });
1468
1469 // Need to run GRBupdatemodel before setting basis and getting the obj sense.
1470 CHECK(gurobi_model_ != nullptr);
1471 RETURN_IF_GUROBI_ERROR(GRBupdatemodel(gurobi_model_));
1472
1473 ASSIGN_OR_RETURN(const int obj_sense, GetIntAttr(GRB_INT_ATTR_MODELSENSE));
1474 const bool is_maximize = obj_sense == GRB_MAXIMIZE;
1475
1476 if (model_parameters.has_initial_basis()) {
1477 RETURN_IF_ERROR(SetGurobiBasis(model_parameters.initial_basis()));

1479

1481 // Check for callback errors when GurobiSolver::GurobiCallback signals their
1482 // existence through GRB_ERROR_CALLBACK and when GRBterminate is called (in
1483 // which case gurobi_error could also be kGrbOk)
1484 if (gurobi_error == kGrbOk && gurobi_cb_data != nullptr &&
1485 gurobi_cb_data->status.ok()) {
1486 gurobi_cb_data->status = GurobiCallbackImplFlush(
1487 gurobi_cb_data->callback_input, gurobi_cb_data->message_callback_data);
1488 }
1489 if ((gurobi_error == GRB_ERROR_CALLBACK || gurobi_error == kGrbOk) &&

1491 RETURN_IF_ERROR(gurobi_cb_data->status) << "Error in callback";
1492 }
1493 RETURN_IF_GUROBI_ERROR(gurobi_error);
1494 if (cb != nullptr) {
1495 std::move(callback_cleanup).Cancel();
1496 RETURN_IF_GUROBI_ERROR(GRBsetcallbackfunc(gurobi_model_, nullptr, nullptr));
1497 }
1498
1499 RETURN_IF_ERROR(
1500 ExtractSolveResultProto(is_maximize, solve_result, model_parameters));
1501 // Reset Gurobi parameters.
1502 // TODO(user): ensure that resetting parameters does not degrade
1503 // incrementalism performance.
1504 RETURN_IF_ERROR(ResetParameters());
1505 CHECK_OK(util_time::EncodeGoogleApiProto(
1506 absl::Now() - start,
1507 solve_result.mutable_solve_stats()->mutable_solve_time()));
1508 return solve_result;
1509}
