33// A value type that references a linear constraint from IndexedModel. Usually
34// this type is passed by copy.
35class LinearConstraint {
36 public:
37 // The typed integer used for ids.
38 using IdType = LinearConstraintId;
39
40 inline LinearConstraint(IndexedModel* model, LinearConstraintId id);
41
42 inline int64_t id() const;
43
44 inline LinearConstraintId typed_id() const;
45 inline IndexedModel* model() const;
46
47 inline double lower_bound() const;
48 inline double upper_bound() const;
49 inline const std::string& name() const;
50
51 inline void set_lower_bound(double lower_bound) const;
52 inline void set_upper_bound(double upper_bound) const;
53
54 // Setting a value to 0.0 will delete the {constraint, variable} pair from the
55 // underlying sparse matrix representation (and has no effect if the pair is
56 // not present).
57 inline void set_coefficient(Variable variable, double value) const;
58
59 inline bool is_coefficient_nonzero(Variable variable) const;
60
61 // Returns 0.0 if the variable is not used in the constraint.
62 inline double coefficient(Variable variable) const;
63
64 std::vector<Variable> RowNonzeros() const;
65
66 BoundedLinearExpression AsBoundedLinearExpression() const;
67
68 friend inline bool operator==(const LinearConstraint& lhs,
69 const LinearConstraint& rhs);
70 friend inline bool operator!=(const LinearConstraint& lhs,
71 const LinearConstraint& rhs);
72 template <typename H>
73 friend H AbslHashValue(H h, const LinearConstraint& linear_constraint);
74 friend std::ostream& operator<<(std::ostream& ostr,
75 const LinearConstraint& linear_constraint);
76
77 private:
78 IndexedModel* model_;
79 LinearConstraintId id_;
80};

116void LinearConstraint::set_coefficient(const Variable variable,
117 const double value) const {
118 CHECK_EQ(variable.model(), model_) << internal::kObjectsFromOtherIndexedModel;
119 model_->set_linear_constraint_coefficient(id_, variable.typed_id(), value);
120}
121bool LinearConstraint::is_coefficient_nonzero(const Variable variable) const {
122 CHECK_EQ(variable.model(), model_) << internal::kObjectsFromOtherIndexedModel;
123 return model_->is_linear_constraint_coefficient_nonzero(id_,
124 variable.typed_id());
125}
126double LinearConstraint::coefficient(const Variable variable) const {
127 CHECK_EQ(variable.model(), model_) << internal::kObjectsFromOtherIndexedModel;
128 return model_->linear_constraint_coefficient(id_, variable.typed_id());
129}
130
131bool operator==(const LinearConstraint& lhs, const LinearConstraint& rhs) {
132 return lhs.id_ == rhs.id_ && lhs.model_ == rhs.model_;
133}
134
135bool operator!=(const LinearConstraint& lhs, const LinearConstraint& rhs) {
136 return !(lhs == rhs);
137}
138
139template <typename H>
140H AbslHashValue(H h, const LinearConstraint& linear_constraint) {
141 return H::combine(std::move(h), linear_constraint.id_.value(),
142 linear_constraint.model_);
143}
144
145std::ostream& operator<<(std::ostream& ostr,
146 const LinearConstraint& linear_constraint) {
147 ostr << linear_constraint.name();
148 return ostr;
149}
