14// This header defines the common properties of "key types" and some related
15// constants.
16//
17// MathOpt provides optimized custom collections for variables and
18// constraints. This file contains implementation details for these custom
19// collections and should not be needed by users.
20//
21// Key types are types that are used as identifiers in the C++ interface where
22// the IndexedModel is using typed integers. They are pairs of (model,
23// typed_index) where `model` is a pointer on an IndexedModel and `typed_index`
24// is the typed integer type used in IndexedModel.
25//
26// A key type K must match the following requirements:
27// - K::IdType is a typed integer used for indices.
28// - K has a constructor K(IndexedModel*, K::IdType).
29// - K is a value-semantic type.
30// - K has a function with signature `K::IdType K::typed_id() const`.
31// - K has a function with signature `IndexedModel* K::model() const`. It
32// must return a non-null pointer.
33// - K::IdType is a valid key for absl::flat_hash_map or absl::flat_hash_set
34// (supports hash and ==).
35//
36// These requirements are met by Variable and LinearConstraint.

50// The CHECK message to use when a KeyType::model() is nullptr.
51inline constexpr absl::string_view kKeyHasNullIndexedModel =
52 "The input key has null model().";
53
54// The CHECK message to use when two KeyType with different models() are used in
55// the same collection.
56inline constexpr absl::string_view kObjectsFromOtherIndexedModel =
57 "The input objects belongs to another model.";
58
59// CHECKs that the non-null models the same, and returns the unique non-null
60// model if it exists, otherwise null.
61inline IndexedModel* ConsistentModel(
62 std::initializer_list<IndexedModel*> models) {
63 IndexedModel* result = nullptr;
64 for (IndexedModel* const model : models) {
65 if (model != nullptr) {
66 if (result == nullptr) {
67 result = model;
68 } else {
69 CHECK_EQ(model, result) << internal::kObjectsFromOtherIndexedModel;
70 }
71 }
72 }
73 return result;
74}
