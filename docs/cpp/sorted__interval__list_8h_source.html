28 // Represents a closed interval [start, end]. We must have start <= end.
29 struct ClosedInterval {
30  ClosedInterval() {}
31  ClosedInterval(int64 s, int64 e) : start(s), end(e) {}
32 
33  std::string DebugString() const;
34  bool operator==(const ClosedInterval& other) const {
35  return start == other.start && end == other.end;
36  }

38  // Because we mainly manipulate vector of disjoint intervals, we only need to
39  // sort by the start. We do not care about the order in which interval with
40  // the same start appear since they will always be merged into one interval.
41  bool operator<(const ClosedInterval& other) const {
42  return start < other.start;
43  }

165  // Advanced usage. Given some "implied" information on this domain that is
166  // assumed to be always true (i.e. only values in the intersection with
167  // implied domain matter), this function will simplify the current domain
168  // without changing the set of "possible values".
169  //
170  // More precisely, this will:
171  // - Take the intersection with implied_domain.
172  // - Minimize the number of intervals. That is, if the
173  // domain is like [1,2][4] and implied is [1][4], then the domain can be
174  // relaxed to [1, 4] to simplify its complexity without changing the set

178  // not be changed.

237 // This class represents a sorted list of disjoint, closed intervals. When an
238 // interval is inserted, all intervals that overlap it or that are even adjacent
239 // to it are merged into one. I.e. [0,14] and [15,30] will be merged to [0,30].
240 //
241 // Iterators returned by this class are invalidated by non-const operations.
242 //
243 // TODO(user): Templatize the class on the type of the bounds.
244 class SortedDisjointIntervalList {
245  public:
246  struct IntervalComparator {
247  bool operator()(const ClosedInterval& a, const ClosedInterval& b) const {
248  return a.start != b.start ? a.start < b.start : a.end < b.end;
249  }
250  };
251  typedef std::set<ClosedInterval, IntervalComparator> IntervalSet;
252  typedef IntervalSet::iterator Iterator;

258  // Creates a SortedDisjointIntervalList and fills it with intervals
259  // [starts[i]..ends[i]]. All intervals must be consistent (starts[i] <=
260  // ends[i]). There's two version, one for int64, one for int.
261  //
262  // TODO(user): Explain why we favored this API to the more natural
263  // input std::vector<ClosedInterval> or std::vector<std::pair<int, int>>.
264  SortedDisjointIntervalList(const std::vector<int64>& starts,
265  const std::vector<int64>& ends);
266  SortedDisjointIntervalList(const std::vector<int>& starts,
267  const std::vector<int>& ends);

269  // Builds the complement of the interval list on the interval [start, end].

272  // Adds the interval [start..end] to the list, and merges overlapping or
273  // immediately adjacent intervals ([2, 5] and [6, 7] are adjacent, but
274  // [2, 5] and [7, 8] are not).
275  //
276  // Returns an iterator to the inserted interval (possibly merged with others).
277  //
278  // If start > end, it does LOG(DFATAL) and returns end() (no interval added).
279  Iterator InsertInterval(int64 start, int64 end);
280 
281  // If value is in an interval, increase its end by one, otherwise insert the
282  // interval [value, value]. In both cases, this returns an iterator to the
283  // new/modified interval (possibly merged with others) and fills newly_covered
284  // with the new value that was just added in the union of all the intervals.
285  //
286  // If this causes an interval ending at kint64max to grow, it will die with a
287  // CHECK fail.
288  Iterator GrowRightByOne(int64 value, int64* newly_covered);
289 
290  // Adds all intervals [starts[i]..ends[i]]. Same behavior as InsertInterval()
291  // upon invalid intervals. There's a version with int64 and int32.
292  void InsertIntervals(const std::vector<int64>& starts,
293  const std::vector<int64>& ends);
294  void InsertIntervals(const std::vector<int>& starts,
295  const std::vector<int>& ends);
296 
297  // Returns the number of disjoint intervals in the list.
298  int NumIntervals() const { return intervals_.size(); }
299 
300  // Returns an iterator to either:
301  // - the first interval containing or above the given value, or
302  // - the last interval containing or below the given value.
303  // Returns end() if no interval fulfils that condition.
304  //
305  // If the value is within an interval, both functions will return it.
306  Iterator FirstIntervalGreaterOrEqual(int64 value) const;
307  Iterator LastIntervalLessOrEqual(int64 value) const;
308 
309  std::string DebugString() const;

314  // for (const ClosedInterval interval : list) {
315  // ...
316  // }
317  const Iterator begin() const { return intervals_.begin(); }

319 
320  // Returns a const& to the last interval. The list must not be empty.
