37 bool IntervalsAreSortedAndNonAdjacent(
38  absl::Span<const ClosedInterval> intervals) {
39  for (int i = 1; i < intervals.size(); ++i) {
40  if (intervals[i - 1].start > intervals[i - 1].end) return false;
41  // First test make sure that intervals[i - 1].end + 1 will not overflow.
42  if (intervals[i - 1].end >= intervals[i].start ||
43  intervals[i - 1].end + 1 >= intervals[i].start) {
44  return false;
45  }
46  }
47  return intervals.empty() ? true
48  : intervals.back().start <= intervals.back().end;
49 }

62 // Transforms a sorted list of intervals in a sorted DISJOINT list for which
63 // IntervalsAreSortedAndNonAdjacent() would return true.
64 void UnionOfSortedIntervals(absl::InlinedVector<ClosedInterval, 1>* intervals) {
65  DCHECK(std::is_sorted(intervals->begin(), intervals->end()));
66  int new_size = 0;
67  for (const ClosedInterval& i : *intervals) {
68  if (new_size > 0 && i.start <= CapAdd((*intervals)[new_size - 1].end, 1)) {
69  (*intervals)[new_size - 1].end =
70  std::max(i.end, (*intervals)[new_size - 1].end);
71  } else {
72  (*intervals)[new_size++] = i;
73  }
74  }
75  intervals->resize(new_size);
76 
77  // This is important for InlinedVector in the case the result is a single
78  // intervals.
79  intervals->shrink_to_fit();
80  DCHECK(IntervalsAreSortedAndNonAdjacent(*intervals));
81 }

86 int64 CeilRatio(int64 value, int64 positive_coeff) {
87  DCHECK_GT(positive_coeff, 0);
88  const int64 result = value / positive_coeff;
89  const int64 adjust = static_cast<int64>(result * positive_coeff < value);
90  return result + adjust;

93 int64 FloorRatio(int64 value, int64 positive_coeff) {
94  DCHECK_GT(positive_coeff, 0);
95  const int64 result = value / positive_coeff;
96  const int64 adjust = static_cast<int64>(result * positive_coeff > value);
97  return result - adjust;
98 }

221 bool Domain::Contains(int64 value) const {
222  // Because we only compare by start and there is no duplicate starts, this
223  // should be the next interval after the one that has a chance to contains
224  // value.
225  auto it = std::upper_bound(intervals_.begin(), intervals_.end(),
226  ClosedInterval(value, value));
227  if (it == intervals_.begin()) return false;
228  --it;
229  return value <= it->end;

233  int i = 0;
234  const auto& others = domain.intervals_;
235  for (const ClosedInterval interval : intervals_) {
236  // Find the unique interval in others that contains interval if any.
237  for (; i < others.size() && interval.end > others[i].end; ++i) {

239  if (i == others.size()) return false;
240  if (interval.start < others[i].start) return false;

242  return true;

245 Domain Domain::Complement() const {
246  Domain result;
247  int64 next_start = kint64min;
248  result.intervals_.reserve(intervals_.size() + 1);
249  for (const ClosedInterval& interval : intervals_) {
250  if (interval.start != kint64min) {
251  result.intervals_.push_back({next_start, interval.start - 1});
252  }
253  if (interval.end == kint64max) return result;
254  next_start = interval.end + 1;
255  }
256  result.intervals_.push_back({next_start, kint64max});
257  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
258  return result;
259 }

267 void Domain::NegateInPlace() {
268  if (intervals_.empty()) return;
269  std::reverse(intervals_.begin(), intervals_.end());
270  if (intervals_.back().end == kint64min) {
271  // corner-case
272  intervals_.pop_back();
273  }
274  for (ClosedInterval& ref : intervals_) {
275  std::swap(ref.start, ref.end);
276  ref.start = ref.start == kint64min ? kint64max : -ref.start;
277  ref.end = ref.end == kint64min ? kint64max : -ref.end;
278  }
279  DCHECK(IntervalsAreSortedAndNonAdjacent(intervals_));
280 }

282 Domain Domain::IntersectionWith(const Domain& domain) const {
283  Domain result;
284  const auto& a = intervals_;
285  const auto& b = domain.intervals_;
286  for (int i = 0, j = 0; i < a.size() && j < b.size();) {
287  if (a[i].start <= b[j].start) {
288  if (a[i].end < b[j].start) {
289  // Empty intersection. We advance past the first interval.
290  ++i;
291  } else { // a[i].end >= b[j].start
292  // Non-empty intersection: push back the intersection of these two, and
293  // advance past the first interval to finish.
294  if (a[i].end <= b[j].end) {
295  result.intervals_.push_back({b[j].start, a[i].end});
296  ++i;
297  } else { // a[i].end > b[j].end.
298  result.intervals_.push_back({b[j].start, b[j].end});
299  ++j;
300  }
301  }
302  } else { // a[i].start > b[i].start.
303  // We do the exact same thing as above, but swapping a and b.
304  if (b[j].end < a[i].start) {
305  ++j;
306  } else { // b[j].end >= a[i].start
307  if (b[j].end <= a[i].end) {
308  result.intervals_.push_back({a[i].start, b[j].end});
309  ++j;
310  } else { // a[i].end > b[j].end.
311  result.intervals_.push_back({a[i].start, a[i].end});
312  ++i;
313  }
314  }
315  }
316  }
317  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
318  return result;
319 }

361 Domain Domain::MultiplicationBy(int64 coeff, bool* exact) const {
362  if (exact != nullptr) *exact = true;
363  if (intervals_.empty() || coeff == 0) return {};
364 
365  const int64 abs_coeff = std::abs(coeff);
366  const int64 size_if_non_trivial = abs_coeff > 1 ? Size() : 0;
367  if (size_if_non_trivial > kDomainComplexityLimit) {
368  if (exact != nullptr) *exact = false;
369  return ContinuousMultiplicationBy(coeff);
370  }
371 
372  Domain result;
373  if (abs_coeff > 1) {
374  result.intervals_.reserve(size_if_non_trivial);
375  for (const ClosedInterval& i : intervals_) {
376  for (int v = i.start; v <= i.end; ++v) {
377  // Because abs_coeff > 1, all new values are disjoint.
378  const int64 new_value = CapProd(v, abs_coeff);
379  result.intervals_.push_back({new_value, new_value});
380  }
381  }
382  } else {
383  result = *this;
384  }
385  if (coeff < 0) result.NegateInPlace();
386  return result;
387 }

389 Domain Domain::ContinuousMultiplicationBy(int64 coeff) const {
390  Domain result = *this;
391  const int64 abs_coeff = std::abs(coeff);
392  for (ClosedInterval& i : result.intervals_) {
393  i.start = CapProd(i.start, abs_coeff);
394  i.end = CapProd(i.end, abs_coeff);
395  }
396  UnionOfSortedIntervals(&result.intervals_);
397  if (coeff < 0) result.NegateInPlace();
398  return result;
399 }

401 Domain Domain::ContinuousMultiplicationBy(const Domain& domain) const {
402  Domain result;
403  for (const ClosedInterval& i : this->intervals_) {
404  for (const ClosedInterval& j : domain.intervals_) {
405  ClosedInterval new_interval;
406  const int64 a = CapProd(i.start, j.start);
407  const int64 b = CapProd(i.end, j.end);
408  const int64 c = CapProd(i.start, j.end);
409  const int64 d = CapProd(i.end, j.start);
410  new_interval.start = std::min({a, b, c, d});
411  new_interval.end = std::max({a, b, c, d});
412  result.intervals_.push_back(new_interval);
413  }
414  }
415  std::sort(result.intervals_.begin(), result.intervals_.end());
416  UnionOfSortedIntervals(&result.intervals_);
417  return result;
418 }

421  CHECK_NE(coeff, 0);
422  Domain result = *this;
423  const int64 abs_coeff = std::abs(coeff);
424  for (ClosedInterval& i : result.intervals_) {
425  i.start = i.start / abs_coeff;

433 Domain Domain::InverseMultiplicationBy(const int64 coeff) const {
434  if (coeff == 0) {
435  return Contains(0) ? Domain::AllValues() : Domain();
436  }
437  Domain result = *this;
438  int new_size = 0;
439  const int64 abs_coeff = std::abs(coeff);
440  for (const ClosedInterval& i : result.intervals_) {
441  const int64 start = CeilRatio(i.start, abs_coeff);
442  const int64 end = FloorRatio(i.end, abs_coeff);
443  if (start > end) continue;
444  if (new_size > 0 && start == result.intervals_[new_size - 1].end + 1) {
445  result.intervals_[new_size - 1].end = end;
446  } else {
447  result.intervals_[new_size++] = {start, end};
448  }
449  }
450  result.intervals_.resize(new_size);
451  result.intervals_.shrink_to_fit();
452  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
453  if (coeff < 0) result.NegateInPlace();
454  return result;
455 }

457 // It is a bit difficult to see, but this code is doing the same thing as
458 // for all interval in this.UnionWith(implied_domain.Complement())):
459 // - Take the two extreme points (min and max) in interval \inter implied.
460 // - Append to result [min, max] if these points exists.
461 Domain Domain::SimplifyUsingImpliedDomain(const Domain& implied_domain) const {
462  Domain result;
463  if (implied_domain.IsEmpty()) return result;
464 
465  int i = 0;
466  int64 min_point;
467  int64 max_point;
468  bool started = false;
469  for (const ClosedInterval interval : intervals_) {
470  // We only "close" the new result interval if it cannot be extended by
471  // implied_domain.Complement(). The only extension possible look like:
472  // interval_: ...] [....
473  // implied : ...] [... i ...]
474  if (started && implied_domain.intervals_[i].start < interval.start) {
475  result.intervals_.push_back({min_point, max_point});
476  started = false;
477  }
478 
479  // Find the two extreme points in interval \inter implied_domain.
480  // Always stop the loop at the first interval with and end strictly greater
481  // that interval.end.
482  for (; i < implied_domain.intervals_.size(); ++i) {
483  const ClosedInterval current = implied_domain.intervals_[i];
484  if (current.end >= interval.start && current.start <= interval.end) {
485  // Current and interval have a non-empty intersection.
486  const int64 inter_max = std::min(interval.end, current.end);
487  if (!started) {
488  started = true;
489  min_point = std::max(interval.start, current.start);
490  max_point = inter_max;
491  } else {
492  // No need to update the min_point here, and the new inter_max must
493  // necessarily be > old one.
494  DCHECK_GE(inter_max, max_point);
495  max_point = inter_max;
496  }
497  }
498  if (current.end > interval.end) break;
499  }
500  if (i == implied_domain.intervals_.size()) break;
501  }
502  if (started) {
503  result.intervals_.push_back({min_point, max_point});
504  }
505  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
506  return result;
507 }

518 bool Domain::operator<(const Domain& other) const {
519  const auto& d1 = intervals_;
520  const auto& d2 = other.intervals_;
521  const int common_size = std::min(d1.size(), d2.size());
522  for (int i = 0; i < common_size; ++i) {
523  const ClosedInterval& i1 = d1[i];
524  const ClosedInterval& i2 = d2[i];
525  if (i1.start < i2.start) return true;
526  if (i1.start > i2.start) return false;

528  if (i1.end > i2.end) return false;
529  }
530  return d1.size() < d2.size();
531 }

535 int64 SumOfKMinValueInDomain(const Domain& domain, int k) {
536  int64 current_sum = 0.0;
537  int current_index = 0;

539  if (current_index >= k) break;

572 SortedDisjointIntervalList
573 SortedDisjointIntervalList::BuildComplementOnInterval(int64 start, int64 end) {
574  SortedDisjointIntervalList interval_list;
575  int64 next_start = start;
576  for (auto it = FirstIntervalGreaterOrEqual(start); it != this->end(); ++it) {
577  const ClosedInterval& interval = *it;
578  const int64 next_end = CapSub(interval.start, 1);
579  if (next_end > end) break;
580  if (next_start <= next_end) {
581  interval_list.InsertInterval(next_start, next_end);
582  }
583  next_start = CapAdd(interval.end, 1);
584  }
585  if (next_start <= end) {
586  interval_list.InsertInterval(next_start, end);
587  }
588  return interval_list;
589 }

591 SortedDisjointIntervalList::Iterator SortedDisjointIntervalList::InsertInterval(
592  int64 start, int64 end) {
593  // start > end could mean an empty interval, but we prefer to LOG(DFATAL)
594  // anyway. Really, the user should never give us that.
595  if (start > end) {
596  LOG(DFATAL) << "Invalid interval: " << ClosedInterval({start, end});
597  return intervals_.end();
598  }
599 
600  auto result = intervals_.insert({start, end});
601  if (!result.second) return result.first; // Duplicate: exit immediately.
602 
603  // TODO(user): tune the algorithm below if it proves to be a bottleneck.
604  // For example, one could try to avoid an insertion if it's not needed
605  // (when the interval merges with a single existing interval or is fully
606  // contained by one).
607 
608  // Iterate over the previous iterators whose end is after (or almost at) our
609  // start. After this, "it1" will point to the first interval that needs to be
610  // merged with the current interval (possibly pointing to the current interval
611  // itself, if no "earlier" interval should be merged).
612  auto it1 = result.first;
613  if (start == kint64min) { // Catch underflows
614  it1 = intervals_.begin();
615  } else {
616  const int64 before_start = start - 1;
617  while (it1 != intervals_.begin()) {
618  auto prev_it = it1;
619  --prev_it;
620  if (prev_it->end < before_start) break;
621  it1 = prev_it;
622  }
623  }
624 
625  // Ditto, on the other side: "it2" will point to the interval *after* the last
626  // one that should be merged with the current interval.
627  auto it2 = result.first;
628  if (end == kint64max) {
629  it2 = intervals_.end();
630  } else {
631  const int64 after_end = end + 1;
632  do {
633  ++it2;
634  } while (it2 != intervals_.end() && it2->start <= after_end);
635  }
636 
637  // [it1..it2) is the range (inclusive on it1, exclusive on it2) of intervals
638  // that should be merged together. We'll set it3 = it2-1 and erase [it1..it3)
639  // and set *it3 to the merged interval.
640  auto it3 = it2;
641  it3--;
642  if (it1 == it3) return it3; // Nothing was merged.
643  const int64 new_start = std::min(it1->start, start);
644  const int64 new_end = std::max(it3->end, end);
645  auto it = intervals_.erase(it1, it3);
646  // HACK(user): set iterators point to *const* values. Which is expected,
647  // because if one alters a set element's value, then it collapses the set
648  // property! But in this very special case, we know that we can just overwrite
649  // it->start, so we do it.
650  const_cast<ClosedInterval*>(&(*it))->start = new_start;
651  const_cast<ClosedInterval*>(&(*it))->end = new_end;
652  return it;
653 }

655 SortedDisjointIntervalList::Iterator SortedDisjointIntervalList::GrowRightByOne(
656  int64 value, int64* newly_covered) {
657  auto it = intervals_.upper_bound({value, kint64max});
658  auto it_prev = it;
659 
660  // No interval containing or adjacent to "value" on the left (i.e. below).
661  if (it != begin()) {
662  --it_prev;
663  }
664  if (it == begin() || ((value != kint64min) && it_prev->end < value - 1)) {
665  *newly_covered = value;
666  if (it == end() || it->start != value + 1) {
667  // No interval adjacent to "value" on the right: insert a singleton.
668  return intervals_.insert(it, {value, value});
669  } else {
670  // There is an interval adjacent to "value" on the right. Extend it by
671  // one. Note that we already know that there won't be a merge with another
672  // interval on the left, since there were no interval adjacent to "value"
673  // on the left.
674  DCHECK_EQ(it->start, value + 1);
675  const_cast<ClosedInterval*>(&(*it))->start = value;
676  return it;
677  }
678  }
679 
680  // At this point, "it_prev" points to an interval containing or adjacent to
681  // "value" on the left: grow it by one, and if it now touches the next
682  // interval, merge with it.
683  CHECK_NE(kint64max, it_prev->end) << "Cannot grow right by one: the interval "
684  "that would grow already ends at "
685  "kint64max";
686  *newly_covered = it_prev->end + 1;
687  if (it != end() && it_prev->end + 2 == it->start) {
688  // We need to merge it_prev with 'it'.
689  const_cast<ClosedInterval*>(&(*it_prev))->end = it->end;
690  intervals_.erase(it);
691  } else {
692  const_cast<ClosedInterval*>(&(*it_prev))->end = it_prev->end + 1;
693  }
694  return it_prev;
695 }
