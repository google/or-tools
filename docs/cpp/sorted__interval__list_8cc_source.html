38 bool IntervalsAreSortedAndNonAdjacent(
39  absl::Span<const ClosedInterval> intervals) {
40  for (int i = 1; i < intervals.size(); ++i) {
41  if (intervals[i - 1].start > intervals[i - 1].end) return false;
42  // First test make sure that intervals[i - 1].end + 1 will not overflow.
43  if (intervals[i - 1].end >= intervals[i].start ||
44  intervals[i - 1].end + 1 >= intervals[i].start) {
45  return false;
46  }
47  }
48  return intervals.empty() ? true
49  : intervals.back().start <= intervals.back().end;
50 }

64 // Transforms a sorted list of intervals in a sorted DISJOINT list for which
65 // IntervalsAreSortedAndNonAdjacent() would return true.
66 void UnionOfSortedIntervals(absl::InlinedVector<ClosedInterval, 1>* intervals) {
67  DCHECK(std::is_sorted(intervals->begin(), intervals->end()));
68  const int size = intervals->size();
69  if (size == 0) return;
70 
71  int new_size = 1;
72  for (int i = 1; i < size; ++i) {
73  const ClosedInterval& current = (*intervals)[i];
74  const int64_t end = (*intervals)[new_size - 1].end;
75  if (end == std::numeric_limits<int64_t>::max() ||
76  current.start <= end + 1) {
77  (*intervals)[new_size - 1].end = std::max(current.end, end);
78  continue;
79  }
80  (*intervals)[new_size++] = current;
81  }
82  intervals->resize(new_size);
83 
84  // This is important for InlinedVector in the case the result is a single
85  // intervals.
86  intervals->shrink_to_fit();
87  DCHECK(IntervalsAreSortedAndNonAdjacent(*intervals));
88 }

93 int64_t CeilRatio(int64_t value, int64_t positive_coeff) {
94  DCHECK_GT(positive_coeff, 0);
95  const int64_t result = value / positive_coeff;
96  const int64_t adjust = static_cast<int64_t>(result * positive_coeff < value);
97  return result + adjust;
98 }

100 int64_t FloorRatio(int64_t value, int64_t positive_coeff) {
101  DCHECK_GT(positive_coeff, 0);
102  const int64_t result = value / positive_coeff;
103  const int64_t adjust = static_cast<int64_t>(result * positive_coeff > value);
104  return result - adjust;
105 }

203  int64_t size = 0;
204  for (const ClosedInterval interval : intervals_) {
205  size = operations_research::CapAdd(

242 bool Domain::Contains(int64_t value) const {
243  // Because we only compare by start and there is no duplicate starts, this
244  // should be the next interval after the one that has a chance to contains
245  // value.
246  auto it = std::upper_bound(intervals_.begin(), intervals_.end(),
247  ClosedInterval(value, value));
248  if (it == intervals_.begin()) return false;
249  --it;
250  return value <= it->end;
251 }

253 bool Domain::IsIncludedIn(const Domain& domain) const {
254  int i = 0;
255  const auto& others = domain.intervals_;
256  for (const ClosedInterval interval : intervals_) {
257  // Find the unique interval in others that contains interval if any.
258  for (; i < others.size() && interval.end > others[i].end; ++i) {
259  }
260  if (i == others.size()) return false;
261  if (interval.start < others[i].start) return false;
262  }
263  return true;
264 }

266 Domain Domain::Complement() const {
267  Domain result;
268  int64_t next_start = kint64min;
269  result.intervals_.reserve(intervals_.size() + 1);
270  for (const ClosedInterval& interval : intervals_) {
271  if (interval.start != kint64min) {
272  result.intervals_.push_back({next_start, interval.start - 1});
273  }
274  if (interval.end == kint64max) return result;
275  next_start = interval.end + 1;
276  }
277  result.intervals_.push_back({next_start, kint64max});
278  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
279  return result;
280 }

288 void Domain::NegateInPlace() {
289  if (intervals_.empty()) return;
290  std::reverse(intervals_.begin(), intervals_.end());
291  if (intervals_.back().end == kint64min) {
292  // corner-case
293  intervals_.pop_back();
294  }
295  for (ClosedInterval& ref : intervals_) {
296  std::swap(ref.start, ref.end);
297  ref.start = ref.start == kint64min ? kint64max : -ref.start;
298  ref.end = ref.end == kint64min ? kint64max : -ref.end;
299  }
300  DCHECK(IntervalsAreSortedAndNonAdjacent(intervals_));
301 }

303 Domain Domain::IntersectionWith(const Domain& domain) const {
304  Domain result;
305  const auto& a = intervals_;
306  const auto& b = domain.intervals_;
307  for (int i = 0, j = 0; i < a.size() && j < b.size();) {
308  if (a[i].start <= b[j].start) {
309  if (a[i].end < b[j].start) {
310  // Empty intersection. We advance past the first interval.
311  ++i;
312  } else { // a[i].end >= b[j].start
313  // Non-empty intersection: push back the intersection of these two, and
314  // advance past the first interval to finish.
315  if (a[i].end <= b[j].end) {
316  result.intervals_.push_back({b[j].start, a[i].end});
317  ++i;
318  } else { // a[i].end > b[j].end.
319  result.intervals_.push_back({b[j].start, b[j].end});
320  ++j;
321  }
322  }
323  } else { // a[i].start > b[i].start.
324  // We do the exact same thing as above, but swapping a and b.
325  if (b[j].end < a[i].start) {
326  ++j;
327  } else { // b[j].end >= a[i].start
328  if (b[j].end <= a[i].end) {
329  result.intervals_.push_back({a[i].start, b[j].end});
330  ++j;
331  } else { // a[i].end > b[j].end.
332  result.intervals_.push_back({a[i].start, a[i].end});
333  ++i;
334  }
335  }
336  }
337  }
338  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
339  return result;
340 }

382 Domain Domain::MultiplicationBy(int64_t coeff, bool* exact) const {
383  if (exact != nullptr) *exact = true;
384  if (intervals_.empty()) return {};
385  if (coeff == 0) return Domain(0);
386 
387  const int64_t abs_coeff = std::abs(coeff);
388  const int64_t size_if_non_trivial = abs_coeff > 1 ? Size() : 0;
389  if (size_if_non_trivial > kDomainComplexityLimit) {
390  if (exact != nullptr) *exact = false;
391  return ContinuousMultiplicationBy(coeff);
392  }
393 
394  Domain result;
395  if (abs_coeff > 1) {
396  const int64_t max_value = kint64max / abs_coeff;
397  const int64_t min_value = kint64min / abs_coeff;
398  result.intervals_.reserve(size_if_non_trivial);
399  for (const ClosedInterval& i : intervals_) {
400  for (int64_t v = i.start;; ++v) {
401  // We ignore anything that overflow.
402  if (v >= min_value && v <= max_value) {
403  // Because abs_coeff > 1, all new values are disjoint.
404  const int64_t new_value = v * abs_coeff;
405  result.intervals_.push_back({new_value, new_value});
406  }
407 
408  // This is to avoid doing ++v when v is kint64max!
409  if (v == i.end) break;
410  }
411  }
412  } else {
413  result = *this;
414  }
415  if (coeff < 0) result.NegateInPlace();
416  return result;
417 }

419 Domain Domain::ContinuousMultiplicationBy(int64_t coeff) const {
420  Domain result = *this;
421  const int64_t abs_coeff = std::abs(coeff);
422  for (ClosedInterval& i : result.intervals_) {
423  i.start = CapProd(i.start, abs_coeff);
424  i.end = CapProd(i.end, abs_coeff);
425  }
426  UnionOfSortedIntervals(&result.intervals_);
427  if (coeff < 0) result.NegateInPlace();
428  return result;
429 }

431 Domain Domain::ContinuousMultiplicationBy(const Domain& domain) const {
432  Domain result;
433  for (const ClosedInterval& i : this->intervals_) {
434  for (const ClosedInterval& j : domain.intervals_) {
435  ClosedInterval new_interval;
436  const int64_t a = CapProd(i.start, j.start);
437  const int64_t b = CapProd(i.end, j.end);
438  const int64_t c = CapProd(i.start, j.end);
439  const int64_t d = CapProd(i.end, j.start);
440  new_interval.start = std::min({a, b, c, d});
441  new_interval.end = std::max({a, b, c, d});
442  result.intervals_.push_back(new_interval);
443  }
444  }
445  std::sort(result.intervals_.begin(), result.intervals_.end());
446  UnionOfSortedIntervals(&result.intervals_);
447  return result;
448 }

463 Domain Domain::InverseMultiplicationBy(const int64_t coeff) const {
464  if (coeff == 0) {
465  return Contains(0) ? Domain::AllValues() : Domain();
466  }
467  Domain result = *this;
468  int new_size = 0;
469  const int64_t abs_coeff = std::abs(coeff);
470  for (const ClosedInterval& i : result.intervals_) {
471  const int64_t start = CeilRatio(i.start, abs_coeff);
472  const int64_t end = FloorRatio(i.end, abs_coeff);
473  if (start > end) continue;
474  if (new_size > 0 && start == result.intervals_[new_size - 1].end + 1) {
475  result.intervals_[new_size - 1].end = end;
476  } else {
477  result.intervals_[new_size++] = {start, end};
478  }
479  }
480  result.intervals_.resize(new_size);
481  result.intervals_.shrink_to_fit();
482  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
483  if (coeff < 0) result.NegateInPlace();
484  return result;
485 }

488 Domain ModuloHelper(int64_t min, int64_t max, const Domain& modulo) {
489  DCHECK_GT(min, 0);
490  DCHECK_GT(modulo.Min(), 0);
491  const int64_t max_mod = modulo.Max() - 1;
492 
493  // The min/max are exact if the modulo is fixed. Note that we could return the
494  // exact domain with a potential hole but we currently don't.
495  if (modulo.Min() == modulo.Max()) {
496  const int64_t size = max - min;
497  const int64_t v1 = min % modulo.Max();
498  if (v1 + size > max_mod) return Domain(0, max_mod);
499  return Domain(v1, v1 + size);
500  }
501 
502  // TODO(user): This is a superset.
503  return Domain(0, std::min(max, max_mod));
504 }

507 Domain Domain::PositiveModuloBySuperset(const Domain& modulo) const {
508  if (IsEmpty()) return Domain();

518  return ModuloHelper(-Max(), -Min(), modulo).Negation();

528 Domain Domain::SquareSuperset() const {
529  if (IsEmpty()) return Domain();
530  const Domain abs_domain =
531  IntersectionWith({0, std::numeric_limits<int64_t>::max()})
532  .UnionWith(Negation().IntersectionWith(
533  {0, std::numeric_limits<int64_t>::max()}));
534  if (abs_domain.Size() >= kDomainComplexityLimit) {

554 // It is a bit difficult to see, but this code is doing the same thing as
555 // for all interval in this.UnionWith(implied_domain.Complement())):
556 // - Take the two extreme points (min and max) in interval \inter implied.
557 // - Append to result [min, max] if these points exists.
558 Domain Domain::SimplifyUsingImpliedDomain(const Domain& implied_domain) const {
559  Domain result;
560  if (implied_domain.IsEmpty()) return result;
561 
562  int i = 0;
563  int64_t min_point;
564  int64_t max_point;
565  bool started = false;
566  for (const ClosedInterval interval : intervals_) {
567  // We only "close" the new result interval if it cannot be extended by
568  // implied_domain.Complement(). The only extension possible look like:
569  // interval_: ...] [....
570  // implied : ...] [... i ...]
571  if (started && implied_domain.intervals_[i].start < interval.start) {
572  result.intervals_.push_back({min_point, max_point});
573  started = false;
574  }
575 
576  // Find the two extreme points in interval \inter implied_domain.
577  // Always stop the loop at the first interval with and end strictly greater
578  // that interval.end.
579  for (; i < implied_domain.intervals_.size(); ++i) {
580  const ClosedInterval current = implied_domain.intervals_[i];
581  if (current.end >= interval.start && current.start <= interval.end) {
582  // Current and interval have a non-empty intersection.
583  const int64_t inter_max = std::min(interval.end, current.end);
584  if (!started) {
585  started = true;
586  min_point = std::max(interval.start, current.start);
587  max_point = inter_max;
588  } else {
589  // No need to update the min_point here, and the new inter_max must
590  // necessarily be > old one.
591  DCHECK_GE(inter_max, max_point);
592  max_point = inter_max;
593  }
594  }
595  if (current.end > interval.end) break;
596  }
597  if (i == implied_domain.intervals_.size()) break;
598  }
599  if (started) {
600  result.intervals_.push_back({min_point, max_point});
601  }
602  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
603  return result;
604 }

615 bool Domain::operator<(const Domain& other) const {
616  const auto& d1 = intervals_;
617  const auto& d2 = other.intervals_;
618  const int common_size = std::min(d1.size(), d2.size());
619  for (int i = 0; i < common_size; ++i) {
620  const ClosedInterval& i1 = d1[i];

670 SortedDisjointIntervalList::BuildComplementOnInterval(int64_t start,
671  int64_t end) {
672  SortedDisjointIntervalList interval_list;
673  int64_t next_start = start;
674  for (auto it = FirstIntervalGreaterOrEqual(start); it != this->end(); ++it) {
675  const ClosedInterval& interval = *it;
676  const int64_t next_end = CapSub(interval.start, 1);
677  if (next_end > end) break;
678  if (next_start <= next_end) {
679  interval_list.InsertInterval(next_start, next_end);
680  }
681  next_start = CapAdd(interval.end, 1);
682  }
683  if (next_start <= end) {
684  interval_list.InsertInterval(next_start, end);
685  }
686  return interval_list;
687 }

689 SortedDisjointIntervalList::Iterator SortedDisjointIntervalList::InsertInterval(
690  int64_t start, int64_t end) {
691  // start > end could mean an empty interval, but we prefer to LOG(DFATAL)
692  // anyway. Really, the user should never give us that.
693  if (start > end) {
694  LOG(DFATAL) << "Invalid interval: " << ClosedInterval({start, end});
695  return intervals_.end();
696  }
697 
698  auto result = intervals_.insert({start, end});
699  if (!result.second) return result.first; // Duplicate: exit immediately.
700 
701  // TODO(user): tune the algorithm below if it proves to be a bottleneck.
702  // For example, one could try to avoid an insertion if it's not needed
703  // (when the interval merges with a single existing interval or is fully
704  // contained by one).
705 
706  // Iterate over the previous iterators whose end is after (or almost at) our
707  // start. After this, "it1" will point to the first interval that needs to be
708  // merged with the current interval (possibly pointing to the current interval
709  // itself, if no "earlier" interval should be merged).
710  auto it1 = result.first;
711  if (start == kint64min) { // Catch underflows
712  it1 = intervals_.begin();
713  } else {
714  const int64_t before_start = start - 1;
715  while (it1 != intervals_.begin()) {
716  auto prev_it = it1;
717  --prev_it;
718  if (prev_it->end < before_start) break;
719  it1 = prev_it;
720  }
721  }
722 
723  // Ditto, on the other side: "it2" will point to the interval *after* the last
724  // one that should be merged with the current interval.
725  auto it2 = result.first;
726  if (end == kint64max) {
727  it2 = intervals_.end();
728  } else {
729  const int64_t after_end = end + 1;
730  do {
731  ++it2;
732  } while (it2 != intervals_.end() && it2->start <= after_end);
733  }
734 
735  // [it1..it2) is the range (inclusive on it1, exclusive on it2) of intervals
736  // that should be merged together. We'll set it3 = it2-1 and erase [it1..it3)
737  // and set *it3 to the merged interval.
738  auto it3 = it2;
739  it3--;
740  if (it1 == it3) return it3; // Nothing was merged.
741  const int64_t new_start = std::min(it1->start, start);
742  const int64_t new_end = std::max(it3->end, end);
743  auto it = intervals_.erase(it1, it3);
744  // HACK(user): set iterators point to *const* values. Which is expected,
745  // because if one alters a set element's value, then it collapses the set
746  // property! But in this very special case, we know that we can just overwrite
747  // it->start, so we do it.
748  const_cast<ClosedInterval*>(&(*it))->start = new_start;
749  const_cast<ClosedInterval*>(&(*it))->end = new_end;
750  return it;
751 }

753 SortedDisjointIntervalList::Iterator SortedDisjointIntervalList::GrowRightByOne(
754  int64_t value, int64_t* newly_covered) {
755  auto it = intervals_.upper_bound({value, kint64max});
756  auto it_prev = it;
757 
758  // No interval containing or adjacent to "value" on the left (i.e. below).
759  if (it != begin()) {
760  --it_prev;
761  }
762  if (it == begin() || ((value != kint64min) && it_prev->end < value - 1)) {
763  *newly_covered = value;
764  if (it == end() || it->start != value + 1) {
765  // No interval adjacent to "value" on the right: insert a singleton.
766  return intervals_.insert(it, {value, value});
767  } else {
768  // There is an interval adjacent to "value" on the right. Extend it by
769  // one. Note that we already know that there won't be a merge with another
770  // interval on the left, since there were no interval adjacent to "value"
771  // on the left.
772  DCHECK_EQ(it->start, value + 1);
773  const_cast<ClosedInterval*>(&(*it))->start = value;
774  return it;
775  }
776  }
777 
778  // At this point, "it_prev" points to an interval containing or adjacent to
779  // "value" on the left: grow it by one, and if it now touches the next
780  // interval, merge with it.
781  CHECK_NE(kint64max, it_prev->end) << "Cannot grow right by one: the interval "
782  "that would grow already ends at "
783  "kint64max";
784  *newly_covered = it_prev->end + 1;
785  if (it != end() && it_prev->end + 2 == it->start) {
786  // We need to merge it_prev with 'it'.
787  const_cast<ClosedInterval*>(&(*it_prev))->end = it->end;
788  intervals_.erase(it);
789  } else {
790  const_cast<ClosedInterval*>(&(*it_prev))->end = it_prev->end + 1;
791  }
792  return it_prev;
793 }
