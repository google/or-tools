37bool IntervalsAreSortedAndNonAdjacent(
38 absl::Span<const ClosedInterval> intervals) {
39 for (int i = 1; i < intervals.size(); ++i) {
40 if (intervals[i - 1].start > intervals[i - 1].end) return false;
41 // First test make sure that intervals[i - 1].end + 1 will not overflow.
42 if (intervals[i - 1].end >= intervals[i].start ||
43 intervals[i - 1].end + 1 >= intervals[i].start) {
44 return false;
45 }
46 }
47 return intervals.empty() ? true
48 : intervals.back().start <= intervals.back().end;
49}

63// Transforms a sorted list of intervals in a sorted DISJOINT list for which
64// IntervalsAreSortedAndNonAdjacent() would return true.
65void UnionOfSortedIntervals(absl::InlinedVector<ClosedInterval, 1>* intervals) {
66 DCHECK(std::is_sorted(intervals->begin(), intervals->end()));
67 int new_size = 0;
68 for (const ClosedInterval& i : *intervals) {
69 if (new_size > 0 && i.start <= CapAdd((*intervals)[new_size - 1].end, 1)) {
70 (*intervals)[new_size - 1].end =
71 std::max(i.end, (*intervals)[new_size - 1].end);
72 } else {
73 (*intervals)[new_size++] = i;
74 }
75 }
76 intervals->resize(new_size);
77
78 // This is important for InlinedVector in the case the result is a single
79 // intervals.
80 intervals->shrink_to_fit();
81 DCHECK(IntervalsAreSortedAndNonAdjacent(*intervals));
82}

86// TODO(user): Use MathUtil::CeilOfRatio / FloorOfRatio instead.
87int64_t CeilRatio(int64_t value, int64_t positive_coeff) {
88 DCHECK_GT(positive_coeff, 0);
89 const int64_t result = value / positive_coeff;
90 const int64_t adjust = static_cast<int64_t>(result * positive_coeff < value);
91 return result + adjust;
92}
93
94int64_t FloorRatio(int64_t value, int64_t positive_coeff) {
95 DCHECK_GT(positive_coeff, 0);
96 const int64_t result = value / positive_coeff;
97 const int64_t adjust = static_cast<int64_t>(result * positive_coeff > value);
98 return result - adjust;
99}

138Domain Domain::FromValues(std::vector<int64_t> values) {
139 std::sort(values.begin(), values.end());
140 Domain result;
141 for (const int64_t v : values) {
142 if (result.intervals_.empty() || v > result.intervals_.back().end + 1) {
143 result.intervals_.push_back({v, v});
144 } else {
145 result.intervals_.back().end = v;
146 }
147 }
148 return result;
149}

159Domain Domain::FromFlatSpanOfIntervals(
160 absl::Span<const int64_t> flat_intervals) {
161 DCHECK(flat_intervals.size() % 2 == 0) << flat_intervals.size();
162 Domain result;
163 result.intervals_.reserve(flat_intervals.size() / 2);
164 for (int i = 0; i < flat_intervals.size(); i += 2) {
165 result.intervals_.push_back({flat_intervals[i], flat_intervals[i + 1]});
166 }
167 std::sort(result.intervals_.begin(), result.intervals_.end());
168 UnionOfSortedIntervals(&result.intervals_);
169 return result;
170}

196int64_t Domain::Size() const {
197 int64_t size = 0;
198 for (const ClosedInterval interval : intervals_) {
199 size = operations_research::CapAdd(
200 size, operations_research::CapSub(interval.end, interval.start));
201 }
202 // Because the intervals are closed on both side above, with miss 1 per
203 // interval.
204 size = operations_research::CapAdd(size, intervals_.size());
205 return size;
206}

218int64_t Domain::SmallestValue() const {
219 DCHECK(!IsEmpty());
220 int64_t result = Min();
221 for (const ClosedInterval interval : intervals_) {
222 if (interval.start <= 0 && interval.end >= 0) return 0;
223 for (const int64_t b : {interval.start, interval.end}) {
224 if (b > 0 && b <= std::abs(result)) result = b;
225 if (b < 0 && -b < std::abs(result)) result = b;

236bool Domain::Contains(int64_t value) const {
237 // Because we only compare by start and there is no duplicate starts, this
238 // should be the next interval after the one that has a chance to contains
239 // value.
240 auto it = std::upper_bound(intervals_.begin(), intervals_.end(),
241 ClosedInterval(value, value));
242 if (it == intervals_.begin()) return false;
243 --it;
244 return value <= it->end;
245}

247bool Domain::IsIncludedIn(const Domain& domain) const {
248 int i = 0;
249 const auto& others = domain.intervals_;
250 for (const ClosedInterval interval : intervals_) {
251 // Find the unique interval in others that contains interval if any.
252 for (; i < others.size() && interval.end > others[i].end; ++i) {
253 }
254 if (i == others.size()) return false;
255 if (interval.start < others[i].start) return false;
256 }
257 return true;
258}

260Domain Domain::Complement() const {
261 Domain result;
262 int64_t next_start = kint64min;
263 result.intervals_.reserve(intervals_.size() + 1);
264 for (const ClosedInterval& interval : intervals_) {
265 if (interval.start != kint64min) {
266 result.intervals_.push_back({next_start, interval.start - 1});
267 }
268 if (interval.end == kint64max) return result;
269 next_start = interval.end + 1;
270 }
271 result.intervals_.push_back({next_start, kint64max});
272 DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
273 return result;
274}

282void Domain::NegateInPlace() {
283 if (intervals_.empty()) return;
284 std::reverse(intervals_.begin(), intervals_.end());
285 if (intervals_.back().end == kint64min) {
286 // corner-case
287 intervals_.pop_back();
288 }
289 for (ClosedInterval& ref : intervals_) {
290 std::swap(ref.start, ref.end);
291 ref.start = ref.start == kint64min ? kint64max : -ref.start;
292 ref.end = ref.end == kint64min ? kint64max : -ref.end;
293 }
294 DCHECK(IntervalsAreSortedAndNonAdjacent(intervals_));
295}

297Domain Domain::IntersectionWith(const Domain& domain) const {
298 Domain result;
299 const auto& a = intervals_;
300 const auto& b = domain.intervals_;
301 for (int i = 0, j = 0; i < a.size() && j < b.size();) {
302 if (a[i].start <= b[j].start) {
303 if (a[i].end < b[j].start) {
304 // Empty intersection. We advance past the first interval.
305 ++i;
306 } else { // a[i].end >= b[j].start
307 // Non-empty intersection: push back the intersection of these two, and
308 // advance past the first interval to finish.
309 if (a[i].end <= b[j].end) {
310 result.intervals_.push_back({b[j].start, a[i].end});
311 ++i;
312 } else { // a[i].end > b[j].end.
313 result.intervals_.push_back({b[j].start, b[j].end});
314 ++j;
315 }
316 }
317 } else { // a[i].start > b[i].start.
318 // We do the exact same thing as above, but swapping a and b.
319 if (b[j].end < a[i].start) {
320 ++j;
321 } else { // b[j].end >= a[i].start
322 if (b[j].end <= a[i].end) {
323 result.intervals_.push_back({a[i].start, b[j].end});
324 ++j;
325 } else { // a[i].end > b[j].end.
326 result.intervals_.push_back({a[i].start, a[i].end});
327 ++i;
328 }
329 }
330 }
331 }
332 DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
333 return result;
334}

336Domain Domain::UnionWith(const Domain& domain) const {
337 Domain result;
338 const auto& a = intervals_;
339 const auto& b = domain.intervals_;
340 result.intervals_.resize(a.size() + b.size());
341 std::merge(a.begin(), a.end(), b.begin(), b.end(), result.intervals_.begin());
342 UnionOfSortedIntervals(&result.intervals_);
343 return result;
344}

347Domain Domain::AdditionWith(const Domain& domain) const {
348 Domain result;
349
350 const auto& a = intervals_;
351 const auto& b = domain.intervals_;
352 result.intervals_.reserve(a.size() * b.size());
353 for (const ClosedInterval& i : a) {

363 }
364 UnionOfSortedIntervals(&result.intervals_);
365 return result;

376Domain Domain::MultiplicationBy(int64_t coeff, bool* exact) const {
377 if (exact != nullptr) *exact = true;
378 if (intervals_.empty()) return {};
379 if (coeff == 0) return Domain(0);
380
381 const int64_t abs_coeff = std::abs(coeff);
382 const int64_t size_if_non_trivial = abs_coeff > 1 ? Size() : 0;
383 if (size_if_non_trivial > kDomainComplexityLimit) {
384 if (exact != nullptr) *exact = false;
385 return ContinuousMultiplicationBy(coeff);
386 }
387
388 Domain result;
389 if (abs_coeff > 1) {
390 const int64_t max_value = kint64max / abs_coeff;
391 const int64_t min_value = kint64min / abs_coeff;
392 result.intervals_.reserve(size_if_non_trivial);
393 for (const ClosedInterval& i : intervals_) {
394 for (int64_t v = i.start;; ++v) {
395 // We ignore anything that overflow.
396 if (v >= min_value && v <= max_value) {
397 // Because abs_coeff > 1, all new values are disjoint.
398 const int64_t new_value = v * abs_coeff;
399 result.intervals_.push_back({new_value, new_value});
400 }
401
402 // This is to avoid doing ++v when v is kint64max!
403 if (v == i.end) break;
404 }
405 }
406 } else {
407 result = *this;
408 }
409 if (coeff < 0) result.NegateInPlace();
410 return result;
411}

413Domain Domain::ContinuousMultiplicationBy(int64_t coeff) const {
414 Domain result = *this;
415 const int64_t abs_coeff = std::abs(coeff);
416 for (ClosedInterval& i : result.intervals_) {
417 i.start = CapProd(i.start, abs_coeff);
418 i.end = CapProd(i.end, abs_coeff);
419 }
420 UnionOfSortedIntervals(&result.intervals_);
421 if (coeff < 0) result.NegateInPlace();
422 return result;
423}

425Domain Domain::ContinuousMultiplicationBy(const Domain& domain) const {
426 Domain result;
427 for (const ClosedInterval& i : this->intervals_) {
428 for (const ClosedInterval& j : domain.intervals_) {
429 ClosedInterval new_interval;
430 const int64_t a = CapProd(i.start, j.start);
431 const int64_t b = CapProd(i.end, j.end);
432 const int64_t c = CapProd(i.start, j.end);
433 const int64_t d = CapProd(i.end, j.start);
434 new_interval.start = std::min({a, b, c, d});
435 new_interval.end = std::max({a, b, c, d});
436 result.intervals_.push_back(new_interval);
437 }
438 }
439 std::sort(result.intervals_.begin(), result.intervals_.end());
440 UnionOfSortedIntervals(&result.intervals_);
441 return result;
442}

444Domain Domain::DivisionBy(int64_t coeff) const {
445 CHECK_NE(coeff, 0);
446 Domain result = *this;
447 const int64_t abs_coeff = std::abs(coeff);
448 for (ClosedInterval& i : result.intervals_) {
449 i.start = i.start / abs_coeff;
450 i.end = i.end / abs_coeff;
451 }
452 UnionOfSortedIntervals(&result.intervals_);
453 if (coeff < 0) result.NegateInPlace();
454 return result;
455}

457Domain Domain::InverseMultiplicationBy(const int64_t coeff) const {
458 if (coeff == 0) {
459 return Contains(0) ? Domain::AllValues() : Domain();
460 }
461 Domain result = *this;
462 int new_size = 0;
463 const int64_t abs_coeff = std::abs(coeff);
464 for (const ClosedInterval& i : result.intervals_) {
465 const int64_t start = CeilRatio(i.start, abs_coeff);
466 const int64_t end = FloorRatio(i.end, abs_coeff);
467 if (start > end) continue;
468 if (new_size > 0 && start == result.intervals_[new_size - 1].end + 1) {
469 result.intervals_[new_size - 1].end = end;
470 } else {
471 result.intervals_[new_size++] = {start, end};
472 }
473 }
474 result.intervals_.resize(new_size);
475 result.intervals_.shrink_to_fit();
476 DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
477 if (coeff < 0) result.NegateInPlace();
478 return result;
479}

481Domain Domain::PositiveModuloBySuperset(const Domain& modulo) const {
482 if (IsEmpty()) return Domain();
483 CHECK_GT(modulo.Min(), 0);
484 const int64_t max_mod = modulo.Max() - 1;

486 const int64_t min = Min() < 0 ? std::max(Min(), -max_mod) : 0;
487 return Domain(min, max);

497// It is a bit difficult to see, but this code is doing the same thing as
498// for all interval in this.UnionWith(implied_domain.Complement())):
499// - Take the two extreme points (min and max) in interval \inter implied.
500// - Append to result [min, max] if these points exists.
501Domain Domain::SimplifyUsingImpliedDomain(const Domain& implied_domain) const {
502 Domain result;
503 if (implied_domain.IsEmpty()) return result;
504
505 int i = 0;
506 int64_t min_point;
507 int64_t max_point;
508 bool started = false;
509 for (const ClosedInterval interval : intervals_) {
510 // We only "close" the new result interval if it cannot be extended by
511 // implied_domain.Complement(). The only extension possible look like:
512 // interval_: ...] [....
513 // implied : ...] [... i ...]
514 if (started && implied_domain.intervals_[i].start < interval.start) {
515 result.intervals_.push_back({min_point, max_point});
516 started = false;
517 }
518
519 // Find the two extreme points in interval \inter implied_domain.
520 // Always stop the loop at the first interval with and end strictly greater
521 // that interval.end.
522 for (; i < implied_domain.intervals_.size(); ++i) {
523 const ClosedInterval current = implied_domain.intervals_[i];
524 if (current.end >= interval.start && current.start <= interval.end) {
525 // Current and interval have a non-empty intersection.
526 const int64_t inter_max = std::min(interval.end, current.end);
527 if (!started) {
528 started = true;
529 min_point = std::max(interval.start, current.start);
530 max_point = inter_max;
531 } else {
532 // No need to update the min_point here, and the new inter_max must
533 // necessarily be > old one.
534 DCHECK_GE(inter_max, max_point);
535 max_point = inter_max;
536 }
537 }
538 if (current.end > interval.end) break;
539 }
540 if (i == implied_domain.intervals_.size()) break;
541 }
542 if (started) {
543 result.intervals_.push_back({min_point, max_point});
544 }
545 DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
546 return result;
547}

558bool Domain::operator<(const Domain& other) const {
559 const auto& d1 = intervals_;
560 const auto& d2 = other.intervals_;
561 const int common_size = std::min(d1.size(), d2.size());
562 for (int i = 0; i < common_size; ++i) {
563 const ClosedInterval& i1 = d1[i];
564 const ClosedInterval& i2 = d2[i];
565 if (i1.start < i2.start) return true;
566 if (i1.start > i2.start) return false;
567 if (i1.end < i2.end) return true;
568 if (i1.end > i2.end) return false;
569 }
570 return d1.size() < d2.size();
571}

575int64_t SumOfKMinValueInDomain(const Domain& domain, int k) {
576 int64_t current_sum = 0.0;
577 int current_index = 0;
578 for (const ClosedInterval interval : domain) {
579 if (current_index >= k) break;
580 for (int v(interval.start); v <= interval.end; ++v) {
581 if (current_index >= k) break;
582 current_index++;
583 current_sum += v;
584 }
585 }
586 return current_sum;
587}

612SortedDisjointIntervalList
613SortedDisjointIntervalList::BuildComplementOnInterval(int64_t start,
614 int64_t end) {
615 SortedDisjointIntervalList interval_list;
616 int64_t next_start = start;
617 for (auto it = FirstIntervalGreaterOrEqual(start); it != this->end(); ++it) {
618 const ClosedInterval& interval = *it;
619 const int64_t next_end = CapSub(interval.start, 1);
620 if (next_end > end) break;
621 if (next_start <= next_end) {
622 interval_list.InsertInterval(next_start, next_end);
623 }
624 next_start = CapAdd(interval.end, 1);
625 }
626 if (next_start <= end) {
627 interval_list.InsertInterval(next_start, end);
628 }
629 return interval_list;
630}

632SortedDisjointIntervalList::Iterator SortedDisjointIntervalList::InsertInterval(
633 int64_t start, int64_t end) {
634 // start > end could mean an empty interval, but we prefer to LOG(DFATAL)
635 // anyway. Really, the user should never give us that.
636 if (start > end) {
637 LOG(DFATAL) << "Invalid interval: " << ClosedInterval({start, end});
638 return intervals_.end();
639 }
640
641 auto result = intervals_.insert({start, end});
642 if (!result.second) return result.first; // Duplicate: exit immediately.
643
644 // TODO(user): tune the algorithm below if it proves to be a bottleneck.
645 // For example, one could try to avoid an insertion if it's not needed
646 // (when the interval merges with a single existing interval or is fully
647 // contained by one).
648
649 // Iterate over the previous iterators whose end is after (or almost at) our
650 // start. After this, "it1" will point to the first interval that needs to be
651 // merged with the current interval (possibly pointing to the current interval
652 // itself, if no "earlier" interval should be merged).
653 auto it1 = result.first;
654 if (start == kint64min) { // Catch underflows
655 it1 = intervals_.begin();
656 } else {
657 const int64_t before_start = start - 1;
658 while (it1 != intervals_.begin()) {
659 auto prev_it = it1;
660 --prev_it;
661 if (prev_it->end < before_start) break;
662 it1 = prev_it;
663 }
664 }
665
666 // Ditto, on the other side: "it2" will point to the interval *after* the last
667 // one that should be merged with the current interval.
668 auto it2 = result.first;
669 if (end == kint64max) {
670 it2 = intervals_.end();
671 } else {
672 const int64_t after_end = end + 1;
673 do {
674 ++it2;
675 } while (it2 != intervals_.end() && it2->start <= after_end);
676 }
677
678 // [it1..it2) is the range (inclusive on it1, exclusive on it2) of intervals
679 // that should be merged together. We'll set it3 = it2-1 and erase [it1..it3)
680 // and set *it3 to the merged interval.
681 auto it3 = it2;
682 it3--;
683 if (it1 == it3) return it3; // Nothing was merged.
684 const int64_t new_start = std::min(it1->start, start);
685 const int64_t new_end = std::max(it3->end, end);
686 auto it = intervals_.erase(it1, it3);
687 // HACK(user): set iterators point to *const* values. Which is expected,
688 // because if one alters a set element's value, then it collapses the set
689 // property! But in this very special case, we know that we can just overwrite
690 // it->start, so we do it.
691 const_cast<ClosedInterval*>(&(*it))->start = new_start;
692 const_cast<ClosedInterval*>(&(*it))->end = new_end;
693 return it;
694}

696SortedDisjointIntervalList::Iterator SortedDisjointIntervalList::GrowRightByOne(
697 int64_t value, int64_t* newly_covered) {
698 auto it = intervals_.upper_bound({value, kint64max});
699 auto it_prev = it;
700
701 // No interval containing or adjacent to "value" on the left (i.e. below).
702 if (it != begin()) {
703 --it_prev;
704 }
705 if (it == begin() || ((value != kint64min) && it_prev->end < value - 1)) {
706 *newly_covered = value;
707 if (it == end() || it->start != value + 1) {
708 // No interval adjacent to "value" on the right: insert a singleton.
709 return intervals_.insert(it, {value, value});
710 } else {
711 // There is an interval adjacent to "value" on the right. Extend it by
712 // one. Note that we already know that there won't be a merge with another
713 // interval on the left, since there were no interval adjacent to "value"
714 // on the left.
715 DCHECK_EQ(it->start, value + 1);
716 const_cast<ClosedInterval*>(&(*it))->start = value;
717 return it;
718 }
719 }
720
721 // At this point, "it_prev" points to an interval containing or adjacent to
722 // "value" on the left: grow it by one, and if it now touches the next
723 // interval, merge with it.
724 CHECK_NE(kint64max, it_prev->end) << "Cannot grow right by one: the interval "
725 "that would grow already ends at "
726 "kint64max";
727 *newly_covered = it_prev->end + 1;
728 if (it != end() && it_prev->end + 2 == it->start) {
729 // We need to merge it_prev with 'it'.
730 const_cast<ClosedInterval*>(&(*it_prev))->end = it->end;
731 intervals_.erase(it);
732 } else {
733 const_cast<ClosedInterval*>(&(*it_prev))->end = it_prev->end + 1;
734 }
735 return it_prev;
736}
