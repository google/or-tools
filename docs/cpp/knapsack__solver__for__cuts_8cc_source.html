30 // Comparator used to sort item in decreasing efficiency order
31 // (see KnapsackCapacityPropagator).
32 struct CompareKnapsackItemsInDecreasingEfficiencyOrder {
33  explicit CompareKnapsackItemsInDecreasingEfficiencyOrder(double _profit_max)
34  : profit_max(_profit_max) {}
35  bool operator()(const KnapsackItemForCutsPtr& item1,
36  const KnapsackItemForCutsPtr& item2) const {
37  return item1->GetEfficiency(profit_max) > item2->GetEfficiency(profit_max);
38  }
39  const double profit_max;
40 };
41 
42 // Comparator used to sort search nodes in the priority queue in order
43 // to pop first the node with the highest profit upper bound
44 // (see KnapsackSearchNodeForCuts). When two nodes have the same upper bound, we
45 // prefer the one with the highest current profit. This is usually the one
46 // closer to a leaf. In practice, the main advantage is to have smaller path.
47 struct CompareKnapsackSearchNodePtrInDecreasingUpperBoundOrder {
48  bool operator()(const KnapsackSearchNodeForCuts* node_1,
49  const KnapsackSearchNodeForCuts* node_2) const {
50  const double profit_upper_bound_1 = node_1->profit_upper_bound();
51  const double profit_upper_bound_2 = node_2->profit_upper_bound();
52  if (profit_upper_bound_1 == profit_upper_bound_2) {
53  return node_1->current_profit() < node_2->current_profit();
54  }
55  return profit_upper_bound_1 < profit_upper_bound_2;
56  }
57 };

155 bool KnapsackPropagatorForCuts::Update(
156  bool revert, const KnapsackAssignmentForCuts& assignment) {
157  if (assignment.is_in) {
158  if (revert) {
159  current_profit_ -= items_[assignment.item_id]->profit;
160  consumed_capacity_ -= items()[assignment.item_id]->weight;
161  } else {
162  current_profit_ += items_[assignment.item_id]->profit;
163  consumed_capacity_ += items()[assignment.item_id]->weight;
164  if (consumed_capacity_ > capacity_) {
165  return false;
166  }
167  }
168  }
169  return true;
170 }

172 void KnapsackPropagatorForCuts::CopyCurrentStateToSolution(
173  std::vector<bool>* solution) const {
174  DCHECK(solution != nullptr);
175  for (int i(0); i < items_.size(); ++i) {
176  const int item_id = items_[i]->id;
177  (*solution)[item_id] = state_->is_bound(item_id) && state_->is_in(item_id);
178  }
179  double remaining_capacity = capacity_ - consumed_capacity_;
180  for (const KnapsackItemForCutsPtr& item : sorted_items_) {
181  if (!state().is_bound(item->id)) {
182  if (remaining_capacity >= item->weight) {
183  remaining_capacity -= item->weight;
184  (*solution)[item->id] = true;
185  } else {
186  return;
187  }
188  }
189  }
190 }

192 void KnapsackPropagatorForCuts::ComputeProfitBounds() {
193  set_profit_lower_bound(current_profit());
194  break_item_id_ = kNoSelection;
195 
196  double remaining_capacity = capacity_ - consumed_capacity_;
197  int break_sorted_item_id = kNoSelection;
198  for (int sorted_id(0); sorted_id < sorted_items_.size(); ++sorted_id) {
199  if (!state().is_bound(sorted_items_[sorted_id]->id)) {
200  const KnapsackItemForCutsPtr& item = sorted_items_[sorted_id];
201  break_item_id_ = item->id;
202  if (remaining_capacity >= item->weight) {
203  remaining_capacity -= item->weight;
204  set_profit_lower_bound(profit_lower_bound() + item->profit);
205  } else {
206  break_sorted_item_id = sorted_id;
207  break;
208  }
209  }
210  }
211 
212  set_profit_upper_bound(profit_lower_bound());
213  // If break_sorted_item_id == kNoSelection, then all remaining items fit into
214  // the knapsack, and thus the lower bound on the profit equals the upper
215  // bound. Otherwise, we compute a tight upper bound by filling the remaining
216  // capacity of the knapsack with "fractional" items, in the decreasing order
217  // of their efficiency.
218  if (break_sorted_item_id != kNoSelection) {
219  const double additional_profit =
220  GetAdditionalProfitUpperBound(remaining_capacity, break_sorted_item_id);
221  set_profit_upper_bound(profit_upper_bound() + additional_profit);
222  }
223 }

225 void KnapsackPropagatorForCuts::InitPropagator() {
226  consumed_capacity_ = 0;
227  break_item_id_ = kNoSelection;
228  sorted_items_.clear();
229  sorted_items_.reserve(items().size());
230  for (int i(0); i < items().size(); ++i) {
231  sorted_items_.emplace_back(absl::make_unique<KnapsackItemForCuts>(
232  i, items()[i]->weight, items()[i]->profit));
233  }
234  profit_max_ = 0;
235  for (const KnapsackItemForCutsPtr& item : sorted_items_) {
236  profit_max_ = std::max(profit_max_, item->profit);
237  }
238  profit_max_ += 1.0;
239  CompareKnapsackItemsInDecreasingEfficiencyOrder compare_object(profit_max_);
240  std::sort(sorted_items_.begin(), sorted_items_.end(), compare_object);
241 }

243 double KnapsackPropagatorForCuts::GetAdditionalProfitUpperBound(
244  double remaining_capacity, int break_item_id) const {
245  const int after_break_item_id = break_item_id + 1;
246  double additional_profit_when_no_break_item = 0;
247  if (after_break_item_id < sorted_items_.size()) {
248  // As items are sorted by decreasing profit / weight ratio, and the current
249  // weight is non-zero, the next_weight is non-zero too.
250  const double next_weight = sorted_items_[after_break_item_id]->weight;
251  const double next_profit = sorted_items_[after_break_item_id]->profit;
252  additional_profit_when_no_break_item =
253  std::max((remaining_capacity * next_profit) / next_weight, 0.0);
254  }
255 
256  const int before_break_item_id = break_item_id - 1;
257  double additional_profit_when_break_item = 0;
258  if (before_break_item_id >= 0) {
259  const double previous_weight = sorted_items_[before_break_item_id]->weight;
260  // Having previous_weight == 0 means the total capacity is smaller than
261  // the weight of the current item. In such a case the item cannot be part
262  // of a solution of the local one dimension problem.
263  if (previous_weight != 0) {
264  const double previous_profit =
265  sorted_items_[before_break_item_id]->profit;
266  const double overused_capacity =
267  sorted_items_[break_item_id]->weight - remaining_capacity;
268  const double lost_profit_from_previous_item =
269  (overused_capacity * previous_profit) / previous_weight;
270  additional_profit_when_break_item = std::max(
271  sorted_items_[break_item_id]->profit - lost_profit_from_previous_item,
272  0.0);
273  }
274  }
275 
276  const double additional_profit = std::max(
277  additional_profit_when_no_break_item, additional_profit_when_break_item);
278  return additional_profit;
279 }

298 void KnapsackSolverForCuts::GetLowerAndUpperBoundWhenItem(int item_id,
299  bool is_item_in,
300  double* lower_bound,
301  double* upper_bound) {
302  DCHECK(lower_bound != nullptr);
303  DCHECK(upper_bound != nullptr);
304  KnapsackAssignmentForCuts assignment(item_id, is_item_in);
305  const bool fail = !IncrementalUpdate(false, assignment);
306  if (fail) {
307  *lower_bound = 0;
308  *upper_bound = 0;
309  } else {
310  *lower_bound = propagator_.profit_lower_bound();
311  *upper_bound = GetAggregatedProfitUpperBound();
312  }
313 
314  const bool fail_revert = !IncrementalUpdate(true, assignment);
315  if (fail_revert) {
316  *lower_bound = 0;
317  *upper_bound = 0;
318  }
319 }

321 double KnapsackSolverForCuts::Solve(TimeLimit* time_limit,
322  bool* is_solution_optimal) {
323  DCHECK(time_limit != nullptr);
324  DCHECK(is_solution_optimal != nullptr);
325  best_solution_profit_ = 0;
326  *is_solution_optimal = true;
327 
328  SearchQueue search_queue;
329  const KnapsackAssignmentForCuts assignment(kNoSelection, true);
330  auto root_node =
331  absl::make_unique<KnapsackSearchNodeForCuts>(nullptr, assignment);
332  root_node->set_current_profit(GetCurrentProfit());
333  root_node->set_profit_upper_bound(GetAggregatedProfitUpperBound());
334  root_node->set_next_item_id(GetNextItemId());
335  search_nodes_.push_back(std::move(root_node));
336  const KnapsackSearchNodeForCuts* current_node =
337  search_nodes_.back().get(); // Start with the root node.
338 
339  if (MakeNewNode(*current_node, false)) {
340  search_queue.push(search_nodes_.back().get());
341  }
342  if (MakeNewNode(*current_node, true)) {
343  search_queue.push(search_nodes_.back().get());
344  }
345 
346  int64_t number_of_nodes_visited = 0;
347  while (!search_queue.empty() &&
348  search_queue.top()->profit_upper_bound() > best_solution_profit_) {
349  if (time_limit->LimitReached()) {
350  *is_solution_optimal = false;
351  break;
352  }
353  if (solution_upper_bound_threshold_ > -kInfinity &&
354  GetAggregatedProfitUpperBound() < solution_upper_bound_threshold_) {
355  *is_solution_optimal = false;
356  break;
357  }
358  if (best_solution_profit_ > solution_lower_bound_threshold_) {
359  *is_solution_optimal = false;
360  break;
361  }
362  if (number_of_nodes_visited >= node_limit_) {
363  *is_solution_optimal = false;
364  break;
365  }
366  KnapsackSearchNodeForCuts* const node = search_queue.top();
367  search_queue.pop();
368 
369  if (node != current_node) {
370  KnapsackSearchPathForCuts path(current_node, node);
371  path.Init();
372  CHECK_EQ(UpdatePropagators(path), true);
373  current_node = node;
374  }
375  number_of_nodes_visited++;
376 
377  if (MakeNewNode(*node, false)) {
378  search_queue.push(search_nodes_.back().get());
379  }
380  if (MakeNewNode(*node, true)) {
381  search_queue.push(search_nodes_.back().get());
382  }
383  }
384  return best_solution_profit_;
385 }

387 // Returns false when at least one propagator fails.
388 bool KnapsackSolverForCuts::UpdatePropagators(
389  const KnapsackSearchPathForCuts& path) {
390  bool no_fail = true;
391  // Revert previous changes.
392  const KnapsackSearchNodeForCuts* node = &path.from();
393  const KnapsackSearchNodeForCuts* const via = &path.via();
394  while (node != via) {
395  no_fail = IncrementalUpdate(true, node->assignment()) && no_fail;
396  node = node->parent();
397  }
398  // Apply current changes.
399  node = &path.to();
400  while (node != via) {
401  no_fail = IncrementalUpdate(false, node->assignment()) && no_fail;
402  node = node->parent();
403  }
404  return no_fail;
405 }

413 bool KnapsackSolverForCuts::MakeNewNode(const KnapsackSearchNodeForCuts& node,
414  bool is_in) {
415  if (node.next_item_id() == kNoSelection) {
416  return false;
417  }
418  KnapsackAssignmentForCuts assignment(node.next_item_id(), is_in);
419  KnapsackSearchNodeForCuts new_node(&node, assignment);
420 
421  KnapsackSearchPathForCuts path(&node, &new_node);
422  path.Init();
423  const bool no_fail = UpdatePropagators(path);
424  if (no_fail) {
425  new_node.set_current_profit(GetCurrentProfit());
426  new_node.set_profit_upper_bound(GetAggregatedProfitUpperBound());
427  new_node.set_next_item_id(GetNextItemId());
428  UpdateBestSolution();
429  }
430 
431  // Revert to be able to create another node from parent.
432  KnapsackSearchPathForCuts revert_path(&new_node, &node);
433  revert_path.Init();
434  UpdatePropagators(revert_path);
435 
436  if (!no_fail || new_node.profit_upper_bound() < best_solution_profit_) {
437  return false;
438  }
439 
440  // The node is relevant.
441  auto relevant_node =
442  absl::make_unique<KnapsackSearchNodeForCuts>(&node, assignment);
443  relevant_node->set_current_profit(new_node.current_profit());
444  relevant_node->set_profit_upper_bound(new_node.profit_upper_bound());
445  relevant_node->set_next_item_id(new_node.next_item_id());
446  search_nodes_.push_back(std::move(relevant_node));
447 
448  return true;
449 }

451 bool KnapsackSolverForCuts::IncrementalUpdate(
452  bool revert, const KnapsackAssignmentForCuts& assignment) {
453  // Do not stop on a failure: To be able to be incremental on the update,
454  // partial solution (state) and propagators must all be in the same state.
455  bool no_fail = state_.UpdateState(revert, assignment);
456  no_fail = propagator_.Update(revert, assignment) && no_fail;
457  return no_fail;
458 }
