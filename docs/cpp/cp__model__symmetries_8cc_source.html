69 // Returns a graph whose automorphisms can be mapped back to the symmetries of
70 // the model described in the given CpModelProto.
71 //
72 // Any permutation of the graph that respects the initial_equivalence_classes
73 // output can be mapped to a symmetry of the given problem simply by taking its
74 // restriction on the first num_variables nodes and interpreting its index as a
75 // variable index. In a sense, a node with a low enough index #i is in
76 // one-to-one correspondence with the variable #i (using the index
77 // representation of variables).
78 //
79 // The format of the initial_equivalence_classes is the same as the one
80 // described in GraphSymmetryFinder::FindSymmetries(). The classes must be dense
81 // in [0, num_classes) and any symmetry will only map nodes with the same class
82 // between each other.
83 template <typename Graph>
84 std::unique_ptr<Graph> GenerateGraphForSymmetryDetection(
85  const CpModelProto& problem,
86  std::vector<int>* initial_equivalence_classes) {
87  CHECK(initial_equivalence_classes != nullptr);
88 
89  const int num_variables = problem.variables_size();
90  auto graph = absl::make_unique<Graph>();
91  initial_equivalence_classes->clear();
92  enum NodeType {
93  VARIABLE_NODE,
94  CONSTRAINT_NODE,
95  CONSTRAINT_COEFFICIENT_NODE,
96  ENFORCEMENT_LITERAL
97  };
98  IdGenerator id_generator;
99 
100  // For two variables to be in the same equivalence class, they need to have
101  // the same objective coefficient, and the same possible bounds.
102  std::vector<int64> objective_by_var(num_variables, 0);
103  for (int i = 0; i < problem.objective().vars_size(); ++i) {
104  objective_by_var[problem.objective().vars(i)] =
105  problem.objective().coeffs(i);
106  }
107 
108  auto new_node = [&initial_equivalence_classes,
109  &id_generator](const std::vector<int64>& key) {
110  // Since we add nodes one by one, initial_equivalence_classes->size() gives
111  // the number of nodes at any point, which we use as the next node index.
112  const int node = initial_equivalence_classes->size();
113  initial_equivalence_classes->push_back(id_generator.GetId(key));
114  return node;
115  };
116 
117  std::vector<int64> tmp_key;
118  for (int v = 0; v < num_variables; ++v) {
119  const IntegerVariableProto& variable = problem.variables(v);
120  tmp_key = {VARIABLE_NODE, objective_by_var[v]};
121  Append(variable.domain(), &tmp_key);
122  CHECK_EQ(v, new_node(tmp_key));
123 
124  // Make sure the graph contains all the variable nodes, even if no edges are
125  // attached to them through constraints.
126  graph->AddNode(v);
127  }
128 
129  auto add_edge = [&graph](int node_1, int node_2) {
130  graph->AddArc(node_1, node_2);
131  graph->AddArc(node_2, node_1);
132  };
133 
134  // We will create a bunch of nodes linked to a variable node. Only one node
135  // per (var, type) will be required so we cache them to not create more node
136  // than necessary.
137  absl::flat_hash_map<std::vector<int64>, int, VectorHash> secondary_var_nodes;
138  auto get_secondary_var_node = [&new_node, &add_edge, &secondary_var_nodes,
139  &tmp_key](int var_node,
140  absl::Span<const int64> type) {
141  tmp_key.assign(type.begin(), type.end());
142  tmp_key.push_back(var_node);
143  const auto insert = secondary_var_nodes.insert({tmp_key, 0});
144  if (!insert.second) return insert.first->second;
145 
146  tmp_key.pop_back();
147  const int secondary_node = new_node(tmp_key);
148  add_edge(var_node, secondary_node);
149  insert.first->second = secondary_node;
150  return secondary_node;
151  };
152 
153  auto add_literal_edge = [&add_edge, &get_secondary_var_node](
154  int ref, int constraint_node) {
155  const int variable_node = PositiveRef(ref);
156  if (RefIsPositive(ref)) {
157  // For all coefficients equal to one, which are the most common, we
158  // can optimize the size of the graph by omitting the coefficient
159  // node altogether.
160  add_edge(variable_node, constraint_node);
161  } else {
162  const int coefficient_node = get_secondary_var_node(
163  variable_node, {CONSTRAINT_COEFFICIENT_NODE, -1});
164  add_edge(coefficient_node, constraint_node);
165  }
166  };
167 
168  // Add constraints to the graph.
169  for (const ConstraintProto& constraint : problem.constraints()) {
170  const int constraint_node = initial_equivalence_classes->size();
171  std::vector<int64> key = {CONSTRAINT_NODE, constraint.constraint_case()};
172 
173  switch (constraint.constraint_case()) {
174  case ConstraintProto::kLinear: {
175  Append(constraint.linear().domain(), &key);
176  CHECK_EQ(constraint_node, new_node(key));
177 
178  for (int i = 0; i < constraint.linear().vars_size(); ++i) {
179  if (constraint.linear().coeffs(i) == 0) continue;
180  const int ref = constraint.linear().vars(i);
181  const int variable_node = PositiveRef(ref);
182  const int64 coeff = RefIsPositive(ref)
183  ? constraint.linear().coeffs(i)
184  : -constraint.linear().coeffs(i);
185  if (coeff == 1) {
186  // For all coefficients equal to one, which are the most common, we
187  // can optimize the size of the graph by omitting the coefficient
188  // node altogether.
189  add_edge(variable_node, constraint_node);
190  } else {
191  const int coefficient_node = get_secondary_var_node(
192  variable_node, {CONSTRAINT_COEFFICIENT_NODE, coeff});
193  add_edge(coefficient_node, constraint_node);
194  }
195  }
196  break;
197  }
198  case ConstraintProto::kBoolOr: {
199  CHECK_EQ(constraint_node, new_node(key));
200  for (const int ref : constraint.bool_or().literals()) {
201  add_literal_edge(ref, constraint_node);
202  }
203  break;
204  }
205  case ConstraintProto::kAtMostOne: {
206  CHECK_EQ(constraint_node, new_node(key));
207  for (const int ref : constraint.at_most_one().literals()) {
208  add_literal_edge(ref, constraint_node);
209  }
210  break;
211  }
212  case ConstraintProto::kExactlyOne: {
213  CHECK_EQ(constraint_node, new_node(key));
214  for (const int ref : constraint.exactly_one().literals()) {
215  add_literal_edge(ref, constraint_node);
216  }
217  break;
218  }
219  case ConstraintProto::kBoolXor: {
220  CHECK_EQ(constraint_node, new_node(key));
221  for (const int ref : constraint.bool_xor().literals()) {
222  add_literal_edge(ref, constraint_node);
223  }
224  break;
225  }
226 
227  // TODO(user): We could directly connect variable node together to
228  // deal more efficiently with this constraint. Make sure not to create
229  // multi-arc since I am not sure the symmetry code works with these
230  // though.
231  case ConstraintProto::kBoolAnd: {
232  if (constraint.enforcement_literal_size() == 0) {
233  // All literals are true in this case.
234  CHECK_EQ(constraint_node, new_node(key));
235  for (const int ref : constraint.bool_and().literals()) {
236  add_literal_edge(ref, constraint_node);
237  }
238  break;
239  }
240 
241  // To make the BoolAnd constraint more generic in the graph, we expand
242  // it into a set of BoolOr constraints where
243  // not(enforcement_literal) OR literal = true
244  // for all constraint's literals. This is equivalent to
245  // enforcement_literal => literal
246  // for all literals.
247  std::vector<int64> key = {CONSTRAINT_NODE, ConstraintProto::kBoolOr};
248  const int non_enforcement_literal =
249  NegatedRef(constraint.enforcement_literal(0));
250  for (const int literal : constraint.bool_and().literals()) {
251  const int constraint_node = new_node(key);
252  add_literal_edge(non_enforcement_literal, constraint_node);
253  add_literal_edge(literal, constraint_node);
254  }
255  break;
256  }
257  default: {
258  // If the model contains any non-supported constraints, return an empty
259  // graph.
260  //
261  // TODO(user): support other types of constraints. Or at least, we
262  // could associate to them an unique node so that their variables can
263  // appear in no symmetry.
264  LOG(ERROR) << "Unsupported constraint type "
265  << ConstraintCaseName(constraint.constraint_case());
266  return nullptr;
267  }
268  }
269 
270  if (constraint.constraint_case() != ConstraintProto::kBoolAnd) {
271  for (const int ref : constraint.enforcement_literal()) {
272  const int enforcement_node = get_secondary_var_node(
273  PositiveRef(ref), {ENFORCEMENT_LITERAL, RefIsPositive(ref)});
274  add_edge(constraint_node, enforcement_node);
275  }
276  }
277  }
278 
279  graph->Build();
280  DCHECK_EQ(graph->num_nodes(), initial_equivalence_classes->size());
281  return graph;
282 }

285 void FindCpModelSymmetries(
286  const SatParameters& params, const CpModelProto& problem,
287  std::vector<std::unique_ptr<SparsePermutation>>* generators,
288  double time_limit_seconds) {
289  const bool log_info = params.log_search_progress() || VLOG_IS_ON(1);
290  CHECK(generators != nullptr);
291  generators->clear();
292 
293  typedef GraphSymmetryFinder::Graph Graph;
294 
295  std::vector<int> equivalence_classes;
296  std::unique_ptr<Graph> graph(
297  GenerateGraphForSymmetryDetection<Graph>(problem, &equivalence_classes));
298  if (graph == nullptr) return;
299 
300  if (log_info) {
301  LOG(INFO) << "Graph has " << graph->num_nodes() << " nodes and "
302  << graph->num_arcs() / 2 << " edges.";
303  }
304  if (graph->num_nodes() == 0) return;
305 
306  GraphSymmetryFinder symmetry_finder(*graph, /*is_undirected=*/true);
307  std::vector<int> factorized_automorphism_group_size;
308  const absl::Status status = symmetry_finder.FindSymmetries(
309  time_limit_seconds, &equivalence_classes, generators,
310  &factorized_automorphism_group_size);
311 
312  // TODO(user): Change the API to not return an error when the time limit is
313  // reached.
314  if (log_info && !status.ok()) {
315  LOG(INFO) << "GraphSymmetryFinder error: " << status.message();
316  }
317 
318  // Remove from the permutations the part not concerning the variables.
319  // Note that some permutations may become empty, which means that we had
320  // duplicate constraints.
321  double average_support_size = 0.0;
322  int num_generators = 0;
323  int num_duplicate_constraints = 0;
324  for (int i = 0; i < generators->size(); ++i) {
325  SparsePermutation* permutation = (*generators)[i].get();
326  std::vector<int> to_delete;
327  for (int j = 0; j < permutation->NumCycles(); ++j) {
328  // Because variable nodes are in a separate equivalence class than any
329  // other node, a cycle can either contain only variable nodes or none, so
330  // we just need to check one element of the cycle.
331  if (*(permutation->Cycle(j).begin()) >= problem.variables_size()) {
332  to_delete.push_back(j);
333  if (DEBUG_MODE) {
334  // Verify that the cycle's entire support does not touch any variable.
335  for (const int node : permutation->Cycle(j)) {
336  DCHECK_GE(node, problem.variables_size());
337  }
338  }
339  }
340  }
341 
342  permutation->RemoveCycles(to_delete);
343  if (!permutation->Support().empty()) {
344  average_support_size += permutation->Support().size();
345  swap((*generators)[num_generators], (*generators)[i]);
346  ++num_generators;
347  } else {
348  ++num_duplicate_constraints;
349  }
350  }
351  generators->resize(num_generators);
352  average_support_size /= num_generators;
353  if (log_info) {
354  LOG(INFO) << "# of generators: " << num_generators;
355  LOG(INFO) << "Average support size: " << average_support_size;
356  if (num_duplicate_constraints > 0) {
357  LOG(INFO) << "The model contains " << num_duplicate_constraints
358  << " duplicate constraints !";
359  }
360  }
361 }
