46 absl::Status VariablesValid(const VariablesProto& variables,
47  const bool check_names) {
48  RETURN_IF_ERROR(CheckIdsNonnegativeAndStrictlyIncreasing(variables.ids()))
49  << "Bad variable ids";
50  RETURN_IF_ERROR(
51  CheckValues(MakeView(variables.ids(), variables.lower_bounds()),
52  {.allow_positive_infinity = false}, "lower_bounds"));
53  RETURN_IF_ERROR(
54  CheckValues(MakeView(variables.ids(), variables.upper_bounds()),
55  {.allow_negative_infinity = false}, "upper_bounds"));
56  RETURN_IF_ERROR(
57  CheckValues(MakeView(variables.ids(), variables.integers()), "integers"));
58  RETURN_IF_ERROR(CheckNameVector(MakeView(variables.ids(), variables.names()),
59  check_names));
60  return absl::OkStatus();
61 }

167 absl::Status LinearConstraintMatrixValid(
168  const SparseDoubleMatrixProto& matrix) {
169  const int nnz = matrix.row_ids_size();
170  if (nnz != matrix.column_ids_size()) {
171  return absl::InvalidArgumentError(
172  absl::StrCat("Expected row_id.size=", nnz,
173  " equal to column_ids.size=", matrix.column_ids_size()));
174  }
175  if (nnz != matrix.coefficients_size()) {
176  return absl::InvalidArgumentError(absl::StrCat(
177  "Expected row_id.size=", nnz,
178  " equal to coefficients.size=", matrix.coefficients_size()));
179  }
180  int64_t previous_row = -1;
181  int64_t previous_col = -1;
182  for (int i = 0; i < nnz; ++i) {
183  const int64_t row = matrix.row_ids(i);
184  if (row < 0) {
185  return absl::InvalidArgumentError(
186  absl::StrCat("row_ids should be nonnegative, but found id: ", row,
187  " (at index: ", i, ")"));
188  }
189  const int64_t col = matrix.column_ids(i);
190  if (col < 0) {
191  return absl::InvalidArgumentError(
192  absl::StrCat("column_ids should be nonnegative, but found id: ", col,
193  " (at index: ", i, ")"));
194  }
195  if (row < previous_row) {
196  return absl::InvalidArgumentError(
197  absl::StrCat("row_ids should nondecreasing, but found ids [",
198  previous_row, row, "] at indices [", i - 1, i, "]"));
199  } else if (row == previous_row) {
200  if (previous_col >= col) {
201  return absl::InvalidArgumentError(
202  absl::StrCat("column_ids should be strictly increasing within a "
203  "row, but for row_id: ",
204  row, " found [", previous_col, ", ", col,
205  "] at indices, [", i - 1, ", ", i, "]"));
206  }
207  }
208  // When row > previous_row, we have a new row, nothing to check.
209  if (!std::isfinite(matrix.coefficients(i))) {
210  return absl::InvalidArgumentError(absl::StrCat(
211  "Expected finite coefficients without NaN, but at row_id: ", row,
212  ", column_id: ", col, " found coefficient: ", matrix.coefficients(i),
213  " (at index: ", i, ")"));
214  }
215  previous_row = row;
216  previous_col = col;
217  }
218  return absl::OkStatus();
219 }

251 absl::Status ValidateModel(const ModelProto& model, const bool check_names) {
252  RETURN_IF_ERROR(VariablesValid(model.variables(), check_names))
253  << "Model.variables are invalid.";
254  RETURN_IF_ERROR(ObjectiveValid(model.objective(), model.variables().ids()))
255  << "Model.objective is invalid";
256  RETURN_IF_ERROR(
257  LinearConstraintsValid(model.linear_constraints(), check_names))
258  << "Model.linear_constraints are invalid";
259  RETURN_IF_ERROR(LinearConstraintMatrixValid(model.linear_constraint_matrix()))
260  << "Model.linear_constraint_matrix invalid";
261  RETURN_IF_ERROR(LinearConstraintMatrixIdsAreKnown(
262  model.linear_constraint_matrix(), model.linear_constraints().ids(),
263  model.variables().ids()))
264  << "Model.linear_constraint_matrix ids are inconsistent";
265  return absl::OkStatus();
266 }

272 absl::Status ValidateModelUpdate(const ModelUpdateProto& model_update,
273  const bool check_names) {
274  RETURN_IF_ERROR(CheckIdsNonnegativeAndStrictlyIncreasing(
275  model_update.deleted_linear_constraint_ids()))
276  << "ModelUpdateProto.deleted_linear_constraint_ids invalid";
277  RETURN_IF_ERROR(CheckIdsNonnegativeAndStrictlyIncreasing(
278  model_update.deleted_variable_ids()))
279  << "ModelUpdateProto.deleted_variable_ids invalid";
280  RETURN_IF_ERROR(VariableUpdatesValid(model_update.variable_updates()))
281  << "ModelUpdateProto.variable_updates invalid";
282  RETURN_IF_ERROR(
283  LinearConstraintUpdatesValid(model_update.linear_constraint_updates()))
284  << "ModelUpdateProto.linear_constraint_updates invalid";
285  RETURN_IF_ERROR(VariablesValid(model_update.new_variables(), check_names))
286  << "ModelUpdateProto.new_variables invalid";
287  RETURN_IF_ERROR(LinearConstraintsValid(model_update.new_linear_constraints(),
288  check_names))
289  << "ModelUpdateProto.new_linear_constraints invalid";
290  RETURN_IF_ERROR(ObjectiveUpdatesValid(model_update.objective_updates()))
291  << "ModelUpdateProto.objective_update invalid";
292  RETURN_IF_ERROR(LinearConstraintMatrixValid(
293  model_update.linear_constraint_matrix_updates()))
294  << "Model.linear_constraint_matrix_updates invalid";
295  return absl::OkStatus();
296 }

298 absl::Status ValidateModelUpdateAndSummary(const ModelUpdateProto& model_update,
299  const ModelSummary& model_summary,
300  const bool check_names) {
301  RETURN_IF_ERROR(ValidateModelUpdate(model_update));
302  const IdUpdateValidator variable_id_validator(
303  model_summary.variables, model_update.deleted_variable_ids(),
304  model_update.new_variables().ids());
305  RETURN_IF_ERROR(variable_id_validator.IsValid())
306  << "Invalid new or deleted variable id";
307  const IdUpdateValidator linear_constraint_id_validator(
308  model_summary.linear_constraints,
309  model_update.deleted_linear_constraint_ids(),
310  model_update.new_linear_constraints().ids());
311  RETURN_IF_ERROR(linear_constraint_id_validator.IsValid())
312  << "Invalid new or deleted linear constraint id";
313 
314  RETURN_IF_ERROR(VariableUpdatesValidForState(model_update.variable_updates(),
315  variable_id_validator))
316  << "Invalid variable update";
317 
318  RETURN_IF_ERROR(LinearConstraintUpdatesValidForState(
319  model_update.linear_constraint_updates(), linear_constraint_id_validator))
320  << "Invalid linear constraint update";
321 
322  RETURN_IF_ERROR(ObjectiveUpdatesValidForModel(
323  model_update.objective_updates(), variable_id_validator))
324  << "Invalid objective update";
325  RETURN_IF_ERROR(LinearConstraintMatrixIdsValidForUpdate(
326  model_update.linear_constraint_matrix_updates(),
327  linear_constraint_id_validator, variable_id_validator))
328  << "Invalid linear constraint matrix update";
329  if (check_names && !model_update.new_variables().names().empty()) {
330  RETURN_IF_ERROR(
331  CheckNewNames(model_summary.variables,
332  MakeView(model_update.new_variables().ids(),
333  model_update.new_variables().names())))
334  << "Bad new variable names";
335  }
336 
337  if (check_names && !model_update.new_linear_constraints().names().empty()) {
338  RETURN_IF_ERROR(
339  CheckNewNames(model_summary.linear_constraints,
340  MakeView(model_update.new_linear_constraints().ids(),
341  model_update.new_linear_constraints().names())))
342  << "Bad new linear constraint names";
343  }
344  return absl::OkStatus();
345 }
