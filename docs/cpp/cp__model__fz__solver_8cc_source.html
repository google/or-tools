59static const int kNoVar = std::numeric_limits<int>::min();

61struct VarOrValue {
62 int var = kNoVar;
63 int64_t value = 0;
64};

66// Returns the true/false literal corresponding to a CpModelProto variable.
67int TrueLiteral(int var) { return var; }
68int FalseLiteral(int var) { return -var - 1; }
69int NegatedCpModelVariable(int var) { return -var - 1; }

71// Helper class to convert a flatzinc model to a CpModelProto.
72struct CpModelProtoWithMapping {
73 // Returns a constant CpModelProto variable created on-demand.
74 int LookupConstant(int64_t value);
75
76 // Convert a flatzinc argument to a variable or a list of variable.
77 // Note that we always encode a constant argument with a constant variable.
78 int LookupVar(const fz::Argument& argument);
79 std::vector<int> LookupVars(const fz::Argument& argument);
80 std::vector<VarOrValue> LookupVarsOrValues(const fz::Argument& argument);
81
82 // Create and return the indices of the IntervalConstraint corresponding
83 // to the flatzinc "interval" specified by a start var and a size var.
84 // This method will cache intervals with the key <start, size>.
85 std::vector<int> CreateIntervals(const std::vector<int>& starts,
86 const std::vector<VarOrValue>& sizes);
87
88 // Create and return the index of the optional IntervalConstraint
89 // corresponding to the flatzinc "interval" specified by a start var, the
90 // size_var, and the Boolean opt_var. This method will cache intervals with
91 // the key <start, size, opt_var>. If opt_var == kNoVar, the interval will not
92 // be optional.
93 int GetOrCreateOptionalInterval(int start_var, VarOrValue size, int opt_var);
94
95 // Adds a constraint to the model, add the enforcement literal if it is
96 // different from kNoVar, and returns a ptr to the ConstraintProto.
97 ConstraintProto* AddEnforcedConstraint(int literal);
98
99 // Helpers to fill a ConstraintProto.
100 void FillAMinusBInDomain(const std::vector<int64_t>& domain,
101 const fz::Constraint& fz_ct, ConstraintProto* ct);
102 void FillLinearConstraintWithGivenDomain(const std::vector<int64_t>& domain,
103 const fz::Constraint& fz_ct,
104 ConstraintProto* ct);
105 void FillConstraint(const fz::Constraint& fz_ct, ConstraintProto* ct);
106 void FillReifOrImpliedConstraint(const fz::Constraint& fz_ct,
107 ConstraintProto* ct);
108
109 // Translates the flatzinc search annotations into the CpModelProto
110 // search_order field.
111 void TranslateSearchAnnotations(
112 const std::vector<fz::Annotation>& search_annotations);
113
114 // The output proto.
115 CpModelProto proto;
116 SatParameters parameters;
117
118 // Mapping from flatzinc variables to CpModelProto variables.
119 absl::flat_hash_map<fz::Variable*, int> fz_var_to_index;
120 absl::flat_hash_map<int64_t, int> constant_value_to_index;
121 absl::flat_hash_map<std::tuple<int, int, int>, int>
122 start_size_opt_tuple_to_interval;
123 absl::flat_hash_map<std::tuple<int, int64_t, int>, int>
124 start_fixed_size_opt_tuple_to_interval;
125};

127int CpModelProtoWithMapping::LookupConstant(int64_t value) {
128 if (constant_value_to_index.contains(value)) {
129 return constant_value_to_index[value];
130 }
131
132 // Create the constant on the fly.
133 const int index = proto.variables_size();
134 IntegerVariableProto* var_proto = proto.add_variables();
135 var_proto->add_domain(value);
136 var_proto->add_domain(value);
137 constant_value_to_index[value] = index;
138 return index;
139}
140
141int CpModelProtoWithMapping::LookupVar(const fz::Argument& argument) {
142 if (argument.HasOneValue()) return LookupConstant(argument.Value());
143 CHECK_EQ(argument.type, fz::Argument::VAR_REF);
144 return fz_var_to_index[argument.Var()];
145}
146
147std::vector<int> CpModelProtoWithMapping::LookupVars(
148 const fz::Argument& argument) {
149 std::vector<int> result;
150 if (argument.type == fz::Argument::VOID_ARGUMENT) return result;
151 if (argument.type == fz::Argument::INT_LIST) {
152 for (int64_t value : argument.values) {
153 result.push_back(LookupConstant(value));
154 }
155 } else if (argument.type == fz::Argument::INT_VALUE) {
156 result.push_back(LookupConstant(argument.Value()));
157 } else {
158 CHECK_EQ(argument.type, fz::Argument::VAR_REF_ARRAY);
159 for (fz::Variable* var : argument.variables) {
160 CHECK(var != nullptr);
161 result.push_back(fz_var_to_index[var]);
162 }
163 }
164 return result;
165}
166
167std::vector<VarOrValue> CpModelProtoWithMapping::LookupVarsOrValues(
168 const fz::Argument& argument) {
169 std::vector<VarOrValue> result;
170 const int no_var = kNoVar;
171 if (argument.type == fz::Argument::VOID_ARGUMENT) return result;
172 if (argument.type == fz::Argument::INT_LIST) {
173 for (int64_t value : argument.values) {
174 result.push_back({no_var, value});
175 }
176 } else if (argument.type == fz::Argument::INT_VALUE) {
177 result.push_back({no_var, argument.Value()});
178 } else {
179 CHECK_EQ(argument.type, fz::Argument::VAR_REF_ARRAY);
180 for (fz::Variable* var : argument.variables) {
181 CHECK(var != nullptr);
182 if (var->domain.HasOneValue()) {
183 result.push_back({no_var, var->domain.Value()});
184 } else {
185 result.push_back({fz_var_to_index[var], 0});
186 }
187 }
188 }
189 return result;
190}

192ConstraintProto* CpModelProtoWithMapping::AddEnforcedConstraint(int literal) {
193 ConstraintProto* result = proto.add_constraints();
194 if (literal != kNoVar) {
195 result->add_enforcement_literal(literal);
196 }
197 return result;
198}

200int CpModelProtoWithMapping::GetOrCreateOptionalInterval(int start_var,
201 VarOrValue size,
202 int opt_var) {
203 const int interval_index = proto.constraints_size();
204 if (size.var == kNoVar) { // Size is fixed.
205 const std::tuple<int, int64_t, int> key =
206 std::make_tuple(start_var, size.value, opt_var);
207 const auto [it, inserted] =
208 start_fixed_size_opt_tuple_to_interval.insert({key, interval_index});
209 if (!inserted) {
210 return it->second;
211 }
212
213 auto* interval = AddEnforcedConstraint(opt_var)->mutable_interval();
214 interval->mutable_start_view()->add_vars(start_var);
215 interval->mutable_start_view()->add_coeffs(1);
216 interval->mutable_size_view()->set_offset(size.value);
217 interval->mutable_end_view()->add_vars(start_var);
218 interval->mutable_end_view()->add_coeffs(1);
219 interval->mutable_end_view()->set_offset(size.value);
220
221 return interval_index;
222 } else { // Size is variable.
223 const std::tuple<int, int, int> key =
224 std::make_tuple(start_var, size.var, opt_var);
225 const auto [it, inserted] =
226 start_size_opt_tuple_to_interval.insert({key, interval_index});
227 if (!inserted) {
228 return it->second;
229 }
230
231 const int end_var = proto.variables_size();
232 FillDomainInProto(
233 ReadDomainFromProto(proto.variables(start_var))
234 .AdditionWith(ReadDomainFromProto(proto.variables(size.var))),
235 proto.add_variables());
236
237 // Create the interval.
238 auto* interval = AddEnforcedConstraint(opt_var)->mutable_interval();
239 interval->mutable_start_view()->add_vars(start_var);
240 interval->mutable_start_view()->add_coeffs(1);
241 interval->mutable_size_view()->add_vars(size.var);
242 interval->mutable_size_view()->add_coeffs(1);
243 interval->mutable_end_view()->add_vars(end_var);
244 interval->mutable_end_view()->add_coeffs(1);
245
246 // Add the linear constraint (after the interval constraint as we have
247 // stored its index).
248 auto* lin = AddEnforcedConstraint(opt_var)->mutable_linear();
249 lin->add_vars(start_var);
250 lin->add_coeffs(1);
251 lin->add_vars(size.var);
252 lin->add_coeffs(1);
253 lin->add_vars(end_var);
254 lin->add_coeffs(-1);
255 lin->add_domain(0);
256 lin->add_domain(0);
257
258 return interval_index;
259 }
260}

262std::vector<int> CpModelProtoWithMapping::CreateIntervals(
263 const std::vector<int>& starts, const std::vector<VarOrValue>& sizes) {
264 std::vector<int> intervals;
265 for (int i = 0; i < starts.size(); ++i) {
266 intervals.push_back(
267 GetOrCreateOptionalInterval(starts[i], sizes[i], kNoVar));
268 }
269 return intervals;
270}

272void CpModelProtoWithMapping::FillAMinusBInDomain(
273 const std::vector<int64_t>& domain, const fz::Constraint& fz_ct,
274 ConstraintProto* ct) {
275 auto* arg = ct->mutable_linear();
276 if (fz_ct.arguments[1].type == fz::Argument::INT_VALUE) {
277 const int64_t value = fz_ct.arguments[1].Value();
278 const int var_a = LookupVar(fz_ct.arguments[0]);
279 for (const int64_t domain_bound : domain) {
280 if (domain_bound == std::numeric_limits<int64_t>::min() ||
281 domain_bound == std::numeric_limits<int64_t>::max()) {
282 arg->add_domain(domain_bound);
283 } else {
284 arg->add_domain(domain_bound + value);
285 }
286 }
287 arg->add_vars(var_a);
288 arg->add_coeffs(1);
289 } else if (fz_ct.arguments[0].type == fz::Argument::INT_VALUE) {
290 const int64_t value = fz_ct.arguments[0].Value();
291 const int var_b = LookupVar(fz_ct.arguments[1]);
292 for (int64_t domain_bound : gtl::reversed_view(domain)) {
293 if (domain_bound == std::numeric_limits<int64_t>::min()) {
294 arg->add_domain(std::numeric_limits<int64_t>::max());
295 } else if (domain_bound == std::numeric_limits<int64_t>::max()) {
296 arg->add_domain(std::numeric_limits<int64_t>::min());
297 } else {
298 arg->add_domain(value - domain_bound);
299 }
300 }
301 arg->add_vars(var_b);
302 arg->add_coeffs(1);
303 } else {
304 for (const int64_t domain_bound : domain) arg->add_domain(domain_bound);
305 arg->add_vars(LookupVar(fz_ct.arguments[0]));
306 arg->add_coeffs(1);
307 arg->add_vars(LookupVar(fz_ct.arguments[1]));
308 arg->add_coeffs(-1);
309 }
310}

312void CpModelProtoWithMapping::FillLinearConstraintWithGivenDomain(
313 const std::vector<int64_t>& domain, const fz::Constraint& fz_ct,
314 ConstraintProto* ct) {
315 auto* arg = ct->mutable_linear();
316 for (const int64_t domain_bound : domain) arg->add_domain(domain_bound);
317 std::vector<int> vars = LookupVars(fz_ct.arguments[1]);
318 for (int i = 0; i < vars.size(); ++i) {
319 arg->add_vars(vars[i]);
320 arg->add_coeffs(fz_ct.arguments[0].values[i]);
321 }
322}

324void CpModelProtoWithMapping::FillConstraint(const fz::Constraint& fz_ct,
325 ConstraintProto* ct) {
326 if (fz_ct.type == "false_constraint") {
327 // An empty clause is always false.
328 ct->mutable_bool_or();
329 } else if (fz_ct.type == "bool_clause") {
330 auto* arg = ct->mutable_bool_or();
331 for (const int var : LookupVars(fz_ct.arguments[0])) {
332 arg->add_literals(TrueLiteral(var));
333 }
334 for (const int var : LookupVars(fz_ct.arguments[1])) {
335 arg->add_literals(FalseLiteral(var));
336 }
337 } else if (fz_ct.type == "bool_xor") {
338 // This is not the same semantics as the array_bool_xor as this constraint
339 // is actually a fully reified xor(a, b) <==> x.
340 const int a = LookupVar(fz_ct.arguments[0]);
341 const int b = LookupVar(fz_ct.arguments[1]);
342 const int x = LookupVar(fz_ct.arguments[2]);
343
344 // not(x) => a == b
345 ct->add_enforcement_literal(NegatedRef(x));
346 auto* const refute = ct->mutable_linear();
347 refute->add_vars(a);
348 refute->add_coeffs(1);
349 refute->add_vars(b);
350 refute->add_coeffs(-1);
351 refute->add_domain(0);
352 refute->add_domain(0);
353
354 // x => a + b == 1
355 auto* enforce = AddEnforcedConstraint(x)->mutable_linear();
356 enforce->add_vars(a);
357 enforce->add_coeffs(1);
358 enforce->add_vars(b);
359 enforce->add_coeffs(1);
360 enforce->add_domain(1);
361 enforce->add_domain(1);
362 } else if (fz_ct.type == "array_bool_or") {
363 auto* arg = ct->mutable_bool_or();
364 for (const int var : LookupVars(fz_ct.arguments[0])) {
365 arg->add_literals(TrueLiteral(var));
366 }
367 } else if (fz_ct.type == "array_bool_or_negated") {
368 auto* arg = ct->mutable_bool_and();
369 for (const int var : LookupVars(fz_ct.arguments[0])) {
370 arg->add_literals(FalseLiteral(var));
371 }
372 } else if (fz_ct.type == "array_bool_and") {
373 auto* arg = ct->mutable_bool_and();
374 for (const int var : LookupVars(fz_ct.arguments[0])) {
375 arg->add_literals(TrueLiteral(var));
376 }
377 } else if (fz_ct.type == "array_bool_and_negated") {
378 auto* arg = ct->mutable_bool_or();
379 for (const int var : LookupVars(fz_ct.arguments[0])) {
380 arg->add_literals(FalseLiteral(var));
381 }
382 } else if (fz_ct.type == "array_bool_xor") {
383 auto* arg = ct->mutable_bool_xor();
384 for (const int var : LookupVars(fz_ct.arguments[0])) {
385 arg->add_literals(TrueLiteral(var));
386 }
387 } else if (fz_ct.type == "bool_le" || fz_ct.type == "int_le") {
388 FillAMinusBInDomain({std::numeric_limits<int64_t>::min(), 0}, fz_ct, ct);
389 } else if (fz_ct.type == "bool_ge" || fz_ct.type == "int_ge") {
390 FillAMinusBInDomain({0, std::numeric_limits<int64_t>::max()}, fz_ct, ct);
391 } else if (fz_ct.type == "bool_lt" || fz_ct.type == "int_lt") {
392 FillAMinusBInDomain({std::numeric_limits<int64_t>::min(), -1}, fz_ct, ct);
393 } else if (fz_ct.type == "bool_gt" || fz_ct.type == "int_gt") {
394 FillAMinusBInDomain({1, std::numeric_limits<int64_t>::max()}, fz_ct, ct);
395 } else if (fz_ct.type == "bool_eq" || fz_ct.type == "int_eq" ||
396 fz_ct.type == "bool2int") {
397 FillAMinusBInDomain({0, 0}, fz_ct, ct);
398 } else if (fz_ct.type == "bool_ne" || fz_ct.type == "bool_not") {
399 auto* arg = ct->mutable_linear();
400 arg->add_vars(LookupVar(fz_ct.arguments[0]));
401 arg->add_coeffs(1);
402 arg->add_vars(LookupVar(fz_ct.arguments[1]));
403 arg->add_coeffs(1);
404 arg->add_domain(1);
405 arg->add_domain(1);
406 } else if (fz_ct.type == "int_ne") {
407 FillAMinusBInDomain({std::numeric_limits<int64_t>::min(), -1, 1,
408 std::numeric_limits<int64_t>::max()},
409 fz_ct, ct);
410 } else if (fz_ct.type == "int_lin_eq") {
411 const int64_t rhs = fz_ct.arguments[2].values[0];
412 FillLinearConstraintWithGivenDomain({rhs, rhs}, fz_ct, ct);
413 } else if (fz_ct.type == "bool_lin_eq") {
414 auto* arg = ct->mutable_linear();
415 const std::vector<int> vars = LookupVars(fz_ct.arguments[1]);
416 for (int i = 0; i < vars.size(); ++i) {
417 arg->add_vars(vars[i]);
418 arg->add_coeffs(fz_ct.arguments[0].values[i]);
419 }
420 if (fz_ct.arguments[2].IsVariable()) {
421 arg->add_vars(LookupVar(fz_ct.arguments[2]));
422 arg->add_coeffs(-1);
423 arg->add_domain(0);
424 arg->add_domain(0);
425 } else {
426 const int64_t v = fz_ct.arguments[2].Value();
427 arg->add_domain(v);
428 arg->add_domain(v);
429 }
430 } else if (fz_ct.type == "int_lin_le" || fz_ct.type == "bool_lin_le") {
431 const int64_t rhs = fz_ct.arguments[2].values[0];
432 FillLinearConstraintWithGivenDomain(
433 {std::numeric_limits<int64_t>::min(), rhs}, fz_ct, ct);
434 } else if (fz_ct.type == "int_lin_lt") {
435 const int64_t rhs = fz_ct.arguments[2].values[0];
436 FillLinearConstraintWithGivenDomain(
437 {std::numeric_limits<int64_t>::min(), rhs - 1}, fz_ct, ct);
438 } else if (fz_ct.type == "int_lin_ge") {
439 const int64_t rhs = fz_ct.arguments[2].values[0];
440 FillLinearConstraintWithGivenDomain(
441 {rhs, std::numeric_limits<int64_t>::max()}, fz_ct, ct);
442 } else if (fz_ct.type == "int_lin_gt") {
443 const int64_t rhs = fz_ct.arguments[2].values[0];
444 FillLinearConstraintWithGivenDomain(
445 {rhs + 1, std::numeric_limits<int64_t>::max()}, fz_ct, ct);
446 } else if (fz_ct.type == "int_lin_ne") {
447 const int64_t rhs = fz_ct.arguments[2].values[0];
448 FillLinearConstraintWithGivenDomain(
449 {std::numeric_limits<int64_t>::min(), rhs - 1, rhs + 1,
450 std::numeric_limits<int64_t>::max()},
451 fz_ct, ct);
452 } else if (fz_ct.type == "set_in") {
453 auto* arg = ct->mutable_linear();
454 arg->add_vars(LookupVar(fz_ct.arguments[0]));
455 arg->add_coeffs(1);
456 if (fz_ct.arguments[1].type == fz::Argument::INT_LIST) {
457 FillDomainInProto(Domain::FromValues(std::vector<int64_t>{
458 fz_ct.arguments[1].values.begin(),
459 fz_ct.arguments[1].values.end()}),
460 arg);
461 } else if (fz_ct.arguments[1].type == fz::Argument::INT_INTERVAL) {
462 FillDomainInProto(
463 Domain(fz_ct.arguments[1].values[0], fz_ct.arguments[1].values[1]),
464 arg);
465 } else {
466 LOG(FATAL) << "Wrong format";
467 }
468 } else if (fz_ct.type == "set_in_negated") {
469 auto* arg = ct->mutable_linear();
470 arg->add_vars(LookupVar(fz_ct.arguments[0]));
471 arg->add_coeffs(1);
472 if (fz_ct.arguments[1].type == fz::Argument::INT_LIST) {
473 FillDomainInProto(
474 Domain::FromValues(
475 std::vector<int64_t>{fz_ct.arguments[1].values.begin(),
476 fz_ct.arguments[1].values.end()})
477 .Complement(),
478 arg);
479 } else if (fz_ct.arguments[1].type == fz::Argument::INT_INTERVAL) {
480 FillDomainInProto(
481 Domain(fz_ct.arguments[1].values[0], fz_ct.arguments[1].values[1])
482 .Complement(),
483 arg);
484 } else {
485 LOG(FATAL) << "Wrong format";
486 }
487 } else if (fz_ct.type == "int_min") {
488 auto* arg = ct->mutable_int_min();
489 arg->set_target(LookupVar(fz_ct.arguments[2]));
490 arg->add_vars(LookupVar(fz_ct.arguments[0]));
491 arg->add_vars(LookupVar(fz_ct.arguments[1]));
492 } else if (fz_ct.type == "array_int_minimum" || fz_ct.type == "minimum_int") {
493 auto* arg = ct->mutable_int_min();
494 arg->set_target(LookupVar(fz_ct.arguments[0]));
495 for (const int var : LookupVars(fz_ct.arguments[1])) arg->add_vars(var);
496 } else if (fz_ct.type == "int_max") {
497 auto* arg = ct->mutable_int_max();
498 arg->set_target(LookupVar(fz_ct.arguments[2]));
499 arg->add_vars(LookupVar(fz_ct.arguments[0]));
500 arg->add_vars(LookupVar(fz_ct.arguments[1]));
501 } else if (fz_ct.type == "array_int_maximum" || fz_ct.type == "maximum_int") {
502 auto* arg = ct->mutable_int_max();
503 arg->set_target(LookupVar(fz_ct.arguments[0]));
504 for (const int var : LookupVars(fz_ct.arguments[1])) arg->add_vars(var);
505 } else if (fz_ct.type == "int_times") {
506 auto* arg = ct->mutable_int_prod();
507 arg->set_target(LookupVar(fz_ct.arguments[2]));
508 arg->add_vars(LookupVar(fz_ct.arguments[0]));
509 arg->add_vars(LookupVar(fz_ct.arguments[1]));
510 } else if (fz_ct.type == "int_abs") {
511 auto* arg = ct->mutable_int_max();
512 arg->set_target(LookupVar(fz_ct.arguments[1]));
513 arg->add_vars(LookupVar(fz_ct.arguments[0]));
514 arg->add_vars(-LookupVar(fz_ct.arguments[0]) - 1);
515 } else if (fz_ct.type == "int_plus") {
516 auto* arg = ct->mutable_linear();
517 FillDomainInProto(Domain(0, 0), arg);
518 arg->add_vars(LookupVar(fz_ct.arguments[0]));
519 arg->add_coeffs(1);
520 arg->add_vars(LookupVar(fz_ct.arguments[1]));
521 arg->add_coeffs(1);
522 arg->add_vars(LookupVar(fz_ct.arguments[2]));
523 arg->add_coeffs(-1);
524 } else if (fz_ct.type == "int_div") {
525 auto* arg = ct->mutable_int_div();
526 arg->add_vars(LookupVar(fz_ct.arguments[0]));
527 arg->add_vars(LookupVar(fz_ct.arguments[1]));
528 arg->set_target(LookupVar(fz_ct.arguments[2]));
529 } else if (fz_ct.type == "int_mod") {
530 auto* arg = ct->mutable_int_mod();
531 arg->set_target(LookupVar(fz_ct.arguments[2]));
532 arg->add_vars(LookupVar(fz_ct.arguments[0]));
533 arg->add_vars(LookupVar(fz_ct.arguments[1]));
534 } else if (fz_ct.type == "array_int_element" ||
535 fz_ct.type == "array_bool_element" ||
536 fz_ct.type == "array_var_int_element" ||
537 fz_ct.type == "array_var_bool_element" ||
538 fz_ct.type == "array_int_element_nonshifted") {
539 if (fz_ct.arguments[0].type == fz::Argument::VAR_REF ||
540 fz_ct.arguments[0].type == fz::Argument::INT_VALUE) {
541 auto* arg = ct->mutable_element();
542 arg->set_index(LookupVar(fz_ct.arguments[0]));
543 arg->set_target(LookupVar(fz_ct.arguments[2]));
544
545 if (!absl::EndsWith(fz_ct.type, "_nonshifted")) {
546 // Add a dummy variable at position zero because flatzinc index start
547 // at 1.
548 // TODO(user): Make sure that zero is not in the index domain...
549 arg->add_vars(LookupConstant(0));
550 }
551 for (const int var : LookupVars(fz_ct.arguments[1])) arg->add_vars(var);
552 } else {
553 // Special case added by the presolve or in flatzinc. We encode this
554 // as a table constraint.
555 CHECK(!absl::EndsWith(fz_ct.type, "_nonshifted"));
556 auto* arg = ct->mutable_table();
557
558 // the constraint is:
559 // values[coeff1 * vars[0] + coeff2 * vars[1] + offset] == target.
560 for (const int var : LookupVars(fz_ct.arguments[0])) arg->add_vars(var);
561 arg->add_vars(LookupVar(fz_ct.arguments[2])); // the target
562
563 const std::vector<int64_t>& values = fz_ct.arguments[1].values;
564 const int64_t coeff1 = fz_ct.arguments[3].values[0];
565 const int64_t coeff2 = fz_ct.arguments[3].values[1];
566 const int64_t offset = fz_ct.arguments[4].values[0] - 1;
567
568 for (const int64_t a : AllValuesInDomain(proto.variables(arg->vars(0)))) {
569 for (const int64_t b :
570 AllValuesInDomain(proto.variables(arg->vars(1)))) {
571 const int index = coeff1 * a + coeff2 * b + offset;
572 CHECK_GE(index, 0);
573 CHECK_LT(index, values.size());
574 arg->add_values(a);
575 arg->add_values(b);
576 arg->add_values(values[index]);
577 }
578 }
579 }
580 } else if (fz_ct.type == "ortools_table_int") {
581 auto* arg = ct->mutable_table();
582 for (const int var : LookupVars(fz_ct.arguments[0])) arg->add_vars(var);
583 for (const int64_t value : fz_ct.arguments[1].values)
584 arg->add_values(value);
585 } else if (fz_ct.type == "ortools_regular") {
586 auto* arg = ct->mutable_automaton();
587 for (const int var : LookupVars(fz_ct.arguments[0])) arg->add_vars(var);
588
589 int count = 0;
590 const int num_states = fz_ct.arguments[1].Value();
591 const int num_values = fz_ct.arguments[2].Value();
592 for (int i = 1; i <= num_states; ++i) {
593 for (int j = 1; j <= num_values; ++j) {
594 CHECK_LT(count, fz_ct.arguments[3].values.size());
595 const int next = fz_ct.arguments[3].values[count++];
596 if (next == 0) continue; // 0 is a failing state.
597 arg->add_transition_tail(i);
598 arg->add_transition_label(j);
599 arg->add_transition_head(next);
600 }
601 }
602
603 arg->set_starting_state(fz_ct.arguments[4].Value());
604 switch (fz_ct.arguments[5].type) {
605 case fz::Argument::INT_VALUE: {
606 arg->add_final_states(fz_ct.arguments[5].values[0]);
607 break;
608 }
609 case fz::Argument::INT_INTERVAL: {
610 for (int v = fz_ct.arguments[5].values[0];
611 v <= fz_ct.arguments[5].values[1]; ++v) {
612 arg->add_final_states(v);
613 }
614 break;
615 }
616 case fz::Argument::INT_LIST: {
617 for (const int v : fz_ct.arguments[5].values) {
618 arg->add_final_states(v);
619 }
620 break;
621 }
622 default: {
623 LOG(FATAL) << "Wrong constraint " << fz_ct.DebugString();
624 }
625 }
626 } else if (fz_ct.type == "fzn_all_different_int") {
627 auto* arg = ct->mutable_all_diff();
628 for (const int var : LookupVars(fz_ct.arguments[0])) arg->add_vars(var);
629 } else if (fz_ct.type == "ortools_circuit" ||
630 fz_ct.type == "ortools_subcircuit") {
631 const int64_t min_index = fz_ct.arguments[1].Value();
632 const int size = std::max(fz_ct.arguments[0].values.size(),
633 fz_ct.arguments[0].variables.size());
634
635 const int64_t max_index = min_index + size - 1;
636 // The arc-based mutable circuit.
637 auto* circuit_arg = ct->mutable_circuit();
638
639 // We fully encode all variables so we can use the literal based circuit.
640 // TODO(user): avoid fully encoding more than once?
641 int64_t index = min_index;
642 const bool is_circuit = (fz_ct.type == "ortools_circuit");
643 for (const int var : LookupVars(fz_ct.arguments[0])) {
644 Domain domain = ReadDomainFromProto(proto.variables(var));
645
646 // Restrict the domain of var to [min_index, max_index]
647 domain = domain.IntersectionWith(Domain(min_index, max_index));
648 if (is_circuit) {
649 // We simply make sure that the variable cannot take the value index.
650 domain = domain.IntersectionWith(Domain::FromIntervals(
651 {{std::numeric_limits<int64_t>::min(), index - 1},
652 {index + 1, std::numeric_limits<int64_t>::max()}}));
653 }
654 FillDomainInProto(domain, proto.mutable_variables(var));
655
656 for (const ClosedInterval interval : domain.intervals()) {
657 for (int64_t value = interval.start; value <= interval.end; ++value) {
658 // Create one Boolean variable for this arc.
659 const int literal = proto.variables_size();
660 {
661 auto* new_var = proto.add_variables();
662 new_var->add_domain(0);
663 new_var->add_domain(1);
664 }
665
666 // Add the arc.
667 circuit_arg->add_tails(index);
668 circuit_arg->add_heads(value);
669 circuit_arg->add_literals(literal);
670
671 // literal => var == value.
672 {
673 auto* lin = AddEnforcedConstraint(literal)->mutable_linear();
674 lin->add_coeffs(1);
675 lin->add_vars(var);
676 lin->add_domain(value);
677 lin->add_domain(value);
678 }
679
680 // not(literal) => var != value
681 {
682 auto* lin =
683 AddEnforcedConstraint(NegatedRef(literal))->mutable_linear();
684 lin->add_coeffs(1);
685 lin->add_vars(var);
686 lin->add_domain(std::numeric_limits<int64_t>::min());
687 lin->add_domain(value - 1);
688 lin->add_domain(value + 1);
689 lin->add_domain(std::numeric_limits<int64_t>::max());
690 }
691 }
692 }
693
694 ++index;
695 }
696 } else if (fz_ct.type == "ortools_inverse") {
697 auto* arg = ct->mutable_inverse();
698
699 const auto direct_variables = LookupVars(fz_ct.arguments[0]);
700 const auto inverse_variables = LookupVars(fz_ct.arguments[1]);
701 const int base_direct = fz_ct.arguments[2].Value();
702 const int base_inverse = fz_ct.arguments[3].Value();
703
704 CHECK_EQ(direct_variables.size(), inverse_variables.size());
705 const int num_variables = direct_variables.size();
706 const int end_direct = base_direct + num_variables;
707 const int end_inverse = base_inverse + num_variables;
708
709 // Any convention that maps the "fixed values" to the one of the inverse and
710 // back works. We decided to follow this one:
711 // There are 3 cases:
712 // (A) base_direct == base_inverse, we fill the arrays
713 // direct = [0, .., base_direct - 1] U [direct_vars]
714 // inverse = [0, .., base_direct - 1] U [inverse_vars]
715 // (B) base_direct == base_inverse + offset (> 0), we fill the arrays
716 // direct = [0, .., base_inverse - 1] U
717 // [end_inverse, .., end_inverse + offset - 1] U
718 // [direct_vars]
719 // inverse = [0, .., base_inverse - 1] U
720 // [inverse_vars] U
721 // [base_inverse, .., base_base_inverse + offset - 1]
722 // (C): base_inverse == base_direct + offset (> 0), we fill the arrays
723 // direct = [0, .., base_direct - 1] U
724 // [direct_vars] U
725 // [base_direct, .., base_direct + offset - 1]
726 // inverse [0, .., base_direct - 1] U
727 // [end_direct, .., end_direct + offset - 1] U
728 // [inverse_vars]
729 const int arity = std::max(base_inverse, base_direct) + num_variables;
730 for (int i = 0; i < arity; ++i) {
731 // Fill the direct array.
732 if (i < base_direct) {
733 if (i < base_inverse) {
734 arg->add_f_direct(LookupConstant(i));
735 } else if (i >= base_inverse) {
736 arg->add_f_direct(LookupConstant(i + num_variables));
737 }
738 } else if (i >= base_direct && i < end_direct) {
739 arg->add_f_direct(direct_variables[i - base_direct]);
740 } else {
741 arg->add_f_direct(LookupConstant(i - num_variables));
742 }
743
744 // Fill the inverse array.
745 if (i < base_inverse) {
746 if (i < base_direct) {
747 arg->add_f_inverse(LookupConstant(i));
748 } else if (i >= base_direct) {
749 arg->add_f_inverse(LookupConstant(i + num_variables));
750 }
751 } else if (i >= base_inverse && i < end_inverse) {
752 arg->add_f_inverse(inverse_variables[i - base_inverse]);
753 } else {
754 arg->add_f_inverse(LookupConstant(i - num_variables));
755 }
756 }
757 } else if (fz_ct.type == "fzn_cumulative") {
758 const std::vector<int> starts = LookupVars(fz_ct.arguments[0]);
759 const std::vector<VarOrValue> sizes =
760 LookupVarsOrValues(fz_ct.arguments[1]);
761 const std::vector<int> demands = LookupVars(fz_ct.arguments[2]);
762 const int capacity = LookupVar(fz_ct.arguments[3]);
763
764 auto* arg = ct->mutable_cumulative();
765 arg->set_capacity(capacity);
766 for (int i = 0; i < starts.size(); ++i) {
767 // Special case for a 0-1 demand, we mark the interval as optional
768 // instead and fix the demand to 1.
769 if (proto.variables(demands[i]).domain().size() == 2 &&
770 proto.variables(demands[i]).domain(0) == 0 &&
771 proto.variables(demands[i]).domain(1) == 1 &&
772 proto.variables(capacity).domain(1) == 1) {
773 arg->add_intervals(
774 GetOrCreateOptionalInterval(starts[i], sizes[i], demands[i]));
775 arg->add_demands(LookupConstant(1));
776 } else {
777 arg->add_intervals(
778 GetOrCreateOptionalInterval(starts[i], sizes[i], kNoVar));
779 arg->add_demands(demands[i]);
780 }
781 }
782 } else if (fz_ct.type == "fzn_diffn" || fz_ct.type == "fzn_diffn_nonstrict") {
783 const std::vector<int> x = LookupVars(fz_ct.arguments[0]);
784 const std::vector<int> y = LookupVars(fz_ct.arguments[1]);
785 const std::vector<VarOrValue> dx = LookupVarsOrValues(fz_ct.arguments[2]);
786 const std::vector<VarOrValue> dy = LookupVarsOrValues(fz_ct.arguments[3]);
787 const std::vector<int> x_intervals = CreateIntervals(x, dx);
788 const std::vector<int> y_intervals = CreateIntervals(y, dy);
789 auto* arg = ct->mutable_no_overlap_2d();
790 for (int i = 0; i < x.size(); ++i) {
791 arg->add_x_intervals(x_intervals[i]);
792 arg->add_y_intervals(y_intervals[i]);
793 }
794 arg->set_boxes_with_null_area_can_overlap(fz_ct.type ==
795 "fzn_diffn_nonstrict");
796 } else if (fz_ct.type == "ortools_network_flow" ||
797 fz_ct.type == "ortools_network_flow_cost") {
798 // Note that we leave ct empty here (with just the name set).
799 // We simply do a linear encoding of this constraint.
800 const bool has_cost = fz_ct.type == "ortools_network_flow_cost";
801 const std::vector<int> flow = LookupVars(fz_ct.arguments[has_cost ? 3 : 2]);
802
803 // Flow conservation constraints.
804 const int num_nodes = fz_ct.arguments[1].values.size();
805 std::vector<std::vector<int>> flows_per_node(num_nodes);
806 std::vector<std::vector<int>> coeffs_per_node(num_nodes);
807 const int num_arcs = fz_ct.arguments[0].values.size() / 2;
808 for (int arc = 0; arc < num_arcs; arc++) {
809 const int tail = fz_ct.arguments[0].values[2 * arc] - 1;
810 const int head = fz_ct.arguments[0].values[2 * arc + 1] - 1;
811 if (tail == head) continue;
812
813 flows_per_node[tail].push_back(flow[arc]);
814 coeffs_per_node[tail].push_back(1);
815 flows_per_node[head].push_back(flow[arc]);
816 coeffs_per_node[head].push_back(-1);
817 }
818 for (int node = 0; node < num_nodes; node++) {
819 auto* arg = proto.add_constraints()->mutable_linear();
820 arg->add_domain(fz_ct.arguments[1].values[node]);
821 arg->add_domain(fz_ct.arguments[1].values[node]);
822 for (int i = 0; i < flows_per_node[node].size(); ++i) {
823 arg->add_vars(flows_per_node[node][i]);
824 arg->add_coeffs(coeffs_per_node[node][i]);
825 }
826 }
827
828 if (has_cost) {
829 auto* arg = proto.add_constraints()->mutable_linear();
830 arg->add_domain(0);
831 arg->add_domain(0);
832 for (int arc = 0; arc < num_arcs; arc++) {
833 const int64_t weight = fz_ct.arguments[2].values[arc];
834 if (weight != 0) {
835 arg->add_vars(flow[arc]);
836 arg->add_coeffs(weight);
837 }
838 }
839 arg->add_vars(LookupVar(fz_ct.arguments[4]));
840 arg->add_coeffs(-1);
841 }
842 } else {
843 LOG(FATAL) << " Not supported " << fz_ct.type;
844 }
845}

847void CpModelProtoWithMapping::FillReifOrImpliedConstraint(
848 const fz::Constraint& fz_ct, ConstraintProto* ct) {
849 // Start by adding a non-reified version of the same constraint.
850 std::string simplified_type;
851 if (absl::EndsWith(fz_ct.type, "_reif")) {
852 // Remove _reif.
853 simplified_type = fz_ct.type.substr(0, fz_ct.type.size() - 5);
854 } else if (absl::EndsWith(fz_ct.type, "_imp")) {
855 // Remove _imp.
856 simplified_type = fz_ct.type.substr(0, fz_ct.type.size() - 4);
857 } else {
858 // Keep name as it is an implicit reified constraint.
859 simplified_type = fz_ct.type;
860 }
861
862 // We need a copy to be able to change the type of the constraint.
863 fz::Constraint copy = fz_ct;
864 copy.type = simplified_type;
865
866 // Create the CP-SAT constraint.
867 FillConstraint(copy, ct);
868
869 // In case of reified constraints, the type of the opposite constraint.
870 std::string negated_type;
871
872 // Fill enforcement_literal and set copy.type to the negated constraint.
873 if (simplified_type == "array_bool_or") {
874 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[1])));
875 negated_type = "array_bool_or_negated";
876 } else if (simplified_type == "array_bool_and") {
877 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[1])));
878 negated_type = "array_bool_and_negated";
879 } else if (simplified_type == "set_in") {
880 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
881 negated_type = "set_in_negated";
882 } else if (simplified_type == "bool_eq" || simplified_type == "int_eq") {
883 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
884 negated_type = "int_ne";
885 } else if (simplified_type == "bool_ne" || simplified_type == "int_ne") {
886 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
887 negated_type = "int_eq";
888 } else if (simplified_type == "bool_le" || simplified_type == "int_le") {
889 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
890 negated_type = "int_gt";
891 } else if (simplified_type == "bool_lt" || simplified_type == "int_lt") {
892 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
893 negated_type = "int_ge";
894 } else if (simplified_type == "bool_ge" || simplified_type == "int_ge") {
895 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
896 negated_type = "int_lt";
897 } else if (simplified_type == "bool_gt" || simplified_type == "int_gt") {
898 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
899 negated_type = "int_le";
900 } else if (simplified_type == "int_lin_eq") {
901 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
902 negated_type = "int_lin_ne";
903 } else if (simplified_type == "int_lin_ne") {
904 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
905 negated_type = "int_lin_eq";
906 } else if (simplified_type == "int_lin_le") {
907 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
908 negated_type = "int_lin_gt";
909 } else if (simplified_type == "int_lin_ge") {
910 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
911 negated_type = "int_lin_lt";
912 } else if (simplified_type == "int_lin_lt") {
913 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
914 negated_type = "int_lin_ge";
915 } else if (simplified_type == "int_lin_gt") {
916 ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
917 negated_type = "int_lin_le";
918 } else {
919 LOG(FATAL) << "Unsupported " << simplified_type;
920 }
921
922 // One way implication. We can stop here.
923 if (absl::EndsWith(fz_ct.type, "_imp")) return;
924
925 // Add the other side of the reification because CpModelProto only support
926 // half reification.
927 ConstraintProto* negated_ct = proto.add_constraints();
928 negated_ct->set_name(fz_ct.type + " (negated)");
929 negated_ct->add_enforcement_literal(
930 sat::NegatedRef(ct->enforcement_literal(0)));
931 copy.type = negated_type;
932 FillConstraint(copy, negated_ct);
933}

935void CpModelProtoWithMapping::TranslateSearchAnnotations(
936 const std::vector<fz::Annotation>& search_annotations) {
937 std::vector<fz::Annotation> flat_annotations;
938 for (const fz::Annotation& annotation : search_annotations) {
939 fz::FlattenAnnotations(annotation, &flat_annotations);
940 }
941
942 for (const fz::Annotation& annotation : flat_annotations) {
943 if (annotation.IsFunctionCallWithIdentifier("int_search") ||
944 annotation.IsFunctionCallWithIdentifier("bool_search")) {
945 const std::vector<fz::Annotation>& args = annotation.annotations;
946 std::vector<fz::Variable*> vars;
947 args[0].AppendAllVariables(&vars);
948
949 DecisionStrategyProto* strategy = proto.add_search_strategy();
950 for (fz::Variable* v : vars) {

961 } else if (choose.id == "anti_first_fail") {
962 strategy->set_variable_selection_strategy(
963 DecisionStrategyProto::CHOOSE_MAX_DOMAIN_SIZE);

965 strategy->set_variable_selection_strategy(
966 DecisionStrategyProto::CHOOSE_LOWEST_MIN);
967 } else if (choose.id == "largest") {
968 strategy->set_variable_selection_strategy(

971 LOG(FATAL) << "Unsupported order: " << choose.id;
972 }

975 if (select.id == "indomain_min" || select.id == "indomain") {

982 strategy->set_domain_reduction_strategy(
983 DecisionStrategyProto::SELECT_LOWER_HALF);
984 } else if (select.id == "indomain_reverse_split") {
985 strategy->set_domain_reduction_strategy(
986 DecisionStrategyProto::SELECT_UPPER_HALF);

993 }

997// The format is fixed in the flatzinc specification.
998std::string SolutionString(
999 const fz::SolutionOutputSpecs& output,
1000 const std::function<int64_t(fz::Variable*)>& value_func) {
1001 if (output.variable != nullptr) {
1002 const int64_t value = value_func(output.variable);
1003 if (output.display_as_boolean) {
1004 return absl::StrCat(output.name, " = ", value == 1 ? "true" : "false",
1005 ";");
1006 } else {
1007 return absl::StrCat(output.name, " = ", value, ";");
1008 }
1009 } else {
1010 const int bound_size = output.bounds.size();
1011 std::string result =
1012 absl::StrCat(output.name, " = array", bound_size, "d(");
1013 for (int i = 0; i < bound_size; ++i) {
1014 if (output.bounds[i].max_value >= output.bounds[i].min_value) {
1015 absl::StrAppend(&result, output.bounds[i].min_value, "..",
1016 output.bounds[i].max_value, ", ");
1017 } else {
1018 result.append("{},");
1019 }
1020 }
1021 result.append("[");
1022 for (int i = 0; i < output.flat_variables.size(); ++i) {
1023 const int64_t value = value_func(output.flat_variables[i]);
1024 if (output.display_as_boolean) {
1025 result.append(value ? "true" : "false");
1026 } else {
1027 absl::StrAppend(&result, value);
1028 }
1029 if (i != output.flat_variables.size() - 1) {
1030 result.append(", ");
1031 }
1032 }
1033 result.append("]);");
1034 return result;
1035 }
1036 return "";
1037}

1091 // their propagation. Because of that, we trim undefined variable
1092 // domains (i.e. int in minizinc) to something hopefully large enough.
1093 LOG_FIRST_N(WARNING, 1)

1095 LOG_FIRST_N(WARNING, 1)

1099 var->add_domain(absl::GetFlag(FLAGS_fz_int_max));

1101 var->add_domain(fz_var->domain.values[0]);
1102 var->add_domain(fz_var->domain.values[1]);

1104 } else {
1105 FillDomainInProto(Domain::FromValues(fz_var->domain.values), var);

1130 NegatedCpModelVariable(m.fz_var_to_index[fz_model.objective()]));

1133 }

1162 // then query the number of cores available. This is complex now as we
1163 // need to still support the expected behabior (no flags -> 1 thread
1164 // fixed search, -f -> 1 thread free search).

1166 "The number of search workers, is not specified. For better "
1167 "performances, please set the number of workers to 8, 16, or "
1168 "more depending on the number of cores of your computer.");

1177 m.parameters.set_search_branching(SatParameters::AUTOMATIC_SEARCH);
1178 m.parameters.set_interleave_search(true);

1193 sat::SatParameters flag_parameters;

1195 &flag_parameters))

1231 fz_model,
1232 [&response, &m](fz::Variable* v) {

1234 },
1235 logger));

1242 if (!p.display_all_solutions) { // Already printed otherwise.
1243 const std::string solution_string =
1244 SolutionString(fz_model, [&response, &m](fz::Variable* v) {
1245 return response.solution(gtl::FindOrDie(m.fz_var_to_index, v));

1247 SOLVER_LOG(solution_logger, solution_string);
1248 SOLVER_LOG(solution_logger, "----------");
