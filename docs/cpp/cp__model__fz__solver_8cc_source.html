117 int CpModelProtoWithMapping::LookupConstant(int64 value) {
118  if (gtl::ContainsKey(constant_value_to_index, value)) {
119  return constant_value_to_index[value];
120  }
121 
122  // Create the constant on the fly.
123  const int index = proto.variables_size();
124  IntegerVariableProto* var_proto = proto.add_variables();
125  var_proto->add_domain(value);
126  var_proto->add_domain(value);
127  constant_value_to_index[value] = index;
128  return index;
129 }

132  if (argument.HasOneValue()) return LookupConstant(argument.Value());
133  CHECK_EQ(argument.type, fz::Argument::INT_VAR_REF);
134  return fz_var_to_index[argument.Var()];

137 std::vector<int> CpModelProtoWithMapping::LookupVars(
138  const fz::Argument& argument) {
139  std::vector<int> result;
140  if (argument.type == fz::Argument::VOID_ARGUMENT) return result;
141  if (argument.type == fz::Argument::INT_LIST) {
142  for (int64 value : argument.values) {
143  result.push_back(LookupConstant(value));
144  }
145 
146  } else if (argument.type == fz::Argument::INT_VALUE) {
147  result.push_back(LookupConstant(argument.Value()));
148  } else {
149  CHECK_EQ(argument.type, fz::Argument::INT_VAR_REF_ARRAY);
150  for (fz::IntegerVariable* var : argument.variables) {
151  CHECK(var != nullptr);
152  result.push_back(fz_var_to_index[var]);
153  }
154  }
155  return result;
156 }

162 int CpModelProtoWithMapping::GetOrCreateOptionalInterval(int start_var,
163  int size_var,
164  int opt_var) {
165  const std::tuple<int, int, int> key =
166  std::make_tuple(start_var, size_var, opt_var);
167  if (gtl::ContainsKey(start_size_opt_tuple_to_interval, key)) {
168  return start_size_opt_tuple_to_interval[key];
169  }
170  const int interval_index = proto.constraints_size();
171 
172  auto* ct = proto.add_constraints();
173  if (opt_var != kint32max) {
174  ct->add_enforcement_literal(opt_var);
175  }
176  auto* interval = ct->mutable_interval();
177  interval->set_start(start_var);
178  interval->set_size(size_var);
179 
180  interval->set_end(proto.variables_size());
181  auto* end_var = proto.add_variables();
182  const auto start_proto = proto.variables(start_var);
183  const auto size_proto = proto.variables(size_var);
184  end_var->add_domain(start_proto.domain(0) + size_proto.domain(0));
185  end_var->add_domain(start_proto.domain(start_proto.domain_size() - 1) +
186  size_proto.domain(size_proto.domain_size() - 1));
187  start_size_opt_tuple_to_interval[key] = interval_index;
188  return interval_index;
189 }

200 void CpModelProtoWithMapping::FillAMinusBInDomain(
201  const std::vector<int64>& domain, const fz::Constraint& fz_ct,
202  ConstraintProto* ct) {
203  auto* arg = ct->mutable_linear();
204  if (fz_ct.arguments[1].type == fz::Argument::INT_VALUE) {
205  const int64 value = fz_ct.arguments[1].Value();
206  const int var_a = LookupVar(fz_ct.arguments[0]);
207  for (const int64 domain_bound : domain) {
208  if (domain_bound == kint64min || domain_bound == kint64max) {
209  arg->add_domain(domain_bound);
210  } else {
211  arg->add_domain(domain_bound + value);
212  }
213  }
214  arg->add_vars(var_a);
215  arg->add_coeffs(1);
216  } else if (fz_ct.arguments[0].type == fz::Argument::INT_VALUE) {
217  const int64 value = fz_ct.arguments[0].Value();
218  const int var_b = LookupVar(fz_ct.arguments[1]);
219  for (int64 domain_bound : gtl::reversed_view(domain)) {
220  if (domain_bound == kint64min) {
221  arg->add_domain(kint64max);
222  } else if (domain_bound == kint64max) {
223  arg->add_domain(kint64min);
224  } else {
225  arg->add_domain(value - domain_bound);
226  }
227  }
228  arg->add_vars(var_b);
229  arg->add_coeffs(1);
230  } else {
231  for (const int64 domain_bound : domain) arg->add_domain(domain_bound);
232  arg->add_vars(LookupVar(fz_ct.arguments[0]));
233  arg->add_coeffs(1);
234  arg->add_vars(LookupVar(fz_ct.arguments[1]));
235  arg->add_coeffs(-1);
236  }
237 }

239 void CpModelProtoWithMapping::FillLinearConstraintWithGivenDomain(
240  const std::vector<int64>& domain, const fz::Constraint& fz_ct,
241  ConstraintProto* ct) {
242  auto* arg = ct->mutable_linear();
243  for (const int64 domain_bound : domain) arg->add_domain(domain_bound);
244  std::vector<int> vars = LookupVars(fz_ct.arguments[1]);
245  for (int i = 0; i < vars.size(); ++i) {
246  arg->add_vars(vars[i]);
247  arg->add_coeffs(fz_ct.arguments[0].values[i]);
248  }

251 void CpModelProtoWithMapping::FillConstraint(const fz::Constraint& fz_ct,
252  ConstraintProto* ct) {
253  if (fz_ct.type == "false_constraint") {
254  // An empty clause is always false.
255  ct->mutable_bool_or();
256  } else if (fz_ct.type == "bool_clause") {
257  auto* arg = ct->mutable_bool_or();
258  for (const int var : LookupVars(fz_ct.arguments[0])) {
259  arg->add_literals(TrueLiteral(var));
260  }
261  for (const int var : LookupVars(fz_ct.arguments[1])) {
262  arg->add_literals(FalseLiteral(var));
263  }
264  } else if (fz_ct.type == "bool_xor") {
265  // This is not the same semantics as the array_bool_xor as this constraint
266  // is actually a fully reified xor(a, b) <==> x.
267  const int a = LookupVar(fz_ct.arguments[0]);
268  const int b = LookupVar(fz_ct.arguments[1]);
269  const int x = LookupVar(fz_ct.arguments[2]);
270 
271  // not(x) => a == b
272  ct->add_enforcement_literal(NegatedRef(x));
273  auto* const refute = ct->mutable_linear();
274  refute->add_vars(a);
275  refute->add_coeffs(1);
276  refute->add_vars(b);
277  refute->add_coeffs(-1);
278  refute->add_domain(0);
279  refute->add_domain(0);
280 
281  // x => a + b == 1
282  auto* ct2 = proto.add_constraints();
283  ct2->add_enforcement_literal(x);
284  auto* const enforce = ct2->mutable_linear();
285  enforce->add_vars(a);
286  enforce->add_coeffs(1);
287  enforce->add_vars(b);
288  enforce->add_coeffs(1);
289  enforce->add_domain(1);
290  enforce->add_domain(1);
291  } else if (fz_ct.type == "array_bool_or") {
292  auto* arg = ct->mutable_bool_or();
293  for (const int var : LookupVars(fz_ct.arguments[0])) {
294  arg->add_literals(TrueLiteral(var));
295  }
296  } else if (fz_ct.type == "array_bool_or_negated") {
297  auto* arg = ct->mutable_bool_and();
298  for (const int var : LookupVars(fz_ct.arguments[0])) {
299  arg->add_literals(FalseLiteral(var));
300  }
301  } else if (fz_ct.type == "array_bool_and") {
302  auto* arg = ct->mutable_bool_and();
303  for (const int var : LookupVars(fz_ct.arguments[0])) {
304  arg->add_literals(TrueLiteral(var));
305  }
306  } else if (fz_ct.type == "array_bool_and_negated") {
307  auto* arg = ct->mutable_bool_or();
308  for (const int var : LookupVars(fz_ct.arguments[0])) {
309  arg->add_literals(FalseLiteral(var));
310  }
311  } else if (fz_ct.type == "array_bool_xor") {
312  auto* arg = ct->mutable_bool_xor();
313  for (const int var : LookupVars(fz_ct.arguments[0])) {
314  arg->add_literals(TrueLiteral(var));
315  }
316  } else if (fz_ct.type == "bool_le" || fz_ct.type == "int_le") {
317  FillAMinusBInDomain({kint64min, 0}, fz_ct, ct);
318  } else if (fz_ct.type == "bool_ge" || fz_ct.type == "int_ge") {
319  FillAMinusBInDomain({0, kint64max}, fz_ct, ct);
320  } else if (fz_ct.type == "bool_lt" || fz_ct.type == "int_lt") {
321  FillAMinusBInDomain({kint64min, -1}, fz_ct, ct);
322  } else if (fz_ct.type == "bool_gt" || fz_ct.type == "int_gt") {
323  FillAMinusBInDomain({1, kint64max}, fz_ct, ct);
324  } else if (fz_ct.type == "bool_eq" || fz_ct.type == "int_eq" ||
325  fz_ct.type == "bool2int") {
326  FillAMinusBInDomain({0, 0}, fz_ct, ct);
327  } else if (fz_ct.type == "bool_ne" || fz_ct.type == "bool_not") {
328  auto* arg = ct->mutable_linear();
329  arg->add_vars(LookupVar(fz_ct.arguments[0]));
330  arg->add_coeffs(1);
331  arg->add_vars(LookupVar(fz_ct.arguments[1]));
332  arg->add_coeffs(1);
333  arg->add_domain(1);
334  arg->add_domain(1);
335  } else if (fz_ct.type == "int_ne") {
336  FillAMinusBInDomain({kint64min, -1, 1, kint64max}, fz_ct, ct);
337  } else if (fz_ct.type == "int_lin_eq") {
338  const int64 rhs = fz_ct.arguments[2].values[0];
339  FillLinearConstraintWithGivenDomain({rhs, rhs}, fz_ct, ct);
340  } else if (fz_ct.type == "bool_lin_eq") {
341  auto* arg = ct->mutable_linear();
342  const std::vector<int> vars = LookupVars(fz_ct.arguments[1]);
343  for (int i = 0; i < vars.size(); ++i) {
344  arg->add_vars(vars[i]);
345  arg->add_coeffs(fz_ct.arguments[0].values[i]);
346  }
347  if (fz_ct.arguments[2].IsVariable()) {
348  arg->add_vars(LookupVar(fz_ct.arguments[2]));
349  arg->add_coeffs(-1);
350  arg->add_domain(0);
351  arg->add_domain(0);
352  } else {
353  const int64 v = fz_ct.arguments[2].Value();
354  arg->add_domain(v);
355  arg->add_domain(v);
356  }
357  } else if (fz_ct.type == "int_lin_le" || fz_ct.type == "bool_lin_le") {
358  const int64 rhs = fz_ct.arguments[2].values[0];
359  FillLinearConstraintWithGivenDomain({kint64min, rhs}, fz_ct, ct);
360  } else if (fz_ct.type == "int_lin_lt") {
361  const int64 rhs = fz_ct.arguments[2].values[0];
362  FillLinearConstraintWithGivenDomain({kint64min, rhs - 1}, fz_ct, ct);
363  } else if (fz_ct.type == "int_lin_ge") {
364  const int64 rhs = fz_ct.arguments[2].values[0];
365  FillLinearConstraintWithGivenDomain({rhs, kint64max}, fz_ct, ct);
366  } else if (fz_ct.type == "int_lin_gt") {
367  const int64 rhs = fz_ct.arguments[2].values[0];
368  FillLinearConstraintWithGivenDomain({rhs + 1, kint64max}, fz_ct, ct);
369  } else if (fz_ct.type == "int_lin_ne") {
370  const int64 rhs = fz_ct.arguments[2].values[0];
371  FillLinearConstraintWithGivenDomain(
372  {kint64min, rhs - 1, rhs + 1, kint64max}, fz_ct, ct);
373  } else if (fz_ct.type == "set_in") {
374  auto* arg = ct->mutable_linear();
375  arg->add_vars(LookupVar(fz_ct.arguments[0]));
376  arg->add_coeffs(1);
377  if (fz_ct.arguments[1].type == fz::Argument::INT_LIST) {
378  FillDomainInProto(Domain::FromValues(std::vector<int64>{
379  fz_ct.arguments[1].values.begin(),
380  fz_ct.arguments[1].values.end()}),
381  arg);
382  } else if (fz_ct.arguments[1].type == fz::Argument::INT_INTERVAL) {
383  FillDomainInProto(
384  Domain(fz_ct.arguments[1].values[0], fz_ct.arguments[1].values[1]),
385  arg);
386  } else {
387  LOG(FATAL) << "Wrong format";
388  }
389  } else if (fz_ct.type == "set_in_negated") {
390  auto* arg = ct->mutable_linear();
391  arg->add_vars(LookupVar(fz_ct.arguments[0]));
392  arg->add_coeffs(1);
393  if (fz_ct.arguments[1].type == fz::Argument::INT_LIST) {
394  FillDomainInProto(
395  Domain::FromValues(
396  std::vector<int64>{fz_ct.arguments[1].values.begin(),
397  fz_ct.arguments[1].values.end()})
398  .Complement(),
399  arg);
400  } else if (fz_ct.arguments[1].type == fz::Argument::INT_INTERVAL) {
401  FillDomainInProto(
402  Domain(fz_ct.arguments[1].values[0], fz_ct.arguments[1].values[1])
403  .Complement(),
404  arg);
405  } else {
406  LOG(FATAL) << "Wrong format";
407  }
408  } else if (fz_ct.type == "int_min") {
409  auto* arg = ct->mutable_int_min();
410  arg->set_target(LookupVar(fz_ct.arguments[2]));
411  arg->add_vars(LookupVar(fz_ct.arguments[0]));
412  arg->add_vars(LookupVar(fz_ct.arguments[1]));
413  } else if (fz_ct.type == "array_int_minimum" || fz_ct.type == "minimum_int") {
414  auto* arg = ct->mutable_int_min();
415  arg->set_target(LookupVar(fz_ct.arguments[0]));
416  for (const int var : LookupVars(fz_ct.arguments[1])) arg->add_vars(var);
417  } else if (fz_ct.type == "int_max") {
418  auto* arg = ct->mutable_int_max();
419  arg->set_target(LookupVar(fz_ct.arguments[2]));
420  arg->add_vars(LookupVar(fz_ct.arguments[0]));
421  arg->add_vars(LookupVar(fz_ct.arguments[1]));
422  } else if (fz_ct.type == "array_int_maximum" || fz_ct.type == "maximum_int") {
423  auto* arg = ct->mutable_int_max();
424  arg->set_target(LookupVar(fz_ct.arguments[0]));
425  for (const int var : LookupVars(fz_ct.arguments[1])) arg->add_vars(var);
426  } else if (fz_ct.type == "int_times") {
427  auto* arg = ct->mutable_int_prod();
428  arg->set_target(LookupVar(fz_ct.arguments[2]));
429  arg->add_vars(LookupVar(fz_ct.arguments[0]));
430  arg->add_vars(LookupVar(fz_ct.arguments[1]));
431  } else if (fz_ct.type == "int_abs") {
432  auto* arg = ct->mutable_int_max();
433  arg->set_target(LookupVar(fz_ct.arguments[1]));
434  arg->add_vars(LookupVar(fz_ct.arguments[0]));
435  arg->add_vars(-LookupVar(fz_ct.arguments[0]) - 1);
436  } else if (fz_ct.type == "int_plus") {
437  auto* arg = ct->mutable_linear();
438  FillDomainInProto(Domain(0, 0), arg);
439  arg->add_vars(LookupVar(fz_ct.arguments[0]));
440  arg->add_coeffs(1);
441  arg->add_vars(LookupVar(fz_ct.arguments[1]));
442  arg->add_coeffs(1);
443  arg->add_vars(LookupVar(fz_ct.arguments[2]));
444  arg->add_coeffs(-1);
445  } else if (fz_ct.type == "int_div") {
446  auto* arg = ct->mutable_int_div();
447  arg->add_vars(LookupVar(fz_ct.arguments[0]));
448  arg->add_vars(LookupVar(fz_ct.arguments[1]));
449  arg->set_target(LookupVar(fz_ct.arguments[2]));
450  } else if (fz_ct.type == "int_mod") {
451  auto* arg = ct->mutable_int_mod();
452  arg->set_target(LookupVar(fz_ct.arguments[2]));
453  arg->add_vars(LookupVar(fz_ct.arguments[0]));
454  arg->add_vars(LookupVar(fz_ct.arguments[1]));
455  } else if (fz_ct.type == "array_int_element" ||
456  fz_ct.type == "array_bool_element" ||
457  fz_ct.type == "array_var_int_element" ||
458  fz_ct.type == "array_var_bool_element" ||
459  fz_ct.type == "array_int_element_nonshifted") {
460  if (fz_ct.arguments[0].type == fz::Argument::INT_VAR_REF ||
461  fz_ct.arguments[0].type == fz::Argument::INT_VALUE) {
462  auto* arg = ct->mutable_element();
463  arg->set_index(LookupVar(fz_ct.arguments[0]));
464  arg->set_target(LookupVar(fz_ct.arguments[2]));
465 
466  if (!absl::EndsWith(fz_ct.type, "_nonshifted")) {
467  // Add a dummy variable at position zero because flatzinc index start
468  // at 1.
469  // TODO(user): Make sure that zero is not in the index domain...
470  arg->add_vars(arg->target());
471  }
472  for (const int var : LookupVars(fz_ct.arguments[1])) arg->add_vars(var);
473  } else {
474  // Special case added by the presolve or in flatzinc. We encode this
475  // as a table constraint.
476  CHECK(!absl::EndsWith(fz_ct.type, "_nonshifted"));
477  auto* arg = ct->mutable_table();
478 
479  // the constraint is:
480  // values[coeff1 * vars[0] + coeff2 * vars[1] + offset] == target.
481  for (const int var : LookupVars(fz_ct.arguments[0])) arg->add_vars(var);
482  arg->add_vars(LookupVar(fz_ct.arguments[2])); // the target
483 
484  const std::vector<int64>& values = fz_ct.arguments[1].values;
485  const int64 coeff1 = fz_ct.arguments[3].values[0];
486  const int64 coeff2 = fz_ct.arguments[3].values[1];
487  const int64 offset = fz_ct.arguments[4].values[0] - 1;
488 
489  for (const int64 a : AllValuesInDomain(proto.variables(arg->vars(0)))) {
490  for (const int64 b : AllValuesInDomain(proto.variables(arg->vars(1)))) {
491  const int index = coeff1 * a + coeff2 * b + offset;
492  CHECK_GE(index, 0);
493  CHECK_LT(index, values.size());
494  arg->add_values(a);
495  arg->add_values(b);
496  arg->add_values(values[index]);
497  }
498  }
499  }
500  } else if (fz_ct.type == "ortools_table_int") {
501  auto* arg = ct->mutable_table();
502  for (const int var : LookupVars(fz_ct.arguments[0])) arg->add_vars(var);
503  for (const int64 value : fz_ct.arguments[1].values) arg->add_values(value);
504  } else if (fz_ct.type == "ortools_regular") {
505  auto* arg = ct->mutable_automaton();
506  for (const int var : LookupVars(fz_ct.arguments[0])) arg->add_vars(var);
507 
508  int count = 0;
509  const int num_states = fz_ct.arguments[1].Value();
510  const int num_values = fz_ct.arguments[2].Value();
511  for (int i = 1; i <= num_states; ++i) {
512  for (int j = 1; j <= num_values; ++j) {
513  CHECK_LT(count, fz_ct.arguments[3].values.size());
514  const int next = fz_ct.arguments[3].values[count++];
515  if (next == 0) continue; // 0 is a failing state.
516  arg->add_transition_tail(i);
517  arg->add_transition_label(j);
518  arg->add_transition_head(next);
519  }
520  }
521 
522  arg->set_starting_state(fz_ct.arguments[4].Value());
523  switch (fz_ct.arguments[5].type) {
524  case fz::Argument::INT_VALUE: {
525  arg->add_final_states(fz_ct.arguments[5].values[0]);
526  break;
527  }
528  case fz::Argument::INT_INTERVAL: {
529  for (int v = fz_ct.arguments[5].values[0];
530  v <= fz_ct.arguments[5].values[1]; ++v) {
531  arg->add_final_states(v);
532  }
533  break;
534  }
535  case fz::Argument::INT_LIST: {
536  for (const int v : fz_ct.arguments[5].values) {
537  arg->add_final_states(v);
538  }
539  break;
540  }
541  default: {
542  LOG(FATAL) << "Wrong constraint " << fz_ct.DebugString();
543  }
544  }
545  } else if (fz_ct.type == "fzn_all_different_int") {
546  auto* arg = ct->mutable_all_diff();
547  for (const int var : LookupVars(fz_ct.arguments[0])) arg->add_vars(var);
548  } else if (fz_ct.type == "fzn_circuit" || fz_ct.type == "fzn_subcircuit") {
549  // Try to auto-detect if it is zero or one based.
550  bool found_zero = false;
551  bool found_size = false;
552  const int size = fz_ct.arguments[0].variables.size();
553  for (fz::IntegerVariable* const var : fz_ct.arguments[0].variables) {
554  if (var->domain.Min() == 0) found_zero = true;
555  if (var->domain.Max() == size) found_size = true;
556  }
557  const bool is_one_based = !found_zero || found_size;
558  const int min_index = is_one_based ? 1 : 0;
559  const int max_index = min_index + fz_ct.arguments[0].variables.size() - 1;
560 
561  // The arc-based mutable circuit.
562  auto* circuit_arg = ct->mutable_circuit();
563 
564  // We fully encode all variables so we can use the literal based circuit.
565  // TODO(user): avoid fully encoding more than once?
566  int index = min_index;
567  const bool is_circuit = (fz_ct.type == "fzn_circuit");
568  for (const int var : LookupVars(fz_ct.arguments[0])) {
569  Domain domain = ReadDomainFromProto(proto.variables(var));
570 
571  // Restrict the domain of var to [min_index, max_index]
572  domain = domain.IntersectionWith(Domain(min_index, max_index));
573  if (is_circuit) {
574  // We simply make sure that the variable cannot take the value index.
575  domain = domain.IntersectionWith(Domain::FromIntervals(
576  {{kint64min, index - 1}, {index + 1, kint64max}}));
577  }
578  FillDomainInProto(domain, proto.mutable_variables(var));
579 
580  for (const ClosedInterval interval : domain.intervals()) {
581  for (int64 value = interval.start; value <= interval.end; ++value) {
582  // Create one Boolean variable for this arc.
583  const int literal = proto.variables_size();
584  {
585  auto* new_var = proto.add_variables();
586  new_var->add_domain(0);
587  new_var->add_domain(1);
588  }
589 
590  // Add the arc.
591  circuit_arg->add_tails(index);
592  circuit_arg->add_heads(value);
593  circuit_arg->add_literals(literal);
594 
595  // literal => var == value.
596  {
597  auto* ct = proto.add_constraints();
598  ct->add_enforcement_literal(literal);
599  ct->mutable_linear()->add_coeffs(1);
600  ct->mutable_linear()->add_vars(var);
601  ct->mutable_linear()->add_domain(value);
602  ct->mutable_linear()->add_domain(value);
603  }
604 
605  // not(literal) => var != value
606  {
607  auto* ct = proto.add_constraints();
608  ct->add_enforcement_literal(NegatedRef(literal));
609  ct->mutable_linear()->add_coeffs(1);
610  ct->mutable_linear()->add_vars(var);
611  ct->mutable_linear()->add_domain(kint64min);
612  ct->mutable_linear()->add_domain(value - 1);
613  ct->mutable_linear()->add_domain(value + 1);
614  ct->mutable_linear()->add_domain(kint64max);
615  }
616  }
617  }
618 
619  ++index;
620  }
621  } else if (fz_ct.type == "fzn_inverse") {
622  auto* arg = ct->mutable_inverse();
623 
624  const auto direct_variables = LookupVars(fz_ct.arguments[0]);
625  const auto inverse_variables = LookupVars(fz_ct.arguments[1]);
626 
627  const int num_variables =
628  std::min(direct_variables.size(), inverse_variables.size());
629 
630  // Try to auto-detect if it is zero or one based.
631  bool found_zero = false;
632  bool found_size = false;
633  for (fz::IntegerVariable* const var : fz_ct.arguments[0].variables) {
634  if (var->domain.Min() == 0) found_zero = true;
635  if (var->domain.Max() == num_variables) found_size = true;
636  }
637  for (fz::IntegerVariable* const var : fz_ct.arguments[1].variables) {
638  if (var->domain.Min() == 0) found_zero = true;
639  if (var->domain.Max() == num_variables) found_size = true;
640  }
641 
642  // Add a dummy constant variable at zero if the indexing is one based.
643  const bool is_one_based = !found_zero || found_size;
644  const int offset = is_one_based ? 1 : 0;
645 
646  if (is_one_based) arg->add_f_direct(LookupConstant(0));
647  for (const int var : direct_variables) {
648  arg->add_f_direct(var);
649  // Intersect domains with offset + [0, num_variables).
650  FillDomainInProto(
651  ReadDomainFromProto(proto.variables(var))
652  .IntersectionWith(Domain(offset, num_variables - 1 + offset)),
653  proto.mutable_variables(var));
654  }
655 
656  if (is_one_based) arg->add_f_inverse(LookupConstant(0));
657  for (const int var : inverse_variables) {
658  arg->add_f_inverse(var);
659  // Intersect domains with offset + [0, num_variables).
660  FillDomainInProto(
661  ReadDomainFromProto(proto.variables(var))
662  .IntersectionWith(Domain(offset, num_variables - 1 + offset)),
663  proto.mutable_variables(var));
664  }
665  } else if (fz_ct.type == "fzn_cumulative") {
666  const std::vector<int> starts = LookupVars(fz_ct.arguments[0]);
667  const std::vector<int> durations = LookupVars(fz_ct.arguments[1]);
668  const std::vector<int> demands = LookupVars(fz_ct.arguments[2]);
669  const int capacity = LookupVar(fz_ct.arguments[3]);
670 
671  auto* arg = ct->mutable_cumulative();
672  arg->set_capacity(capacity);
673  for (int i = 0; i < starts.size(); ++i) {
674  // Special case for a 0-1 demand, we mark the interval as optional
675  // instead and fix the demand to 1.
676  if (proto.variables(demands[i]).domain().size() == 2 &&
677  proto.variables(demands[i]).domain(0) == 0 &&
678  proto.variables(demands[i]).domain(1) == 1 &&
679  proto.variables(capacity).domain(1) == 1) {
680  arg->add_intervals(
681  GetOrCreateOptionalInterval(starts[i], durations[i], demands[i]));
682  arg->add_demands(LookupConstant(1));
683  } else {
684  arg->add_intervals(GetOrCreateInterval(starts[i], durations[i]));
685  arg->add_demands(demands[i]);
686  }
687  }
688  } else if (fz_ct.type == "fzn_diffn" || fz_ct.type == "fzn_diffn_nonstrict") {
689  const std::vector<int> x = LookupVars(fz_ct.arguments[0]);
690  const std::vector<int> y = LookupVars(fz_ct.arguments[1]);
691  const std::vector<int> dx = LookupVars(fz_ct.arguments[2]);
692  const std::vector<int> dy = LookupVars(fz_ct.arguments[3]);
693  const std::vector<int> x_intervals = CreateIntervals(x, dx);
694  const std::vector<int> y_intervals = CreateIntervals(y, dy);
695  auto* arg = ct->mutable_no_overlap_2d();
696  for (int i = 0; i < x.size(); ++i) {
697  arg->add_x_intervals(x_intervals[i]);
698  arg->add_y_intervals(y_intervals[i]);
699  }
700  arg->set_boxes_with_null_area_can_overlap(fz_ct.type ==
701  "fzn_diffn_nonstrict");
702  } else if (fz_ct.type == "ortools_network_flow" ||
703  fz_ct.type == "ortools_network_flow_cost") {
704  // Note that we leave ct empty here (with just the name set).
705  // We simply do a linear encoding of this constraint.
706  const bool has_cost = fz_ct.type == "ortools_network_flow_cost";
707  const std::vector<int> flow = LookupVars(fz_ct.arguments[has_cost ? 3 : 2]);
708 
709  // Flow conservation constraints.
710  const int num_nodes = fz_ct.arguments[1].values.size();
711  std::vector<std::vector<int>> flows_per_node(num_nodes);
712  std::vector<std::vector<int>> coeffs_per_node(num_nodes);
713  const int num_arcs = fz_ct.arguments[0].values.size() / 2;
714  for (int arc = 0; arc < num_arcs; arc++) {
715  const int tail = fz_ct.arguments[0].values[2 * arc] - 1;
716  const int head = fz_ct.arguments[0].values[2 * arc + 1] - 1;
717  if (tail == head) continue;
718 
719  flows_per_node[tail].push_back(flow[arc]);
720  coeffs_per_node[tail].push_back(1);
721  flows_per_node[head].push_back(flow[arc]);
722  coeffs_per_node[head].push_back(-1);
723  }
724  for (int node = 0; node < num_nodes; node++) {
725  auto* arg = proto.add_constraints()->mutable_linear();
726  arg->add_domain(fz_ct.arguments[1].values[node]);
727  arg->add_domain(fz_ct.arguments[1].values[node]);
728  for (int i = 0; i < flows_per_node[node].size(); ++i) {
729  arg->add_vars(flows_per_node[node][i]);
730  arg->add_coeffs(coeffs_per_node[node][i]);
731  }
732  }
733 
734  if (has_cost) {
735  auto* arg = proto.add_constraints()->mutable_linear();
736  arg->add_domain(0);
737  arg->add_domain(0);
738  for (int arc = 0; arc < num_arcs; arc++) {
739  const int64 weight = fz_ct.arguments[2].values[arc];
740  if (weight != 0) {
741  arg->add_vars(flow[arc]);
742  arg->add_coeffs(weight);
743  }
744  }
745  arg->add_vars(LookupVar(fz_ct.arguments[4]));
746  arg->add_coeffs(-1);
747  }
748  } else {
749  LOG(FATAL) << " Not supported " << fz_ct.type;
750  }

753 void CpModelProtoWithMapping::FillReifOrImpliedConstraint(
754  const fz::Constraint& fz_ct, ConstraintProto* ct) {
755  // Start by adding a non-reified version of the same constraint.
756  std::string simplified_type;
757  if (absl::EndsWith(fz_ct.type, "_reif")) {
758  // Remove _reif.
759  simplified_type = fz_ct.type.substr(0, fz_ct.type.size() - 5);
760  } else if (absl::EndsWith(fz_ct.type, "_imp")) {
761  // Remove _imp.
762  simplified_type = fz_ct.type.substr(0, fz_ct.type.size() - 4);
763  } else {
764  // Keep name as it is an implicit reified constraint.
765  simplified_type = fz_ct.type;
766  }
767 
768  // We need a copy to be able to change the type of the constraint.
769  fz::Constraint copy = fz_ct;
770  copy.type = simplified_type;
771 
772  // Create the CP-SAT constraint.
773  FillConstraint(copy, ct);
774 
775  // In case of reified constraints, the type of the opposite constraint.
776  std::string negated_type;
777 
778  // Fill enforcement_literal and set copy.type to the negated constraint.
779  if (simplified_type == "array_bool_or") {
780  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[1])));
781  negated_type = "array_bool_or_negated";
782  } else if (simplified_type == "array_bool_and") {
783  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[1])));
784  negated_type = "array_bool_and_negated";
785  } else if (simplified_type == "set_in") {
786  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
787  negated_type = "set_in_negated";
788  } else if (simplified_type == "bool_eq" || simplified_type == "int_eq") {
789  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
790  negated_type = "int_ne";
791  } else if (simplified_type == "bool_ne" || simplified_type == "int_ne") {
792  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
793  negated_type = "int_eq";
794  } else if (simplified_type == "bool_le" || simplified_type == "int_le") {
795  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
796  negated_type = "int_gt";
797  } else if (simplified_type == "bool_lt" || simplified_type == "int_lt") {
798  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
799  negated_type = "int_ge";
800  } else if (simplified_type == "bool_ge" || simplified_type == "int_ge") {
801  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
802  negated_type = "int_lt";
803  } else if (simplified_type == "bool_gt" || simplified_type == "int_gt") {
804  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
805  negated_type = "int_le";
806  } else if (simplified_type == "int_lin_eq") {
807  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
808  negated_type = "int_lin_ne";
809  } else if (simplified_type == "int_lin_ne") {
810  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
811  negated_type = "int_lin_eq";
812  } else if (simplified_type == "int_lin_le") {
813  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
814  negated_type = "int_lin_gt";
815  } else if (simplified_type == "int_lin_ge") {
816  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
817  negated_type = "int_lin_lt";
818  } else if (simplified_type == "int_lin_lt") {
819  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
820  negated_type = "int_lin_ge";
821  } else if (simplified_type == "int_lin_gt") {
822  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
823  negated_type = "int_lin_le";
824  } else {
825  LOG(FATAL) << "Unsupported " << simplified_type;
826  }
827 
828  // One way implication. We can stop here.
829  if (absl::EndsWith(fz_ct.type, "_imp")) return;
830 
831  // Add the other side of the reification because CpModelProto only support
832  // half reification.
833  ConstraintProto* negated_ct = proto.add_constraints();
834  negated_ct->set_name(fz_ct.type + " (negated)");
835  negated_ct->add_enforcement_literal(
836  sat::NegatedRef(ct->enforcement_literal(0)));
837  copy.type = negated_type;
838  FillConstraint(copy, negated_ct);
839 }

905  const fz::SolutionOutputSpecs& output,
906  const std::function<int64(fz::IntegerVariable*)>& value_func) {
907  if (output.variable != nullptr) {

931  result.append(value ? "true" : "false");

933  absl::StrAppend(&result, value);

938  }

941  }

982 void SolveFzWithCpModelProto(const fz::Model& fz_model,
983  const fz::FlatzincSatParameters& p,
984  const std::string& sat_params) {
985  if (!absl::GetFlag(FLAGS_use_flatzinc_format)) {
986  LOG(INFO) << "*** Starting translation to CP-SAT";
987  } else if (p.verbose_logging) {
988  FZLOG << "*** Starting translation to CP-SAT" << FZENDL;
989  }
990 
991  CpModelProtoWithMapping m;
992  m.proto.set_name(fz_model.name());
993 
994  // The translation is easy, we create one variable per flatzinc variable,
995  // plus eventually a bunch of constant variables that will be created
996  // lazily.
997  int num_variables = 0;
