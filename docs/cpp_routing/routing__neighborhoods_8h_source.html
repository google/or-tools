608 template <bool swap_first>
609 int64 PairNodeSwapActiveOperator<swap_first>::GetBaseNodeRestartPosition(
610  int base_index) {
611  // Base node 1 must be after base node 0 if they are both on the same path.
612  if (base_index == 0 || StartNode(base_index) != StartNode(base_index - 1)) {
613  return StartNode(base_index);

615  return BaseNode(base_index - 1);

630 template <bool swap_first>
631 bool PairNodeSwapActiveOperator<swap_first>::MakeNextNeighbor(
632  Assignment* delta, Assignment* deltadelta) {
633  while (inactive_pair_ < pairs_.size()) {
634  if (!IsInactive(pairs_[inactive_pair_].first[0]) ||
635  !IsInactive(pairs_[inactive_pair_].second[0]) ||
636  !PathOperator::MakeNextNeighbor(delta, deltadelta)) {
637  ResetPosition();
638  ++inactive_pair_;
639  } else {
640  return true;
641  }
642  }
643  return false;
644 }

646 template <bool swap_first>
647 bool PairNodeSwapActiveOperator<swap_first>::MakeNeighbor() {
648  const int64 base = BaseNode(0);
649  if (IsPathEnd(base)) {
650  return false;
651  }
652  const int64 pair_first = pairs_[inactive_pair_].first[0];
653  const int64 pair_second = pairs_[inactive_pair_].second[0];
654  if (swap_first) {
655  return MakeActive(pair_second, BaseNode(1)) &&
656  MakeActive(pair_first, base) &&
657  MakeChainInactive(pair_first, Next(pair_first));
658  } else {
659  return MakeActive(pair_second, BaseNode(1)) &&
660  MakeActive(pair_first, base) &&
661  MakeChainInactive(pair_second, Next(pair_second));
662  }
663 }
