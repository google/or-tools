67  // [START solution_printer]
69  static void printSolution(
70  DataModel data, RoutingModel routing, RoutingIndexManager manager, Assignment solution) {
71  // Inspect solution.
72  long maxRouteDistance = 0;
73  for (int i = 0; i < data.vehicleNumber; ++i) {
74  long index = routing.start(i);
75  logger.info("Route for Vehicle " + i + ":");
76  long routeDistance = 0;
77  String route = "";
78  while (!routing.isEnd(index)) {
79  route += manager.indexToNode(index) + " -> ";
80  long previousIndex = index;
81  index = solution.value(routing.nextVar(index));
82  routeDistance += routing.getArcCostForVehicle(previousIndex, index, i);
83  }
84  logger.info(route + manager.indexToNode(index));
85  logger.info("Distance of the route: " + routeDistance + "m");
86  maxRouteDistance = Math.max(routeDistance, maxRouteDistance);
87  }
88  logger.info("Maximum of the route distances: " + maxRouteDistance + "m");
89  }

116  return data.distanceMatrix[fromNode][toNode];
