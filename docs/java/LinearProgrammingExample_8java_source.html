43  c0.setCoefficient(x, 1);
44  c0.setCoefficient(y, 2);

48  c1.setCoefficient(x, 3);

50

52  MPConstraint c2 = solver.makeConstraint(-infinity, 2.0, "c2");
53  c2.setCoefficient(x, 1);

57
