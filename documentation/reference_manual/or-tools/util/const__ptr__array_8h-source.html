00032 // This class is used to store an const array of T*.
00033 //
00034 // This is useful inside constraints and expressions. The constructors,
00035 // except the one with the pointer to a vector of cells will copy the
00036 // data internally and will not take ownership of the data passed in
00037 // argument.
00038 // Its goals are:
00039 // - to unify the construction code across the optimization libraries.
00040 // - to provide one code to modify these mappings and apply
00041 //    transformations like sorting.
00042 // It requires T to implement a method string T::DebugString() const;
00043 // This is linked to @ref ConstIntArray.
00044 template <class T> class ConstPtrArray {
00045  public:
00046   // Build from one vector. Copy the data internally.
00047   explicit ConstPtrArray(const std::vector<T*>& ptrs)
00048       : data_(new std::vector<T*>(ptrs)) {}
00049 
00050   // Build from one data vector. Takes ownership of the vector.
00051   explicit ConstPtrArray(std::vector<T*>* const data) : data_(data) {}
00052 
00053   // This code releases the ownership of the data into the returned vector.
00054   // After this method is called, data_ points to a null vector.
00055   std::vector<T*>* Release() {
00056     return data_.release();
00057   }
00058 
00059   // Size of the array. This is not valid after Release() has been called.
00060   int size() const {
00061     if (data_.get() == NULL) {
00062       return 0;
00063     }
00064     return data_->size();
00065   }
00066 
00067   // Checks for deep equality with other array.
00068   bool Equals(const ConstPtrArray<T>& other) const {
00069     const int current_size = size();
00070     if (current_size != other.size()) {
00071       return false;
00072     }
00073     for (int i = 0; i < current_size; ++i) {
00074       if (get(i) != other.get(i)) {
00075         return false;
00076       }
00077     }
00078     return true;
00079   }
00080 
00081   // Returns the instance of T* at position index. This is not valid
00082   // after Release() has been called.  @see operator[]().
00083   T* operator[](int64 index) const {
00084     CHECK_NOTNULL(data_.get());
00085     return (*data_)[index];
00086   }
00087 
00088   // Returns the instance of T* at position index. This is not valid
00089   // after Release() has been called. @see operator[]().
00090   T* get(int64 index) const {
00091     CHECK_NOTNULL(data_.get());
00092     return (*data_)[index];
00093   }
00094 
00095   // Returns a copy of the data. Usually used to create a new ConstPtrArray.
00096   std::vector<T*>* Copy() const {
00097     CHECK_NOTNULL(data_.get());
00098     return new std::vector<T*>(*data_);
00099   }
00100 
00101   // Pretty print.
00102   string DebugString() const {
00103     if (data_.get() == NULL) {
00104       return "Released ConstPtrArray";
00105     }
00106     return StringPrintf("[%s]", DebugStringArray(data_->data(),
00107                                                  data_->size(), ", ").c_str());
00108   }
00109 
00110  private:
00111   scoped_ptr<std::vector<T*> > data_;
00112 };
