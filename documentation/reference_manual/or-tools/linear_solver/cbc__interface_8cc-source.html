00280   timer.Start();
00281 
00282   // Note that CBC does not provide any incrementality.
00283   if (param.GetIntegerParam(MPSolverParameters::INCREMENTALITY) ==
00284       MPSolverParameters::INCREMENTALITY_OFF) {
00285     Reset();
00286   }

00288   // Special case if the model is empty since CBC is not able to
00289   // handle this special case by itself.

00291     sync_status_ = SOLUTION_SYNCHRONIZED;
00292     result_status_ = MPSolver::OPTIMAL;

00295     return result_status_;

00298   // Finish preparing the problem.
00299   // Define variables.
00300   switch (sync_status_) {
00301     case MUST_RELOAD: {
00302       Reset();

00304       // Create dummy variable for objective offset.
00305       build.addColumn(0, NULL, NULL, 1.0, 1.0,
00306                       solver_->Objective().offset(), "dummy", false);
00307       const int nb_vars = solver_->variables_.size();
00308       for (int i = 0; i < nb_vars; ++i) {
00309         MPVariable* const var = solver_->variables_[i];
00310         var->set_index(i + 1);  // offset by 1 because of dummy variable.
00311         const double obj_coeff = solver_->Objective().GetCoefficient(var);
00312         if (var->name().empty()) {
00313           build.addColumn(0, NULL, NULL, var->lb(), var->ub(), obj_coeff,
00314                           NULL, var->integer());
00315         } else {
00316           build.addColumn(0, NULL, NULL, var->lb(), var->ub(), obj_coeff,
00317                           var->name().c_str(), var->integer());
00318         }
00319       }
00320 
00321       // Define constraints.
00322       int max_row_length = 0;
00323       int constraint_index = 0;
00324       for (int i = 0; i < solver_->constraints_.size(); ++i) {
00325         MPConstraint* const ct = solver_->constraints_[i];
00326         ct->set_index(constraint_index++);
00327         if (ct->coefficients_.size() > max_row_length) {
00328           max_row_length = ct->coefficients_.size();
00329         }
00330       }
00331       scoped_array<int> indices(new int[max_row_length]);
00332       scoped_array<double> coefs(new double[max_row_length]);
00333 
00334       for (int i = 0; i < solver_->constraints_.size(); ++i) {
00335         MPConstraint* const  ct = solver_->constraints_[i];
00336         const int size = ct->coefficients_.size();
00337         int j = 0;
00338         for (hash_map<const MPVariable*, double>::const_iterator it =
00339                  ct->coefficients_.begin();
00340              it != ct->coefficients_.end();
00341              ++it) {
00342           const int index = it->first->index();
00343           DCHECK_NE(kNoIndex, index);
00344           indices[j] = index;
00345           coefs[j] = it->second;
00346           j++;
00347         }
00348         if (ct->name().empty()) {
00349           build.addRow(size, indices.get(), coefs.get(), ct->lb(), ct->ub());
00350         } else {
00351           build.addRow(size, indices.get(), coefs.get(), ct->lb(), ct->ub(),
00352                        ct->name().c_str());
00353         }
00354       }
00355       osi_.loadFromCoinModel(build);
00356       break;
00357     }
00358     case MODEL_SYNCHRONIZED: {
00359       break;
00360     }
00361     case SOLUTION_SYNCHRONIZED: {
00362       break;
00363     }
00364   }

00366   // Changing optimization direction through OSI so that the model file

00390     message_handler.setLogLevel(1, 0);  // Clp messages
