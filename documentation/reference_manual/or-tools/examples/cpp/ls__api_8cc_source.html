00051       : IntVarLocalSearchOperator(variables.data(), variables.size()),

00054 
00055   virtual ~MoveOneVar() {}
00056 
00057  protected:
00058   // Make a neighbor assigning one variable to its target value.
00059   virtual bool MakeOneNeighbor() {
00060     const int64 current_value = OldValue(variable_index_);
00061     if (move_up_) {
00062       SetValue(variable_index_, current_value  + 1);
00063       variable_index_ = (variable_index_ + 1) % Size();
00064     } else {
00065       SetValue(variable_index_, current_value  - 1);
00066     }
00067     move_up_ = !move_up_;
00068     return true;
00069   }
00070 
00071  private:
00072   virtual void OnStart() {
00073     CHECK_GE(variable_index_, 0);
00074     CHECK_LT(variable_index_, Size());
00075   }
00076 
00077   // Index of the next variable to try to restore
00078   int64 variable_index_;
00079   // Direction of the modification.
00080   bool move_up_;

00083 class SumFilter : public IntVarLocalSearchFilter {
00084  public:
00085   SumFilter(const IntVar* const* vars, int size) :
00086       IntVarLocalSearchFilter(vars, size), sum_(0) {}
00087 
00088   ~SumFilter() {}
00089 
00090   virtual void OnSynchronize() {
00091     sum_ = 0;
00092     for (int index = 0; index < Size(); ++index) {
00093       sum_ += Value(index);
00094     }
00095   }
00096 
00097   virtual bool Accept(const Assignment* delta,
00098                       const Assignment* unused_deltadelta) {
00099     const Assignment::IntContainer& solution_delta = delta->IntVarContainer();
00100     const int solution_delta_size = solution_delta.Size();
00101 
00102     // The input const Assignment* delta given to Accept() may
00103     // actually contain "Deactivated" elements, which represent
00104     // variables that have been freed -- they are not bound to a
00105     // single value anymore. This happens with LNS-type (Large
00106     // Neighborhood Search) LocalSearchOperator, which are not used in
00107     // this example as of 2012-01; and we refer the reader to
00108     // ./routing.cc for an example of such LNS-type operators.
00109     //
00110     // For didactical purposes, we will assume for a moment that a
00111     // LNS-type operator might be applied. The Filter will still be
00112     // called, but our filter here won't be able to work, since
00113     // it needs every variable to be bound (i.e. have a fixed value),
00114     // in the assignment that it considers. Therefore, we include here
00115     // a snippet of code that will detect if the input assignment is
00116     // not fully bound. For further details, read ./routing.cc -- but
00117     // we strongly advise the reader to first try and understand all
00118     // of this file.
00119     for (int i = 0; i < solution_delta_size; ++i) {
00120       if (!solution_delta.Element(i).Activated()) {
00121         VLOG(1)
00122             << "Element #" << i << " of the delta assignment given to"
00123             << " SumFilter::Accept() is not activated (i.e. its variable"
00124             << " is not bound to a single value anymore). This means that"
00125             << " we are in a LNS phase, and the DobbleFilter won't be able"
00126             << " to filter anything. Returning true.";
00127         return true;
00128       }
00129     }
00130     int64 new_sum = sum_;
00131     VLOG(1) << "No LNS, size = " << solution_delta_size;
00132     for (int index = 0; index < solution_delta_size; ++index) {
00133       int64 touched_var = -1;
00134       FindIndex(solution_delta.Element(index).Var(), &touched_var);
00135       const int64 old_value = Value(touched_var);
00136       const int64 new_value = solution_delta.Element(index).Value();
00137       new_sum += new_value - old_value;
00138     }
00139     VLOG(1) << "new sum = " << new_sum << ", old sum = " << sum_;
00140     return new_sum < sum_;
00141   }

00143  private:
00144   int64 sum_;
00145 };

00147 void BasicLns() {
00148   LOG(INFO) << "Basic LNS";
00149   Solver s("Sample");
00150   std::vector<IntVar*> vars;
00151   s.MakeIntVarArray(4, 0, 4, &vars);
00152   IntVar* const sum_var = s.MakeSum(vars)->Var();
00153   OptimizeVar* const obj = s.MakeMinimize(sum_var, 1);
00154   DecisionBuilder* const db =
00155       s.MakePhase(vars,
00156                   Solver::CHOOSE_FIRST_UNBOUND,
00157                   Solver::ASSIGN_MAX_VALUE);
00158   OneVarLns one_var_lns(vars.data(), vars.size());
00159   LocalSearchPhaseParameters* const ls_params =
00160       s.MakeLocalSearchPhaseParameters(&one_var_lns, db);
00161   DecisionBuilder* const ls = s.MakeLocalSearchPhase(vars, db, ls_params);
00162   SolutionCollector* const collector = s.MakeLastSolutionCollector();
00163   collector->Add(vars);
00164   collector->AddObjective(sum_var);
00165   SearchMonitor* const log = s.MakeSearchLog(1000, obj);
00166   s.Solve(ls, collector, obj, log);
00167   LOG(INFO) << "Objective value = " << collector->objective_value(0);
00168 }
