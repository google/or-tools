00014 // Costas Array Problem
00015 //
00016 // Finds an NxN matrix of 0s and 1s, with only one 1 per row,
00017 // and one 1 per column, such that all displacement vectors
00018 // between each pair of 1s are distinct.
00019 //
00020 // This example contains two separate implementations. CostasHard()
00021 // uses hard constraints, whereas CostasSoft() uses a minimizer to
00022 // minimize the number of duplicates.

00047 // Checks that all pairwise distances are unique and returns all violators
00048 void CheckConstraintViolators(const std::vector<int64>& vars,
00049                               std::vector<int>* const violators) {
00050   int dim = vars.size();
00051 
00052   // Check that all indices are unique
00053   for (int i = 0; i < dim; ++i) {
00054     for (int k = i + 1; k < dim; ++k) {
00055       if (vars[i] == vars[k]) {
00056         violators->push_back(i);
00057         violators->push_back(k);
00058       }
00059     }
00060   }
00061 
00062   // Check that all differences are unique for each level
00063   for (int level = 1; level < dim; ++level) {
00064     for (int i = 0; i < dim - level; ++i) {
00065       const int difference = vars[i + level] - vars[i];
00066 
00067       for (int k = i + 1; k < dim - level; ++k) {
00068         if (difference == vars[k + level] - vars[k]) {
00069           violators->push_back(k + level);
00070           violators->push_back(k);
00071           violators->push_back(i + level);
00072           violators->push_back(i);
00073         }
00074       }
00075     }
00076   }
00077 }

00079 // Check that all pairwise differences are unique
00080 bool CheckCostas(const std::vector<int64>& vars) {
00081   std::vector<int> violators;
00082 
00083   CheckConstraintViolators(vars, &violators);
00084 
00085   return violators.empty();
00086 }

00088 // Cycles all possible permutations
00089 class OrderedLNS: public BaseLNS {
00090  public:
00091   OrderedLNS(const std::vector<IntVar*>& vars, int free_elements) :
00092     BaseLNS(vars.data(), vars.size()),
00093     free_elements_(free_elements) {
00094     index_ = 0;
00095 
00096     // Start of with the first free_elements_ as a permutations, eg. 0,1,2,3,...
00097     for (int i = 0; i < free_elements; ++i) {
00098       index_ += i * pow(static_cast<double>(vars.size()),
00099                         free_elements - i - 1);
00100     }
00101   }
00102 
00103   virtual bool NextFragment(std::vector<int>* const fragment) {
00104     int dim = Size();
00105     std::set<int> fragment_set;
00106 
00107     do {
00108       int work_index = index_;
00109 
00110       fragment->clear();
00111       fragment_set.clear();
00112 
00113       for (int i = 0; i < free_elements_; ++i) {
00114         int current_index = work_index % dim;
00115         work_index = work_index / dim;
00116 
00117         std::pair<std::set<int>::iterator, bool> ret =
00118             fragment_set.insert(current_index);
00119 
00120         // Check if element has been used before
00121         if (ret.second) {
00122           fragment->push_back(current_index);
00123         } else {
00124           break;
00125         }
00126       }
00127 
00128       // Go to next possible permutation
00129       ++index_;
00130     // Try again if a duplicate index is used
00131     } while (fragment_set.size() < free_elements_);
00132 
00133     return true;
00134   }
00135 
00136  private:
00137   int index_;
00138   const int free_elements_;
00139 };

00141 // RandomLNS is used for the local search and frees the
00142 // number of elements specified in 'free_elements' randomly.
00143 class RandomLNS: public BaseLNS {
00144  public:
00145   RandomLNS(const std::vector<IntVar*>& vars, int free_elements) :
00146     BaseLNS(vars.data(), vars.size()),
00147     free_elements_(free_elements),
00148     rand_(ACMRandom::HostnamePidTimeSeed()) {
00149   }
00150 
00151   virtual bool NextFragment(std::vector<int>* const fragment) {
00152     std::vector<int> weighted_elements;
00153     std::vector<int64> values;
00154 
00155     // Create weighted vector for randomizer. Add one of each possible elements
00156     // to the weighted vector and then add more elements depending on the
00157     // number of conflicts
00158     for (int i = 0; i < Size(); ++i) {
00159       values.push_back(Value(i));
00160 
00161       // Insert each variable at least once.
00162       weighted_elements.push_back(i);
00163     }
00164 
00165     CheckConstraintViolators(values, &weighted_elements);
00166     int size = weighted_elements.size();
00167 
00168     // Randomly insert elements in vector until no more options remain
00169     while (fragment->size() < std::min(free_elements_, size)) {
00170       const int index = weighted_elements[rand_.Next() % size];
00171       fragment->push_back(index);
00172 
00173       // Remove all elements with this index from weighted_elements
00174       for (std::vector<int>::iterator pos = weighted_elements.begin();
00175            pos != weighted_elements.end(); ) {
00176         if (*pos == index) {
00177           // Try to erase as many elements as possible at the same time
00178           std::vector<int>::iterator end = pos;
00179 
00180           while ((end + 1) != weighted_elements.end() && *(end + 1) == *pos) {
00181             ++end;
00182           }
00183 
00184           pos = weighted_elements.erase(pos, end);
00185         } else {
00186           ++pos;
00187         }
00188       }
00189 
00190       size = weighted_elements.size();
00191     }
00192 
00193     return true;
00194   }
00195 
00196  private:
00197   const int free_elements_;
00198   ACMRandom rand_;
00199 };

00227 // Computes a Costas Array using soft constraints.
00228 // Instead of enforcing that all distance vectors are unique, we
00229 // minimize the number of duplicate distance vectors.
00230 void CostasSoft(const int dim) {
00231   Solver solver("Costas");
00232 
00233   // For the matrix and for the count of occurrences of each possible distance
00234   // for each stage
00235   const int num_elements = dim + (2 * dim + 1) * (dim);
00236 
00237   // create the variables
00238   std::vector<IntVar*> vars;
00239   solver.MakeIntVarArray(num_elements, -dim, dim, "var_", &vars);
00240 
00241   // the costas matrix
00242   std::vector<IntVar*> matrix(dim);
00243   // number of occurrences per stage
00244   std::vector<IntVar*> occurences;
00245 
00246   // All possible values of the distance vector
00247   // used to count the occurrence of all these values
00248   std::vector<int64> possible_values(2 * dim + 1);
00249 
00250   for (int64 i = -dim; i <= dim; ++i) {
00251     possible_values[i + dim] = i;
00252   }
00253 
00254   int index = 0;
00255 
00256   // Initialize the matrix that contains the coordinates of all '1' per row
00257   for (; index < dim; ++index) {
00258     matrix[index] = vars[index];
00259     vars[index]->SetMin(1);
00260   }
00261 
00262   // First constraint for the elements in the Costas Matrix. We want
00263   // them to be unique.
00264   std::vector<IntVar*> matrix_count;
00265   solver.MakeIntVarArray(2 * dim + 1, 0, dim, "matrix_count_", &matrix_count);
00266   solver.AddConstraint(solver.MakeDistribute(matrix, possible_values,
00267                                              matrix_count));
00268 
00269   // Here we only consider the elements from 1 to dim.
00270   for (int64 j = dim + 1; j <= 2 * dim; ++j) {
00271     // Penalize if an element occurs more than once.
00272     vars[index]
00273          = solver.MakeSemiContinuousExpr(solver.MakeSum(matrix_count[j], -1),
00274                                          0, 1)->Var();
00275 
00276     occurences.push_back(vars[index++]);
00277   }
00278 
00279   // Count the number of duplicates for each stage
00280   for (int i = 1; i < dim; ++i) {
00281     std::vector<IntVar*> subset(dim - i);
00282 
00283     // Initialize each stage
00284     for (int j = 0; j < dim - i; ++j) {
00285       IntVar* const diff = solver.MakeDifference(vars[j + i], vars[j])->Var();
00286       subset[j] = diff;
00287     }
00288 
00289     // Count the number of occurrences for all possible values
00290     std::vector<IntVar*> domain_count;
00291     solver.MakeIntVarArray(2 * dim + 1, 0, dim, "domain_count_", &domain_count);
00292     solver.AddConstraint(solver.MakeDistribute(subset,
00293                                                possible_values,
00294                                                domain_count));
00295 
00296     // Penalize occurrences of more than one
00297     for (int64 j = 0; j <= 2 * dim; ++j) {
00298       vars[index] =
00299           solver.MakeSemiContinuousExpr(solver.MakeSum(domain_count[j], -1),
00300                                         0, dim - i)->Var();
00301 
00302       occurences.push_back(vars[index++]);
00303     }
00304   }
00305 
00306   // We would like to minimize the penalties that we just computed
00307   IntVar* const objective_var = solver.MakeSum(occurences)->Var();
00308   OptimizeVar* const total_duplicates = solver.MakeMinimize(objective_var, 1);
00309 
00310   SearchMonitor* const log = solver.MakeSearchLog(1000, objective_var);
00311 
00312   // Out of all solutions, we just want to store the last one.
00313   SolutionCollector* const collector = solver.MakeLastSolutionCollector();
00314   collector->Add(vars);
00315 
00316   // The first solution that the local optimization is based on
00317   Evaluator evaluator(matrix);
00318   DecisionBuilder* const first_solution =
00319       solver.MakePhase(matrix,
00320                        NewPermanentCallback(&evaluator,
00321                                             &Evaluator::VarEvaluator),
00322                        NewPermanentCallback(&evaluator,
00323                                             &Evaluator::ValueEvaluator));
00324 
00325   SearchLimit* const search_time_limit =
00326       solver.MakeLimit(FLAGS_timelimit, kint64max, kint64max, kint64max);
00327 
00328   // Locally optimize solutions for LNS
00329   SearchLimit* const fail_limit = solver.MakeLimit(kint64max, kint64max,
00330                                                    FLAGS_sublimit, kint64max);
00331 
00332   DecisionBuilder* const subdecision_builder =
00333       solver.MakeSolveOnce(first_solution, fail_limit);
00334 
00335   std::vector<LocalSearchOperator*> localSearchOperators;
00336 
00337   // Apply RandomLNS to free FLAGS_freevar variables at each stage
00338   localSearchOperators.push_back(
00339       solver.RevAlloc(new OrderedLNS(matrix, FLAGS_freevar)));
00340 
00341   // Go through all possible permutations one by one
00342   localSearchOperators.push_back(
00343       solver.RevAlloc(new OrderedLNS(matrix, FLAGS_freeorderedvar)));
00344 
00345   LocalSearchPhaseParameters* const ls_params =
00346       solver.MakeLocalSearchPhaseParameters(
00347           solver.ConcatenateOperators(localSearchOperators),
00348           subdecision_builder);
00349 
00350   DecisionBuilder* const second_phase =
00351       solver.MakeLocalSearchPhase(matrix, first_solution, ls_params);
00352 
00353   // Try to find a solution
00354   solver.Solve(second_phase,
00355                collector,
00356                log,
00357                total_duplicates,
00358                search_time_limit);
00359 
00360   if (collector->solution_count() > 0) {
00361     std::vector<int64> costas_matrix;
00362     string output;
00363 
00364     for (int n = 0; n < dim; ++n) {
00365       const int64 v = collector->Value(0, vars[n]);
00366       costas_matrix.push_back(v);
00367       StringAppendF(&output, "%3lld", v);
00368     }
00369 
00370     if (!CheckCostas(costas_matrix)) {
00371       LOG(INFO) << "No Costas Matrix found, closest solution displayed.";
00372     }
00373 
00374     LOG(INFO) << output;
00375   } else {
00376     LOG(INFO) << "No solution found";
00377   }
00378 }

00380 // Computes a Costas Array.
00381 void CostasHard(const int dim) {
00382   SolverParameters parameters;
00383   if (!FLAGS_export_profile.empty()) {
00384     parameters.profile_level = SolverParameters::NORMAL_PROFILING;
00385   }
00386   Solver solver("costas", parameters);
00387 
00388   // create the variables
00389   std::vector<IntVar*> vars;
00390   solver.MakeIntVarArray(dim, -dim, dim, "var", &vars);
00391 
00392   // Initialize the matrix that contains the coordinates of all '1' per row
00393   for (int m = 0; m < dim; ++m) {
00394     vars[m]->SetMin(1);
00395   }
00396 
00397   solver.AddConstraint(solver.MakeAllDifferent(vars));
00398 
00399   // Check that the pairwise difference is unique
00400   for (int i = 1; i < dim; ++i) {
00401     std::vector<IntVar*> subset(dim - i);
00402 
00403     for (int j = 0; j < dim - i; ++j) {
00404       subset[j] = solver.MakeDifference(vars[j + i], vars[j])->Var();
00405     }
00406 
00407     solver.AddConstraint(solver.MakeAllDifferent(subset));
00408   }
00409 
00410   DecisionBuilder* const db = solver.MakePhase(vars,
00411                                                Solver::CHOOSE_FIRST_UNBOUND,
00412                                                Solver::ASSIGN_MIN_VALUE);
00413   solver.NewSearch(db);
00414 
00415   if (solver.NextSolution()) {
00416     std::vector<int64> costas_matrix;
00417     string output;
00418 
00419     for (int n = 0; n < dim; ++n) {
00420       const int64 v = vars[n]->Value();
00421       costas_matrix.push_back(v);
00422       StringAppendF(&output, "%3lld", v);
00423     }
00424 
00425     LOG(INFO) << output << " (" << solver.wall_time() << "ms)";
00426 
00427     CHECK(CheckCostas(costas_matrix)) <<
00428         ": Solution is not a valid Costas Matrix.";
00429   } else {
00430     LOG(INFO) << "No solution found.";
00431   }
00432   if (!FLAGS_export_profile.empty()) {
00433     solver.ExportProfilingOverview(FLAGS_export_profile);
00434   }
00435 }
