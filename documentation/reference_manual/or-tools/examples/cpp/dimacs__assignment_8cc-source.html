00048 template<typename GraphType> CostValue BuildAndSolveHungarianInstance(
00049     const LinearSumAssignment<GraphType>& assignment) {
00050   const GraphType& graph = assignment.Graph();
00051   typedef std::vector<double> HungarianRow;
00052   typedef std::vector<HungarianRow> HungarianProblem;
00053   HungarianProblem hungarian_cost;
00054   hungarian_cost.resize(assignment.NumLeftNodes());
00055   // First we have to find the biggest cost magnitude so we can
00056   // initialize the arc costs that aren't really there.
00057   CostValue largest_cost_magnitude = 0;
00058   for (typename GraphType::ArcIterator arc_it(graph);
00059        arc_it.Ok();
00060        arc_it.Next()) {
00061     ArcIndex arc = arc_it.Index();
00062     CostValue cost_magnitude = ::std::abs(assignment.ArcCost(arc));
00063     largest_cost_magnitude = ::std::max(largest_cost_magnitude, cost_magnitude);
00064   }
00065   double missing_arc_cost = static_cast<double>((assignment.NumLeftNodes() *
00066                                                  largest_cost_magnitude) +
00067                                                 1);
00068   for (HungarianProblem::iterator row = hungarian_cost.begin();
00069        row != hungarian_cost.end();
00070        ++row) {
00071     row->resize(assignment.NumNodes() - assignment.NumLeftNodes(),
00072                 missing_arc_cost);
00073   }
00074   // We're using a graph representation without forward arcs, so in
00075   // order to use the generic GraphType::ArcIterator we would
00076   // need to increase our memory footprint by building the array of
00077   // arc tails (since we need tails to build the input to the
00078   // hungarian algorithm). We opt for the alternative of iterating
00079   // over hte arcs via adjacency lists, which gives us the arc tails
00080   // implicitly.
00081   for (typename GraphType::NodeIterator node_it(graph);
00082        node_it.Ok();
00083        node_it.Next()) {
00084     NodeIndex node = node_it.Index();
00085     NodeIndex tail = (node - GraphType::kFirstNode);
00086     for (typename GraphType::OutgoingArcIterator arc_it(graph, node);
00087          arc_it.Ok();
00088          arc_it.Next()) {
00089       ArcIndex arc = arc_it.Index();
00090       NodeIndex head = (graph.Head(arc) - assignment.NumLeftNodes() -
00091                         GraphType::kFirstNode);
00092       double cost = static_cast<double>(assignment.ArcCost(arc));
00093       hungarian_cost[tail][head] = cost;
00094     }
00095   }
00096   hash_map<int, int> result;
00097   hash_map<int, int> wish_this_could_be_null;
00098   WallTimer timer;
00099   VLOG(1) << "Beginning Hungarian method.";
00100   timer.Start();
00101   MinimizeLinearAssignment(hungarian_cost, &result, &wish_this_could_be_null);
00102   double elapsed = timer.GetInMs() / 1000.0;
00103   LOG(INFO) << "Hungarian result computed in " << elapsed << " seconds.";
00104   double result_cost = 0.0;
00105   for (int i = 0; i < assignment.NumLeftNodes(); ++i) {
00106     int mate = result[i];
00107     result_cost += hungarian_cost[i][mate];
00108   }
00109   return static_cast<CostValue>(result_cost);
00110 }

00112 template<typename GraphType> void DisplayAssignment(
00113     const LinearSumAssignment<GraphType>& assignment) {
00114   for (typename LinearSumAssignment<GraphType>::BipartiteLeftNodeIterator
00115            node_it(assignment);
00116        node_it.Ok();
00117        node_it.Next()) {
00118     const NodeIndex left_node = node_it.Index();
00119     const ArcIndex matching_arc = assignment.GetAssignmentArc(left_node);
00120     const NodeIndex right_node = assignment.Head(matching_arc);
00121     VLOG(5) << "assigned (" << left_node << ", " << right_node << "): "
00122             << assignment.ArcCost(matching_arc);
00123   }
00124 }

00126 template<typename GraphType>
00127 int SolveDimacsAssignment(int argc, char* argv[]) {
00128   string error_message;
00129   // Handle on the graph we will need to delete because the
00130   // LinearSumAssignment object does not take ownership of it.
00131   GraphType* graph = NULL;
00132   DimacsAssignmentParser<GraphType> parser(argv[1]);
00133   LinearSumAssignment<GraphType>* assignment =
00134       parser.Parse(&error_message, &graph);
00135   if (assignment == NULL) {
00136     LOG(FATAL) << error_message;
00137   }
00138   if (!FLAGS_assignment_problem_output_file.empty()) {
00139     // The following tail array management stuff is done in a generic
00140     // way so we can plug in different types of graphs for which the
00141     // TailArrayManager template can be instantiated, even though we
00142     // know the type of the graph explicitly. In this way, the type of
00143     // the graph can be switched just by changing the graph type in
00144     // this file and making no other changes to the code.
00145     TailArrayManager<GraphType> tail_array_manager(graph);
00146     PrintDimacsAssignmentProblem<GraphType>(
00147         *assignment, tail_array_manager, FLAGS_assignment_problem_output_file);
00148     tail_array_manager.ReleaseTailArrayIfForwardGraph();
00149   }
00150   CostValue hungarian_cost = 0.0;
00151   bool hungarian_solved = false;
00152   if (FLAGS_assignment_compare_hungarian) {
00153     hungarian_cost = BuildAndSolveHungarianInstance(*assignment);
00154     hungarian_solved = true;
00155   }
00156   WallTimer timer;
00157   timer.Start();
00158   bool success = assignment->ComputeAssignment();
00159   double elapsed = timer.GetInMs() / 1000.0;
00160   if (success) {
00161     CostValue cost = assignment->GetCost();
00162     DisplayAssignment(*assignment);
00163     LOG(INFO) << "Cost of optimum assignment: " << cost;
00164     LOG(INFO) << "Computed in " << elapsed << " seconds.";
00165     LOG(INFO) << assignment->StatsString();
00166     if (hungarian_solved && (cost != hungarian_cost)) {
00167       LOG(ERROR) << "Optimum cost mismatch: " << cost << " vs. "
00168                  << hungarian_cost << ".";
00169     }
00170   } else {
00171     LOG(WARNING) << "Given problem is infeasible.";
00172   }
00173   delete assignment;
00174   delete graph;
00175   return 0;
00176 }
