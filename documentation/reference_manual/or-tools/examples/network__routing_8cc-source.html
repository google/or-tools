00014 // This model solves a multicommodity mono-routing problem with
00015 // capacity constraints and a max usage cost structure.  This means
00016 // that given a graph with capacity on edges, and a set of demands
00017 // (source, destination, traffic), the goal is to assign one unique
00018 // path for each demand such that the cost is minimized.  The cost is
00019 // defined by the maximum ratio utilization (traffic/capacity) for all
00020 // arcs.  There is also a penalty associated with an traffic of an arc
00021 // being above the comfort zone, 85% of the capacity by default.
00022 // Please note that constraint programming is well suited here because
00023 // we cannot have multiple active paths for a single demand.
00024 // Otherwise, a approach based on a linear solver is a better match.
00025 
00026 // A random problem generator is also included.

00249   void BuildGraph(int num_clients,
00250                   int num_backbones,
00251                   int min_client_degree,
00252                   int max_client_degree,
00253                   int min_backbone_degree,
00254                   int max_backbone_degree) {
00255     const int size = num_backbones + num_clients;
00256 
00257     // First we create the backbone nodes.
00258     for (int i = 1; i < num_backbones; ++i) {
00259       int j = random_.Uniform(i);
00260       CHECK_LT(j, i);
00261       AddEdge(i, j);
00262     }
00263 
00264     hash_set<int> to_complete;
00265     hash_set<int> not_full;
00266     for (int i = 0; i < num_backbones; ++i) {
00267       if (degrees_[i] < min_backbone_degree) {
00268         to_complete.insert(i);
00269       }
00270       if (degrees_[i] < max_backbone_degree) {
00271         not_full.insert(i);
00272       }
00273     }
00274     while (!to_complete.empty() && not_full.size() > 1) {
00275       const int node1 = *(to_complete.begin());
00276       int node2 = node1;
00277       while (node2 == node1 || degrees_[node2] >= max_backbone_degree) {
00278         node2 = random_.Uniform(num_backbones);
00279       }
00280       AddEdge(node1, node2);
00281       if (degrees_[node1] >= min_backbone_degree) {
00282         to_complete.erase(node1);
00283       }
00284       if (degrees_[node2] >= min_backbone_degree) {
00285         to_complete.erase(node2);
00286       }
00287       if (degrees_[node1] >= max_backbone_degree) {
00288         not_full.erase(node1);
00289       }
00290       if (degrees_[node2] >= max_backbone_degree) {
00291         not_full.erase(node2);
00292       }
00293     }
00294 
00295     // Then create the client nodes connected to the backbone nodes.
00296     // If num_client is 0, then backbone nodes are also client nodes.
00297     for (int i = num_backbones; i < size; ++i) {
00298       const int degree = RandomInInterval(min_client_degree, max_client_degree);
00299       while (degrees_[i] < degree) {
00300         const int j = random_.Uniform(num_backbones);
00301         if (!network_[i][j]) {
00302           AddEdge(i, j);
00303         }
00304       }
00305     }
00306   }

00413   void ComputeAllPathsForOneDemandAndOnePathLength(int demand_index,
00414                                                    int max_length,
00415                                                    int max_paths) {
00416     // We search for paths of length exactly 'max_length'.
00417     Solver solver("Counting");
00418     std::vector<IntVar*> arc_vars;
00419     std::vector<IntVar*> node_vars;
00420     solver.MakeIntVarArray(max_length, 0, num_nodes_ - 1, &node_vars);
00421     solver.MakeIntVarArray(max_length - 1, -1, count_arcs() - 1, &arc_vars);
00422 
00423     for (int i = 0; i < max_length - 1; ++i) {
00424       std::vector<IntVar*> tmp_vars;
00425       tmp_vars.push_back(node_vars[i]);
00426       tmp_vars.push_back(node_vars[i + 1]);
00427       tmp_vars.push_back(arc_vars[i]);
00428       solver.AddConstraint(solver.MakeAllowedAssignments(tmp_vars, arcs_data_));
00429     }
00430 
00431     const Demand& demand = demands_array_[demand_index];
00432     solver.AddConstraint(solver.MakeEquality(node_vars[0], demand.source));
00433     solver.AddConstraint(solver.MakeEquality(node_vars[max_length - 1],
00434                                              demand.destination));
00435     solver.AddConstraint(solver.MakeAllDifferent(arc_vars));
00436     solver.AddConstraint(solver.MakeAllDifferent(node_vars));
00437     DecisionBuilder* const db = solver.MakePhase(node_vars,
00438                                                  Solver::CHOOSE_FIRST_UNBOUND,
00439                                                  Solver::ASSIGN_MIN_VALUE);
00440     solver.NewSearch(db);
00441     while (solver.NextSolution()) {
00442       const int path_id = all_paths_[demand_index].size();
00443       all_paths_[demand_index].resize(path_id + 1);
00444       for (int arc_index = 0; arc_index < max_length - 1; ++arc_index) {
00445         const int arc = arc_vars[arc_index]->Value();
00446         all_paths_[demand_index].back().insert(arc);
00447       }
00448       if (all_paths_[demand_index].size() > max_paths) {
00449         break;
00450       }
00451     }
00452     solver.EndSearch();
00453   }

00473       }

00604   // Build the AllowedAssignment constraint with one tuple per path
00605   // for a given demand.
00606   void BuildNodePathConstraint(Solver* const solver,
00607                                const std::vector<IntVar*>& path_vars,
00608                                int demand_index,
00609                                std::vector<IntVar*>* decision_vars) {
00610     // Fill Tuple Set for AllowedAssignment constraint.
00611     const std::vector<OnePath> paths = all_paths_[demand_index];
00612     std::vector<std::vector<int64> > tuple_set;
00613     tuple_set.resize(paths.size());
00614     for (int path_id = 0; path_id < paths.size(); ++path_id) {
00615       tuple_set[path_id].resize(count_arcs() + 1, false);
00616       tuple_set[path_id][0] = path_id;
00617       for (ConstIter<OnePath> it(paths[path_id]); !it.at_end(); ++it) {
00618         const int arc = *it;
00619         // + 1 because tuple_set.back()[0] contains path_id.
00620         tuple_set[path_id][arc + 1] = true;
00621       }
00622     }
00623 
00624     const string name = StringPrintf("PathDecision_%i", demand_index);
00625     IntVar* const var = solver->MakeIntVar(0, tuple_set.size() - 1, name);
00626     std::vector<IntVar*> tmp_vars;
00627     tmp_vars.push_back(var);
00628     for (int i = 0; i < count_arcs(); ++i) {
00629       tmp_vars.push_back(path_vars[i]);
00630     }
00631     solver->AddConstraint(solver->MakeAllowedAssignments(tmp_vars, tuple_set));
00632     decision_vars->push_back(var);
00633   }

00635   // Build traffic variable summing all traffic from all demands
00636   // going through a single arc.
00637   void BuildTrafficVariable(Solver* const solver,
00638                             int arc_index,
00639                             const std::vector<std::vector<IntVar*> >& path_vars,
00640                             IntVar** const traffic) {
00641     std::vector<IntVar*> terms;
00642     for (int i = 0; i < path_vars.size(); ++i) {
00643       terms.push_back(solver->MakeProd(path_vars[i][arc_index],
00644                                        demands_array_[i].traffic)->Var());
00645     }
00646     *traffic = solver->MakeSum(terms)->Var();
00647   }

00651   class PathBasedLns : public BaseLNS {
00652    public:
00653     PathBasedLns(const IntVar* const* vars,
00654                  int size,
00655                  int fragment_size,
00656                  const std::vector<std::vector<OnePath> >& all_paths,
00657                  int num_arcs,
00658                  const std::vector<int64>& actual_usage_costs)
00659         : BaseLNS(vars, size),
00660           rand_(FLAGS_lns_seed),
00661           fragment_size_(fragment_size),
00662           all_paths_(all_paths),
00663           num_arcs_(num_arcs),
00664           actual_usage_costs_(actual_usage_costs) {
00665       CHECK_GT(fragment_size_, 0);
00666     }
00667 
00668     virtual ~PathBasedLns() {}
00669 
00670     virtual void InitFragments() {
00671       // We factorize computations that need to be updated only when
00672       // we have a new solution and not at each fragment.
00673       arc_wrappers_.clear();
00674       for (int i = 0; i < actual_usage_costs_.size(); ++i) {
00675         const int64 cost = actual_usage_costs_[i];
00676         if (cost != 0) {
00677           arc_wrappers_.push_back(ArcWrapper(i, cost));
00678         }
00679       }
00680       if (arc_wrappers_.size() > fragment_size_) {
00681         std::stable_sort(arc_wrappers_.begin(), arc_wrappers_.end());
00682       }
00683     }
00684 
00685     virtual bool NextFragment(std::vector<int>* fragment) {
00686       // First we select a set of arcs to release.
00687       hash_set<int> arcs_to_release;
00688       if (arc_wrappers_.size() <= fragment_size_) {
00689         // There are not enough used arcs, we will release all of them.
00690         for (int index = 0; index < arc_wrappers_.size(); ++index) {
00691           arcs_to_release.insert(arc_wrappers_[index].arc_id);
00692         }
00693       } else {
00694         if (FLAGS_focus_lns) {
00695           // We select 'fragment_size / 2' most costly arcs.
00696           for (int index = 0; index < fragment_size_ / 2; ++index) {
00697             arcs_to_release.insert(arc_wrappers_[index].arc_id);
00698           }
00699         }
00700 
00701         // We fill 'arcs_to_release' until we have chosen 'fragment_size_' arcs
00702         // to release.
00703         while (arcs_to_release.size() < fragment_size_) {
00704           const int candidate = rand_.Uniform(arc_wrappers_.size());
00705           arcs_to_release.insert(arc_wrappers_[candidate].arc_id);
00706         }
00707       }
00708 
00709       // We actually free all paths going through any of the selected arcs.
00710       const int demands = all_paths_.size();
00711       for (int i = 0; i < demands; ++i) {
00712         const OnePath& path = all_paths_[i][Value(i)];
00713         for (ConstIter<hash_set<int> > it(arcs_to_release);
00714              !it.at_end();
00715              ++it) {
00716           if (ContainsKey(path, *it)) {
00717             fragment->push_back(i);
00718             break;
00719           }
00720         }
00721       }
00722       return true;
00723     }
00724 
00725    private:
00726     struct ArcWrapper {
00727      public:
00728       ArcWrapper(int i, int64 c) : arc_id(i), cost(c) {}
00729       int arc_id;
00730       int64 cost;
00731       bool operator<(const ArcWrapper& other_arc_wrapper) const {
00732         return cost > other_arc_wrapper.cost
00733             || (cost == other_arc_wrapper.cost
00734                 && arc_id < other_arc_wrapper.arc_id);
00735       }
00736     };
00737 
00738     ACMRandom rand_;
00739     const int fragment_size_;
00740     const std::vector<std::vector<OnePath> >& all_paths_;
00741     const int num_arcs_;
00742     const std::vector<int64>& actual_usage_costs_;
00743     std::vector<ArcWrapper> arc_wrappers_;
00744   };

00746   // ----- Evaluator for the Decision Builder -----
00747 
00748   static const int kOneThousand = 1000;
00749 
00750   int64 EvaluateMarginalCost(std::vector<IntVar*>* path_costs,
00751                              int64 var,
00752                              int64 val) {
00753     int64 best_cost = 0;
00754     const int64 traffic = demands_array_[var].traffic;
00755     const OnePath& path = all_paths_[var][val];
00756     for (ConstIter<OnePath> it(path); !it.at_end(); ++it) {
00757       const int64 current_percent = (*path_costs)[*it]->Min();
00758       const int64 current_capacity = arc_capacity_[*it];
00759       const int64 expected_percent =
00760           current_percent + traffic * kOneThousand / current_capacity;
00761       if (expected_percent > best_cost) {
00762         best_cost = expected_percent;
00763       }
00764     }
00765     return best_cost;
00766   }

00770   static Solver::DecisionModification MaxDiscrepancy1(Solver* const solver) {
00771     if (solver->SearchDepth() - solver->SearchLeftDepth() > 1) {

00822 
00823   int64 LnsSolve(int time_limit, int fail_limit) {
00824     LOG(INFO) << "Solving model";
00825     const int num_demands = demands_array_.size();
00826     const int num_arcs = count_arcs();
00827     // ----- Build Model -----
00828     Solver solver("MultiPathSolver");
00829     std::vector<std::vector<IntVar*> > path_vars(num_demands);
00830     std::vector<IntVar*> decision_vars;
00831 
00832     // Node - Graph Constraint.
00833     for (int demand_index = 0; demand_index < num_demands; ++demand_index) {
00834       solver.MakeBoolVarArray(num_arcs,
00835                               StringPrintf("path_vars_%i_", demand_index),
00836                               &path_vars[demand_index]);
00837       BuildNodePathConstraint(&solver,
00838                               path_vars[demand_index],
00839                               demand_index,
00840                               &decision_vars);
00841     }
00842     // Traffic variables.
00843     std::vector<IntVar*> vtraffic(num_arcs);
00844     for (int arc_index = 0; arc_index < num_arcs; ++arc_index) {
00845       BuildTrafficVariable(&solver,
00846                            arc_index,
00847                            path_vars,
00848                            &vtraffic[arc_index]);
00849       vtraffic[arc_index]->set_name(StringPrintf("traffic_%i", arc_index));
00850     }
00851 
00852     // Objective Function.
00853     std::vector<IntVar*> costs;
00854     std::vector<IntVar*> usage_costs;
00855     std::vector<IntVar*> comfort_costs;
00856     for (int arc_index = 0; arc_index < num_arcs; ++arc_index) {
00857       const int capacity =
00858           capacity_[arcs_data_[2 * arc_index][0]][arcs_data_[2 * arc_index][1]];
00859       IntVar* const usage_cost =
00860           solver.MakeDiv(solver.MakeProd(vtraffic[arc_index], kOneThousand),
00861                          capacity)->Var();
00862       usage_costs.push_back(usage_cost);
00863       IntVar* const comfort_cost =
00864           solver.MakeIsGreaterCstVar(
00865               vtraffic[arc_index],
00866               capacity * FLAGS_comfort_zone / kOneThousand);
00867       comfort_costs.push_back(comfort_cost);
00868     }
00869     IntVar* const max_usage_cost = solver.MakeMax(usage_costs)->Var();
00870     IntVar* const sum_comfort_cost = solver.MakeSum(comfort_costs)->Var();
00871     IntVar* const objective_var =
00872         solver.MakeSum(max_usage_cost, sum_comfort_cost)->Var();
00873     std::vector<SearchMonitor*> monitors;
00874     OptimizeVar* const objective = solver.MakeMinimize(objective_var, 1);
00875     monitors.push_back(objective);
00876 
00877     // Search Log.
00878     if (FLAGS_report == 0) {
00879       SearchMonitor* const search_log =
00880           solver.MakeSearchLog(FLAGS_log_period, objective);
00881       monitors.push_back(search_log);
00882     }
00883 
00884     // DecisionBuilder.
00885     DecisionBuilder* const db =
00886         solver.MakePhase(decision_vars,
00887                          Solver::CHOOSE_RANDOM,
00888                          NewPermanentCallback(
00889                              this,
00890                              &NetworkRoutingSolver::EvaluateMarginalCost,
00891                              &usage_costs));
00892 
00893     // Limits.
00894     if (time_limit != 0 || fail_limit != 0) {
00895       if (time_limit != 0) {
00896         LOG(INFO) << "adding time limit of " << time_limit
00897                   << " ms";
00898       }
00899       if (fail_limit != 0) {
00900         LOG(INFO) << "adding fail limit of " << fail_limit;
00901       }
00902       monitors.push_back(
00903           solver.MakeLimit(time_limit != 0 ?
00904                            time_limit : kint64max,
00905                            kint64max,
00906                            fail_limit != 0 ?
00907                            fail_limit : kint64max,
00908                            kint64max));
00909     }
00910 
00911     // Lns Decision Builder.
00912     LOG(INFO) << "Using Lns with a fragment size of " << FLAGS_lns_size
00913               << ", and fail limit of " << FLAGS_lns_limit;
00914     std::vector<int64> actual_usage_costs(num_arcs);
00915     DecisionBuilder* const store_info =
00916         solver.RevAlloc(new StoreUsageCosts(usage_costs, &actual_usage_costs));
00917 
00918     LocalSearchOperator* const local_search_operator =
00919         solver.RevAlloc(new PathBasedLns(decision_vars.data(),
00920                                          decision_vars.size(),
00921                                          FLAGS_lns_size,
00922                                          all_paths_,
00923                                          num_arcs,
00924                                          actual_usage_costs));
00925     SearchLimit* const lns_limit =
00926         solver.MakeLimit(kint64max, kint64max, FLAGS_lns_limit, kint64max);
00927     DecisionBuilder* const inner_db =
00928         solver.MakePhase(decision_vars,
00929                          Solver::CHOOSE_RANDOM,
00930                          NewPermanentCallback(
00931                              this,
00932                              &NetworkRoutingSolver::EvaluateMarginalCost,
00933                              &usage_costs));
00934 
00935 
00936     DecisionBuilder* const apply = solver.RevAlloc(new ApplyMaxDiscrepancy);
00937     DecisionBuilder* const max_discrepency_db = solver.Compose(apply, inner_db);
00938     DecisionBuilder* const ls_db = solver.MakeSolveOnce(max_discrepency_db,
00939                                                         lns_limit);
00940     LocalSearchPhaseParameters* const parameters =
00941         solver.MakeLocalSearchPhaseParameters(local_search_operator,
00942                                               solver.Compose(ls_db,
00943                                                              store_info));
00944     DecisionBuilder* const final_db =
00945         solver.Compose(solver.MakeLocalSearchPhase(decision_vars,
00946                                                    db,
00947                                                    parameters),
00948                        store_info);
00949 
00950     // And Now Solve.
00951     int64 best_cost = kint64max;
00952     solver.NewSearch(final_db, monitors);
00953     while (solver.NextSolution()) {
00954       // Solution Found: Report it.
00955       const double percent = max_usage_cost->Value() / 10.0;
00956       const int64 non_comfort = sum_comfort_cost->Value();
00957       if (non_comfort > 0) {
00958         LOG(INFO) << "*** Found a solution with a max usage of " << percent
00959                   << "%, and " << non_comfort
00960                   << " links above the comfort zone";
00961       } else {
00962         LOG(INFO) << "*** Found a solution with a max usage of " << percent
00963                   << "%";
00964       }
00965       best_cost = objective_var->Value();
00966       if (FLAGS_report > 1) {
00967         DisplaySolution(num_arcs,
00968                         max_usage_cost->Value(),
00969                         usage_costs,
00970                         path_vars,
00971                         FLAGS_report > 2,
00972                         FLAGS_comfort_zone);
00973       }
00974     }
00975     solver.EndSearch();
00976 
00977     return best_cost;
00978   }
00979   void DisplaySolution(int num_arcs,
00980                        int64 max_usage_cost,
00981                        const std::vector<IntVar*>& usage_costs,
00982                        const std::vector<std::vector<IntVar*> >& path_vars,
00983                        bool precise,
00984                        int64 comfort_zone) {
00985     // We will show paths above the comfort zone, or above the max
00986     // utilization minus 5%.
00987     const int64 kFivePercentInThousandth = 50;
00988     const int64 cutoff =
00989         std::min(max_usage_cost - kFivePercentInThousandth, comfort_zone);
00990     for (int i = 0; i < num_arcs; ++i) {
00991       const int64 arc_usage = usage_costs[i]->Value();
00992       if (arc_usage >= cutoff) {
00993         const int source_index = arcs_data_[2 * i][0];
00994         const int destination_index = arcs_data_[2 * i][1];
00995         LOG(INFO) << " + Arc " << source_index
00996                   << " <-> " << destination_index
00997                   << " has a usage = " << arc_usage / 10.0
00998                   << "%, capacity = "
00999                   << capacity_[source_index][destination_index];
01000         if (precise) {
01001           const int num_demands = demands_array_.size();
01002           for (int demand_index = 0;
01003                demand_index < num_demands;
01004                ++demand_index) {
01005             if (path_vars[demand_index][i]->Value() == 1) {
01006               const Demand& demand = demands_array_[demand_index];
01007               LOG(INFO) << "   - "
01008                         << StringPrintf("%i -> %i (%i)",
01009                                         demand.source,
01010                                         demand.destination,
01011                                         demand.traffic);
01012             }
01013           }
01014         }
01015       }
01016     }
01017   }
