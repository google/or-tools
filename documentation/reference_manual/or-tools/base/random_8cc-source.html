00032 int32 ACMRandom::Next() {
00033   if (seed_ == 0) {
00034     seed_ = 0x14fd4603;  // Arbitrary random constant
00035   }
00036   const int32 M = 2147483647L;   // 2^31-1
00037   const int32 A = 16807;
00038   // In effect, we are computing seed_ = (seed_ * A) % M, where M = 2^31-1
00039   uint32 lo = A * static_cast<int32>(seed_ & 0xFFFF);
00040   uint32 hi = A * static_cast<int32>(static_cast<uint32>(seed_) >> 16);
00041   lo += (hi & 0x7FFF) << 16;
00042   if (lo > M) {
00043     lo &= M;
00044     ++lo;
00045   }
00046   lo += hi >> 15;
00047   if (lo > M) {
00048     lo &= M;
00049     ++lo;
00050   }
00051   return (seed_ = static_cast<int32>(lo));
00052 }
