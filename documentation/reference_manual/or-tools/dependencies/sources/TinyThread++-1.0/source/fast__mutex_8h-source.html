00061 /// Fast mutex class.
00062 /// This is a mutual exclusion object for synchronizing access to shared
00063 /// memory areas for several threads. It is similar to the tthread::mutex class,
00064 /// but instead of using system level functions, it is implemented as an atomic
00065 /// spin lock with very low CPU overhead.
00066 ///
00067 /// The \c fast_mutex class is NOT compatible with the \c condition_variable
00068 /// class (however, it IS compatible with the \c lock_guard class). It should
00069 /// also be noted that the \c fast_mutex class typically does not provide
00070 /// as accurate thread scheduling as a the standard \c mutex class does.
00071 ///
00072 /// Because of the limitations of the class, it should only be used in
00073 /// situations where the mutex needs to be locked/unlocked very frequently.
00074 ///
00075 /// @note The "fast" version of this class relies on inline assembler language,
00076 /// which is currently only supported for 32/64-bit Intel x86/AMD64 and
00077 /// PowerPC architectures on a limited number of compilers (GNU g++ and MS
00078 /// Visual C++).
00079 /// For other architectures/compilers, system functions are used instead.
00080 class fast_mutex {
00081   public:
00082     /// Constructor.
00083 #if defined(_FAST_MUTEX_ASM_)
00084     fast_mutex() : mLock(0) {}
00085 #else
00086     fast_mutex()
00087     {
00088   #if defined(_TTHREAD_WIN32_)
00089       InitializeCriticalSection(&mHandle);
00090   #elif defined(_TTHREAD_POSIX_)
00091       pthread_mutex_init(&mHandle, NULL);
00092   #endif
00093     }
00094 #endif

00108     /// Lock the mutex.
00109     /// The method will block the calling thread until a lock on the mutex can
00110     /// be obtained. The mutex remains locked until \c unlock() is called.
00111     /// @see lock_guard
00112     inline void lock()
00113     {
00114 #if defined(_FAST_MUTEX_ASM_)
00115       bool gotLock;
00116       do {
00117         gotLock = try_lock();
00118         if(!gotLock)
00119         {
00120   #if defined(_TTHREAD_WIN32_)
00121           Sleep(0);
00122   #elif defined(_TTHREAD_POSIX_)
00123           sched_yield();
00124   #endif
00125         }
00126       } while(!gotLock);
00127 #else
00128   #if defined(_TTHREAD_WIN32_)
00129       EnterCriticalSection(&mHandle);
00130   #elif defined(_TTHREAD_POSIX_)
00131       pthread_mutex_lock(&mHandle);
00132   #endif
00133 #endif
00134     }

00136     /// Try to lock the mutex.
00137     /// The method will try to lock the mutex. If it fails, the function will
00138     /// return immediately (non-blocking).
00139     /// @return \c true if the lock was acquired, or \c false if the lock could
00140     /// not be acquired.
00141     inline bool try_lock()
00142     {
00143 #if defined(_FAST_MUTEX_ASM_)
00144       int oldLock;
00145   #if defined(__GNUC__) && (defined(__i386__) || defined(__x86_64__))
00146       asm volatile (
00147         "movl $1,%%eax\n\t"
00148         "xchg %%eax,%0\n\t"
00149         "movl %%eax,%1\n\t"
00150         : "=m" (mLock), "=m" (oldLock)
00151         :
00152         : "%eax", "memory"
00153       );
00154   #elif defined(_MSC_VER) && (defined(_M_IX86) || defined(_M_X64))
00155       int *ptrLock = &mLock;
00156       __asm {
00157         mov eax,1
00158         mov ecx,ptrLock
00159         xchg eax,[ecx]
00160         mov oldLock,eax
00161       }
00162   #elif defined(__GNUC__) && (defined(__ppc__))
00163       int newLock = 1;
00164       asm volatile (
00165         "\n1:\n\t"
00166         "lwarx  %0,0,%1\n\t"
00167         "cmpwi  0,%0,0\n\t"
00168         "bne-   2f\n\t"
00169         "stwcx. %2,0,%1\n\t"
00170         "bne-   1b\n\t"
00171         "isync\n"
00172         "2:\n\t"
00173         : "=&r" (oldLock)
00174         : "r" (&mLock), "r" (newLock)
00175         : "cr0", "memory"
00176       );
00177   #endif
00178       return (oldLock == 0);
00179 #else
00180   #if defined(_TTHREAD_WIN32_)
00181       return TryEnterCriticalSection(&mHandle) ? true : false;
00182   #elif defined(_TTHREAD_POSIX_)
00183       return (pthread_mutex_trylock(&mHandle) == 0) ? true : false;
00184   #endif
00185 #endif
00186     }

00188     /// Unlock the mutex.
00189     /// If any threads are waiting for the lock on this mutex, one of them will
00190     /// be unblocked.
00191     inline void unlock()
00192     {
00193 #if defined(_FAST_MUTEX_ASM_)
00194   #if defined(__GNUC__) && (defined(__i386__) || defined(__x86_64__))
00195       asm volatile (
00196         "movl $0,%%eax\n\t"
00197         "xchg %%eax,%0\n\t"
00198         : "=m" (mLock)
00199         :
00200         : "%eax", "memory"
00201       );
00202   #elif defined(_MSC_VER) && (defined(_M_IX86) || defined(_M_X64))
00203       int *ptrLock = &mLock;
00204       __asm {
00205         mov eax,0
00206         mov ecx,ptrLock
00207         xchg eax,[ecx]
00208       }
00209   #elif defined(__GNUC__) && (defined(__ppc__))
00210       asm volatile (
00211         "sync\n\t"  // Replace with lwsync where possible?
00212         : : : "memory"
00213       );
00214       mLock = 0;
00215   #endif
00216 #else
00217   #if defined(_TTHREAD_WIN32_)
00218       LeaveCriticalSection(&mHandle);
00219   #elif defined(_TTHREAD_POSIX_)
00220       pthread_mutex_unlock(&mHandle);
00221   #endif
00222 #endif
00223     }
