00014 // Solves the Shortest Hamiltonian Path Problem using a complete algorithm.
00015 // The algorithm was first described in
00016 // M. Held, R.M. Karp, A dynamic programming approach to sequencing problems,
00017 // J. SIAM 10 (1962) 196â€“210
00018 //
00019 // The Shortest Hamiltonian Path Problem (SHPP) is similar to the Traveling
00020 // Salesperson Problem (TSP).
00021 // You have to visit all the cities, starting from a given one and you
00022 // do not need to return to your starting point. With the TSP, you can start
00023 // anywhere, but you have to return to your start location.
00024 //
00025 // By complete we mean that the algorithm guarantees to compute the optimal
00026 // solution.
00027 // The algorithm uses dynamic programming. Its time complexity is O(n*2^n),
00028 // where n is the number of nodes to be visited.
00029 // Its space complexity is also O(n*2^n).
00030 //
00031 // Note that the naive implementation of the SHPP
00032 // exploring all permutations without memorizing intermediate results would
00033 // have a complexity of (n-1)! (factorial of (n-1) ), which is much higher
00034 // than n*2^n. To convince oneself of this, just use Stirling's formula:
00035 // n! ~ sqrt(2*pi*n)*(n/exp(1))^n .
00036 // Because of these complexity figures, the algorithm is not practical for
00037 // problems with more than 20 nodes.
00038 //
00039 // Here is how the algorithm works:
00040 // Let us denote the nodes to be visited by their indices 0..n-1
00041 // Let us pick 0 as the starting node.
00042 // Let d(i,j) denote the distance (or cost) from i to j.
00043 // f(S,j) where S is a set of nodes and j is a node is defined as follows:
00044 // f(S,j) = min(i in S, f(S\{i}, i) + d(i,j))
00045 //
00046 // The set S can be represented by an integer where bit i corresponds to
00047 // element i in the set. In the following S denotes the integer corresponding
00048 // to set S.
00049 //
00050 // The computation of f(S,j) is implemented in the method ComputeShortestPath.
00051 //
00052 // To implement dynamic programming, we store the preceding results of
00053 // computing f(S,i) in an array M[S][i]. Since there are 2^n subsets of the
00054 // set {0..n-1}, there are 2^n rows and n columns in this array. This
00055 // explains the complexity figures.
00056 //
00057 // The array M is initialized as follows:
00058 // for i in (0..n-1) M[0][i] = d(0,i)
00059 // The dynamic programming iteration is as follows:
00060 // for S in (0..2^n-2)
00061 //   for i in (0..n-1) M[S][i]=f(S,i)
00062 //
00063 // The dynamic programming iteration is implemented in the method Solve.
00064 // The optimal value of the Hamiltonian path from 0 to n-1 is given by
00065 // f(2^n-1,n-1).
00066 // The optimal value of the Traveling Salesman tour is given by f(2^n-1,0).
00067 // (There is actually no need to duplicate the first node.)
00068 //
00069 // There are a few tricks in the efficient implementation of this algorithm
00070 // which are explained below.
00071 //
00072 // Keywords: Traveling Salesman, Hamiltonian Path, Dynamic Programming,
00073 //           Held, Karp.

00249 template <typename T>
00250 void HamiltonianPathSolver<T>::CheckRobustness() {
00251   T min_cost = std::numeric_limits<T>::max();
00252   T max_cost = std::numeric_limits<T>::min();
00253 
00254   // We compute the min and max for the cost matrix.
00255   for (int i = 0; i < num_nodes_; ++i) {
00256     for (int j = 0; j < num_nodes_; ++j) {
00257       if (i == j) break;
00258       min_cost = std::min(min_cost, cost_[i][j]);
00259       max_cost = std::max(max_cost, cost_[i][j]);
00260     }
00261   }
00262   // We determine if the range of the cost matrix is going to
00263   // make the algorithm not robust because of precision issues.
00264   if (min_cost < 0) {
00265     robust_ = false;
00266   } else {
00267     robust_ = (min_cost >
00268                num_nodes_ * max_cost * std::numeric_limits<T>::epsilon());
00269   }
00270   robustness_checked_ = true;
00271 }

00290 template <typename T>
00291 void HamiltonianPathSolver<T>::Init(const std::vector<std::vector<T> >& cost) {
00292   num_nodes_ = cost.size();
00293   if (num_nodes_ > 0) {
00294     cost_ = new T*[num_nodes_];
00295     for (int i = 0; i < num_nodes_; ++i) {
00296       cost_[i] = new T[num_nodes_];
00297     }
00298 
00299     CopyCostMatrix(cost);
00300 
00301     two_power_num_nodes_ = 1 << num_nodes_;
00302 
00303     // M[S][i] as defined in the header file is transposed and stored as
00304     // memory_[i][S]. This is due to the fact that the patterns generated
00305     // by the iteration on the subsets have very similar low bits and thus
00306     // yield cache misses.
00307     // To avoid cache misses, we pad the arrays and extend them by a magic
00308     // number.
00309     // This results in a 20 to 30% gain depending on architectures.
00310 
00311     const int padded_size = two_power_num_nodes_
00312                           + kHamiltonianPathSolverPadValue;
00313     memory_ = new T * [num_nodes_];
00314     memory_[0] = new T[num_nodes_ * padded_size];
00315     for (int i = 1; i < num_nodes_; ++i) {
00316       memory_[i] = memory_[i-1] + padded_size;
00317     }
00318   }
00319 }

00321 template <typename T>
00322 void HamiltonianPathSolver<T>::ComputeShortestPath(NodeSet subset,
00323                                                    PathNodeIndex dest) {
00324   // We iterate on the set bits in the NodeSet subset, instead of checking
00325   // which bits are set as in the loop:
00326   // for (int src = 0; src < num_nodes_; ++src) {
00327   //   const NodeSet singleton = (1 << src);
00328   //   if (subset & singleton) { ...
00329   // This results in a 30% gain.
00330 
00331   const NodeSet first_singleton = LeastSignificantBitWord32(subset);
00332   const PathNodeIndex first_src =
00333                                  LeastSignificantBitPosition32(first_singleton);
00334   NodeSet start_subset = subset - first_singleton;
00335   T min_cost = memory_[first_src][start_subset] + cost_[first_src][dest];
00336   NodeSet copy = start_subset;
00337   while (copy != 0) {
00338     const NodeSet singleton = LeastSignificantBitWord32(copy);
00339     const PathNodeIndex src = LeastSignificantBitPosition32(singleton);
00340     const NodeSet new_subset = subset - singleton;
00341     const T cost = memory_[src][new_subset] + cost_[src][dest];
00342     if (cost < min_cost) {
00343       min_cost = cost;
00344     }
00345     copy -= singleton;
00346   }
00347   memory_[dest][subset] = min_cost;
00348 }

00350 template <typename T> void HamiltonianPathSolver<T>::Solve() {
00351   if (solved_) return;
00352   for (PathNodeIndex dest = 0; dest < num_nodes_; ++dest) {
00353     memory_[dest][0] = cost_[0][dest];
00354   }
00355   for (NodeSet subset = 1; subset < two_power_num_nodes_; ++subset) {
00356     for (PathNodeIndex dest = 0; dest < num_nodes_; ++dest) {
00357       ComputeShortestPath(subset, dest);
00358     }
00359   }
00360   solved_ = true;

00384 // The naive way of getting the optimal path would be to store the subpath
00385 // in the same way the cost is stored in the matrix memory_.
00386 // The end of ComputeShortestPair (int i = 0; i < size; ++i)
00387 // would look like:
00388 // ...
00389 //   if (cost < min_cost) {
00390 //     min_cost = cost;
00391 //     best_path = new_subset;
00392 //   }
00393 //   copy -= singleton;
00394 //   }
00395 //   memory_[dest][subset] = min_cost;
00396 //   path_[dest][subset] = best_path;
00397 // }
00398 //
00399 // One can however get the optimal path from the matrix memory_, by
00400 // considering that:
00401 //     min_cost = memory_[src][subset - singleton] + cost_[src][dest]
00402 // The goal is thus to find a singleton satisfying the above equality.
00403 // The algorithm is therefore straightforward, apart from the fact that
00404 // care has to be taken about precision in case the type parameter T
00405 // is float our double (see below).
00406 //
00407 // The net result of this is that we gain a factor 2 over storing the paths
00408 // in a path_ matrix. Since the algorithm is memory-bound and due to the fact
00409 // that there are also less cache misses, there is a factor 2.5x in speed
00410 // when using this trick.

00412 template <typename T>
00413 void HamiltonianPathSolver<T>::Path(PathNodeIndex end,
00414                                     std::vector<PathNodeIndex>* path) {
00415   PathNodeIndex dest = end;
00416   NodeSet current_set = two_power_num_nodes_ - 1;
00417   // It may happen that node 0 be on a segment (node i, node j), in which case
00418   // it would appear in the middle of the path. We explicitly remove it from
00419   // the set to be explored to avoid this.
00420   current_set -= 1;
00421   int i = num_nodes_ - 1;
00422   while (current_set != 0) {
00423     NodeSet copy = current_set;
00424     while (copy != 0) {
00425       const NodeSet singleton = LeastSignificantBitWord32(copy);
00426       const PathNodeIndex src = LeastSignificantBitPosition32(singleton);
00427       const NodeSet incumbent_set = current_set - singleton;
00428       const double current_cost = memory_[dest][current_set];
00429       const double incumbent_cost = memory_[src][incumbent_set]
00430                                   + cost_[src][dest];
00431       // We take precision into account in case T is float or double.
00432       // There is no visible penalty in the case T is an integer type.
00433       if (fabs(current_cost - incumbent_cost)
00434           <=  std::numeric_limits<T>::epsilon() * current_cost) {
00435         current_set = incumbent_set;
00436         dest = src;
00437         (*path)[i] = dest;
00438         break;
00439       }
00440       copy -= singleton;
00441     }
00442     --i;
00443   }
00444 }
