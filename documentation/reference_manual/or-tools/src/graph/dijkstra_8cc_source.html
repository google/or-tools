00082 void DijkstraSP::Initialize() {
00083   for (int i = 0; i < node_count_; i++) {
00084     elements_[i].set_node(i);
00085     if (i == start_node_) {
00086       predecessor_[i] = -1;
00087       elements_[i].set_distance(0);
00088       frontier_.Add(&elements_[i]);
00089     } else {
00090       elements_[i].set_distance(kInfinity);
00091       predecessor_[i] = start_node_;
00092       not_visited_.insert(i);
00093     }
00094   }

00097 int DijkstraSP::SelectClosestNode(int64* distance) {
00098   const int node = frontier_.Top()->node();
00099   *distance = frontier_.Top()->distance();
00100   frontier_.Pop();
00101   not_visited_.erase(node);
00102   added_to_the_frontier_.erase(node);
00103   return node;

00106 void DijkstraSP::Update(int node) {
00107   for (hash_set<int>::const_iterator it = not_visited_.begin();
00108        it != not_visited_.end();
00109        ++it) {
00110     const int other_node = *it;
00111     const int64 graph_node_i = graph_->Run(node, other_node);
00112     if (graph_node_i != disconnected_distance_) {
00113       if (added_to_the_frontier_.find(other_node) ==
00114           added_to_the_frontier_.end()) {
00115         frontier_.Add(&elements_[other_node]);
00116         added_to_the_frontier_.insert(other_node);
00117       }
00118       const int64 other_distance = elements_[node].distance() + graph_node_i;
00119       if (elements_[other_node].distance() > other_distance) {
00120         elements_[other_node].set_distance(other_distance);
00121         frontier_.NoteChangedPriority(&elements_[other_node]);
00122         predecessor_[other_node] = node;
00123       }
00124     }
00125   }

00128 void DijkstraSP::FindPath(int dest, std::vector<int>* nodes) {

00137 bool DijkstraSP::ShortestPath(int end_node, std::vector<int>* nodes) {
00138   Initialize();
00139   bool found = false;
00140   while (!frontier_.IsEmpty()) {
00141     int64 distance;
00142     int node = SelectClosestNode(&distance);
00143     if (distance == kInfinity) {
00144       found = false;
00145       break;
00146     } else if (node == end_node) {
00147       found = true;
00148       break;

00152   if (found) {
