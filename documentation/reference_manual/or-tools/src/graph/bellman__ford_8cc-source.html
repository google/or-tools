00054 void BellmanFord::Initialize() {
00055   for (int i = 0; i < node_count_; i++) {
00056     distance_[i] = kint64max / 2;

00062 void BellmanFord::Update() {
00063   for (int i = 0; i < node_count_ - 1; i++) {
00064     for (int u = 0; u < node_count_; u++) {
00065       for (int v = 0; v < node_count_; v++) {
00066         const int64 graph_u_v = graph_->Run(u, v);
00067         if (graph_u_v != disconnected_distance_) {
00068           const int64 other_distance = distance_[u] + graph_u_v;
00069           if (distance_[v] > other_distance) {
00070             distance_[v] = other_distance;
00071             predecessor_[v] = u;
00072           }
00073         }
00074       }
00075     }
00076   }
00077 }

00079 bool BellmanFord::Check() const {
00080   for (int u = 0; u < node_count_; u++) {
00081     for (int v = 0; v < node_count_; v++) {
00082       const int graph_u_v = graph_->Run(u, v);
00083       if (graph_u_v != disconnected_distance_) {
00084         if (distance_[v] > distance_[u] + graph_u_v) {
00085           return false;

00093 void BellmanFord::FindPath(int dest, std::vector<int>* nodes) const {

00097     nodes->push_back(predecessor_[j]);
