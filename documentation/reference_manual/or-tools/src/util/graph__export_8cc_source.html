00173 GraphExporter* GraphExporter::MakeFileExporter(
00174     File* const file,
00175     GraphExporter::GraphFormat format) {
00176   GraphSyntax* syntax = NULL;
00177   switch (format) {
00178     case GraphExporter::DOT_FORMAT: {
00179       syntax = new DotSyntax();
