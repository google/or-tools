00025 // This class stores a vector of distinct elements, as well as a map
00026 // from elements to index to find the index in the vector.
00027 // This is useful to store mapping between objects and indices.
00028 template <class T> class VectorMap {
00029  public:
00030   // Adds an element if not already present, and returns its index in
00031   // the vector-map.
00032   int Add(const T& element) {
00033     int current_index = Index(element);
00034     if (current_index != -1) {
00035       return current_index;
00036     }
00037     const int index = list_.size();
00038     CHECK_EQ(index, map_.size());
00039     list_.push_back(element);
00040     map_[element] = index;
00041     return index;
00042   }
00043   // TODO(user): Use ArraySlice.
00044 
00045   // Adds all elements of the vector.
00046   void Add(const std::vector<T>& elements) {
00047     for (int i = 0; i < elements.size(); ++i) {
00048       Add(elements[i]);
00049     }
00050   }
00051 
00052   // Returns -1 if the element is not in the vector, or its unique
00053   // index if it is.
00054   int Index(const T& element) const {
00055     return FindWithDefault(map_, element, -1);
00056   }
00057   // TODO(user): explore a int-type version.
00058 
00059   // Returns wether the element has already been added to the vector-map.
00060   bool Contains(const T& element) const {
00061     return ContainsKey(map_, element);
00062   }
00063 
00064   // Returns the element at position index.
00065   const T& Element(int index) const {
00066     CHECK_GE(index, 0);
00067     CHECK_LT(index, list_.size());
00068     return list_[index];
00069   }
00070 
00071   const T& operator[](int index) const {
00072     return Element(index);
00073   }
00074 
00075   // Returns the number of distinct elements added to the vector-map.
00076   int size() const { return list_.size(); }
00077 
00078   // Clears all the elements added to the vector-map.
00079   void clear() {
00080     list_.clear();
00081     map_.clear();
00082   }
00083 
00084   // Returns a read-only access to the vector of elements.
00085   const std::vector<T>& list() const { return list_; }
00086 
00087   // Standard STL container boilerplate.
00088   typedef T value_type;
00089   typedef const T* pointer;
00090   typedef const T& reference;
00091   typedef const T& const_reference;
00092   typedef size_t size_type;
00093   typedef ptrdiff_t difference_type;
00094   static const size_type npos;
00095   typedef const T* const_iterator;
00096   typedef std::reverse_iterator<const_iterator> const_reverse_iterator;
00097   const_iterator begin() const { return list_.data(); }
00098   const_iterator end() const { return list_.data() + list_.size(); }
00099   const_reverse_iterator rbegin() const {
00100     return const_reverse_iterator(list_.data() + list_.size());
00101   }
00102   const_reverse_iterator rend() const {
00103     return const_reverse_iterator(list_.data());
00104   }
00105 
00106  private:
00107   std::vector<T> list_;
00108   hash_map<T, int> map_;
00109 };
