00614     ct->set_activity(row_activity);
00615 
00616     if (mip_) {
00617       VLOG(4) << "row " << ct->index()
00618               << ": activity = " << row_activity;
00619     } else {
00620       double dual_value;
00621       CheckReturnKey(SlmGetSolDualConsI(model_,ct->index(),&dual_value));
00622       ct->set_dual_value(dual_value);
00623       VLOG(4) << "row " << ct->index()
00624               << ": activity = " << row_activity
00625               << ": dual value = " << dual_value;
00626     }
00627   }
00628 
00629   // Check the status: optimal, infeasible, etc.
00630   SlmSolStatus tmp_status;
00631   CheckReturnKey(SlmGetSolStatus(model_,&tmp_status));
00632 
00633   switch(tmp_status)
00634   {
00635     case SlmSolStatUnk :
00636       VLOG(1) << "slm result status: SlmSolStatUnk";
00637       result_status_ = MPSolver::INFEASIBLE; /* What ever that means.. */
00638       break;
00639     case SlmSolStatOpt :
00640       VLOG(1) << "slm result status: SlmSolStatOpt";
00641       result_status_ = MPSolver::OPTIMAL;
00642       break;
00643     case SlmSolStatPrimFeas :
00644       VLOG(1) << "slm result status: SlmSolStatPrimFeas";
00645       result_status_ = MPSolver::FEASIBLE; /* What ever that means.. */
00646       break;
00647     case SlmSolStatDualFeas :
00648       VLOG(1) << "slm result status: SlmSolStatDualFeas";
00649       result_status_ = MPSolver::FEASIBLE; /* What ever that means.. */
00650       break;
00651     case SlmSolStatPrimInf :
00652       VLOG(1) << "slm result status: SlmSolStatPrimInf";
00653       result_status_ = MPSolver::INFEASIBLE;
00654       break;
00655     case SlmSolStatDualInf :
00656       VLOG(1) << "slm result status: SlmSolStatDualInf";
00657       result_status_ = MPSolver::UNBOUNDED; /* Theoretically not correct, you need a primal feasible point in LP */
00658       break;
00659     case SlmSolStatIntFeas :
00660       VLOG(1) << "slm result status: SlmSolStatIntFeas";
00661       result_status_ = MPSolver::FEASIBLE;
00662       break;
00663     case SlmSolStatIntInf :
00664       VLOG(1) << "slm result status: SlmSolStatIntInf";
00665       result_status_ = MPSolver::INFEASIBLE;
00666       break;
00667   }
00668 
00669   sync_status_ = SOLUTION_SYNCHRONIZED;
00670 
00671   return result_status_;
00672 }

00783 double SLMInterface::ComputeExactConditionNumber() const {
00784   CHECK(IsContinuous()) <<
00785       "Condition number only available for continuous problems";
00786   CheckSolutionIsSynchronized();
00787   // Simplex is the only LP algorithm supported in the wrapper for
00788   // SLM, so when a solution exists, a basis exists.
00789   CheckSolutionExists();
00790   int num_rows;
00791   int num_cols;
00792 
00793   CheckReturnKey(SlmGetCons(model_,&num_rows));
00794   CheckReturnKey(SlmGetVars(model_,&num_cols));
00795 
00796   scoped_array<double> row_scaling_factor(new double[num_rows]);
00797   scoped_array<double> column_scaling_factor(new double[num_cols]);
00798 
00799   for (int row = 0; row < num_rows; ++row) {
00800     row_scaling_factor[row] = 1.0;
00801   }
00802   for (int col = 0; col < num_cols; ++col) {
00803     column_scaling_factor[col] = 1.0;
00804   }
00805 
00806   return
00807       ComputeInverseScaledBasisL1Norm(
00808           num_rows, num_cols,
00809           row_scaling_factor.get(), column_scaling_factor.get()) *
00810       ComputeScaledBasisL1Norm(
00811           num_rows, num_cols,
00812           row_scaling_factor.get(), column_scaling_factor.get());
00813 }
00814 
00815 double SLMInterface::ComputeScaledBasisL1Norm(
00816     int num_rows, int num_cols,
00817     double* row_scaling_factor, double* column_scaling_factor) const {
00818   double norm = 0.0;
00819 
00820   scoped_array<double> values(new double[num_rows]);
00821   scoped_array<int> indices(new int[num_rows]);
00822   for (int col = 0; col < num_cols; ++col) {
00823     SlmStatusKey slm_basis_status;
00824 
00825     CheckReturnKey(SlmGetSolKeyPrimVarsI(model_,col,&slm_basis_status));
00826 
00827     // Take into account only basic columns.
00828     if (slm_basis_status == SlmStaBa) {
00829       // Compute L1-norm of column 'col': sum_row |a_row,col|.
00830       int num_nz;
00831 
00832       CheckReturnKey(SlmGetAVarsI(model_, col,&num_nz,indices.get(), values.get()));
00833 
00834       double column_norm = 0.0;
00835       for (int k = 0; k < num_nz; k++) {
00836         column_norm += fabs(values[k] * row_scaling_factor[indices[k]]);
00837       }
00838       column_norm *= fabs(column_scaling_factor[col]);
00839       // Compute max_col column_norm
00840       norm = std::max(norm, column_norm);
00841     }
00842   }
00843   // Slack variables.
00844   for (int row = 0; row < num_rows; ++row) {
00845     SlmStatusKey slm_basis_status;
00846 
00847     CheckReturnKey(SlmGetSolKeyPrimConsI(model_,row,&slm_basis_status));
00848 
00849     // Take into account only basic slack variables.
00850     if (slm_basis_status == SlmStaBa) {
00851       // Only one non-zero coefficient: +/- 1.0 in the corresponding
00852       // row. The row has a scaling coefficient but the slack variable
00853       // is never scaled on top of that.
00854       const double column_norm = fabs(row_scaling_factor[row]);
00855       // Compute max_col column_norm
00856       norm = std::max(norm, column_norm);
00857     }
00858   }
00859 
00860   return norm;
00861 }
00862 
00863 double SLMInterface::ComputeInverseScaledBasisL1Norm(
00864     int num_rows, int num_cols,
00865     double* row_scaling_factor, double* column_scaling_factor) const {
00866 
00867   // Currently we just refactor each time
00868   int ret = SlmInitBasisSolves(model_);
00869 
00870   // Compute the LU factorization if it doesn't exist yet.
00871   if ( ret != SlmRetOk  ) {
00872     switch (ret) {
00873       case SlmRetBasisSingular: {
00874         LOG(WARNING)
00875             << "Not able to factorize: "
00876             << "the basis matrix is singular within the working precision.";
00877         return MPSolver::infinity();
00878       }
00879       default:
00880         CheckReturnKey(ret);
00881         break;
00882     }
00883   }
00884 
00885   scoped_array<double> right_hand_side(new double[num_rows]);
00886   scoped_array<int> basidx(new int[num_rows]);
00887 
00888   CheckReturnKey(SlmGetBasisHead(model_,basidx.get()));
00889 
00890   double norm = 0.0;
00891 
00892   // Iteratively solve B x = e_k, where e_k is the kth unit vector.
00893   // The result of this computation is the kth column of B^-1.
00894 
00895   for (int k = 0; k < num_rows; ++k) {
00896     for (int row = 0; row < num_rows; ++row) {
00897       right_hand_side[row] = 0.0;
00898     }
00899     right_hand_side[k] = 1.0;
00900     // Multiply input by inv(R).
00901     for (int row = 0; row < num_rows; ++row) {
00902       right_hand_side[row] /= row_scaling_factor[row];
00903     }
00904 
00905     CheckReturnKey(SlmSolveFtranDense(model_, right_hand_side.get()));
00906 
00907     // stores the result in the same vector where the right
00908     // hand side was provided.
00909     // Multiply result by inv(SB).
00910     for (int row = 0; row < num_rows; ++row) {
00911       const int k = basidx[row];
00912       if (k <= num_rows) {
00913         // Auxiliary variable.
00914         right_hand_side[row] *= row_scaling_factor[k];
00915       } else {
00916         // Structural variable.
00917         right_hand_side[row] /= column_scaling_factor[k - num_rows];
00918       }
00919     }
00920 
00921     // Compute sum_row |vector_row|.
00922     double column_norm = 0.0;
00923     for (int row = 0; row < num_rows; ++row) {
00924       column_norm += fabs(right_hand_side[row]);
00925     }
00926 
00927     // Compute max_col column_norm
00928     norm = std::max(norm, column_norm);
00929   }
00930 
00931   return norm;
00932 }
