00291                                    const MPVariable* const variable,
00292                                    double new_value,
00293                                    double old_value) {
00294   InvalidateSolutionSynchronization();
00295   const int constraint_index = constraint->index();

00297   if (constraint_index != kNoIndex && variable_index != kNoIndex) {
00298     // The modification of the coefficient for an extracted row and

00301     DCHECK_LE(variable_index, last_variable_index_);

00305     ORTOOLS_SCIP_CALL(SCIPaddCoefLinear(
00306         scip_, scip_constraints_[constraint_index],
00307         scip_variables_[variable_index], new_value - old_value));
00308   } else {
00309     // The modification of an unextracted row or variable is cached
00310     // and handled in ExtractModel.
00311     sync_status_ = MUST_RELOAD;

00374 // Define new variables and add them to existing constraints.
00375 void SCIPInterface::ExtractNewVariables() {
00376   int total_num_vars = solver_->variables_.size();
00377   if (total_num_vars > last_variable_index_) {
00378     ORTOOLS_SCIP_CALL(SCIPfreeTransform(scip_));
00379     // Define new variables
00380     for (int j = last_variable_index_; j < total_num_vars; ++j) {
00381       MPVariable* const var = solver_->variables_[j];
00382       DCHECK_EQ(kNoIndex, var->index());
00383       var->set_index(j + 1);  // offset by 1 because of dummy variable.
00384       SCIP_VAR* scip_var = NULL;
00385       // The true objective coefficient will be set later in ExtractObjective.
00386       double tmp_obj_coef = 0.0;
00387       ORTOOLS_SCIP_CALL(SCIPcreateVar(
00388           scip_, &scip_var, var->name().c_str(),
00389           var->lb(), var->ub(), tmp_obj_coef,
00390           var->integer() ? SCIP_VARTYPE_INTEGER : SCIP_VARTYPE_CONTINUOUS,
00391           true, false, NULL, NULL, NULL, NULL, NULL));
00392       ORTOOLS_SCIP_CALL(SCIPaddVar(scip_, scip_var));
00393       scip_variables_.push_back(scip_var);
00394     }
00395     // Add new variables to existing constraints.
00396     for (int i = 0; i < last_constraint_index_; i++) {
00397       MPConstraint* const ct = solver_->constraints_[i];
00398       for (ConstIter<hash_map<const MPVariable*, double> > it(
00399                ct->coefficients_);
00400            !it.at_end(); ++it) {
00401         const int var_index = it->first->index();
00402         DCHECK_NE(kNoIndex, var_index);
00403         if (var_index >= last_variable_index_ + 1) {
00404           // The variable is new (index offset by 1 because of the
00405           // dummy variable), so we know the previous coefficient
00406           // value was 0 and we can directly add the coefficient.
00407           ORTOOLS_SCIP_CALL(SCIPaddCoefLinear(
00408               scip_, scip_constraints_[i],
00409               scip_variables_[var_index], it->second));
00410         }
00411       }
00412     }
00413   }

00416 // Define new constraints on old and new variables.
00417 void SCIPInterface::ExtractNewConstraints() {
00418   int total_num_rows = solver_->constraints_.size();
00419   if (last_constraint_index_ < total_num_rows) {
00420     ORTOOLS_SCIP_CALL(SCIPfreeTransform(scip_));
00421     // Find the length of the longest row.
00422     int max_row_length = 0;
00423     for (int i = last_constraint_index_; i < total_num_rows; ++i) {
00424       MPConstraint* const ct = solver_->constraints_[i];
00425       DCHECK_EQ(kNoIndex, ct->index());
00426       ct->set_index(i);
00427       if (ct->coefficients_.size() > max_row_length) {
00428         max_row_length = ct->coefficients_.size();
00429       }
00430     }
00431     scoped_array<SCIP_VAR*> vars(new SCIP_VAR*[max_row_length]);
00432     scoped_array<double> coefs(new double[max_row_length]);
00433     // Add each new constraint.
00434     for (int i = last_constraint_index_; i < total_num_rows; ++i) {
00435       MPConstraint* const ct = solver_->constraints_[i];
00436       DCHECK_NE(kNoIndex, ct->index());
00437       int size = ct->coefficients_.size();
00438       int j = 0;
00439       for (ConstIter<hash_map<const MPVariable*, double> > it(
00440                ct->coefficients_);
00441            !it.at_end(); ++it) {
00442         const int index = it->first->index();
00443         DCHECK_NE(kNoIndex, index);
00444         vars[j] = scip_variables_[index];
00445         coefs[j] = it->second;
00446         j++;
00447       }
00448       SCIP_CONS* scip_constraint = NULL;
00449       ORTOOLS_SCIP_CALL(SCIPcreateConsLinear(
00450           scip_, &scip_constraint,
00451           // TODO(user)
00452           ct->name().empty() ? "" : ct->name().c_str(),
00453           size, vars.get(), coefs.get(),
00454           ct->lb(), ct->ub(),
00455           true, true, true, true, true, false, false, false, false, false));
00456       ORTOOLS_SCIP_CALL(SCIPaddCons(scip_, scip_constraint));
00457       scip_constraints_.push_back(scip_constraint);
00458     }
00459   }

00503 #endif

00505 
00506   // Special case if the model is empty since SCIP expects a non-empty model
00507   if (solver_->variables_.size() == 0 && solver_->constraints_.size() == 0) {
00508     sync_status_ = SOLUTION_SYNCHRONIZED;
00509     result_status_ = MPSolver::OPTIMAL;

00511     return result_status_;

00513 
00514   ExtractModel();
00515   VLOG(1) << StringPrintf("Model built in %.3f seconds.", timer.Get());
00516 
00517   WriteModelToPredefinedFiles();
00518

00521     VLOG(1) << "Setting time limit = " << solver_->time_limit() << " ms.";
00522     ORTOOLS_SCIP_CALL(SCIPsetRealParam(
00523         scip_, "limits/time", solver_->time_limit() / 1000.0));
00524   } else {
00525     ORTOOLS_SCIP_CALL(SCIPresetParam(scip_, "limits/time"));
00526   }

00540     VLOG(1) << "objective=" << objective_value_;
00541     for (int i = 0; i < solver_->variables_.size(); ++i) {
00542       MPVariable* const var = solver_->variables_[i];
00543       const int var_index = var->index();

00545                                        scip_variables_[var_index]);
00546       var->set_solution_value(val);

00550       MPConstraint* const ct = solver_->constraints_[i];
00551       const int constraint_index = ct->index();

00553           SCIPgetActivityLinear(scip_,
00554                                 scip_constraints_[constraint_index],
00555                                 solution);
00556       ct->set_activity(row_activity);

00562   }
00563 
00564   // Check the status: optimal, infeasible, etc.
00565   SCIP_STATUS scip_status = SCIPgetStatus(scip_);
00566   switch (scip_status) {
00567     case SCIP_STATUS_OPTIMAL:
00568       result_status_ = MPSolver::OPTIMAL;
00569       break;
00570     case SCIP_STATUS_GAPLIMIT:
00571       // To be consistent with the other solvers.
00572       result_status_ = MPSolver::OPTIMAL;
00573       break;
00574     case SCIP_STATUS_INFEASIBLE:
00575       result_status_ = MPSolver::INFEASIBLE;
00576       break;
00577     case SCIP_STATUS_UNBOUNDED:
00578       result_status_ = MPSolver::UNBOUNDED;
00579       break;
00580     case SCIP_STATUS_INFORUNBD:
00581       // TODO(user): We could introduce our own "infeasible or
00582       // unbounded" status.
00583       result_status_ = MPSolver::INFEASIBLE;
00584       break;
00585     default:
00586       if (solution != NULL) {
00587         result_status_ = MPSolver::FEASIBLE;
00588       } else {
00589         // TODO(user): We could introduce additional values for the
00590         // status: for example, stopped because of time limit.
00591         result_status_ = MPSolver::ABNORMAL;
00592       }
00593       break;
00594   }
00595 
00596   ORTOOLS_SCIP_CALL(SCIPresetParams(scip_));
00597 
00598   sync_status_ = SOLUTION_SYNCHRONIZED;
00599   return result_status_;
00600 }
