00014 // A string-like object that points to a sized piece of memory.
00015 //
00016 // Functions or methods may use const StringPiece& parameters to accept either
00017 // a "const char*" or a "string" value that will be implicitly converted to
00018 // a StringPiece.  The implicit conversion means that it is often appropriate
00019 // to include this .h file in other files rather than forward-declaring
00020 // StringPiece as would be appropriate for most other Google classes.
00021 //
00022 // Systematic usage of StringPiece is encouraged as it will reduce unnecessary
00023 // conversions from "const char*" to "string" and back again.
00024

00041  public:
00042   // We provide non-explicit singleton constructors so users can pass
00043   // in a "const char*" or a "string" wherever a "StringPiece" is
00044   // expected.
00045   StringPiece() : ptr_(NULL), length_(0) { }
00046   StringPiece(const char* str)  // NOLINT
00047     : ptr_(str), length_((str == NULL) ? 0 : static_cast<int>(strlen(str))) { }
00048   StringPiece(const std::string& str)  // NOLINT
00049     : ptr_(str.data()), length_(static_cast<int>(str.size())) { }
00050   StringPiece(const char* offset, int len) : ptr_(offset), length_(len) { }
00051 
00052   // data() may return a pointer to a buffer with embedded NULs, and the
00053   // returned buffer may or may not be null terminated.  Therefore it is
00054   // typically a mistake to pass data() to a routine that expects a NUL
00055   // terminated string.
00056   const char* data() const { return ptr_; }
00057   int size() const { return length_; }
00058   int length() const { return length_; }
00059   bool empty() const { return length_ == 0; }
00060

00063     length_ = 0;
00064   }
00065   void set(const char* data, int len) {
00066     ptr_ = data;
00067     length_ = len;
00068   }
00069   void set(const char* str) {
00070     ptr_ = str;
00071     if (str != NULL)
00072       length_ = static_cast<int>(strlen(str));
00073     else

00109   // Does "this" start with "x"
00110   bool starts_with(const StringPiece& x) const {
00111     return ((length_ >= x.length_) &&
00112             (memcmp(ptr_, x.ptr_, x.length_) == 0));
