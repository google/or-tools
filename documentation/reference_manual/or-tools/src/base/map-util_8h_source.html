00022 // Perform a lookup in a map or hash_map.
00023 // If the key is present in the map then the value associated with that
00024 // key is returned, otherwise the value passed as a default is returned.
00025 template <class Collection>
00026 const typename Collection::value_type::second_type&
00027 FindWithDefault(const Collection& collection,
00028                 const typename Collection::value_type::first_type& key,
00029                 const typename Collection::value_type::second_type& value) {
00030   typename Collection::const_iterator it = collection.find(key);
00031   if (it == collection.end()) {
00032     return value;
00033   }
00034   return it->second;
00035 }

00081 // Change the value associated with a particular key in a map or hash_map.
00082 // If the key is not present in the map the key and value are inserted,
00083 // otherwise the value is updated to be a copy of the value provided.
00084 // True indicates that an insert took place, false indicates an update.
00085 template <class Collection, class Key, class Value>
00086 bool InsertOrUpdate(Collection * const collection,
00087                    const Key& key, const Value& value) {
00088   std::pair<typename Collection::iterator, bool> ret =
00089       collection->insert(typename Collection::value_type(key, value));
00090   if (!ret.second) {
00091     // update
00092     ret.first->second = value;
00093     return false;
00094   }
00095   return true;
00096 }

00130 // Perform a lookup in map or hash_map.
00131 // If the key is present and value is non-NULL then a copy of the value
00132 // associated with the key is made into *value. Returns whether key was present.
00133 template <class Collection, class Key, class Value>
00134 bool FindCopy(const Collection& collection,

00136               Value* const value) {
00137   typename Collection::const_iterator it = collection.find(key);
00138   if (it == collection.end()) {
00139     return false;
