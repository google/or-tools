00052       if (left > (kint64max / right)) {
00053         return kint64max;

00055     }  else { /* left positive, right non-positive */
00056       if (right < (kint64min / left)) {
00057         return kint64min;
00058       }
00059     } /* left positive, right non-positive */
00060   } else { /* left is non-positive */
00061     if (right > 0) { /* left is non-positive, right is positive */
00062       if (left < (kint64min / right)) {
00063         return kint64min;
00064       }
00065     } else { /* left and right are non-positive */

00067         return kint64max;
00068       }
00069     } /* end if left and right are non-positive */

01723   CHECK(!in_process_);

01732     for (SimpleRevFIFO<Demon*>::Iterator it(&bound_demons_); it.ok(); ++it) {

01737     for (SimpleRevFIFO<Demon*>::Iterator it(&range_demons_); it.ok(); ++it) {

01742     Enqueue(*it);

01745   clear_queue_action_on_fail();
01746   ClearInProcess();

01748   old_max_ = max_.Value();
01749   if (min_.Value() < new_min_) {
01750     SetMin(new_min_);
01751   }
01752   if (max_.Value() > new_max_) {
01753     SetMax(new_max_);
01754   }
01755   if (bits_ != NULL) {

03732 // Propagates set_min on left * right, left and right >= 0.
03733 void SetPosPosMinExpr(IntExpr* const left, IntExpr* const right, int64 m) {
03734   DCHECK_GE(left->Min(), 0);
03735   DCHECK_GE(right->Min(), 0);
03736   const int64 lmax = left->Max();
03737   const int64 rmax = right->Max();
03738   if (m > CapProd(lmax, rmax)) {
03739     left->solver()->Fail();
03740   }
03741   if (m > CapProd(left->Min(), right->Min())) {
03742     // Ok for m == 0 due to left and right being positive
03743     if (0 != rmax) {
03744       left->SetMin(PosIntDivUp(m, rmax));
03745     }
03746     if (0 != lmax) {
03747       right->SetMin(PosIntDivUp(m, lmax));
03748     }
03749   }
03750 }
03751 
03752 // Propagates set_max on left * right, left and right >= 0.
03753 void SetPosPosMaxExpr(IntExpr* const left, IntExpr* const right, int64 m) {
03754   DCHECK_GE(left->Min(), 0);
03755   DCHECK_GE(right->Min(), 0);
03756   const int64 lmin = left->Min();
03757   const int64 rmin = right->Min();
03758   if (m < CapProd(lmin, rmin)) {
03759     left->solver()->Fail();
03760   }
03761   if (m < CapProd(left->Max(), right->Max())) {
03762     if (0 != lmin) {
03763       right->SetMax(PosIntDivDown(m, lmin));
03764     }
03765     if (0 != rmin) {
03766       left->SetMax(PosIntDivDown(m, rmin));
03767     }
03768     // else do nothing: 0 is supporting any value from other expr.
03769   }
03770 }
03771 
03772 // Propagates set_min on left * right, left >= 0, right across 0.
03773 void SetPosGenMinExpr(IntExpr* const left, IntExpr* const right, int64 m) {
03774   DCHECK_GE(left->Min(), 0);
03775   DCHECK_GT(right->Max(), 0);
03776   DCHECK_LT(right->Min(), 0);
03777   const int64 lmax = left->Max();
03778   const int64 rmax = right->Max();
03779   if (m > CapProd(lmax, rmax)) {
03780     left->solver()->Fail();
03781   }
03782   DCHECK_GT(left->Max(), 0);
03783   if (m > 0) {  // We deduce right > 0.
03784     left->SetMin(PosIntDivUp(m, rmax));
03785     right->SetMin(PosIntDivUp(m, lmax));
03786   } else if (m == 0) {
03787     const int64 lmin = left->Min();
03788     if (lmin > 0) {
03789       right->SetMin(0);
03790     }
03791   } else {  // m < 0
03792     const int64 lmin = left->Min();
03793     if (0 != lmin) {  // We cannot deduce anything if 0 is in the domain.
03794       right->SetMin(-PosIntDivDown(-m, lmin));
03795     }
03796   }
03797 }
03798 
03799 // Propagates set_min on left * right, left and right across 0.
03800 void SetGenGenMinExpr(IntExpr* const left, IntExpr* const right, int64 m) {
03801   DCHECK_LT(left->Min(), 0);
03802   DCHECK_GT(left->Max(), 0);
03803   DCHECK_GT(right->Max(), 0);
03804   DCHECK_LT(right->Min(), 0);
03805   const int64 lmin = left->Min();
03806   const int64 lmax = left->Max();
03807   const int64 rmin = right->Min();
03808   const int64 rmax = right->Max();
03809   if (m > std::max(CapProd(lmin, rmin), CapProd(lmax, rmax))) {
03810     left->solver()->Fail();
03811   }
03812   if (m > lmin * rmin) {  // Must be positive section * positive section.
03813     left->SetMin(PosIntDivUp(m, rmax));
03814     right->SetMin(PosIntDivUp(m, lmax));
03815   } else if (m > CapProd(lmax, rmax)) {  // Negative section * negative section.
03816     left->SetMax(-PosIntDivUp(m, -rmin));
03817     right->SetMax(-PosIntDivUp(m, -lmin));
03818   }
03819 }

03821 void TimesSetMin(IntExpr* const left,
03822                  IntExpr* const right,
03823                  IntExpr* const minus_left,
03824                  IntExpr* const minus_right,
03825                  int64 m) {
03826   if (left->Min() >= 0) {
03827     if (right->Min() >= 0) {
03828       SetPosPosMinExpr(left, right, m);
03829     } else if (right->Max() <= 0) {
03830       SetPosPosMaxExpr(left, minus_right, -m);
03831     } else {  // right->Min() < 0 && right->Max() > 0
03832       SetPosGenMinExpr(left, right, m);
03833     }
03834   } else if (left->Max() <= 0) {
03835     if (right->Min() >= 0) {
03836       SetPosPosMaxExpr(right, minus_left, -m);
03837     } else if (right->Max() <= 0) {
03838       SetPosPosMinExpr(minus_left, minus_right, m);
03839     } else {  // right->Min() < 0 && right->Max() > 0
03840       SetPosGenMinExpr(minus_left, minus_right, m);
03841     }
03842   } else if (right->Min() >= 0) {  // left->Min() < 0 && left->Max() > 0
03843     SetPosGenMinExpr(right, left, m);
03844   } else if (right->Max() <= 0) {  // left->Min() < 0 && left->Max() > 0
03845     SetPosGenMinExpr(minus_right, minus_left, m);
03846   } else {  // left->Min() < 0 && left->Max() > 0 &&
03847             // right->Min() < 0 && right->Max() > 0
03848     SetGenGenMinExpr(left, right, m);
03849   }
03850 }

05267 // ----- Convex Piecewise -----
05268 
05269 // This class is a very simple convex piecewise linear function.  The
05270 // argument of the function is the expression.  Between early_date and
05271 // late_date, the value of the function is 0.  Before early date, it
05272 // is affine and the cost is early_cost * (early_date - x). After
05273 // late_date, the cost is late_cost * (x - late_date).
05274 
05275 class SimpleConvexPiecewiseExpr : public BaseIntExpr {
05276  public:
05277   SimpleConvexPiecewiseExpr(Solver* const s,
05278                             IntExpr* const e,
05279                             int64 ec, int64 ed,
05280                             int64 ld, int64 lc)
05281       : BaseIntExpr(s),
05282     expr_(e),
05283     early_cost_(ec),
05284     early_date_(ec == 0 ? kint64min : ed),
05285     late_date_(lc == 0 ? kint64max : ld),
05286     late_cost_(lc) {
05287         DCHECK_GE(ec, 0LL);
05288         DCHECK_GE(lc, 0LL);
05289         DCHECK_GE(ld, ed);
05290 
05291         // If the penalty is 0, we can push the "confort zone or zone
05292         // of no cost towards infinity.
05293   }
05294 
05295   virtual ~SimpleConvexPiecewiseExpr() {}
05296 
05297   virtual int64 Min() const {
05298     const int64 vmin = expr_->Min();
05299     const int64 vmax = expr_->Max();
05300     if (vmin >= late_date_) {
05301       return (vmin - late_date_) * late_cost_;
05302     } else if (vmax <= early_date_) {
05303       return (early_date_ - vmax) * early_cost_;
05304     } else {
05305       return 0LL;
05306     }
05307   }
05308 
05309   virtual void SetMin(int64 m) {
05310     if (m <= 0) {
05311       return;
05312     }
05313     const int64 vmin = expr_->Min();
05314     const int64 vmax = expr_->Max();
05315     const int64 rb = (late_cost_ == 0 ?
05316                       vmax :
05317                       late_date_ + PosIntDivUp(m , late_cost_) - 1);
05318     const int64 lb = (early_cost_ == 0 ?
05319                       vmin :
05320                       early_date_ - PosIntDivUp(m , early_cost_) + 1);
05321     if (expr_->IsVar()) {
05322       expr_->Var()->RemoveInterval(lb, rb);
05323     }
05324   }
05325 
05326   virtual int64 Max() const {
05327     const int64 vmin = expr_->Min();
05328     const int64 vmax = expr_->Max();
05329     const int64 mr = vmax > late_date_ ? (vmax - late_date_) * late_cost_ : 0;
05330     const int64 ml = vmin < early_date_
05331         ? (early_date_ - vmin) * early_cost_
05332         : 0;
05333     return std::max(mr, ml);
05334   }
05335 
05336   virtual void SetMax(int64 m) {
05337     if (m < 0) {
05338       solver()->Fail();
05339     }
05340     if (late_cost_ != 0LL) {
05341       const int64 rb = late_date_ + PosIntDivDown(m, late_cost_);
05342       if (early_cost_ != 0LL) {
05343         const int64 lb = early_date_ - PosIntDivDown(m, early_cost_);
05344         expr_->SetRange(lb, rb);
05345       } else {
05346         expr_->SetMax(rb);
05347       }
05348     } else {
05349       if (early_cost_ != 0LL) {
05350         const int64 lb = early_date_ - PosIntDivDown(m, early_cost_);
05351         expr_->SetMin(lb);
05352       }
05353     }
05354   }
05355 
05356   virtual string name() const {
05357     return StringPrintf(
05358         "ConvexPiecewiseExpr(%s, ec = %" GG_LL_FORMAT "d, ed = %"
05359         GG_LL_FORMAT "d, ld = %" GG_LL_FORMAT "d, lc = %" GG_LL_FORMAT "d)",
05360         expr_->name().c_str(),
05361         early_cost_, early_date_, late_date_, late_cost_);
05362   }

05365     return StringPrintf(
05366         "ConvexPiecewiseExpr(%s, ec = %" GG_LL_FORMAT "d, ed = %"
05367         GG_LL_FORMAT "d, ld = %" GG_LL_FORMAT "d, lc = %" GG_LL_FORMAT "d)",
05368         expr_->DebugString().c_str(),
05369         early_cost_, early_date_, late_date_, late_cost_);
05370   }
05371 
05372   virtual void WhenRange(Demon* d) {
05373     expr_->WhenRange(d);
05374   }
05375 
05376   virtual void Accept(ModelVisitor* const visitor) const {
05377     visitor->BeginVisitIntegerExpression(ModelVisitor::kConvexPiecewise, this);
05378     visitor->VisitIntegerExpressionArgument(ModelVisitor::kExpressionArgument,
05379                                             expr_);
05380     visitor->VisitIntegerArgument(ModelVisitor::kEarlyCostArgument,
05381                                   early_cost_);
05382     visitor->VisitIntegerArgument(ModelVisitor::kEarlyDateArgument,
05383                                   early_date_);
05384     visitor->VisitIntegerArgument(ModelVisitor::kLateCostArgument,
05385                                   late_cost_);
05386     visitor->VisitIntegerArgument(ModelVisitor::kLateDateArgument,
05387                                   late_date_);
05388     visitor->EndVisitIntegerExpression(ModelVisitor::kConvexPiecewise, this);

05933 IntExpr* Solver::MakeSum(IntExpr* const l, IntExpr* const r) {
05934   CHECK_EQ(this, l->solver());
05935   CHECK_EQ(this, r->solver());
05936   if (r->Bound()) {
05937     return MakeSum(l, r->Min());
05938   }
05939   if (l->Bound()) {
05940     return MakeSum(r, l->Min());
05941   }
05942   if (l == r) {
05943     return MakeProd(l, 2);
05944   }
05945   IntExpr* cache =
05946       model_cache_->FindExprExprExpression(l, r, ModelCache::EXPR_EXPR_SUM);
05947   if (cache == NULL) {
05948     cache =
05949         model_cache_->FindExprExprExpression(r, l, ModelCache::EXPR_EXPR_SUM);
05950   }
05951   if (cache != NULL) {
05952     return cache;
05953   } else {
05954     IntExpr* const result =
05955         RegisterIntExpr(RevAlloc(new PlusIntExpr(this, l, r)));
05956     model_cache_->InsertExprExprExpression(
05957         result,
05958         l,
05959         r,
05960         ModelCache::EXPR_EXPR_SUM);
05961     return result;
05962   }

05965 IntExpr* Solver::MakeSum(IntExpr* const e, int64 v) {
05966   CHECK_EQ(this, e->solver());
05967   if (e->Bound()) {
05968     return MakeIntConst(e->Min() + v);
05969   }
05970   if (v == 0) {
05971     return e;
05972   }
05973   IntExpr* result = Cache()->FindExprConstantExpression(
05974       e, v, ModelCache::EXPR_CONSTANT_SUM);
05975   if (result == NULL) {
05976     if (e->IsVar() &&
05977         !AddOverflows(v, e->Max()) &&
05978         !AddUnderflows(v, e->Min())) {
05979       IntVar* const var = e->Var();
05980       switch (var->VarType()) {
05981         case DOMAIN_INT_VAR: {
05982           result = RegisterIntExpr(RevAlloc(
05983               new PlusCstDomainIntVar(
05984                   this, reinterpret_cast<DomainIntVar*>(var), v)));
05985           break;
05986         }
05987         case CONST_VAR: {
05988           result = RegisterIntExpr(MakeIntConst(var->Min() + v));
05989           break;
05990         }
05991         case VAR_ADD_CST: {
05992           PlusCstVar* const add_var = reinterpret_cast<PlusCstVar*>(var);
05993           IntVar* const sub_var = add_var->SubVar();
05994           const int64 new_constant = v + add_var->Constant();
05995           if (new_constant == 0) {
05996             result = sub_var;
05997           } else {
05998             if (sub_var->VarType() == DOMAIN_INT_VAR) {
05999               DomainIntVar* const dvar =
06000                   reinterpret_cast<DomainIntVar*>(sub_var);
06001               result = RegisterIntExpr(
06002                   RevAlloc(new PlusCstDomainIntVar(this, dvar, new_constant)));
06003             } else {
06004               result = RegisterIntExpr(
06005                   RevAlloc(new PlusCstIntVar(this, sub_var, new_constant)));
06006             }
06007           }
06008           break;
06009         }
06010         case CST_SUB_VAR: {
06011           SubCstIntVar* const add_var = reinterpret_cast<SubCstIntVar*>(var);
06012           IntVar* const sub_var = add_var->SubVar();
06013           const int64 new_constant = v - add_var->Constant();
06014           result =
06015               RegisterIntExpr(new SubCstIntVar(this, sub_var, new_constant));
06016           break;
06017         }
06018         case OPP_VAR: {
06019           OppIntVar* const add_var = reinterpret_cast<OppIntVar*>(var);
06020           IntVar* const sub_var = add_var->SubVar();
06021           result = RegisterIntExpr(new SubCstIntVar(this, sub_var, v));
06022           break;
06023         }
06024         default:
06025           result = RegisterIntExpr(RevAlloc(new PlusCstIntVar(this, var, v)));
06026       }
06027     } else {
06028       result = RegisterIntExpr(RevAlloc(new PlusIntCstExpr(this, e, v)));
06029     }
06030     Cache()->InsertExprConstantExpression(
06031         result, e, v, ModelCache::EXPR_CONSTANT_SUM);
06032   }
06033   return result;
06034 }

06058   if (e->Bound()) {
06059     return MakeIntConst(v - e->Min());
06060   }
06061   if (v == 0) {
06062     return MakeOpposite(e);
06063   }
06064   IntExpr* result = Cache()->FindExprConstantExpression(
06065       e, v, ModelCache::EXPR_CONSTANT_DIFFERENCE);
06066   if (result == NULL) {
06067     if (e->IsVar() &&
06068         e->Min() != kint64min &&
06069         !SubOverflows(v, e->Min()) &&
06070         !SubUnderflows(v, e->Max())) {
06071       IntVar* const var = e->Var();
06072       switch (var->VarType()) {
06073         case VAR_ADD_CST: {
06074           PlusCstVar* const add_var = reinterpret_cast<PlusCstVar*>(var);
06075           IntVar* const sub_var = add_var->SubVar();
06076           const int64 new_constant = v - add_var->Constant();
06077           if (new_constant == 0) {
06078             result = sub_var;
06079           } else {
06080             result = RegisterIntExpr(
06081                 RevAlloc(new SubCstIntVar(this, sub_var, new_constant)));
06082           }
06083           break;
06084         }
06085         case CST_SUB_VAR: {
06086           SubCstIntVar* const add_var = reinterpret_cast<SubCstIntVar*>(var);
06087           IntVar* const sub_var = add_var->SubVar();
06088           const int64 new_constant = v - add_var->Constant();
06089           result = MakeSum(sub_var, new_constant);
06090           break;
06091         }
06092         case OPP_VAR: {
06093           OppIntVar* const add_var = reinterpret_cast<OppIntVar*>(var);
06094           IntVar* const sub_var = add_var->SubVar();
06095           result = MakeSum(sub_var, v);
06096           break;
06097         }
06098         default:
06099           result = RegisterIntExpr(RevAlloc(new SubCstIntVar(this, var, v)));
06100       }
06101     } else {
06102       result = RegisterIntExpr(RevAlloc(new SubIntCstExpr(this, e, v)));
06103     }
06104     Cache()->InsertExprConstantExpression(
06105         result, e, v, ModelCache::EXPR_CONSTANT_DIFFERENCE);
06106   }
06107   return result;
06108 }

06131   if (result != NULL) {
06132     return result;
06133   } else {
06134     if (e->Bound()) {
06135       return MakeIntConst(v * e->Min());
06136     } else if (v == 1) {
06137       return e;
06138     } else if (v == -1) {
06139       return MakeOpposite(e);
06140     } else if (v > 0) {
06141       if (e->Max() > kint64max / v || e->Min() < kint64min / v) {

06143             RegisterIntExpr(RevAlloc(new SafeTimesIntPosCstExpr(this, e, v)));

06149     } else {  // v < 0.
06150       result = RegisterIntExpr(RevAlloc(new TimesIntNegCstExpr(this, e, v)));
06151     }
06152     if (e->IsVar() && !FLAGS_cp_disable_expression_optimization) {
06153       result = result->Var();

06155     Cache()->InsertExprConstantExpression(
06156         result, e, v, ModelCache::EXPR_CONSTANT_PROD);
06157     return result;

06161 IntExpr* Solver::MakeProd(IntExpr* const l, IntExpr* const r) {
06162   if (l->Bound()) {
06163     return MakeProd(r, l->Min());
06164   }
06165 
06166   if (r->Bound()) {
06167     return MakeProd(l, r->Min());
06168   }
06169 
06170   // ----- Discover squares and powers -----
06171 
06172   IntExpr* left = l;
06173   IntExpr* right = r;
06174   int64 left_exponant = 1;
06175   int64 right_exponant = 1;
06176   if (dynamic_cast<BasePower*>(l) != NULL) {
06177     BasePower* const left_power = dynamic_cast<BasePower*>(l);
06178     left = left_power->expr();
06179     left_exponant = left_power->exponant();
06180   }
06181   if (dynamic_cast<IntSquare*>(l) != NULL) {
06182     IntSquare* const left_power = dynamic_cast<IntSquare*>(l);
06183     left = left_power->expr();
06184     left_exponant = 2;
06185   }
06186   if (left->IsVar()) {
06187     IntVar* const left_var = left->Var();
06188     IntExpr* const left_sub = CastExpression(left_var);
06189     if (left_sub != NULL && dynamic_cast<BasePower*>(left_sub) != NULL) {
06190       BasePower* const left_power = dynamic_cast<BasePower*>(left_sub);
06191       left = left_power->expr();
06192       left_exponant = left_power->exponant();
06193     }
06194     if (left_sub != NULL && dynamic_cast<IntSquare*>(left_sub) != NULL) {
06195       IntSquare* const left_power = dynamic_cast<IntSquare*>(left_sub);
06196       left = left_power->expr();
06197       left_exponant = 2;
06198     }
06199   }
06200   if (dynamic_cast<BasePower*>(r) != NULL) {
06201     BasePower* const right_power = dynamic_cast<BasePower*>(l);
06202     right = right_power->expr();
06203     right_exponant = right_power->exponant();
06204   }
06205   if (dynamic_cast<IntSquare*>(r) != NULL) {
06206     IntSquare* const right_power = dynamic_cast<IntSquare*>(l);
06207     right = right_power->expr();
06208     right_exponant = 2;
06209   }
06210   if (right->IsVar()) {
06211     IntVar* const right_var = right->Var();
06212     IntExpr* const right_sub = CastExpression(right_var);
06213     if (right_sub != NULL && dynamic_cast<BasePower*>(right_sub) != NULL) {
06214       BasePower* const right_power = dynamic_cast<BasePower*>(right_sub);
06215       right = right_power->expr();
06216       right_exponant = right_power->exponant();
06217     }
06218     if (right_sub != NULL && dynamic_cast<IntSquare*>(right_sub) != NULL) {
06219       IntSquare* const right_power = dynamic_cast<IntSquare*>(right_sub);
06220       right = right_power->expr();
06221       right_exponant = 2;
06222     }
06223   }
06224 
06225   if (left == right) {
06226     return MakePower(left, left_exponant + right_exponant);
06227   }
06228 
06229   // ----- Discover nested products -----
06230 
06231   left = l;
06232   right = r;
06233   int64 coefficient = 1;
06234   bool modified = false;
06235   if (dynamic_cast<TimesCstIntVar*>(left) != NULL) {
06236     TimesCstIntVar* const left_prod = dynamic_cast<TimesCstIntVar*>(left);
06237     coefficient *= left_prod->Constant();
06238     left = left_prod->Var();
06239     modified = true;
06240   }
06241   if (dynamic_cast<TimesIntCstExpr*>(left) != NULL) {
06242     TimesIntCstExpr* const left_prod = dynamic_cast<TimesIntCstExpr*>(left);
06243     coefficient *= left_prod->Constant();
06244     left = left_prod->Expr();
06245     modified = true;
06246   }
06247   if (dynamic_cast<TimesCstIntVar*>(right) != NULL) {
06248     TimesCstIntVar* const right_prod = dynamic_cast<TimesCstIntVar*>(right);
06249     coefficient *= right_prod->Constant();
06250     right = right_prod->Var();
06251     modified = true;
06252   }
06253   if (dynamic_cast<TimesIntCstExpr*>(right) != NULL) {
06254     TimesIntCstExpr* const right_prod = dynamic_cast<TimesIntCstExpr*>(right);
06255     coefficient *= right_prod->Constant();
06256     right = right_prod->Expr();
06257     modified = true;
06258   }
06259   if (modified) {
06260     return MakeProd(MakeProd(left, right), coefficient);
06261   }
06262 
06263   // ----- Standart buuild -----
06264 
06265   CHECK_EQ(this, l->solver());
06266   CHECK_EQ(this, r->solver());
06267   IntExpr* result =
06268       model_cache_->FindExprExprExpression(l, r, ModelCache::EXPR_EXPR_PROD);
06269   if (result == NULL) {
06270     result =
06271         model_cache_->FindExprExprExpression(r, l, ModelCache::EXPR_EXPR_PROD);
06272   }
06273   if (result != NULL) {
06274     return result;
06275   }
06276   if (l->IsVar() && l->Var()->VarType() == BOOLEAN_VAR) {
06277     if (r->Min() >= 0) {
06278       result = RegisterIntExpr(RevAlloc(
06279           new TimesBooleanPosIntExpr(this,
06280                                      reinterpret_cast<BooleanVar*>(l), r)));
06281     } else {
06282       result = RegisterIntExpr(RevAlloc(
06283           new TimesBooleanIntExpr(this, reinterpret_cast<BooleanVar*>(l), r)));
06284     }
06285   } else if (r->IsVar() &&
06286       reinterpret_cast<IntVar*>(r)->VarType() == BOOLEAN_VAR) {
06287     if (l->Min() >= 0) {
06288       result = RegisterIntExpr(RevAlloc(
06289           new TimesBooleanPosIntExpr(this,
06290                                      reinterpret_cast<BooleanVar*>(r), l)));
06291     } else {
06292       result = RegisterIntExpr(RevAlloc(
06293           new TimesBooleanIntExpr(this, reinterpret_cast<BooleanVar*>(r), l)));
06294     }
06295   } else if (l->Min() >= 0 && r->Min() >= 0) {
06296     if (CapProd(l->Max(), r->Max()) == kint64max) {  // Potential overflow.
06297       result = RegisterIntExpr(RevAlloc(new SafeTimesIntPosExpr(this, l, r)));
06298     } else {
06299       result = RegisterIntExpr(RevAlloc(new TimesIntPosExpr(this, l, r)));
06300     }
06301   } else {
06302     result = RegisterIntExpr(RevAlloc(new TimesIntExpr(this, l, r)));
06303   }
06304   model_cache_->InsertExprExprExpression(
06305       result,
06306       l,
06307       r,
06308       ModelCache::EXPR_EXPR_PROD);
06309   return result;
06310 }

06312 IntExpr* Solver::MakeDiv(IntExpr* const numerator, IntExpr* const denominator) {
06313   if (denominator->Min() > 0) {
06314     return RevAlloc(new DivIntPosExpr(this, numerator, denominator));
06315   }
06316 
06317   // Both numerator and denominator are positive.
06318   // Denominator needs to be != 0.
06319   AddConstraint(MakeGreater(denominator, 0));
06320   IntVar* const result = MakeIntVar(0, numerator->Max());
06321   IntExpr* const product = MakeProd(denominator, result);
06322   // The integer division result = numerator / denominator means
06323   // numerator = result * denominator + quotient, with quotient < denominator.
06324   // This translates into:
06325   //     numerator >= denominator * result
06326   AddConstraint(MakeGreaterOrEqual(numerator->Var(), product->Var()));
06327   //     numerator < denominator * result + denominator
06328   IntExpr* const product_up = RegisterIntExpr(MakeSum(product, denominator));
06329   AddConstraint(MakeLess(numerator->Var(), product_up->Var()));
06330   return result;
06331 }

06541 void IntVar::RemoveValues(const int64* const values, int size) {
06542   DCHECK_GE(size, 0);
06543   if (size == 0) {
06544     return;
06545   }
06546   int start_index = 0;
06547   int64 new_min = Min();
06548   if (values[start_index] <= new_min) {
06549     while (start_index < size - 1 &&
06550            values[start_index + 1] == values[start_index] + 1) {
06551       new_min = values[start_index + 1] + 1;
06552       start_index++;
06553     }
06554   }
06555   int end_index = size - 1;
06556   int64 new_max = Max();
06557   if (values[end_index] >= new_max) {
06558     while (end_index > start_index + 1 &&
06559            values[end_index - 1] == values[end_index] - 1) {
06560       new_max = values[end_index - 1] - 1;
06561       end_index--;
06562     }
06563   }
06564   SetRange(new_min, new_max);
06565   for (int i = start_index; i <= end_index; ++i) {
06566     RemoveValue(values[i]);
06567   }
06568 }
