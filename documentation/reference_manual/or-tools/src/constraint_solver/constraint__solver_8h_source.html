00022 //   Knowing that we see 56 legs and 20 heads, how many pheasants and rabbits

00031 //     IntExpr* const heads = s.MakeSum(p, r);
00032 //     Constraint* const ct_legs = s.MakeEquality(legs, 56);

00037 //                                             Solver::CHOOSE_FIRST_UNBOUND,
00038 //                                             Solver::ASSIGN_MIN_VALUE);

00040 //     CHECK(s.NextSolution());

04501 template <class V, class E> class AssignmentContainer {
04502  public:
04503   AssignmentContainer() {}
04504   E& Add(V* const var) {
04505     CHECK_NOTNULL(var);
04506     int index = -1;
04507     if (!Find(var, &index)) {
04508       return FastAdd(var);
04509     } else {
04510       return elements_[index];
04511     }
04512   }
04513   // Adds element without checking its presence in the container.
04514   E& FastAdd(V* const var) {
04515     DCHECK(var != NULL);
04516     E e(var);
04517     elements_.push_back(e);
04518     return elements_.back();
04519   }
04520   void Clear() {
04521     elements_.clear();
04522     if (!elements_map_.empty()) {  // 2x speedup on or-tools.
04523       elements_map_.clear();
04524     }
04525   }
04526   bool Empty() const {
04527     return elements_.empty();
04528   }
04529   // Copies intersection of containers.
04530   void Copy(const AssignmentContainer<V, E>& container) {
04531     for (int i = 0; i < container.elements_.size(); ++i) {
04532       const E& element = container.elements_[i];
04533       const V* const var = element.Var();
04534       int index = -1;
04535       if (i < elements_.size() && elements_[i].Var() == var) {
04536         index = i;
04537       } else if (!Find(var, &index)) {
04538         continue;
04539       }
04540       DCHECK_GE(index, 0);
04541       E& local_element(elements_[index]);
04542       local_element.Copy(element);
04543       if (element.Activated()) {
04544         local_element.Activate();
04545       } else {
04546         local_element.Deactivate();
04547       }
04548     }
04549   }
04550   bool Contains(const V* const var) const {
04551     int index;
04552     return Find(var, &index);
04553   }
04554   E& MutableElement(const V* const var) {
04555     int index = -1;
04556     const bool found = Find(var, &index);
04557     CHECK(found) << "Unknown variable " << var->DebugString() << " in solution";
04558     return MutableElement(index);
04559   }
04560   const E& Element(const V* const var) const {
04561     int index = -1;
04562     const bool found = Find(var, &index);
04563     CHECK(found) << "Unknown variable " << var->DebugString() << " in solution";
04564     return Element(index);
04565   }
04566   const std::vector<E>& elements() const { return elements_; }
04567   E& MutableElement(int index) { return elements_[index]; }
04568   const E& Element(int index) const { return elements_[index]; }
04569   int Size() const { return elements_.size(); }
04570   void Store() {
04571     for (int i = 0; i < elements_.size(); ++i) {
04572       elements_[i].Store();
04573     }
04574   }
04575   void Restore() {
04576     for (int i = 0; i < elements_.size(); ++i) {
04577       E& element = elements_[i];
04578       if (element.Activated()) {
04579         element.Restore();
04580       }
04581     }
04582   }
04583 
04584   // Returns true if this and 'container' both represent the same V* -> E map.
04585   // Runs in linear time; requires that the == operator on the type E is well
04586   // defined.
04587   bool operator==(const AssignmentContainer<V, E>& container) const {
04588     // We may not have any work to do
04589     if (Size() != container.Size()) {
04590       return false;
04591     }
04592     // The == should be order-independent
04593     EnsureMapIsUpToDate();
04594     // Do not use the hash_map::== operator! It does not just compare content,
04595     // but also how the map is hashed (e.g., number of buckets). This is not
04596     // what we want.
04597     typedef ConstIter<std::vector<E> > Iterator;
04598     for (Iterator it(container.elements_); !it.at_end(); ++it) {
04599       const int position = FindWithDefault(elements_map_, it->Var(), -1);
04600       if (position < 0 || elements_[position] != *it) {
04601         return false;
04602       }
04603     }
04604     return true;
04605   }
04606   bool operator!=(const AssignmentContainer<V, E>& container) const {
04607     return !(*this == container);
04608   }
04609 
04610  private:
04611   void EnsureMapIsUpToDate() const {
04612     hash_map<const V*, int>* map =
04613         const_cast<hash_map<const V*, int>* >(&elements_map_);
04614     for (int i = map->size(); i < elements_.size(); ++i) {
04615       (*map)[elements_[i].Var()] = i;
04616     }
04617   }
04618   bool Find(const V* const var, int* index) const {
04619     EnsureMapIsUpToDate();
04620     DCHECK_EQ(elements_map_.size(), elements_.size());
04621     return FindCopy(elements_map_, var, index);
04622   }
04623 
04624   std::vector<E> elements_;
04625   hash_map<const V*, int> elements_map_;
04626 };
