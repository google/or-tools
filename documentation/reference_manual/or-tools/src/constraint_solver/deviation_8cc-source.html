00096   // Builds an assignment with minimal deviation and assign it to
00097   // scaled_vars_assigned_value_. It returns the minimal deviation:
00098   //   sum_i |scaled_vars_assigned_value_[i] - total_sum_|.

00108   // Propagates the upper and lower bounds of x[i]'s.
00109   // It assumes the constraint is consistent:
00110   //   - the sum constraint is consistent
00111   //   - min deviation smaller than max allowed deviation

00113   void PropagateBounds(int64 min_delta) {

00118   // Prunes the upper/lower-bound of vars. We apply a mirroing of the
00119   // domains wrt 0 to prune the lower bounds such that we can use the
00120   // same algo to prune both sides of the domains.  upperBounds = true
00121   // to prune the upper bounds of vars, false to prune the lower
00122   // bounds.
00123   void PropagateBounds(int64 min_delta, bool upper_bound) {
00124     // Builds greedy assignment.
00125     const int64 greedy_sum = BuildGreedySum(upper_bound);
00126     // Repairs assignment and store information to be used when pruning.
00127     RepairSumAndComputeInfo(greedy_sum);
00128     // Does the actual pruning.

00132   // Cache min and max values of variables.
00133   void ComputeData(bool upper_bound) {
00134     scaled_sum_max_ = 0;
00135     scaled_sum_min_ = 0;
00136     for (int i = 0; i < size_; ++i) {
00137       scaled_vars_max_[i] =
00138           size_ * (upper_bound ? vars_[i]->Max() : -vars_[i]->Min());
00139       scaled_vars_min_[i] =
00140           size_ * (upper_bound ? vars_[i]->Min() : -vars_[i]->Max());
00141       scaled_sum_max_ += scaled_vars_max_[i];
00142       scaled_sum_min_ += scaled_vars_min_[i];
00143     }
00144     active_sum_ = (!upper_bound ? -total_sum_ : total_sum_);
00145     // down is <= sum.
00146     active_sum_rounded_down_ =
00147         size_ * MathUtil::FloorOfRatio<int64>(active_sum_, size_);
00148     // up is > sum, always.
00149     active_sum_rounded_up_ = active_sum_rounded_down_ + size_;
00150     active_sum_nearest_ =
00151         (active_sum_rounded_up_ - active_sum_ <=
00152          active_sum_ - active_sum_rounded_down_) ?
00153         active_sum_rounded_up_ :
00154         active_sum_rounded_down_;
00155   }

00157   // Builds an approximate sum in a greedy way.
00158   int64 BuildGreedySum(bool upper_bound) {
00159     // Update data structure.
00160     ComputeData(upper_bound);
00161 
00162     // Number of constraint should be consistent.
00163     DCHECK_GE(size_ * active_sum_, scaled_sum_min_);
00164     DCHECK_LE(size_ * active_sum_, scaled_sum_max_);
00165 
00166     int64 sum = 0;
00167     // Greedily assign variable to nearest value to average.
00168     overlaps_.clear();
00169     for (int i = 0; i < size_; ++i) {
00170       if (scaled_vars_min_[i] >= active_sum_) {
00171         scaled_vars_assigned_value_[i] = scaled_vars_min_[i];
00172       } else if (scaled_vars_max_[i] <= active_sum_) {
00173         scaled_vars_assigned_value_[i] = scaled_vars_max_[i];
00174       } else {
00175         // Overlapping variable scaled_vars_min_[i] < active_sum_ <
00176         // scaled_vars_max_[i].
00177         scaled_vars_assigned_value_[i] = active_sum_nearest_;
00178         if (active_sum_ % size_ != 0) {
00179           overlaps_.push_back(i);
00180         }
00181       }
00182       sum += scaled_vars_assigned_value_[i];
00183     }
00184     DCHECK_EQ(0, active_sum_rounded_down_ % size_);
00185     DCHECK_LE(active_sum_rounded_down_, active_sum_);
00186     DCHECK_LT(active_sum_ - active_sum_rounded_down_, size_);
00187 
00188     return sum;
00189   }

00196   // Repairs the greedy sum obtained above to get the correct sum.
00197   void RepairGreedySum(int64 greedy_sum) {
00198     // Useful constant: scaled version of the sum.
00199     const int64 scaled_total_sum = size_ * active_sum_;
00200     // Step used to make the repair.
00201     const int64 delta = greedy_sum > scaled_total_sum ? -size_ : size_;
00202 
00203     // Change overlapping variables as long as the sum is not
00204     // satisfied and there are overlapping vars, we use that ones to
00205     // repair.
00206     for (int j = 0;
00207          j < overlaps_.size() && greedy_sum != scaled_total_sum;
00208          j++) {
00209       scaled_vars_assigned_value_[overlaps_[j]] += delta;
00210       greedy_sum += delta;
00211     }
00212     // Change other variables if the sum is still not satisfied.
00213     for (int i = 0; i < size_ && greedy_sum != scaled_total_sum; ++i) {
00214       const int64 old_scaled_vars_i = scaled_vars_assigned_value_[i];
00215       if (greedy_sum < scaled_total_sum) {
00216         // Increase scaled_vars_assigned_value_[i] as much as
00217         // possible to fix the too low sum.
00218         scaled_vars_assigned_value_[i] += scaled_total_sum - greedy_sum;
00219         scaled_vars_assigned_value_[i] =
00220             std::min(scaled_vars_assigned_value_[i], scaled_vars_max_[i]);
00221       } else {
00222         // Decrease scaled_vars_assigned_value_[i] as much as
00223         // possible to fix the too high sum.
00224         scaled_vars_assigned_value_[i] -= (greedy_sum - scaled_total_sum);
00225         scaled_vars_assigned_value_[i] =
00226             std::max(scaled_vars_assigned_value_[i], scaled_vars_min_[i]);
00227       }
00228       // Maintain the sum.
00229       greedy_sum += scaled_vars_assigned_value_[i] - old_scaled_vars_i;
00230     }
00231   }

00233   // Computes the maximum values of variables in the case the repaired
00234   // greedy sum is actually the active sum.
00235   void ComputeMaxWhenNoRepair() {
00236     int num_overlap_sum_rounded_up = 0;
00237     if (active_sum_nearest_ == active_sum_rounded_up_) {
00238       num_overlap_sum_rounded_up = overlaps_.size();
00239     }
00240     for (int i = 0; i < size_; ++i) {
00241       maximum_[i] = scaled_vars_assigned_value_[i];
00242       if (Overlap(i) &&
00243           active_sum_nearest_ == active_sum_rounded_up_ &&
00244           active_sum_ % size_ != 0) {
00245         overlaps_sup_[i] = num_overlap_sum_rounded_up - 1;
00246       } else {
00247         overlaps_sup_[i] = num_overlap_sum_rounded_up;
00248       }
00249     }

00252   // Returns the number of variables overlapping the average value,
00253   // assigned to // the average value rounded up that we can/need to
00254   // move.
00255   int ComputeNumOverlapsVariableRoundedUp() {
00256     if (active_sum_ % size_ == 0) {
00257       return 0;
00258     }
00259     int num_overlap_sum_rounded_up = 0;
00260     for (int i = 0; i < size_; ++i) {
00261       if (scaled_vars_assigned_value_[i] > scaled_vars_min_[i] &&
00262           scaled_vars_assigned_value_[i] == active_sum_rounded_up_) {
00263         num_overlap_sum_rounded_up++;
00264       }
00265     }
00266     return num_overlap_sum_rounded_up;

00269   // Returns whether we can push the greedy sum across the scaled
00270   // total sum in the same direction as going from the nearest rounded
00271   // sum to the farthest one.
00272   bool CanPushSumAcrossMean(int64 greedy_sum, int64 scaled_total_sum) const {
00273     return (greedy_sum > scaled_total_sum &&
00274             active_sum_nearest_ == active_sum_rounded_up_) ||
00275         (greedy_sum < scaled_total_sum &&
00276          active_sum_nearest_ == active_sum_rounded_down_);

00279   // Repairs the sum and store intermediate information to be used
00280   // during pruning.
00281   void RepairSumAndComputeInfo(int64 greedy_sum) {
00282     const int64 scaled_total_sum = size_ * active_sum_;
00283     // Computation of key values for the pruning:
00284     // - overlaps_sup_
00285     // - maximum_[i]
00286     if (greedy_sum == scaled_total_sum) {  // No repair needed.
00287       ComputeMaxWhenNoRepair();
00288     } else {  // Repair and compute maximums.
00289       // Try to repair the sum greedily.
00290       if (CanPushSumAcrossMean(greedy_sum, scaled_total_sum)) {
00291         const int64 delta = greedy_sum > scaled_total_sum ? -size_ : size_;
00292         for (int j = 0;
00293              j < overlaps_.size() && greedy_sum != scaled_total_sum;
00294              ++j) {
00295           scaled_vars_assigned_value_[overlaps_[j]] += delta;
00296           greedy_sum += delta;
00297         }
00298       }
00299 
00300       const int num_overlap_sum_rounded_up =
00301           ComputeNumOverlapsVariableRoundedUp();
00302 
00303       if (greedy_sum == scaled_total_sum) {
00304         // Greedy sum is repaired.
00305         for (int i = 0; i < size_; ++i) {
00306           if (Overlap(i) && num_overlap_sum_rounded_up > 0) {
00307             maximum_[i] = active_sum_rounded_up_;
00308             overlaps_sup_[i] = num_overlap_sum_rounded_up - 1;
00309           } else {
00310             maximum_[i] = scaled_vars_assigned_value_[i];
00311             overlaps_sup_[i] = num_overlap_sum_rounded_up;
00312           }
00313         }
00314       } else if (greedy_sum > scaled_total_sum) {
00315         // scaled_vars_assigned_value_[i] = active_sum_rounded_down_ or
00316         // scaled_vars_assigned_value_[i] <= total_sum
00317         // (there is no more num_overlap_sum_rounded_up).
00318         for (int i = 0; i < size_; ++i) {
00319           maximum_[i] = scaled_vars_assigned_value_[i];
00320           overlaps_sup_[i] = 0;
00321         }
00322       } else {  // greedy_sum < scaled_total_sum.
00323         for (int i = 0; i < size_; ++i) {
00324           if (Overlap(i) && num_overlap_sum_rounded_up > 0) {
00325             overlaps_sup_[i] = num_overlap_sum_rounded_up - 1;
00326           } else {
00327             overlaps_sup_[i] = num_overlap_sum_rounded_up;
00328           }
00329 
00330           if (scaled_vars_assigned_value_[i] < scaled_vars_max_[i]) {
00331             maximum_[i] =
00332                 scaled_vars_assigned_value_[i] + scaled_total_sum - greedy_sum;
00333           } else {
00334             maximum_[i] = scaled_vars_assigned_value_[i];
00335           }
00336         }
00337       }
00338     }
00339   }

00341   // Propagates onto variables with all computed data.
00342   void PruneVars(int64 min_delta, bool upper_bound) {
00343     // Pruning of upper bound of vars_[i] for var_index in [1..n].
00344     const int64 increase_down_up = (active_sum_rounded_up_ - active_sum_) -
00345         (active_sum_ - active_sum_rounded_down_);
00346     for (int var_index = 0; var_index < size_; ++var_index) {
00347       // Not bound, and a compatible new max.
00348       if (scaled_vars_max_[var_index] != scaled_vars_min_[var_index] &&
00349           maximum_[var_index] < scaled_vars_max_[var_index]) {
00350         const int64 new_max = ComputeNewMax(var_index,
00351                                             min_delta,
00352                                             increase_down_up);
00353         PruneBound(var_index, new_max, upper_bound);
00354       }
00355     }
00356   }

00358   // Computes new max for a variable.
00359   int64 ComputeNewMax(int var_index,
00360                       int64 min_delta,
00361                       int64 increase_down_up) {
00362     int64 maximum_value = maximum_[var_index];
00363     int64 current_min_delta = min_delta;
00364 
00365     if (overlaps_sup_[var_index] > 0 &&
00366         (current_min_delta +
00367          overlaps_sup_[var_index] * (size_ - increase_down_up) >=
00368          deviation_var_->Max())) {
00369       const int64 delta = deviation_var_->Max() - current_min_delta;
00370       maximum_value += (size_ * delta) / (size_ - increase_down_up);
00371       return MathUtil::FloorOfRatio<int64>(maximum_value, size_);
00372     } else {
00373       if (maximum_value == active_sum_rounded_down_ &&
00374           active_sum_rounded_down_ < active_sum_) {
00375         DCHECK_EQ(0, overlaps_sup_[var_index]);
00376         current_min_delta += size_ + increase_down_up;
00377         if (current_min_delta > deviation_var_->Max()) {
00378           DCHECK_EQ(0, maximum_value % size_);
00379           return maximum_value / size_;
00380         }
00381         maximum_value += size_;
00382       }
00383       current_min_delta +=
00384           overlaps_sup_[var_index] * (size_ - increase_down_up);
00385       maximum_value += size_ * overlaps_sup_[var_index];
00386       // Slope of 2 x n.
00387       const int64 delta = deviation_var_->Max() - current_min_delta;
00388       maximum_value +=  delta / 2;  // n * delta / (2 * n);
00389       return MathUtil::FloorOfRatio<int64>(maximum_value, size_);
00390     }
00391   }
