00533       const int64 target_value = target_.at(variable_index_);
00534       const int64 current_value = OldValue(variable_index_);
00535       if (current_value != target_value) {

00687 bool PathOperator::MoveChain(int64 before_chain,
00688                              int64 chain_end,
00689                              int64 destination) {
00690   if (CheckChainValidity(before_chain, chain_end, destination)
00691       && !IsPathEnd(chain_end)
00692       && !IsPathEnd(destination)) {
00693     const int64 destination_path = Path(destination);
00694     const int64 after_chain = Next(chain_end);
00695     SetNext(chain_end, Next(destination), destination_path);
00696     if (!ignore_path_vars_) {

00698       int next = Next(before_chain);
00699       while (current != chain_end) {
00700         SetNext(current, next, destination_path);
00701         current = next;
00702         next = Next(next);
00703       }
00704     } else {
00705       SetNext(destination, Next(before_chain), destination_path);
00706     }
00707     SetNext(before_chain, after_chain, Path(before_chain));
00708     return true;
00709   }

00713 bool PathOperator::ReverseChain(int64 before_chain,
00714                                 int64 after_chain,
00715                                 int64* chain_last) {
00716   if (CheckChainValidity(before_chain, after_chain, -1)) {
00717     int64 path = Path(before_chain);
00718     int64 current = Next(before_chain);
00719     if (current == after_chain) {
00720       return false;
00721     }
00722     int64 current_next = Next(current);
00723     SetNext(current, after_chain, path);
00724     while (current_next != after_chain) {
00725       const int64 next = Next(current_next);
00726       SetNext(current_next, current, path);
00727       current = current_next;
00728       current_next = next;
00729     }
00730     SetNext(before_chain, current, path);
00731     *chain_last = current;
00732     return true;
00733   }
00734   return false;
00735 }

00775 bool PathOperator::IncrementPosition() {
00776   const int base_node_size = base_nodes_.size();
00777   if (!just_started_) {
00778     const int number_of_paths = path_starts_.size();
00779     // Finding next base node positions.
00780     // Increment the position of inner base nodes first (higher index nodes);
00781     // if a base node is at the end of a path, reposition it at the start
00782     // of the path and increment the position of the preceding base node (this
00783     // action is called a restart).
00784     int last_restarted = base_node_size;
00785     for (int i = base_node_size - 1; i >= 0; --i) {
00786       if (base_nodes_[i] < number_of_nexts_) {
00787         base_nodes_[i] = OldNext(base_nodes_[i]);
00788         break;
00789       }
00790       base_nodes_[i] = StartNode(i);
00791       last_restarted = i;
00792     }
00793     // At the end of the loop, base nodes with indexes in
00794     // [last_restarted, base_node_size[ have been restarted.
00795     // Restarted base nodes are then repositioned by the virtual
00796     // GetBaseNodeRestartPosition to reflect position constraints between
00797     // base nodes (by default GetBaseNodeRestartPosition leaves the nodes
00798     // at the start of the path).
00799     // Base nodes are repositioned in ascending order to ensure that all
00800     // base nodes "below" the node being repositioned have their final
00801     // position.
00802     for (int i = last_restarted; i < base_node_size; ++i) {
00803       base_nodes_[i] = GetBaseNodeRestartPosition(i);
00804     }
00805     if (last_restarted > 0) {
00806       return CheckEnds();
00807     }
00808     // If all base nodes have been restarted, base nodes are moved to new paths.
00809     for (int i = base_node_size - 1; i >= 0; --i) {
00810       const int next_path_index = base_paths_[i] + 1;
00811       if (next_path_index < number_of_paths) {
00812         base_paths_[i] = next_path_index;
00813         base_nodes_[i] = path_starts_[next_path_index];
00814         if (i == 0 || !OnSamePathAsPreviousBase(i)) {
00815           return CheckEnds();
00816         }
00817       } else {
00818         base_paths_[i] = 0;
00819         base_nodes_[i] = path_starts_[0];
00820       }
00821     }
00822   } else {
00823     just_started_ = false;
00824     return true;
00825   }
00826   return CheckEnds();
00827 }

00829 void PathOperator::InitializePathStarts() {
00830   path_starts_.clear();
00831   Bitmap has_prevs(number_of_nexts_, false);
00832   for (int i = 0; i < number_of_nexts_; ++i) {
00833     const int next = OldNext(i);
00834     if (next < number_of_nexts_) {
00835       has_prevs.Set(next, true);
00836     }
00837   }
00838   for (int i = 0; i < number_of_nexts_; ++i) {
00839     if (!has_prevs.Get(i)) {
00840       path_starts_.push_back(i);
00841     }
00842   }
00843 }

00852 void PathOperator::InitializeBaseNodes() {
00853   InitializePathStarts();
00854   InitializeInactives();
00855   if (first_start_ || InitPosition()) {
00856     // Only do this once since the following starts will continue from the
00857     // preceding position
00858     for (int i = 0; i < base_nodes_.size(); ++i) {
00859       base_paths_[i] = 0;
00860       base_nodes_[i] = path_starts_[0];
00861     }
00862     first_start_ = false;
00863   }
00864   for (int i = 0; i < base_nodes_.size(); ++i) {
00865     // If base node has been made inactive, restart from path start.
00866     int64 base_node = base_nodes_[i];
00867     if (RestartAtPathStartOnSynchronize() || IsInactive(base_node)) {
00868       base_node = path_starts_[base_paths_[i]];
00869       base_nodes_[i] = base_node;
00870     }
00871     end_nodes_[i] = base_node;
00872   }
00873   // Repair end_nodes_ in case some must be on the same path and are not anymore
00874   // (due to other operators moving these nodes).
00875   for (int i = 1; i < base_nodes_.size(); ++i) {
00876     if (OnSamePathAsPreviousBase(i)
00877         && !OnSamePath(base_nodes_[i - 1], base_nodes_[i])) {
00878       const int64 base_node = base_nodes_[i -1];
00879       base_nodes_[i] = base_node;
00880       end_nodes_[i] = base_node;
00881     }
00882   }
00883   just_started_ = true;
00884 }

00886 bool PathOperator::OnSamePath(int64 node1, int64 node2) const {
00887   if (IsInactive(node1) != IsInactive(node2)) {
00888     return false;
00889   }
00890   for (int node = node1; !IsPathEnd(node); node = OldNext(node)) {
00891     if (node == node2) {
00892       return true;
00893     }
00894   }
00895   for (int node = node2; !IsPathEnd(node); node = OldNext(node)) {
00896     if (node == node1) {
00897       return true;
00898     }
00899   }
00900   return false;
00901 }

00903 // Rejects chain if chain_end is not after before_chain on the path or if
00904 // the chain contains exclude. Given before_chain is the node before the
00905 // chain, if before_chain and chain_end are the same the chain is rejected too.
00906 // Also rejects cycles (cycle detection is detected through chain length
00907 //  overflow).
00908 bool PathOperator::CheckChainValidity(int64 before_chain,
00909                                       int64 chain_end,
00910                                       int64 exclude) const {
00911   if (before_chain == chain_end || before_chain == exclude) return false;
00912   int64 current = before_chain;
00913   int chain_size = 0;
00914   while (current != chain_end) {
00915     if (chain_size > number_of_nexts_) {
00916       return false;
00917     }
00918     if (IsPathEnd(current)) {
00919       return false;
00920     }
00921     current = Next(current);
00922     ++chain_size;
00923     if (current == exclude) {
00924       return false;
00925     }
00926   }
00927   return true;
00928 }

01285 // Uses an exact dynamic programming algorithm to solve the TSP corresponding

01405 bool TSPLns::MakeNeighbor() {
01406   const int64 base_node = BaseNode(0);
01407   if (IsPathEnd(base_node)) {
01408     return false;
01409   }
01410   std::vector<int64> nodes;
01411   for (int64 node = StartNode(0); !IsPathEnd(node); node = Next(node)) {
01412     nodes.push_back(node);
01413   }
01414   if (nodes.size() <= tsp_size_) {
01415     return false;
01416   }
01417   // Randomly select break nodes (final nodes of a meta-node, after which
01418   // an arc is relaxed.
01419   hash_set<int64> breaks_set;
01420   // Always add base node to break nodes (diversification)
01421   breaks_set.insert(base_node);
01422   while (breaks_set.size() < tsp_size_) {
01423     const int64 one_break = nodes[rand_.Uniform(nodes.size())];
01424     if (!ContainsKey(breaks_set, one_break)) {
01425       breaks_set.insert(one_break);
01426     }
01427   }
01428   CHECK_EQ(breaks_set.size(), tsp_size_);
01429   // Setup break node indexing and internal meta-node cost (cost of partial
01430   // route starting at first node of the meta-node and ending at its last node);
01431   // this cost has to be added to the TSP matrix cost in order to respect the
01432   // triangle inequality.
01433   std::vector<int> breaks;
01434   std::vector<int64> meta_node_costs;
01435   int64 cost = 0;
01436   int64 node = StartNode(0);
01437   int64 node_path = Path(node);
01438   while (!IsPathEnd(node)) {
01439     int64 next = Next(node);
01440     if (ContainsKey(breaks_set, node)) {
01441       breaks.push_back(node);
01442       meta_node_costs.push_back(cost);
01443       cost = 0;
01444     } else {
01445       cost += evaluator_->Run(node, next, node_path);
01446     }
01447     node = next;
01448   }
01449   meta_node_costs[0] += cost;
01450   CHECK_EQ(breaks.size(), tsp_size_);
01451   // Setup TSP cost matrix
01452   CHECK_EQ(meta_node_costs.size(), tsp_size_);
01453   for (int i = 0; i < tsp_size_; ++i) {
01454     cost_[i][0] = meta_node_costs[i]
01455         + evaluator_->Run(breaks[i], Next(breaks[tsp_size_ - 1]), node_path);
01456     for (int j = 1; j < tsp_size_; ++j) {
01457       cost_[i][j] = meta_node_costs[i]
01458           + evaluator_->Run(breaks[i], Next(breaks[j - 1]), node_path);
01459     }
01460     cost_[i][i] = 0;
01461   }
01462   // Solve TSP and inject solution in delta (only if it leads to a new solution)
01463   hamiltonian_path_solver_.ChangeCostMatrix(cost_);
01464   std::vector<PathNodeIndex> path;
01465   hamiltonian_path_solver_.TravelingSalesmanPath(&path);
01466   bool nochange = true;
01467   for (int i = 0; i < path.size() - 1; ++i) {
01468     if (path[i] != i) {
01469       nochange = false;
01470       break;
01471     }
01472   }
01473   if (nochange) {
01474     return false;
01475   }
01476   CHECK_EQ(0, path[path.size() - 1]);
01477   for (int i = 0; i < tsp_size_ - 1; ++i) {
01478     SetNext(breaks[path[i]], OldNext(breaks[path[i + 1] - 1]), node_path);
01479   }
01480   SetNext(breaks[path[tsp_size_ - 1]],
01481           OldNext(breaks[tsp_size_ - 1]),
01482           node_path);
01483   return true;
01484 }

01542 void NearestNeighbors::ComputeNearest(int row) {
01543   // Find size_ nearest neighbors for row of index 'row'.
01544   const int path = path_operator_.Path(row);
01545   const IntVar* var = path_operator_.Var(row);
01546   const int64 var_min = var->Min();
01547   const int var_size = var->Max() - var_min + 1;
01548   scoped_array<int> neighbors(new int[var_size]);
01549   scoped_array<int64> row_data(new int64[var_size]);
01550   for (int i = 0; i < var_size; ++i) {
01551     const int index = i + var_min;
01552     neighbors[i] = index;
01553     row_data[i] = evaluator_->Run(row, index, path);
01554   }
01555 
01556   if (var_size > size_) {
01557     int start = 0;
01558     int end = var_size;
01559     int size = size_;
01560     while (size > 0) {
01561       int index = (end - start) / 2;
01562       Pivot(start, end, neighbors.get(), row_data.get(), &index);
01563       if (index - start >= size) {
01564         end = index;
01565       } else {
01566         start = index + 1;
01567         size -= start;
01568       }
01569     }
01570   }
01571 
01572   // Setup global neighbor matrix for row row_index
01573   for (int i = 0; i < std::min(size_, var_size); ++i) {
01574     neighbors_[row].push_back(neighbors[i]);
01575     std::sort(neighbors_[row].begin(), neighbors_[row].end());
01576   }
01577 }

01579 void NearestNeighbors::Pivot(int start,
01580                              int end,
01581                              int* neighbors,
01582                              int64* row,
01583                              int* index) {
01584   Swap(start, *index, neighbors, row);
01585   int j = start;
01586   for (int i = start + 1; i < end; ++i) {
01587     if (row[i] < row[j]) {
01588       Swap(j, i, neighbors, row);
01589       ++j;
01590       Swap(i, j, neighbors, row);
01591     }
01592   }
01593   *index = j;
01594 }

01596 void NearestNeighbors::Swap(int i, int j, int* neighbors, int64* row) {
01597   const int64 row_i = row[i];
01598   const int neighbor_i = neighbors[i];
01599   row[i] = row[j];
01600   neighbors[i] = neighbors[j];
01601   row[j] = row_i;
01602   neighbors[j] = neighbor_i;
01603 }

01655 bool LinKernighan::MakeNeighbor() {
01656   marked_.clear();
01657   int64 node = BaseNode(0);
01658   if (IsPathEnd(node)) return false;
01659   int64 path = Path(node);
01660   int64 base = node;
01661   int64 next = Next(node);
01662   if (IsPathEnd(next)) return false;
01663   int64 out = -1;
01664   int64 gain = 0;
01665   marked_.insert(node);
01666   if (topt_) {  // Try a 3opt first
01667     if (InFromOut(node, next, &out, &gain)) {
01668       marked_.insert(next);
01669       marked_.insert(out);
01670       const int64 node1 = out;
01671       if (IsPathEnd(node1)) return false;
01672       const int64 next1 = Next(node1);
01673       if (IsPathEnd(next1)) return false;
01674       if (InFromOut(node1, next1, &out, &gain)) {
01675         marked_.insert(next1);
01676         marked_.insert(out);
01677         if (MoveChain(out, node1, node)) {
01678           const int64 next_out = Next(out);
01679           int64 in_cost = evaluator_->Run(node, next_out, path);
01680           int64 out_cost = evaluator_->Run(out, next_out, path);
01681           if (gain - in_cost + out_cost > 0)
01682             return true;
01683           node = out;
01684           if (IsPathEnd(node)) {
01685             return false;
01686           }
01687           next = next_out;
01688           if (IsPathEnd(next)) {
01689             return false;
01690           }
01691         } else {
01692           return false;
01693         }
01694       } else {
01695         return false;
01696       }
01697     } else {
01698       return false;
01699     }
01700   }
01701   // Try 2opts
01702   while (InFromOut(node, next, &out, &gain)) {
01703     marked_.insert(next);
01704     marked_.insert(out);
01705     int64 chain_last;
01706     if (!ReverseChain(node, out, &chain_last)) {
01707       return false;
01708     }
01709     int64 in_cost = evaluator_->Run(base, chain_last, path);
01710     int64 out_cost = evaluator_->Run(chain_last, out, path);
01711     if (gain - in_cost + out_cost > 0) {
01712       return true;
01713     }
01714     node = chain_last;
01715     if (IsPathEnd(node)) {
01716       return false;
01717     }
01718     next = out;
01719     if (IsPathEnd(next)) {
01720       return false;
01721     }
01722   }
01723   return false;
01724 }

01727 
01728 bool LinKernighan::InFromOut(int64 in_i, int64 in_j, int64* out, int64* gain) {
01729   const std::vector<int>& nexts = neighbors_.Neighbors(in_j);
01730   int64 best_gain = kint64min;
01731   int64 path = Path(in_i);
01732   int64 out_cost = evaluator_->Run(in_i, in_j, path);
01733   const int64 current_gain = *gain + out_cost;
01734   for (int k = 0; k < nexts.size(); ++k) {
01735     const int64 next = nexts[k];
01736     if (next != in_j) {
01737       int64 in_cost = evaluator_->Run(in_j, next, path);
01738       int64 new_gain = current_gain - in_cost;
01739       if (new_gain > 0
01740           && next != Next(in_j)
01741           && marked_.count(in_j) == 0
01742           && marked_.count(next) == 0) {
01743         if (best_gain < new_gain) {
01744           *out = next;
01745           best_gain = new_gain;
01746         }
01747       }
01748     }
01749   }
01750   *gain = best_gain;
01751   return (best_gain > kint64min);
01752 }

02455 bool ObjectiveFilter::Accept(const Assignment* delta,
02456                              const Assignment* deltadelta) {
02457   if (delta == NULL) {
02458     return false;
02459   }
02460   int64 value = 0;
02461   if (!deltadelta->Empty()) {
02462     if (!incremental_) {
02463       value = Evaluate(delta, old_value_, cache_, true);
02464     } else {
02465       value = Evaluate(deltadelta, old_delta_value_, delta_cache_, true);
02466     }
02467     incremental_ = true;
02468   } else {
02469     if (incremental_) {
02470       for (int i = 0; i < primary_vars_size_; ++i) {
02471         delta_cache_[i] = cache_[i];
02472       }
02473       old_delta_value_ = old_value_;
02474     }
02475     incremental_ = false;
02476     value = Evaluate(delta, old_value_, cache_, false);
02477   }
02478   old_delta_value_ = value;
02479   int64 var_min = objective_->Min();
02480   int64 var_max = objective_->Max();
02481   if (delta->Objective() == objective_) {
02482     var_min = std::max(var_min, delta->ObjectiveMin());
02483     var_max = std::min(var_max, delta->ObjectiveMax());
02484   }
02485   switch (filter_enum_) {
02486     case Solver::LE: {
02487       return value <= var_max;
02488     }
02489     case Solver::GE: {
02490       return value >= var_min;
02491     }
02492     case Solver::EQ: {
02493       return value <= var_max && value >= var_min;
02494     }
02495     default: {
02496       LOG(ERROR) << "Unknown local search filter enum value";
02497       return false;
02498     }
02499   }
02500 }

02515 int64 ObjectiveFilter::Evaluate(const Assignment* delta,
02516                                 int64 current_value,
02517                                 const int64* const out_values,
02518                                 bool cache_delta_values) {
02519   if (current_value == kint64max) return current_value;
02520   op_->set_value(current_value);
02521   const Assignment::IntContainer& container = delta->IntVarContainer();
02522   const int size = container.Size();
02523   for (int i = 0; i < size; ++i) {
02524     const IntVarElement& new_element = container.Element(i);
02525     const IntVar* var = new_element.Var();
02526     int64 index = -1;
02527     if (FindIndex(var, &index) && index < primary_vars_size_) {
02528       op_->Remove(out_values[index]);
02529       int64 obj_value = 0LL;
02530       if (EvaluateElementValue(container, index, &i, &obj_value)) {
02531         op_->Update(obj_value);
02532         if (cache_delta_values) {
02533           delta_cache_[index] = obj_value;
02534         }
02535       }
02536     }
02537   }
02538   return op_->value();
02539 }

02638     const Assignment::IntContainer& container,
02639     int index,

02650         && secondary_var == container.Element(hint_index).Var()) {

02652           value_evaluator_->Run(index,
02653                                 value,
02654                                 container.Element(hint_index).Value());

02659                                 value,

02661     }

02664     const IntVar* var = element.Var();
02665     if (var->Bound() && secondary_var->Bound()) {
02666       *obj_value = value_evaluator_->Run(index,
02667                                          var->Min(),
02668                                          secondary_var->Min());

02670     }
02671   }

02797 Decision* FindOneNeighbor::Next(Solver* const solver) {
02798   CHECK(NULL != solver);
02799 
02800   if (original_limit_ != NULL) {
02801     limit_->Copy(original_limit_);
02802   }
02803 
02804   if (!neighbor_found_) {
02805     // Only called on the first call to Next(), reference_assignment_ has not
02806     // been synced with assignment_ yet
02807 
02808     // Keeping the code in case a performance problem forces us to
02809     // use the old code with a zero test on pool_.
02810     // reference_assignment_->Copy(assignment_);
02811     pool_->Initialize(assignment_);
02812     SynchronizeAll();
02813   }
02814 
02815   {
02816     // Another assignment is needed to apply the delta
02817     Assignment* assignment_copy =
02818         solver->MakeAssignment(reference_assignment_.get());
02819     int counter = 0;
02820 
02821     DecisionBuilder* restore =
02822         solver->MakeRestoreAssignment(assignment_copy);
02823     if (sub_decision_builder_) {
02824       restore = solver->Compose(restore, sub_decision_builder_);
02825     }
02826     Assignment* delta = solver->MakeAssignment();
02827     Assignment* deltadelta = solver->MakeAssignment();
02828     while (true) {
02829       delta->Clear();
02830       deltadelta->Clear();
02831       solver->TopPeriodicCheck();
02832       if (++counter >= FLAGS_cp_local_search_sync_frequency &&
02833           pool_->SyncNeeded(reference_assignment_.get())) {
02834         // TODO(user) : SyncNeed(assignment_) ?
02835         counter = 0;
02836         SynchronizeAll();
02837       }
02838 
02839       if (!limit_->Check()
02840           && ls_operator_->MakeNextNeighbor(delta, deltadelta)) {
02841         solver->neighbors_ += 1;
02842         // All filters must be called for incrementality reasons.
02843         // Empty deltas must also be sent to incremental filters; can be needed
02844         // to resync filters on non-incremental (empty) moves.
02845         // TODO(user): Don't call both if no filter is incremental and one
02846         // of them returned false.
02847         const bool mh_filter =
02848             AcceptDelta(solver->ParentSearch(), delta, deltadelta);
02849         const bool move_filter = FilterAccept(delta, deltadelta);
02850         if (mh_filter && move_filter) {
02851           solver->filtered_neighbors_ += 1;
02852           assignment_copy->Copy(reference_assignment_.get());
02853           assignment_copy->Copy(delta);
02854           if (solver->SolveAndCommit(restore)) {
02855             solver->accepted_neighbors_ += 1;
02856             assignment_->Store();
02857             neighbor_found_ = true;
02858             return NULL;
02859           }
02860         }
02861       } else {
02862         if (neighbor_found_) {
02863           AcceptNeighbor(solver->ParentSearch());
02864           // Keeping the code in case a performance problem forces us to
02865           // use the old code with a zero test on pool_.
02866           //          reference_assignment_->Copy(assignment_);
02867           pool_->RegisterNewSolution(assignment_);
02868           SynchronizeAll();
02869         } else {
02870           break;
02871         }
02872       }
02873     }
02874   }
02875   solver->Fail();
02876   return NULL;
02877 }

02879 bool FindOneNeighbor::FilterAccept(const Assignment* delta,
02880                                    const Assignment* deltadelta) {
02881   bool ok = true;
02882   for (int i = 0; i < filters_.size(); ++i) {
02883     if (filters_[i]->IsIncremental()) {
02884       ok = filters_[i]->Accept(delta, deltadelta) && ok;
02885     } else {
02886       ok = ok && filters_[i]->Accept(delta, deltadelta);
02887     }
02888   }
02889   return ok;
02890 }

02897   SynchronizeFilters(reference_assignment_.get());

03220 Decision* LocalSearch::Next(Solver* const solver) {
03221   CHECK(NULL != solver);
03222   CHECK_LT(0, nested_decisions_.size());
03223   if (!has_started_) {
03224     nested_decision_index_ = 0;
03225     solver->SaveAndSetValue(&has_started_, true);
03226   } else if (nested_decision_index_ < 0) {
03227     solver->Fail();
03228   }
03229   NestedSolveDecision* decision = nested_decisions_[nested_decision_index_];
03230   const int state = decision->state();
03231   switch (state) {
03232     case NestedSolveDecision::DECISION_FAILED: {
03233       if (!LocalOptimumReached(solver->ActiveSearch())) {
03234         nested_decision_index_ = -1;  // Stop the search
03235       }
03236       solver->Fail();
03237       return NULL;
03238     }
03239     case NestedSolveDecision::DECISION_PENDING: {
03240       // TODO(user): Find a way to make this balancing invisible to the
03241       // user (no increase in branch or fail counts for instance).
03242       const int32 kLocalSearchBalancedTreeDepth = 32;
03243       const int depth = solver->SearchDepth();
03244       if (depth < kLocalSearchBalancedTreeDepth) {
03245         return solver->balancing_decision();
03246       } else if (depth > kLocalSearchBalancedTreeDepth) {
03247         solver->Fail();
03248       }
03249       return decision;
03250     }
03251     case NestedSolveDecision::DECISION_FOUND: {
03252       // Next time go to next decision
03253       if (nested_decision_index_ + 1 < nested_decisions_.size()) {
03254         ++nested_decision_index_;
03255       }
03256       return NULL;
03257     }
03258     default: {
03259       LOG(ERROR) << "Unknown local search state";
03260       return NULL;
03261     }
03262   }
