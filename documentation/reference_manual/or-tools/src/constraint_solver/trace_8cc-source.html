00805   void PushDelayedInfo(const string& delayed) {
00806     contexes_.top().delayed_info.push_back(Info(delayed));
00807   }
00808 
00809   void PopDelayedInfo() {
00810     CHECK(!contexes_.top().delayed_info.empty());
00811     if (contexes_.top().delayed_info.back().displayed &&
00812         !contexes_.top().TopLevel()) {
00813       DecreaseIndent();
00814       LOG(INFO) << Indent() << "}";
00815     } else {
00816       contexes_.top().delayed_info.pop_back();
00817     }
00818   }
00819 
00820   void CheckNoDelayed() {
00821     CHECK(contexes_.top().delayed_info.empty());
00822   }
00823 
00824   void PrintDelayedString() {
00825     const std::vector<Info>& infos = contexes_.top().delayed_info;
00826     for (int i = 0; i < infos.size(); ++i) {
00827       const Info& info = infos[i];
00828       if (!info.displayed) {
00829         LOG(INFO) << Indent() << info.message << " {";
00830         IncreaseIndent();
00831         // Marks it as displayed.
00832         contexes_.top().delayed_info[i].displayed = true;
00833       }
00834     }
00835   }
00836 
00837   void DisplayModification(const string& to_print) {
00838     PrintDelayedString();
00839     if (contexes_.top().in_demon ||
00840         contexes_.top().in_constraint ||
00841         contexes_.top().in_decision_builder ||
00842         contexes_.top().in_decision ||
00843         contexes_.top().in_objective) {
00844       // Inside a demon, constraint, decision builder -> normal print.
00845       LOG(INFO) << Indent() << to_print;
00846     } else {
00847       // Top level, modification pushed by the objective.  This is a
00848       // hack. The SetMax or SetMin done by the objective happens in
00849       // the RefuteDecision callback of search monitors.  We cannot
00850       // easily differentiate that from the actual modifications done
00851       // by the Refute() call itself.  To distinguish that, we force
00852       // the print trace to be last in the list of monitors. Thus
00853       // modifications that happens at the top level before the
00854       // RefuteDecision() callbacks must be from the objective.
00855       // In that case, we push the in_objective context.
00856       CHECK(contexes_.top().TopLevel());
00857       DisplaySearch(StringPrintf("Objective -> %s", to_print.c_str()));
00858       IncreaseIndent();
00859       contexes_.top().in_objective = true;
00860     }
00861   }
00862 
00863   void DisplaySearch(const string& to_print) {
00864     const int solve_depth = solver()->SolveDepth();
00865     if (solve_depth <= 1) {
00866       LOG(INFO) << Indent() << "######## Top Level Search: " << to_print;
00867     } else {
00868       LOG(INFO) << Indent() << "######## Nested Search(" << solve_depth - 1
00869                 << "): " << to_print;
00870     }
00871   }

00873   string Indent() {
00874     string output = " @ ";
00875     for (int i = 0; i < contexes_.top().indent; ++i) {
00876       output.append("    ");
00877     }
00878     return output;
00879   }

00882     contexes_.top().indent++;
00883   }
00884 
00885   void DecreaseIndent() {
00886     contexes_.top().indent--;
00887   }

00889   void PushNestedContext() {
00890     const int initial_indent = contexes_.top().indent;
00891     contexes_.push(Context(initial_indent));
00892   }
