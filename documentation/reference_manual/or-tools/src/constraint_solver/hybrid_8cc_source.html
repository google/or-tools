00598   void SolveProblem() {
00599     if (objective_ != NULL) {
00600       switch (mp_solver_.Solve()) {
00601         case MPSolver::OPTIMAL: {
00602           const double obj_value = mp_solver_.objective_value();
00603           if (maximize_) {
00604             const int64 int_obj_value = static_cast<int64>(ceil(obj_value));
00605             objective_->SetMax(int_obj_value);
00606           } else {
00607             const int64 int_obj_value = static_cast<int64>(floor(obj_value));
00608             objective_->SetMin(int_obj_value);
00609           }
00610           break;
00611         }
00612         case MPSolver::FEASIBLE:
00613           break;
00614         case MPSolver::INFEASIBLE:
00615           solver()->Fail();
00616           break;
00617         case MPSolver::UNBOUNDED:
00618           LOG(INFO) << "Error: unbounded LP status.";
00619           break;
00620         case MPSolver::ABNORMAL:
00621           LOG(INFO) << "Error: abnormal LP status.";
00622           break;
00623         default:
