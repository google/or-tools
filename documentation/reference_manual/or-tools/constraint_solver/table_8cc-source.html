00048 
00049 // Tuples are indexed, we maintain a bitset for active tuples.
00050 
00051 // For each var and each value, we maintain a bitset mask of tuples
00052 // containing this value for this variable.
00053 
00054 // Propagation: When a value is removed, blank all active tuples using the

00294   virtual void InitialPropagate() {
00295     // Build active_ structure.
00296     for (int var_index = 0; var_index < arity_; ++var_index) {
00297       for (ConstIter<ValueBitset> it(masks_[var_index]); !it.at_end(); ++it) {
00298         if (!vars_[var_index]->Contains(it->first)) {
00299           for (int i = 0; i < length_; ++i) {
00300             active_tuples_[i] &= ~it->second[i];
00301           }

00306     for (int i = 0; i < length_; ++i) {
00307       if (active_tuples_[i] != 0) {

00313       solver()->Fail();

00315     // Remove unreached values.

00326       }

00635   bool Supported(int var_index, int64 value_index) {
00636     DCHECK_GE(var_index, 0);
00637     DCHECK_LT(var_index, arity_);
00638     DCHECK_GE(value_index, 0);
00639     DCHECK(masks_[var_index][value_index]);
00640     const uint64* const mask = masks_[var_index][value_index];
00641     const int support = supports_[var_index][value_index];
00642     if ((mask[support] & active_tuples_[support]) != 0) {
00643       return true;
00644     }
00645     const int loop_end = ends_[var_index][value_index];
00646     for (int offset = starts_[var_index][value_index];
00647          offset <= loop_end;
00648          ++offset) {
00649       if ((mask[offset] & active_tuples_[offset]) != 0) {
00650         supports_[var_index][value_index] = offset;
00651         return true;
00652       }
00653     }
00654     return false;
00655   }

00657   void Propagate() {
00658     // This methods scans all values of all variables to see if they
00659     // are still supported.
00660     // This method is not attached to any particular variable, but is pushed
00661     // at a delayed priority when Update(var_index) deems it necessary.
00662     memset(temp_mask_.get(), 0, length_ * sizeof(*temp_mask_.get()));
00663     for (int var_index = 0; var_index < arity_; ++var_index) {
00664       to_remove_.clear();
00665       IntVarIterator* const it = iterators_[var_index];
00666       for (it->Init(); it->Ok(); it->Next()) {
00667         const int64 value_index = it->Value() - original_min_[var_index];
00668         if (!Supported(var_index, value_index)) {
00669           to_remove_.push_back(it->Value());
00670         }
00671       }
00672       if (to_remove_.size() > 0) {
00673         vars_[var_index]->RemoveValues(to_remove_);
00674         // Actively remove unsupported bitsets from active_tuples_.
00675         for (int offset = 0; offset < length_; ++offset) {
00676           temp_mask_[offset] = 0;
00677         }
00678         for (ConstIter<std::vector<int64> > it(to_remove_); !it.at_end(); ++it) {
00679           const int64 value_index = (*it) - original_min_[var_index];
00680           const uint64* const mask = masks_[var_index][value_index];
00681           DCHECK(mask);
00682           UpdateTempMask(mask,
00683                          starts_[var_index][value_index],
00684                          ends_[var_index][value_index]);
00685         }
00686         for (int offset = 0; offset < length_; ++offset) {
00687           if ((temp_mask_[offset] & active_tuples_[offset]) != 0) {
00688             AndActiveTuples(offset, ~temp_mask_[offset]);
00689           }
00690         }
00691       }
00692     }
00693     for (int offset = 0; offset < length_; ++offset) {
00694       if (active_tuples_[offset]) {
00695         return;
00696       }
00697     }
00698     solver()->Fail();
00699   }

00701   void UpdateTempMask(const uint64* const mask, int start, int end) {
00702     for (int offset = start; offset <= end; ++offset) {
00703       temp_mask_[offset] |= mask[offset];
00704     }
00705   }

00707   void Update(int var_index) {
00708     // This method will update the set of active tuples by masking out all
00709     // tuples attached to values of the variables that have been removed.
00710 
00711     // We first collect the complete set of tuples to blank out in temp_mask_.
00712     IntVar* const var = vars_[var_index];
00713     const int64 omin = original_min_[var_index];
00714     bool changed = false;
00715     memset(temp_mask_.get(), 0, length_ * sizeof(*temp_mask_.get()));
00716     const int64 oldmax = var->OldMax();
00717     const int64 vmin = var->Min();
00718     const int64 vmax = var->Max();
00719     for (int64 value = var->OldMin(); value < vmin; ++value) {
00720       const int64 value_index = value  - omin;
00721       const uint64* const mask = masks_[var_index][value_index];
00722       if (mask) {
00723         UpdateTempMask(mask,
00724                        starts_[var_index][value_index],
00725                        ends_[var_index][value_index]);
00726       }
00727     }
00728     IntVarIterator* const hole = holes_[var_index];
00729     for (hole->Init(); hole->Ok(); hole->Next()) {
00730       const int64 value_index = hole->Value() - omin;
00731       const uint64* const mask = masks_[var_index][value_index];
00732       if (mask) {
00733         UpdateTempMask(mask,
00734                        starts_[var_index][value_index],
00735                        ends_[var_index][value_index]);
00736       }
00737     }
00738     for (int64 value = vmax + 1; value <= oldmax; ++value) {
00739       const int64 value_index = value  - omin;
00740       const uint64* const mask = masks_[var_index][value_index];
00741       if (mask) {
00742         UpdateTempMask(mask,
00743                        starts_[var_index][value_index],
00744                        ends_[var_index][value_index]);
00745       }
00746     }
00747     // Then we apply this mask to active_tuples_.
00748     for (int offset = 0; offset < length_; ++offset) {
00749       if ((temp_mask_[offset] & active_tuples_[offset]) != 0) {
00750         AndActiveTuples(offset, ~temp_mask_[offset]);
00751         changed = true;
00752       }
00753     }
00754     // And check active_tuples_ is still not empty, we fail otherwise.
00755     for (int offset = 0; offset < length_; ++offset) {
00756       if (active_tuples_[offset]) {
00757         // We push the propagate method only if something has changed.
00758         if (changed) {
00759           Enqueue(demon_);
00760         }
00761         return;
00762       }
00763     }
00764     solver()->Fail();
00765   }
00766 
00767  private:
00768   void AndActiveTuples(int offset, uint64 mask) {
00769     const uint64 current_stamp = solver()->stamp();
00770     if (stamps_[offset] < current_stamp) {
00771       stamps_[offset] = current_stamp;
00772       solver()->SaveValue(&active_tuples_[offset]);
00773     }
00774     active_tuples_[offset] &= mask;
00775   }
00776 
00777   // Length of bitsets in double words.
00778   const int length_;
00779   // Bitset of active tuples.
00780   scoped_array<uint64> active_tuples_;
00781   // Array of stamps, one per 64 tuples.
00782   scoped_array<uint64> stamps_;
00783   // The masks per value per variable.
00784   std::vector<std::vector<uint64*> > masks_;
00785   // The min on the vars at creation time.
00786   scoped_array<int64> original_min_;
00787   // The starts of active bitsets.
00788   std::vector<std::vector<int> > starts_;
00789   // The ends of the active bitsets.x
00790   std::vector<std::vector<int> > ends_;
00791   // A temporary mask use for computation.
00792   scoped_array<uint64> temp_mask_;
00793   // The portion of the active tuples supporting each value per variable.
00794   std::vector<std::vector<int> > supports_;
00795   Demon* demon_;
00796 };

00965     // are still supported.

00985           const int64 value = it->Value();
00986           if ((var_mask[value - original_min] & actives) == 0) {
00987             to_remove_.push_back(value);
00988           }

01151 // This constraint implements a finite automaton when transitions are
01152 // the values of the variables in the array.
01153 // that is state[i+1] = transition[var[i]][state[i]] if
01154 // (state[i], var[i],  state[i+1]) in the transition table.

01161   TransitionConstraint(Solver* const s,

01164                        int64 initial_state,
01165                        const std::vector<int64>& final_states)
01166       : Constraint(s),
01167         vars_(vars),
01168         transition_count_(transition_table.size()),
01169         transition_table_(new int64*[transition_count_]),
01170         initial_state_(initial_state),
01171         final_states_(final_states) {
01172     // Copy tuples
01173     for (int i = 0; i < transition_table.size(); ++i) {
01174       CHECK_EQ(kTransitionTupleSize, transition_table[i].size());
01175       transition_table_[i] = new int64[kTransitionTupleSize];
01176       memcpy(transition_table_[i],
01177              transition_table[i].data(),
01178              kTransitionTupleSize * sizeof(transition_table[i][0]));
01179     }
01180   }

01183                        const std::vector<IntVar*>& vars,

01191         initial_state_(initial_state),

01193     // Copy tuples
01194     for (int i = 0; i < transition_count_; ++i) {
01195       transition_table_[i] = new int64[kTransitionTupleSize];
01196       for (int j = 0; j < kTransitionTupleSize; ++j) {

01204   }

01213   virtual void Post() {
01214     Solver* const s = solver();
01215     int64 state_min = kint64max;
01216     int64 state_max = kint64min;
01217     const int nb_vars = vars_.size();
01218     for (int i = 0; i < transition_count_; ++i) {
01219       state_max = std::max(state_max, transition_table_[i][kStatePosition]);
01220       state_max = std::max(state_max, transition_table_[i][kNextStatePosition]);
01221       state_min = std::min(state_min, transition_table_[i][kStatePosition]);
01222       state_min = std::min(state_min, transition_table_[i][kNextStatePosition]);
01223     }
01224 
01225     std::vector<IntVar*> states;
01226     states.push_back(s->MakeIntConst(initial_state_));
01227     for (int var_index = 1; var_index < nb_vars; ++var_index) {
01228       states.push_back(s->MakeIntVar(state_min, state_max));
01229     }
01230     states.push_back(s->MakeIntVar(final_states_));
01231     CHECK_EQ(nb_vars + 1, states.size());
01232 
01233     for (int var_index = 0; var_index < nb_vars; ++var_index) {
01234       std::vector<IntVar*> tmp_vars;
01235       tmp_vars.push_back(states[var_index]);
01236       tmp_vars.push_back(vars_[var_index]);
01237       tmp_vars.push_back(states[var_index + 1]);
01238       s->AddConstraint(s->MakeAllowedAssignments(tmp_vars.data(),
01239                                                  transition_table_.get(),
01240                                                  transition_count_,
01241                                                  kTransitionTupleSize));
01242     }
01243   }
01244 
01245   virtual void InitialPropagate() {}
01246 
01247   virtual void Accept(ModelVisitor* const visitor) const {
01248     visitor->BeginVisitConstraint(ModelVisitor::kTransition, this);
01249     visitor->VisitIntegerVariableArrayArgument(ModelVisitor::kVarsArgument,
01250                                                vars_.data(),
01251                                                vars_.size());
01252     visitor->VisitIntegerArgument(ModelVisitor::kInitialState,
01253                                   initial_state_);
01254     visitor->VisitIntegerArrayArgument(ModelVisitor::kFinalStatesArgument,
01255                                        final_states_.data(),
01256                                        final_states_.size());
01257     visitor->VisitIntegerMatrixArgument(ModelVisitor::kTuplesArgument,
01258                                         transition_table_.get(),
01259                                         transition_count_,
01260                                         kTransitionTupleSize);
01261     visitor->EndVisitConstraint(ModelVisitor::kTransition, this);
01262   }
