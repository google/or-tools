00451 IntExpr* BuildElement(Solver* const solver,
00452                       ConstIntArray* values,
00453                       IntVar* const index) {
00454   // Various checks.
00455   // Is array constant?
00456   if (values->HasProperty(ConstIntArray::IS_CONSTANT)) {
00457     solver->AddConstraint(solver->MakeBetweenCt(index, 0, values->size() - 1));
00458     return solver->MakeIntConst(values->get(0));
00459   }
00460   // Is array built with booleans only?
00461   // TODO(user): We could maintain the index of the first one.
00462   if (values->HasProperty(ConstIntArray::IS_BOOLEAN)) {
00463     std::vector<int64> ones;
00464     int first_zero = -1;
00465     for (int i = 0; i < values->size(); ++i) {
00466       if (values->get(i) == 1LL) {
00467         ones.push_back(i);
00468       } else {
00469         first_zero = i;
00470       }
00471     }
00472     if (ones.size() == 1) {
00473       DCHECK_EQ(1LL, values->get(ones.back()));
00474       solver->AddConstraint(
00475           solver->MakeBetweenCt(index, 0, values->size() - 1));
00476       return solver->MakeIsEqualCstVar(index, ones.back());
00477     } else if (ones.size() == values->size() - 1) {
00478       solver->AddConstraint(
00479           solver->MakeBetweenCt(index, 0, values->size() - 1));
00480       return solver->MakeIsDifferentCstVar(index, first_zero);
00481     } else if (ones.size() == ones.back() - ones.front() + 1) {  // contiguous.
00482       solver->AddConstraint(
00483           solver->MakeBetweenCt(index, 0, values->size() - 1));
00484       IntVar* const b = solver->MakeBoolVar("ContiguousBooleanElementVar");
00485       solver->AddConstraint(
00486           solver->MakeIsBetweenCt(index, ones.front(), ones.back(), b));
00487       return b;
00488     } else {
00489       IntVar* const b = solver->MakeBoolVar("NonContiguousBooleanElementVar");
00490       solver->AddConstraint(
00491           solver->MakeBetweenCt(index, 0, values->size() - 1));
00492       solver->AddConstraint(solver->MakeIsMemberCt(index, ones, b));
00493       return b;
00494     }
00495   }
00496   // Is Array increasing
00497   if (values->HasProperty(ConstIntArray::IS_INCREASING)) {
00498     return solver->RegisterIntExpr(solver->RevAlloc(
00499         new IncreasingIntExprElement(solver, values->Release(), index)));
00500   }
00501   return solver->RegisterIntExpr(solver->RevAlloc(
00502       new IntExprElement(solver, values->Release(), index)));
00503 }

01283   const int64 emax = std::min(size_ - 1LL, var_->Max());
01284   if (emin == emax) {
01285     var_->SetValue(emin);  // in case it was reduced by the above min/max.
01286     vars_[emin]->SetMax(m);
01287   } else {
01288     int64 nmin = emin;
01289     while (nmin <= emax && vars_[nmin]->Min() > m) {
01290       nmin++;
01291     }
01292     if (nmin > emax) {
01293       solver()->Fail();
01294     }
01295     int64 nmax = emax;
01296     while (nmax >= nmin && vars_[nmax]->Min() > m) {
01297       nmax--;
01298     }
01299     var_->SetRange(nmin, nmax);
01300     if (var_->Bound()) {
01301       vars_[var_->Min()]->SetMax(m);
01302     }
01303   }
01304 }

01306 void IntExprArrayElement::SetRange(int64 mi, int64 ma) {
01307   if (mi > ma) {
01308     solver()->Fail();
01309   }
01310   const int64 emin = std::max(0LL, var_->Min());
01311   const int64 emax = std::min(size_ - 1LL, var_->Max());
01312   if (emin == emax) {
01313     var_->SetValue(emin);  // in case it was reduced by the above min/max.
01314     vars_[emin]->SetRange(mi, ma);
01315   } else {
01316     int64 nmin = emin;
01317     while (nmin <= emax && (vars_[nmin]->Min() > ma ||
01318                             vars_[nmin]->Max() < mi)) {
01319       nmin++;
01320     }
01321     if (nmin > emax) {
01322       solver()->Fail();
01323     }
01324     int64 nmax = emax;
01325     while (nmax >= nmin && (vars_[nmax]->Max() < mi ||
01326                             vars_[nmax]->Min() > ma)) {
01327       nmax--;
01328     }
01329     if (nmax < emin) {
01330       solver()->Fail();
01331     }
01332     var_->SetRange(nmin, nmax);
01333     if (var_->Bound()) {
01334       vars_[var_->Min()]->SetRange(mi, ma);
01335     }
01336   }

01339 bool IntExprArrayElement::Bound() const {
01340   const int64 emin = std::max(0LL, var_->Min());
01341   const int64 emax = std::min(size_ - 1LL, var_->Max());
01342   const int64 v = vars_[emin]->Min();
01343   for (int i = emin; i <= emax; ++i) {
01344     if (var_->Contains(i) && (!vars_[i]->Bound() || vars_[i]->Value() != v)) {
01345       return false;
01346     }
01347   }
01348   return true;
