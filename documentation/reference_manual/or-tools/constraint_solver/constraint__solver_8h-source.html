00899   // defined as a leaf of the search tree with respect to the given decision

00902   // the responsibility of the decision builder to keep returning decisions
00903   // until all variables are indeed bound. The most extreme counterexample is
00904   // calling Solve with a trivial decision builder whose Next() method always

00906   // assigning any variable to any value is a solution, unless the root node

04459 template <class V, class E> class AssignmentContainer {
04460  public:
04461   AssignmentContainer() {}
04462   E& Add(V* const var) {
04463     CHECK_NOTNULL(var);
04464     int index = -1;
04465     if (!Find(var, &index)) {
04466       return FastAdd(var);
04467     } else {
04468       return elements_[index];
04469     }
04470   }
04471   // Adds element without checking its presence in the container.
04472   E& FastAdd(V* const var) {
04473     DCHECK(var != NULL);
04474     E e(var);
04475     elements_.push_back(e);
04476     return elements_.back();
04477   }
04478   void Clear() {
04479     elements_.clear();
04480     if (!elements_map_.empty()) {  // 2x speedup on or-tools.
04481       elements_map_.clear();
04482     }
04483   }
04484   bool Empty() const {
04485     return elements_.empty();
04486   }
04487   // Copies intersection of containers.
04488   void Copy(const AssignmentContainer<V, E>& container) {
04489     for (int i = 0; i < container.elements_.size(); ++i) {
04490       const E& element = container.elements_[i];
04491       const V* const var = element.Var();
04492       int index = -1;
04493       if (i < elements_.size() && elements_[i].Var() == var) {
04494         index = i;
04495       } else if (!Find(var, &index)) {
04496         continue;
04497       }
04498       DCHECK_GE(index, 0);
04499       E& local_element(elements_[index]);
04500       local_element.Copy(element);
04501       if (element.Activated()) {
04502         local_element.Activate();
04503       } else {
04504         local_element.Deactivate();
04505       }
04506     }
04507   }
04508   bool Contains(const V* const var) const {
04509     int index;
04510     return Find(var, &index);
04511   }
04512   E& MutableElement(const V* const var) {
04513     int index = -1;
04514     const bool found = Find(var, &index);
04515     CHECK(found) << "Unknown variable " << var->DebugString() << " in solution";
04516     return MutableElement(index);
04517   }
04518   const E& Element(const V* const var) const {
04519     int index = -1;
04520     const bool found = Find(var, &index);
04521     CHECK(found) << "Unknown variable " << var->DebugString() << " in solution";
04522     return Element(index);
04523   }
04524   const std::vector<E>& elements() const { return elements_; }
04525   E& MutableElement(int index) { return elements_[index]; }
04526   const E& Element(int index) const { return elements_[index]; }
04527   int Size() const { return elements_.size(); }
04528   void Store() {
04529     for (int i = 0; i < elements_.size(); ++i) {
04530       elements_[i].Store();
04531     }
04532   }
04533   void Restore() {
04534     for (int i = 0; i < elements_.size(); ++i) {
04535       E& element = elements_[i];
04536       if (element.Activated()) {
04537         element.Restore();
04538       }
04539     }
04540   }
04541 
04542   // Returns true if this and 'container' both represent the same V* -> E map.
04543   // Runs in linear time; requires that the == operator on the type E is well
04544   // defined.
04545   bool operator==(const AssignmentContainer<V, E>& container) const {
04546     // We may not have any work to do
04547     if (Size() != container.Size()) {
04548       return false;
04549     }
04550     // The == should be order-independent
04551     EnsureMapIsUpToDate();
04552     // Do not use the hash_map::== operator! It does not just compare content,
04553     // but also how the map is hashed (e.g., number of buckets). This is not
04554     // what we want.
04555     typedef ConstIter<std::vector<E> > Iterator;
04556     for (Iterator it(container.elements_); !it.at_end(); ++it) {
04557       const int position = FindWithDefault(elements_map_, it->Var(), -1);
04558       if (position < 0 || elements_[position] != *it) {
04559         return false;
04560       }
04561     }
04562     return true;
04563   }
04564   bool operator!=(const AssignmentContainer<V, E>& container) const {
04565     return !(*this == container);
04566   }
04567 
04568  private:
04569   void EnsureMapIsUpToDate() const {
04570     hash_map<const V*, int>* map =
04571         const_cast<hash_map<const V*, int>* >(&elements_map_);
04572     for (int i = map->size(); i < elements_.size(); ++i) {
04573       (*map)[elements_[i].Var()] = i;
04574     }
04575   }
04576   bool Find(const V* const var, int* index) const {
04577     EnsureMapIsUpToDate();
04578     DCHECK_EQ(elements_map_.size(), elements_.size());
04579     return FindCopy(elements_map_, var, index);
04580   }
04581 
04582   std::vector<E> elements_;
